def _compute_signature():
    # Compute signature of ufc header files
    h = hashlib.sha1()
    with open(os.path.join(get_include_path(), "ufcx.h")) as f:
        h.update(f.read().encode("utf-8"))
    return h.hexdigest()


_signature = _compute_signature()
