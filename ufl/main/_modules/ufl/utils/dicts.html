[docs]def split_dict(d, criteria):
    """Split a dict d into two dicts based on a criteria on the keys."""
    a = {}
    b = {}
    for (k, v) in d.items():
        if criteria(k):
            a[k] = v
        else:
            b[k] = v
    return a, b

[docs]def dict_sum(items):
    """Construct a dict, in between dict(items) and sum(items), by accumulating items for each key."""
    d = {}
    for k, v in items:
        if k not in d:
            d[k] = v
        else:
            d[k] += v
    return d
