class StackDict(dict):
    """A dictionary type.

    A dict that can be changed incrementally with 'd.push(k,v)' and have
    changes rolled back with 'k,v = d.pop()'.
    """

    def __init__(self, *args, **kwargs):
        """Initialise."""
        dict.__init__(self, *args, **kwargs)
        self._l = []



[docs]
    def push(self, k, v):
        """Store previous state for this key."""
        self._l.append((k, self.get(k, None)))
        if v is None:
            if k in self:
                del self[k]
        else:
            self[k] = v

    def pop(self):
        """Restore previous state for this key."""
        k, v = self._l.pop()
        if v is None:
            if k in self:
                del self[k]
        else:
            self[k] = v
        return k, v
