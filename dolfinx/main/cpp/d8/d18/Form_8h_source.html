112  switch (type)

358  // Helper function to get a std::vector of (cell, local_facet) pairs
359  // corresponding to a given facet index.
360  // @param[in] f Facet index
361  // @param[in] f_to_c Facet to cell connectivity
362  // @param[in] c_to_f Cell to facet connectivity
363  // @return Vector of (cell, local_facet) pairs
364  template <int num_cells>
365  static std::array<std::pair<std::int32_t, int>, num_cells>
366  get_cell_local_facet_pairs(
367  std::int32_t f, const xtl::span<const std::int32_t>& cells,
368  const dolfinx::graph::AdjacencyList<std::int32_t>& c_to_f)
369  {
370  // Loop over cells sharing facet
371  assert(cells.size() == num_cells);
372  std::array<std::pair<std::int32_t, int>, num_cells> cell_local_facet_pairs;
373  for (int c = 0; c < num_cells; ++c)
374  {
375  // Get local index of facet with respect to the cell
376  std::int32_t cell = cells[c];
377  auto cell_facets = c_to_f.links(cell);
378  auto facet_it = std::find(cell_facets.begin(), cell_facets.end(), f);
379  assert(facet_it != cell_facets.end());
380  int local_f = std::distance(cell_facets.begin(), facet_it);
381  cell_local_facet_pairs[c] = {cell, local_f};
382  }
383 
384  return cell_local_facet_pairs;
385  }

387  // Set cell domains
388  template <typename iterator>
389  void set_cell_domains(
390  std::map<int, std::pair<kern, std::vector<std::int32_t>>>& integrals,
391  const iterator& tagged_cells_begin, const iterator& tagged_cells_end,
392  const std::vector<int>& tags)
393  {
394  // For cell integrals use all markers (but not on ghost entities)
395  for (auto c = tagged_cells_begin; c != tagged_cells_end; ++c)
396  {
397  const std::size_t pos = std::distance(tagged_cells_begin, c);
398  if (auto it = integrals.find(tags[pos]); it != integrals.end())
399  it->second.second.push_back(*c);
400  }
401  }

403  // Set exterior facet domains
404  template <typename iterator>
405  void set_exterior_facet_domains(
406  const mesh::Topology& topology,
407  std::map<int, std::pair<kern, std::vector<std::pair<std::int32_t, int>>>>&
408  integrals,
409  const iterator& tagged_facets_begin, const iterator& tagged_facets_end,
410  const std::vector<int>& tags)
411  {
412  // When a mesh is not ghosted by cell, it is not straightforward
413  // to distinguish between (i) exterior facets and (ii) interior
414  // facets that are on a partition boundary. If there are no
415  // ghost cells, build a set of owned facts that are ghosted on
416  // another process to help determine if a facet is on an
417  // exterior boundary.
418  int tdim = topology.dim();
419  assert(topology.index_map(tdim));
420  std::set<std::int32_t> fwd_shared_facets;
421  if (topology.index_map(tdim)->num_ghosts() == 0)
422  {
423  const std::vector<std::int32_t>& fwd_indices
424  = topology.index_map(tdim - 1)->scatter_fwd_indices().array();
425  fwd_shared_facets.insert(fwd_indices.begin(), fwd_indices.end());
426  }
427 
428  auto f_to_c = topology.connectivity(tdim - 1, tdim);
429  assert(f_to_c);
430  auto c_to_f = topology.connectivity(tdim, tdim - 1);
431  assert(c_to_f);
432  for (auto f = tagged_facets_begin; f != tagged_facets_end; ++f)
433  {
434  // All "owned" facets connected to one cell, that are not
435  // shared, should be external
436  // TODO: Consider removing this check and integrating over all
437  // tagged facets. This may be useful in a few cases.
438  if (f_to_c->num_links(*f) == 1
439  and fwd_shared_facets.find(*f) == fwd_shared_facets.end())
440  {
441  const std::size_t pos = std::distance(tagged_facets_begin, f);
442  if (auto it = integrals.find(tags[pos]); it != integrals.end())
443  {
444  // There will only be one pair for an exterior facet integral
445  std::pair<std::int32_t, int> pair = get_cell_local_facet_pairs<1>(
446  *f, f_to_c->links(*f), *c_to_f)[0];
447  it->second.second.push_back(pair);
448  }
449  }
450  }
451  }

453  // Set interior facet domains
454  template <typename iterator>
455  static void set_interior_facet_domains(
456  const mesh::Topology& topology,
457  std::map<int,
458  std::pair<kern, std::vector<std::tuple<std::int32_t, int,
459  std::int32_t, int>>>>&
460  integrals,
461  const iterator& tagged_facets_begin, const iterator& tagged_facets_end,
462  const std::vector<int>& tags)
463  {
464  int tdim = topology.dim();
465  auto f_to_c = topology.connectivity(tdim - 1, tdim);
466  assert(f_to_c);
467  auto c_to_f = topology.connectivity(tdim, tdim - 1);
468  assert(c_to_f);
469  for (auto f = tagged_facets_begin; f != tagged_facets_end; ++f)
470  {
471  if (f_to_c->num_links(*f) == 2)
472  {
473  const std::size_t pos = std::distance(tagged_facets_begin, f);
474  if (auto it = integrals.find(tags[pos]); it != integrals.end())
475  {
476  std::array<std::pair<std::int32_t, int>, 2> pairs
477  = get_cell_local_facet_pairs<2>(*f, f_to_c->links(*f), *c_to_f);
478  it->second.second.emplace_back(pairs[0].first, pairs[0].second,
479  pairs[1].first, pairs[1].second);
480  }
481  }
482  }
483  }

485  // Sets the entity indices to assemble over for kernels with a domain
486  // ID
487  // @param[in] type Integral type
488  // @param[in] marker MeshTags with domain ID. Entities with marker 'i'
489  // will be assembled over using the kernel with ID 'i'. The MeshTags
490  // is not stored.
491  void set_domains(IntegralType type, const mesh::MeshTags<int>& marker)
492  {
493  std::shared_ptr<const mesh::Mesh> mesh = marker.mesh();
494  const mesh::Topology& topology = mesh->topology();
495  const int tdim = topology.dim();
496  int dim = type == IntegralType::cell ? tdim : tdim - 1;
497  if (dim != marker.dim())
498  {
499  throw std::runtime_error("Invalid MeshTags dimension: "
500  + std::to_string(marker.dim()));
501  }
502 
503  // Get mesh tag data
504  const std::vector<int>& tags = marker.values();
505  const std::vector<std::int32_t>& tagged_entities = marker.indices();
506  assert(topology.index_map(dim));
507  const auto entity_end
508  = std::lower_bound(tagged_entities.begin(), tagged_entities.end(),
509  topology.index_map(dim)->size_local());
510  switch (type)
511  {
512  case IntegralType::cell:
513  set_cell_domains(_cell_integrals, tagged_entities.cbegin(), entity_end,
514  tags);
515  break;
516  default:
517  mesh->topology_mutable().create_connectivity(dim, tdim);
518  mesh->topology_mutable().create_connectivity(tdim, dim);
519  switch (type)
520  {
521  case IntegralType::exterior_facet:
522  set_exterior_facet_domains(topology, _exterior_facet_integrals,
523  tagged_entities.cbegin(), entity_end, tags);
524  break;
525  case IntegralType::interior_facet:
526  set_interior_facet_domains(topology, _interior_facet_integrals,
527  tagged_entities.cbegin(), entity_end, tags);
528  break;
529  default:
530  throw std::runtime_error(
531  "Cannot set domains. Integral type not supported.");
532  }
533  }
534  }

541  void set_default_domains(const mesh::Mesh& mesh)
542  {
543  const mesh::Topology& topology = mesh.topology();
544  const int tdim = topology.dim();
545 
546  // Cells. If there is a default integral, define it on all owned
547  // cells
548  for (auto& [domain_id, kernel_cells] : _cell_integrals)
549  {
550  if (domain_id == -1)
551  {
552  std::vector<std::int32_t>& cells = kernel_cells.second;
553  const int num_cells = topology.index_map(tdim)->size_local();
554  cells.resize(num_cells);
555  std::iota(cells.begin(), cells.end(), 0);
556  }
557  }
558 
559  // Exterior facets. If there is a default integral, define it only
560  // on owned surface facets.
561  for (auto& [domain_id, kernel_facets] : _exterior_facet_integrals)
562  {
563  if (domain_id == -1)
564  {
565  std::vector<std::pair<std::int32_t, int>>& facets
566  = kernel_facets.second;
567  facets.clear();
568 
569  mesh.topology_mutable().create_connectivity(tdim - 1, tdim);
570  auto f_to_c = topology.connectivity(tdim - 1, tdim);
571  assert(f_to_c);
572 
573  mesh.topology_mutable().create_connectivity(tdim, tdim - 1);
574  auto c_to_f = mesh.topology().connectivity(tdim, tdim - 1);
575  assert(c_to_f);
576 
577  // Only need to consider shared facets when there are no ghost
578  // cells
579  std::set<std::int32_t> fwd_shared_facets;
580  assert(topology.index_map(tdim - 1));
581  if (topology.index_map(tdim)->num_ghosts() == 0)
582  {
583  const std::vector<std::int32_t>& fwd_indices
584  = topology.index_map(tdim - 1)->scatter_fwd_indices().array();
585  fwd_shared_facets.insert(fwd_indices.begin(), fwd_indices.end());
586  }
587 
588  const int num_facets = topology.index_map(tdim - 1)->size_local();
589  for (int f = 0; f < num_facets; ++f)
590  {
591  if (f_to_c->num_links(f) == 1
592  and fwd_shared_facets.find(f) == fwd_shared_facets.end())
593  {
594  // There will only be one pair for an exterior facet
595  // integral
596  std::pair<std::int32_t, int> pair = get_cell_local_facet_pairs<1>(
597  f, f_to_c->links(f), *c_to_f)[0];
598  facets.push_back(pair);
599  }
600  }
601  }
602  }
603 
604  // Interior facets. If there is a default integral, define it only on
605  // owned interior facets.
606  for (auto& [domain_id, kernel_facets] : _interior_facet_integrals)
607  {
608  if (domain_id == -1)
609  {
610  std::vector<std::tuple<std::int32_t, int, std::int32_t, int>>& facets
611  = kernel_facets.second;
612  facets.clear();
613 
614  mesh.topology_mutable().create_connectivity(tdim - 1, tdim);
615  auto f_to_c = topology.connectivity(tdim - 1, tdim);
616  assert(f_to_c);
617  mesh.topology_mutable().create_connectivity(tdim, tdim - 1);
618  auto c_to_f = mesh.topology().connectivity(tdim, tdim - 1);
619  assert(c_to_f);
620 
621  // Get number of facets owned by this process
622  assert(topology.index_map(tdim - 1));
623  const int num_facets = topology.index_map(tdim - 1)->size_local();
624  facets.reserve(num_facets);
625  for (int f = 0; f < num_facets; ++f)
626  {
627  if (f_to_c->num_links(f) == 2)
628  {
629  std::array<std::pair<std::int32_t, int>, 2> pairs
630  = get_cell_local_facet_pairs<2>(f, f_to_c->links(f), *c_to_f);
631  facets.emplace_back(pairs[0].first, pairs[0].second, pairs[1].first,
632  pairs[1].second);
633  }
634  }
635  }
636  }
637  }
