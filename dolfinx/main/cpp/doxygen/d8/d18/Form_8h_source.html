157 template <typename X>
158 requires std::is_convertible_v<

160 std::map<IntegralType, std::vector<integral_data<
161 scalar_type, geometry_type>>>>
162 Form(
163 const std::vector<std::shared_ptr<const FunctionSpace<geometry_type>>>& V,
164 X&& integrals,
165 const std::vector<
166 std::shared_ptr<const Function<scalar_type, geometry_type>>>&
167 coefficients,
168 const std::vector<std::shared_ptr<const Constant<scalar_type>>>&
169 constants,
170 bool needs_facet_permutations,
171 const std::map<std::shared_ptr<const mesh::Mesh<geometry_type>>,
172 std::span<const std::int32_t>>& entity_maps,
173 std::shared_ptr<const mesh::Mesh<geometry_type>> mesh = nullptr)
174 : _function_spaces(V), _coefficients(coefficients), _constants(constants),
175 _mesh(mesh), _needs_facet_permutations(needs_facet_permutations)
176 {
177 // Extract _mesh from FunctionSpace, and check they are the same
178 if (!_mesh and !V.empty())
179 _mesh = V[0]->mesh();
180 for (auto& space : V)
181 {
182 if (_mesh != space->mesh()
183 and entity_maps.find(space->mesh()) == entity_maps.end())
184 {
185 throw std::runtime_error(
186 "Incompatible mesh. entity_maps must be provided.");
187 }
188 }
189 if (!_mesh)
190 throw std::runtime_error("No mesh could be associated with the Form.");
191
192 // Store kernels, looping over integrals by domain type (dimension)
193 for (auto&& [domain_type, data] : integrals)
194 {
195 if (!std::is_sorted(data.begin(), data.end(),
196 [](auto& a, auto& b) { return a.id < b.id; }))
197 {
198 throw std::runtime_error("Integral IDs not sorted");
199 }
200
201 std::vector<integral_data<scalar_type, geometry_type>>& itg
202 = _integrals[static_cast<std::size_t>(domain_type)];
203 for (auto&& [id, kern, e] : data)
204 itg.emplace_back(id, kern, std::move(e));
205 }
206
207 // Store entity maps
208 for (auto [msh, map] : entity_maps)
209 _entity_maps.insert({msh, std::vector(map.begin(), map.end())});
210 }

250 kernel(IntegralType type, int i) const
251 {
252 const auto& integrals = _integrals[static_cast<std::size_t>(type)];
253 auto it = std::lower_bound(integrals.begin(), integrals.end(), i,
254 [](auto& itg_data, int i)
255 { return itg_data.id < i; });

257 return it->kernel;

317 std::span<const std::int32_t> domain(IntegralType type, int i) const
318 {
319 const auto& integrals = _integrals[static_cast<std::size_t>(type)];
320 auto it = std::lower_bound(integrals.begin(), integrals.end(), i,
321 [](auto& itg_data, int i)
322 { return itg_data.id < i; });

337 std::vector<std::int32_t> domain(IntegralType type, int i,
338 const mesh::Mesh<geometry_type>& mesh) const
339 {
340 // Hack to avoid passing shared pointer to this function
341 std::shared_ptr<const mesh::Mesh<geometry_type>> msh_ptr(
342 &mesh, [](const mesh::Mesh<geometry_type>*) {});
343
344 std::span<const std::int32_t> entities = domain(type, i);
345 if (msh_ptr == _mesh)
346 return std::vector(entities.begin(), entities.end());
347 else
348 {
349 std::span<const std::int32_t> entity_map = _entity_maps.at(msh_ptr);
350 std::vector<std::int32_t> mapped_entities;
351 mapped_entities.reserve(entities.size());
352 switch (type)
353 {
354 case IntegralType::cell:
355 {
356 std::transform(entities.begin(), entities.end(),
357 std::back_inserter(mapped_entities),
358 [&entity_map](auto e) { return entity_map[e]; });
359 break;
360 }
361 case IntegralType::exterior_facet:
362 {
363 // Get the codimension of the mesh
364 const int tdim = _mesh->topology()->dim();
365 const int codim = tdim - mesh.topology()->dim();
366 assert(codim >= 0);
367 if (codim == 0)
368 {
369 for (std::size_t i = 0; i < entities.size(); i += 2)
370 {
371 // Add cell and the local facet index
372 mapped_entities.insert(mapped_entities.end(),
373 {entity_map[entities[i]], entities[i + 1]});
374 }
375 }
376 else if (codim == 1)
377 {
378 // In this case, the entity maps take facets in (`_mesh`) to cells in
379 // `mesh`, so we need to get the facet number from the (cell,
380 // local_facet pair) first.
381 auto c_to_f = _mesh->topology()->connectivity(tdim, tdim - 1);
382 assert(c_to_f);
383 for (std::size_t i = 0; i < entities.size(); i += 2)
384 {
385 // Get the facet index
386 const std::int32_t facet
387 = c_to_f->links(entities[i])[entities[i + 1]];
388 // Add cell and the local facet index
389 mapped_entities.insert(mapped_entities.end(),
390 {entity_map[facet], entities[i + 1]});
391 }
392 }
393 else
394 throw std::runtime_error("Codimension > 1 not supported.");
395
396 break;
397 }
398 case IntegralType::interior_facet:
399 {
400 for (std::size_t i = 0; i < entities.size(); i += 2)
401 {
402 // Add cell and the local facet index
403 mapped_entities.insert(mapped_entities.end(),
404 {entity_map[entities[i]], entities[i + 1]});
405 }
406 break;
407 }
408 default:
409 throw std::runtime_error("Integral type not supported.");
410 }
411
412 return mapped_entities;
413 }
