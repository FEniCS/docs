374  // Helper function to get a std::vector of (cell, local_facet) pairs
375  // corresponding to a given facet index.
376  // @param[in] f Facet index
377  // @param[in] f_to_c Facet to cell connectivity
378  // @param[in] c_to_f Cell to facet connectivity
379  // @return Vector of (cell, local_facet) pairs
380  template <int num_cells>
381  static std::array<std::array<std::int32_t, 2>, num_cells>
382  get_cell_local_facet_pairs(
383  std::int32_t f, const xtl::span<const std::int32_t>& cells,
384  const dolfinx::graph::AdjacencyList<std::int32_t>& c_to_f)
385  {
386  // Loop over cells sharing facet
387  assert(cells.size() == num_cells);
388  std::array<std::array<std::int32_t, 2>, num_cells> cell_local_facet_pairs;
389  for (int c = 0; c < num_cells; ++c)
390  {
391  // Get local index of facet with respect to the cell
392  std::int32_t cell = cells[c];
393  auto cell_facets = c_to_f.links(cell);
394  auto facet_it = std::find(cell_facets.begin(), cell_facets.end(), f);
395  assert(facet_it != cell_facets.end());
396  int local_f = std::distance(cell_facets.begin(), facet_it);
397  cell_local_facet_pairs[c] = {cell, local_f};
398  }
399 
400  return cell_local_facet_pairs;
401  }

403  // Set cell domains
404  template <typename iterator>
405  void set_cell_domains(
406  std::map<int, std::pair<kern, std::vector<std::int32_t>>>& integrals,
407  const iterator& tagged_cells_begin, const iterator& tagged_cells_end,
408  const std::vector<int>& tags)
409  {
410  // For cell integrals use all markers (but not on ghost entities)
411  for (auto c = tagged_cells_begin; c != tagged_cells_end; ++c)
412  {
413  const std::size_t pos = std::distance(tagged_cells_begin, c);
414  if (auto it = integrals.find(tags[pos]); it != integrals.end())
415  it->second.second.push_back(*c);
416  }
417  }

419  // Set exterior facet domains
420  template <typename iterator>
421  void set_exterior_facet_domains(
422  const mesh::Topology& topology,
423  std::map<int, std::pair<kern, std::vector<std::int32_t>>>& integrals,
424  const iterator& tagged_facets_begin, const iterator& tagged_facets_end,
425  const std::vector<int>& tags)
426  {
427  // When a mesh is not ghosted by cell, it is not straightforward
428  // to distinguish between (i) exterior facets and (ii) interior
429  // facets that are on a partition boundary. If there are no
430  // ghost cells, build a set of owned facts that are ghosted on
431  // another process to help determine if a facet is on an
432  // exterior boundary.
433  int tdim = topology.dim();
434  assert(topology.index_map(tdim));
435  assert(topology.index_map(tdim - 1));
436  const std::vector<std::int32_t> fwd_shared_facets
437  = topology.index_map(tdim)->overlapped()
438  ? std::vector<std::int32_t>()
439  : topology.index_map(tdim - 1)->shared_indices();
440 
441  auto f_to_c = topology.connectivity(tdim - 1, tdim);
442  assert(f_to_c);
443  auto c_to_f = topology.connectivity(tdim, tdim - 1);
444  assert(c_to_f);
445  for (auto f = tagged_facets_begin; f != tagged_facets_end; ++f)
446  {
447  // All "owned" facets connected to one cell, that are not
448  // shared, should be external
449  // TODO: Consider removing this check and integrating over all
450  // tagged facets. This may be useful in a few cases.
451  if (f_to_c->num_links(*f) == 1)
452  {
453  if (!std::binary_search(fwd_shared_facets.begin(),
454  fwd_shared_facets.end(), *f))
455  {
456  const std::size_t pos = std::distance(tagged_facets_begin, f);
457  if (auto it = integrals.find(tags[pos]); it != integrals.end())
458  {
459  // There will only be one pair for an exterior facet integral
460  const std::array<std::int32_t, 2> pair
461  = get_cell_local_facet_pairs<1>(*f, f_to_c->links(*f),
462  *c_to_f)[0];
463  it->second.second.insert(it->second.second.end(), pair.cbegin(),
464  pair.cend());
465  }
466  }
467  }
468  }
469  }

471  // Set interior facet domains
472  template <typename iterator>
473  static void set_interior_facet_domains(
474  const mesh::Topology& topology,
475  std::map<int, std::pair<kern, std::vector<std::int32_t>>>& integrals,
476  const iterator& tagged_facets_begin, const iterator& tagged_facets_end,
477  const std::vector<int>& tags)
478  {
479  int tdim = topology.dim();
480  auto f_to_c = topology.connectivity(tdim - 1, tdim);
481  assert(f_to_c);
482  auto c_to_f = topology.connectivity(tdim, tdim - 1);
483  assert(c_to_f);
484  for (auto f = tagged_facets_begin; f != tagged_facets_end; ++f)
485  {
486  if (f_to_c->num_links(*f) == 2)
487  {
488  const std::size_t pos = std::distance(tagged_facets_begin, f);
489  if (auto it = integrals.find(tags[pos]); it != integrals.end())
490  {
491  const std::array<std::array<std::int32_t, 2>, 2> pairs
492  = get_cell_local_facet_pairs<2>(*f, f_to_c->links(*f), *c_to_f);
493  it->second.second.insert(it->second.second.end(), pairs[0].cbegin(),
494  pairs[0].cend());
495  it->second.second.insert(it->second.second.end(), pairs[1].cbegin(),
496  pairs[1].cend());
497  }
498  }
499  }
500  }

502  // Sets the entity indices to assemble over for kernels with a domain
503  // ID
504  // @param[in] type Integral type
505  // @param[in] marker MeshTags with domain ID. Entities with marker 'i'
506  // will be assembled over using the kernel with ID 'i'. The MeshTags
507  // is not stored.
508  void set_domains(IntegralType type, const mesh::MeshTags<int>& marker)
509  {
510  std::shared_ptr<const mesh::Mesh> mesh = marker.mesh();
511  const mesh::Topology& topology = mesh->topology();
512  const int tdim = topology.dim();
513  int dim = type == IntegralType::cell ? tdim : tdim - 1;
514  if (dim != marker.dim())
515  {
516  throw std::runtime_error("Invalid MeshTags dimension: "
517  + std::to_string(marker.dim()));
518  }
519 
520  // Get mesh tag data
521  const std::vector<int>& tags = marker.values();
522  const std::vector<std::int32_t>& tagged_entities = marker.indices();
523  assert(topology.index_map(dim));
524  const auto entity_end
525  = std::lower_bound(tagged_entities.begin(), tagged_entities.end(),
526  topology.index_map(dim)->size_local());
527  switch (type)
528  {
529  case IntegralType::cell:
530  set_cell_domains(_cell_integrals, tagged_entities.cbegin(), entity_end,
531  tags);
532  break;
533  default:
534  mesh->topology_mutable().create_connectivity(dim, tdim);
535  mesh->topology_mutable().create_connectivity(tdim, dim);
536  switch (type)
537  {
538  case IntegralType::exterior_facet:
539  set_exterior_facet_domains(topology, _exterior_facet_integrals,
540  tagged_entities.cbegin(), entity_end, tags);
541  break;
542  case IntegralType::interior_facet:
543  set_interior_facet_domains(topology, _interior_facet_integrals,
544  tagged_entities.cbegin(), entity_end, tags);
545  break;
546  default:
547  throw std::runtime_error(
548  "Cannot set domains. Integral type not supported.");
549  }
550  }
551  }

558  void set_default_domains(const mesh::Mesh& mesh)
559  {
560  const mesh::Topology& topology = mesh.topology();
561  const int tdim = topology.dim();
562 
563  // Cells. If there is a default integral, define it on all owned
564  // cells
565  for (auto& [domain_id, kernel_cells] : _cell_integrals)
566  {
567  if (domain_id == -1)
568  {
569  std::vector<std::int32_t>& cells = kernel_cells.second;
570  const int num_cells = topology.index_map(tdim)->size_local();
571  cells.resize(num_cells);
572  std::iota(cells.begin(), cells.end(), 0);
573  }
574  }
575 
576  // Exterior facets. If there is a default integral, define it only
577  // on owned surface facets.
578 
579  if (!_exterior_facet_integrals.empty())
580  {
581  mesh.topology_mutable().create_connectivity(tdim - 1, tdim);
582  mesh.topology_mutable().create_connectivity(tdim, tdim - 1);
583  }
584  const std::vector<std::int32_t> boundary_facets
585  = _exterior_facet_integrals.empty()
586  ? std::vector<std::int32_t>()
587  : mesh::exterior_facet_indices(topology);
588  for (auto& [domain_id, kernel_facets] : _exterior_facet_integrals)
589  {
590  if (domain_id == -1)
591  {
592  std::vector<std::int32_t>& facets = kernel_facets.second;
593  facets.clear();
594 
595  auto f_to_c = topology.connectivity(tdim - 1, tdim);
596  assert(f_to_c);
597  auto c_to_f = topology.connectivity(tdim, tdim - 1);
598  assert(c_to_f);
599  for (std::int32_t f : boundary_facets)
600  {
601  // There will only be one pair for an exterior facet integral
602  std::array<std::int32_t, 2> pair
603  = get_cell_local_facet_pairs<1>(f, f_to_c->links(f), *c_to_f)[0];
604  facets.insert(facets.end(), pair.cbegin(), pair.cend());
605  }
606  }
607  }
608 
609  // Interior facets. If there is a default integral, define it only on
610  // owned interior facets.
611  for (auto& [domain_id, kernel_facets] : _interior_facet_integrals)
612  {
613  if (domain_id == -1)
614  {
615  std::vector<std::int32_t>& facets = kernel_facets.second;
616  facets.clear();
617 
618  mesh.topology_mutable().create_connectivity(tdim - 1, tdim);
619  auto f_to_c = topology.connectivity(tdim - 1, tdim);
620  assert(f_to_c);
621  mesh.topology_mutable().create_connectivity(tdim, tdim - 1);
622  auto c_to_f = mesh.topology().connectivity(tdim, tdim - 1);
623  assert(c_to_f);
624 
625  // Get number of facets owned by this process
626  assert(topology.index_map(tdim - 1));
627  const int num_facets = topology.index_map(tdim - 1)->size_local();
628  facets.reserve(num_facets);
629  for (int f = 0; f < num_facets; ++f)
630  {
631  if (f_to_c->num_links(f) == 2)
632  {
633  const std::array<std::array<std::int32_t, 2>, 2> pairs
634  = get_cell_local_facet_pairs<2>(f, f_to_c->links(f), *c_to_f);
635  facets.insert(facets.end(), pairs[0].cbegin(), pairs[0].cend());
636  facets.insert(facets.end(), pairs[1].cbegin(), pairs[1].cend());
637  }
638  }
639  }
640  }
641  }
