175 template <typename X>
176 requires std::is_convertible_v<
177 std::remove_cvref_t<X>,
178 std::map<IntegralType, std::vector<integral_data<
179 scalar_type, geometry_type>>>>
180 Form(
181 const std::vector<std::shared_ptr<const FunctionSpace<geometry_type>>>& V,
182 X&& integrals,
183 const std::vector<
184 std::shared_ptr<const Function<scalar_type, geometry_type>>>&
185 coefficients,
186 const std::vector<std::shared_ptr<const Constant<scalar_type>>>&
187 constants,
188 bool needs_facet_permutations,
189 const std::map<std::shared_ptr<const mesh::Mesh<geometry_type>>,
190 std::span<const std::int32_t>>& entity_maps,
191 std::shared_ptr<const mesh::Mesh<geometry_type>> mesh = nullptr)
192 : _function_spaces(V), _coefficients(coefficients), _constants(constants),
193 _mesh(mesh), _needs_facet_permutations(needs_facet_permutations)
194 {
195 // Extract _mesh from FunctionSpace, and check they are the same
196 if (!_mesh and !V.empty())
197 _mesh = V[0]->mesh();
198 for (auto& space : V)
199 {
200 if (_mesh != space->mesh()
201 and entity_maps.find(space->mesh()) == entity_maps.end())
202 {
203 throw std::runtime_error(
204 "Incompatible mesh. entity_maps must be provided.");
205 }
206 }
207 if (!_mesh)
208 throw std::runtime_error("No mesh could be associated with the Form.");
209
210 // Store kernels, looping over integrals by domain type (dimension)
211 for (auto&& [domain_type, data] : integrals)
212 {
213 if (!std::ranges::is_sorted(data,
214 [](auto& a, auto& b) { return a.id < b.id; }))
215 {
216 throw std::runtime_error("Integral IDs not sorted");
217 }
218
219 std::vector<integral_data<scalar_type, geometry_type>>& itg
220 = _integrals[static_cast<std::size_t>(domain_type)];
221 for (auto&& [id, kern, e, c] : data)
222 itg.emplace_back(id, kern, std::move(e), std::move(c));
223 }
224
225 // Store entity maps
226 for (auto [msh, map] : entity_maps)
227 _entity_maps.insert({msh, std::vector(map.begin(), map.end())});
228 }

268 std::function<void(scalar_type*, const scalar_type*, const scalar_type*,
269 const geometry_type*, const int*, const uint8_t*)>
270 kernel(IntegralType type, int i, int kernel_idx) const
271 {
272 const std::vector<integral_data<scalar_type, geometry_type>>& integrals
273 = _integrals[static_cast<std::size_t>(type)];
274
275 // Get the range of integrals with a given ID
276 auto get_id = [](const auto& a) { return a.id; };
277 auto start = std::ranges::lower_bound(integrals, i, std::less<>{}, get_id);
278 auto end = std::ranges::upper_bound(integrals, i, std::less<>{}, get_id);
279
280 // Check that the kernel is valid and return it if so
281 if (start == integrals.end() or start->id != i
282 or std::distance(start, end) <= kernel_idx)
283 {
284 throw std::runtime_error("No kernel for requested domain index.");
285 }
286
287 return std::next(start, kernel_idx)->kernel;
288 }

346 std::vector<int> integral_ids(IntegralType type) const
347 {
348 std::vector<int> ids;
349 const std::vector<integral_data<scalar_type, geometry_type>>& integrals
350 = _integrals[static_cast<std::size_t>(type)];
351 std::ranges::transform(integrals, std::back_inserter(ids),
352 [](auto& integral) { return integral.id; });
353
354 // IDs may be repeated in mixed-topology meshes, so remove duplicates
355 std::sort(ids.begin(), ids.end());
356 auto it = std::unique(ids.begin(), ids.end());
357 ids.erase(it, ids.end());
358 return ids;
359 }

427 std::vector<std::int32_t> domain(IntegralType type, int i, int kernel_idx,
428 const mesh::Mesh<geometry_type>& mesh) const
429 {
430 // Hack to avoid passing shared pointer to this function
431 std::shared_ptr<const mesh::Mesh<geometry_type>> msh_ptr(
432 &mesh, [](const mesh::Mesh<geometry_type>*) {});
433
434 std::vector<std::int32_t> entities = domain(type, i, kernel_idx);
435 if (msh_ptr == _mesh)
436 return entities;
437 else
438 {
439 std::span<const std::int32_t> entity_map = _entity_maps.at(msh_ptr);
440 std::vector<std::int32_t> mapped_entities;
441 mapped_entities.reserve(entities.size());
442 switch (type)
443 {
444 case IntegralType::cell:
445 {
446 std::ranges::transform(entities, std::back_inserter(mapped_entities),
447 [&entity_map](auto e) { return entity_map[e]; });
448 break;
449 }
450 case IntegralType::exterior_facet:
451 {
452 // Get the codimension of the mesh
453 const int tdim = _mesh->topology()->dim();
454 const int codim = tdim - mesh.topology()->dim();
455 assert(codim >= 0);
456 if (codim == 0)
457 {
458 for (std::size_t i = 0; i < entities.size(); i += 2)
459 {
460 // Add cell and the local facet index
461 mapped_entities.insert(mapped_entities.end(),
462 {entity_map[entities[i]], entities[i + 1]});
463 }
464 }
465 else if (codim == 1)
466 {
467 // In this case, the entity maps take facets in (`_mesh`) to cells in
468 // `mesh`, so we need to get the facet number from the (cell,
469 // local_facet pair) first.
470 auto c_to_f = _mesh->topology()->connectivity(tdim, tdim - 1);
471 assert(c_to_f);
472 for (std::size_t i = 0; i < entities.size(); i += 2)
473 {
474 // Get the facet index
475 const std::int32_t facet
476 = c_to_f->links(entities[i])[entities[i + 1]];
477
478 // Add cell and the local facet index
479 mapped_entities.insert(mapped_entities.end(),
480 {entity_map[facet], entities[i + 1]});
481 }
482 }
483 else
484 throw std::runtime_error("Codimension > 1 not supported.");
485
486 break;
487 }
488 case IntegralType::interior_facet:
489 {
490 // Get the codimension of the mesh
491 const int tdim = _mesh->topology()->dim();
492 const int codim = tdim - mesh.topology()->dim();
493 assert(codim >= 0);
494 if (codim == 0)
495 {
496 for (std::size_t i = 0; i < entities.size(); i += 2)
497 {
498 // Add cell and the local facet index
499 mapped_entities.insert(mapped_entities.end(),
500 {entity_map[entities[i]], entities[i + 1]});
501 }
502 }
503 else if (codim == 1)
504 {
505 // In this case, the entity maps take facets in (`_mesh`) to
506 // cells in `mesh`, so we need to get the facet number from
507 // the (cell, local_facet pair) first.
508 auto c_to_f = _mesh->topology()->connectivity(tdim, tdim - 1);
509 assert(c_to_f);
510 for (std::size_t i = 0; i < entities.size(); i += 2)
511 {
512 // Get the facet index
513 const std::int32_t facet
514 = c_to_f->links(entities[i])[entities[i + 1]];
515
516 // Add cell and the local facet index
517 mapped_entities.insert(mapped_entities.end(),
518 {entity_map[facet], entities[i + 1]});
519 }
520 }
521 break;
522 }
523 default:
524 throw std::runtime_error("Integral type not supported.");
525 }
526
527 return mapped_entities;
528 }
