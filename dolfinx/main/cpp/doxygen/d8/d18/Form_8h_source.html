104template <
105 dolfinx::scalar T, std::floating_point U = dolfinx::scalar_value_type_t<T>,
106 FEkernel<T> Kern
107 = std::function<void(T*, const T*, const T*, const scalar_value_type_t<T>*,
108 const int*, const std::uint8_t*)>>
109class Form
110{
111 using kern_t = Kern;
112
113public:
115 using scalar_type = T;
116
134 template <typename X>
135 Form(const std::vector<std::shared_ptr<const FunctionSpace<U>>>& V,
136 X&& integrals,
137 const std::vector<std::shared_ptr<const Function<scalar_type, U>>>&
138 coefficients,
139 const std::vector<std::shared_ptr<const Constant<scalar_type>>>&
140 constants,
141 bool needs_facet_permutations,
142 std::shared_ptr<const mesh::Mesh<U>> mesh = nullptr)
143 : _function_spaces(V), _coefficients(coefficients), _constants(constants),
144 _mesh(mesh), _needs_facet_permutations(needs_facet_permutations)
145 {
146 // Extract _mesh from FunctionSpace, and check they are the same
147 if (!_mesh and !V.empty())
148 _mesh = V[0]->mesh();
149 for (auto& space : V)
150 if (_mesh != space->mesh())
151 throw std::runtime_error("Incompatible mesh");
152 if (!_mesh)
153 throw std::runtime_error("No mesh could be associated with the Form.");
154
155 // Store kernels, looping over integrals by domain type (dimension)
156 for (auto&& [domain_type, data] : integrals)
157 {
158 if (!std::is_sorted(data.begin(), data.end(),
159 [](auto& a, auto& b) { return a.id < b.id; }))
160 {
161 throw std::runtime_error("Integral IDs not sorted");
162 }
163
164 std::vector<integral_data<T, kern_t>>& itg
165 = _integrals[static_cast<std::size_t>(domain_type)];
166 for (auto&& [id, kern, e] : data)
167 itg.emplace_back(id, kern, std::move(e));
168 }
169 }

202 kern_t kernel(IntegralType type, int i) const
203 {
204 const auto& integrals = _integrals[static_cast<std::size_t>(type)];
205 auto it = std::lower_bound(integrals.begin(), integrals.end(), i,
206 [](auto& itg_data, int i)
207 { return itg_data.id < i; });
208 if (it != integrals.end() and it->id == i)
209 return it->kernel;
210 else
211 throw std::runtime_error("No kernel for requested domain index.");
212 }
213
216 std::set<IntegralType> integral_types() const
217 {
218 std::set<IntegralType> set;
219 for (std::size_t i = 0; i < _integrals.size(); ++i)
220 {
221 if (!_integrals[i].empty())
222 set.insert(static_cast<IntegralType>(i));
223 }
224
225 return set;
226 }

242 std::vector<int> integral_ids(IntegralType type) const
243 {
244 std::vector<int> ids;

246 std::transform(integrals.begin(), integrals.end(), std::back_inserter(ids),
247 [](auto& integral) { return integral.id; });
248 return ids;
249 }

268 std::span<const std::int32_t> domain(IntegralType type, int i) const
269 {
270 const auto& integrals = _integrals[static_cast<std::size_t>(type)];
271 auto it = std::lower_bound(integrals.begin(), integrals.end(), i,
272 [](auto& itg_data, int i)
273 { return itg_data.id < i; });
274 if (it != integrals.end() and it->id == i)
275 return it->entities;
276 else
277 throw std::runtime_error("No mesh entities for requested domain index.");
278 }

294 std::vector<int> coefficient_offsets() const
295 {
296 std::vector<int> n = {0};
297 for (auto& c : _coefficients)
298 {
299 if (!c)
300 throw std::runtime_error("Not all form coefficients have been set.");
301 n.push_back(n.back() + c->function_space()->element()->space_dimension());
302 }
303 return n;
304 }

307 const std::vector<std::shared_ptr<const Constant<T>>>& constants() const
308 {
309 return _constants;
310 }

312private:
313 // Function spaces (one for each argument)
314 std::vector<std::shared_ptr<const FunctionSpace<U>>> _function_spaces;
315
316 // Integrals. Array index is

318 std::array<std::vector<integral_data<T, kern_t>>, 4> _integrals;
319
320 // Form coefficients
321 std::vector<std::shared_ptr<const Function<T, U>>> _coefficients;
322
323 // Constants associated with the Form
324 std::vector<std::shared_ptr<const Constant<T>>> _constants;
325
326 // The mesh
327 std::shared_ptr<const mesh::Mesh<U>> _mesh;
328
329 // True if permutation data needs to be passed into these integrals
330 bool _needs_facet_permutations;
