33template <std::floating_point T>
34std::tuple<graph::AdjacencyList<std::int64_t>, std::vector<T>,
35 std::array<std::size_t, 2>, std::vector<std::int32_t>>
36compute_interval_refinement(const mesh::Mesh<T>& mesh,
37 std::optional<std::span<const std::int32_t>> cells)
38{
39 auto topology = mesh.topology();
40 assert(topology);
41 assert(topology->dim() == 1);
42 auto map_c = topology->index_map(1);
43 assert(map_c);
44
45 // TODO: creation of sharing ranks in external function? Also same
46 // code in use for plaza
47 // Get sharing ranks for each cell
48 graph::AdjacencyList<int> cell_ranks = map_c->index_to_dest_ranks();
49
50 // Create unique list of ranks that share cells (owners of ghosts plus
51 // ranks that ghost owned indices)
52 std::vector<int> ranks = cell_ranks.array();
53 std::ranges::sort(ranks);
54 auto to_remove = std::ranges::unique(ranks);
55 ranks.erase(to_remove.begin(), to_remove.end());
56
57 // Convert cell_ranks from global rank to to neighbourhood ranks
58 std::ranges::transform(cell_ranks.array(), cell_ranks.array().begin(),
59 [&ranks](auto r)
60 {
61 auto it = std::lower_bound(ranks.begin(),
62 ranks.end(), r);
63 assert(it != ranks.end() and *it == r);
64 return std::distance(ranks.begin(), it);
65 });
66
67 // Create refinement flag for cells
68 std::vector<std::int8_t> refinement_marker(
69 map_c->size_local() + map_c->num_ghosts(), !cells.has_value());
70
71 // Mark cells for refinement
72 std::vector<std::vector<std::int32_t>> marked_for_update(ranks.size());
73 if (cells.has_value())
74 {
75 std::ranges::for_each(cells.value(),
76 [&](auto cell)
77 {
78 if (!refinement_marker[cell])
79 {
80 refinement_marker[cell] = true;
81 for (int rank : cell_ranks.links(cell))
82 marked_for_update[rank].push_back(cell);
83 }
84 });
85 }
86
87 // Create neighborhood communicator for vertex creation
88 MPI_Comm neighbor_comm;
89 MPI_Dist_graph_create_adjacent(
90 mesh.comm(), ranks.size(), ranks.data(), MPI_UNWEIGHTED, ranks.size(),
91 ranks.data(), MPI_UNWEIGHTED, MPI_INFO_NULL, false, &neighbor_comm);
92
93 // Communicate ghost cells that might have been marked. This is not
94 // necessary for a uniform refinement.
95 if (cells.has_value())
96 {
97 update_logical_edgefunction(neighbor_comm, marked_for_update,
98 refinement_marker, *map_c);
99 }
100
101 // Construct the new vertices
102 const auto [new_vertex_map, new_vertex_coords, xshape]
103 = create_new_vertices(neighbor_comm, cell_ranks, mesh, refinement_marker);
104 MPI_Comm_free(&neighbor_comm);
105
106 auto c_to_v = mesh.topology()->connectivity(1, 0);
107 assert(c_to_v);
108
109 // Get the count of cells to refine, note: we only consider non-ghost
110 // cells
111 const std::int32_t number_of_refined_cells
112 = std::count(refinement_marker.begin(),
113 std::next(refinement_marker.begin(),
114 mesh.topology()->index_map(1)->size_local()),
115 true);
116
117 // Produce local global indices, by padding out the previous index map
118 std::vector<std::int64_t> global_indices
119 = adjust_indices(*mesh.topology()->index_map(0), number_of_refined_cells);
120
121 // Build the topology on the new vertices
122 const std::int32_t refined_cell_count
123 = mesh.topology()->index_map(1)->size_local() + number_of_refined_cells;
124
125 std::vector<std::int64_t> cell_topology;
126 cell_topology.reserve(refined_cell_count * 2);
127 std::vector<std::int32_t> parent_cell;
128 parent_cell.reserve(refined_cell_count);
129 for (std::int32_t cell = 0; cell < map_c->size_local(); ++cell)
130 {
131 const auto& vertices = c_to_v->links(cell);
132 assert(vertices.size() == 2);
133
134 // We consider a cell (defined by global vertices)
135 // a ----------- b
136 const std::int64_t a = global_indices[vertices[0]];
137 const std::int64_t b = global_indices[vertices[1]];
138 if (refinement_marker[cell])
139 {
140 // Find (global) index of new midpoint vertex:
141 // a --- c --- b
142 auto it = new_vertex_map.find(cell);
143 assert(it != new_vertex_map.end());
144 const std::int64_t c = it->second;
145
146 // Add new cells/edges to refined topology
147 cell_topology.insert(cell_topology.end(), {a, c, c, b});
148 parent_cell.insert(parent_cell.end(), {cell, cell});
149 }
150 else
151 {
152 // Copy the previous cell
153 cell_topology.insert(cell_topology.end(), {a, b});
154 parent_cell.push_back(cell);
155 }
156 }
157
158 assert(cell_topology.size() == 2 * refined_cell_count);
159 assert(parent_cell.size() == refined_cell_count);
160
161 std::vector<std::int32_t> offsets(refined_cell_count + 1);
162 std::ranges::generate(offsets, [i = 0]() mutable { return 2 * i++; });
163
164 return {graph::AdjacencyList(std::move(cell_topology), std::move(offsets)),
165 std::move(new_vertex_coords), xshape, std::move(parent_cell)};
166}

186template <std::floating_point T>
187std::tuple<mesh::Mesh<T>, std::vector<std::int32_t>>
188refine_interval(const mesh::Mesh<T>& mesh,
189 std::optional<std::span<const std::int32_t>> cells,
190 bool redistribute,
191 mesh::GhostMode ghost_mode = mesh::GhostMode::shared_facet)
192{
193 if (mesh.topology()->cell_type() != mesh::CellType::interval)
194 throw std::runtime_error("Cell type not supported");
195 assert(mesh.topology()->dim() == 1);
196 assert(mesh.topology()->index_map(1));
197
198 auto [cell_adj, new_coords, xshape, parent_cell]
199 = impl::compute_interval_refinement(mesh, cells);
200 if (dolfinx::MPI::size(mesh.comm()) == 1)
201 {
202 return {mesh::create_mesh(mesh.comm(), cell_adj.array(),
203 mesh.geometry().cmap(), new_coords, xshape,
204 mesh::GhostMode::none),
205 std::move(parent_cell)};
206 }
207 else
208 {
209 return {partition<T>(mesh, cell_adj, std::span(new_coords), xshape,
210 redistribute, ghost_mode),
211 std::move(parent_cell)};
212 }
213}
