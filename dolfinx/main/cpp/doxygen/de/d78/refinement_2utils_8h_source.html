53template <std::floating_point T>
54std::pair<std::vector<T>, std::array<std::size_t, 2>> create_new_geometry(
55 const mesh::Mesh<T>& mesh,
56 const std::map<std::int32_t, std::int64_t>& local_edge_to_new_vertex)
57{
58 // Build map from vertex -> geometry dof
59 auto x_dofmap = mesh.geometry().dofmap();
60 const int tdim = mesh.topology()->dim();
61 auto c_to_v = mesh.topology()->connectivity(tdim, 0);
62 assert(c_to_v);
63 auto map_v = mesh.topology()->index_map(0);
64 assert(map_v);
65 std::vector<std::int32_t> vertex_to_x(map_v->size_local()
66 + map_v->num_ghosts());
67 auto map_c = mesh.topology()->index_map(tdim);
68
69 assert(map_c);
70 auto dof_layout = mesh.geometry().cmap().create_dof_layout();
71 auto entity_dofs_all = dof_layout.entity_dofs_all();
72 for (int c = 0; c < map_c->size_local() + map_c->num_ghosts(); ++c)
73 {
74 auto vertices = c_to_v->links(c);
75 auto dofs = md::submdspan(x_dofmap, c, md::full_extent);
76 for (std::size_t i = 0; i < vertices.size(); ++i)
77 {
78 auto vertex_pos = entity_dofs_all[0][i][0];
79 vertex_to_x[vertices[i]] = dofs[vertex_pos];
80 }
81 }
82
83 // Copy over existing mesh vertices
84 std::span<const T> x_g = mesh.geometry().x();
85 const std::size_t gdim = mesh.geometry().dim();
86 const std::size_t num_vertices = map_v->size_local();
87 const std::size_t num_new_vertices = local_edge_to_new_vertex.size();
88
89 std::array<std::size_t, 2> shape = {num_vertices + num_new_vertices, gdim};
90 std::vector<T> new_vertex_coords(shape[0] * shape[1]);
91 for (std::size_t v = 0; v < num_vertices; ++v)
92 {
93 std::size_t pos = 3 * vertex_to_x[v];
94 for (std::size_t j = 0; j < gdim; ++j)
95 new_vertex_coords[gdim * v + j] = x_g[pos + j];
96 }
97
98 // Compute new vertices
99 if (num_new_vertices > 0)
100 {
101 std::vector<int> edges(num_new_vertices);
102 int i = 0;
103 for (auto& e : local_edge_to_new_vertex)
104 edges[i++] = e.first;
105
106 // Compute midpoint of each edge (padded to 3D)
107 const std::vector<T> midpoints = mesh::compute_midpoints(mesh, 1, edges);
108 for (std::size_t i = 0; i < num_new_vertices; ++i)
109 for (std::size_t j = 0; j < gdim; ++j)
110 new_vertex_coords[gdim * (num_vertices + i) + j] = midpoints[3 * i + j];
111 }
112
113 return {std::move(new_vertex_coords), shape};
114}

146template <std::floating_point T>
147std::tuple<std::map<std::int32_t, std::int64_t>, std::vector<T>,
148 std::array<std::size_t, 2>>
149create_new_vertices(MPI_Comm comm,
150 const graph::AdjacencyList<int>& shared_edges,
151 const mesh::Mesh<T>& mesh,
152 std::span<const std::int8_t> marked_edges)
153{
154 // Take marked_edges and use to create new vertices
155 std::shared_ptr<const common::IndexMap> edge_index_map
156 = mesh.topology()->index_map(1);
157
158 // Add new edge midpoints to list of vertices. The new vertex will be owned by
159 // the process owning the edge.
160 int n = 0;
161 std::map<std::int32_t, std::int64_t> local_edge_to_new_vertex;
162 for (int local_i = 0; local_i < edge_index_map->size_local(); ++local_i)
163 {
164 if (marked_edges[local_i])
165 {
166 [[maybe_unused]] auto it = local_edge_to_new_vertex.insert({local_i, n});
167 assert(it.second);
168 ++n;
169 }
170 }
171
172 const std::int64_t num_local = n;
173 std::int64_t global_offset = 0;
174 MPI_Exscan(&num_local, &global_offset, 1, MPI_INT64_T, MPI_SUM, mesh.comm());
175 global_offset += mesh.topology()->index_map(0)->local_range()[1];
176 std::for_each(local_edge_to_new_vertex.begin(),
177 local_edge_to_new_vertex.end(),
178 [global_offset](auto& e) { e.second += global_offset; });
179
180 // Create actual points
181 auto [new_vertex_coords, xshape]
182 = impl::create_new_geometry(mesh, local_edge_to_new_vertex);
183
184 // If they are shared, then the new global vertex index needs to be
185 // sent off-process.
186
187 // Get number of neighbors
188 int indegree(-1), outdegree(-2), weighted(-1);
189 MPI_Dist_graph_neighbors_count(comm, &indegree, &outdegree, &weighted);
190 assert(indegree == outdegree);
191 const int num_neighbors = indegree;
192
193 std::vector<std::vector<std::int64_t>> values_to_send(num_neighbors);
194 for (auto& local_edge : local_edge_to_new_vertex)
195 {
196 const std::size_t local_i = local_edge.first;
197 // shared, but locally owned : remote owned are not in list.
198
199 for (int remote_process : shared_edges.links(local_i))
200 {
201 // Send (global edge index) -> (new global vertex index) map
202 values_to_send[remote_process].push_back(
203 impl::local_to_global(local_i, *edge_index_map));
204 values_to_send[remote_process].push_back(local_edge.second);
205 }
206 }
207
208 // Send all shared edges marked for update and receive from other
209 // processes
210 std::vector<std::int64_t> received_values;
211 {
212 int indegree(-1), outdegree(-2), weighted(-1);
213 MPI_Dist_graph_neighbors_count(comm, &indegree, &outdegree, &weighted);
214 assert(indegree == outdegree);
215
216 std::vector<std::int64_t> send_buffer;
217 std::vector<int> send_sizes;
218 for (auto& x : values_to_send)
219 {
220 send_sizes.push_back(x.size());
221 send_buffer.insert(send_buffer.end(), x.begin(), x.end());
222 }
223 assert((int)send_sizes.size() == outdegree);
224
225 std::vector<int> recv_sizes(outdegree);
226 send_sizes.reserve(1);
227 recv_sizes.reserve(1);
228 MPI_Neighbor_alltoall(send_sizes.data(), 1, MPI_INT, recv_sizes.data(), 1,
229 MPI_INT, comm);
230
231 // Build displacements
232 std::vector<int> send_disp = {0};
233 std::partial_sum(send_sizes.begin(), send_sizes.end(),
234 std::back_inserter(send_disp));
235 std::vector<int> recv_disp = {0};
236 std::partial_sum(recv_sizes.begin(), recv_sizes.end(),
237 std::back_inserter(recv_disp));
238
239 received_values.resize(recv_disp.back());
240 MPI_Neighbor_alltoallv(send_buffer.data(), send_sizes.data(),
241 send_disp.data(), MPI_INT64_T,
242 received_values.data(), recv_sizes.data(),
243 recv_disp.data(), MPI_INT64_T, comm);
244 }
245
246 // Add received remote global vertex indices to map
247 std::vector<std::int64_t> recv_global_edge;
248 assert(received_values.size() % 2 == 0);
249 for (std::size_t i = 0; i < received_values.size() / 2; ++i)
250 recv_global_edge.push_back(received_values[i * 2]);
251 std::vector<std::int32_t> recv_local_edge(recv_global_edge.size());
252 mesh.topology()->index_map(1)->global_to_local(recv_global_edge,
253 recv_local_edge);
254 for (std::size_t i = 0; i < received_values.size() / 2; ++i)
255 {
256 assert(recv_local_edge[i] != -1);
257 [[maybe_unused]] auto it = local_edge_to_new_vertex.insert(
258 {recv_local_edge[i], received_values[i * 2 + 1]});
259 assert(it.second);
260 }
261
262 return {std::move(local_edge_to_new_vertex), std::move(new_vertex_coords),
263 xshape};
264}
