266 template <typename U>
267 std::function<void(std::span<U>, std::span<const std::uint32_t>, std::int32_t,
268 int)>
269 dof_transformation_function(doftransform ttype = doftransform::standard,
270 bool scalar_element = false) const
271 {
272 if (!needs_dof_transformations())
273 {
274 // If no permutation needed, return function that does nothing
275 return [](std::span<U>, std::span<const std::uint32_t>, std::int32_t, int)
276 {
277 // Do nothing
278 };
279 }
280
281 if (_sub_elements.size() != 0)
282 {
283 if (_is_mixed)
284 {
285 // Mixed element
286 std::vector<std::function<void(
287 std::span<U>, std::span<const std::uint32_t>, std::int32_t, int)>>
288 sub_element_functions;
289 std::vector<int> dims;
290 for (std::size_t i = 0; i < _sub_elements.size(); ++i)
291 {
292 sub_element_functions.push_back(
293 _sub_elements[i]->template dof_transformation_function<U>(
294 ttype));
295 dims.push_back(_sub_elements[i]->space_dimension());
296 }
297
298 return [dims, sub_element_functions](
299 std::span<U> data, std::span<const std::uint32_t> cell_info,
300 std::int32_t cell, int block_size)
301 {
302 std::size_t offset = 0;
303 for (std::size_t e = 0; e < sub_element_functions.size(); ++e)
304 {
305 const std::size_t width = dims[e] * block_size;
306 sub_element_functions[e](data.subspan(offset, width), cell_info,
307 cell, block_size);
308 offset += width;
309 }
310 };
311 }
312 else if (!scalar_element)
313 {
314 // Vector element
315 const std::function<void(std::span<U>, std::span<const std::uint32_t>,
316 std::int32_t, int)>
317 sub_function
318 = _sub_elements[0]->template dof_transformation_function<U>(
319 ttype);
320 const int ebs = _bs;
321 return [ebs, sub_function](std::span<U> data,
322 std::span<const std::uint32_t> cell_info,
323 std::int32_t cell, int data_block_size)
324 { sub_function(data, cell_info, cell, ebs * data_block_size); };
325 }
326 }
327 switch (ttype)
328 {
329 case doftransform::inverse_transpose:
330 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
331 std::int32_t cell, int block_size)
332 {
333 pre_apply_inverse_transpose_dof_transformation(data, cell_info[cell],
334 block_size);
335 };
336 case doftransform::transpose:
337 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
338 std::int32_t cell, int block_size) {
339 pre_apply_transpose_dof_transformation(data, cell_info[cell],
340 block_size);
341 };
342 case doftransform::inverse:
343 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
344 std::int32_t cell, int block_size) {
345 pre_apply_inverse_dof_transformation(data, cell_info[cell], block_size);
346 };
347 case doftransform::standard:
348 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
349 std::int32_t cell, int block_size)
350 { pre_apply_dof_transformation(data, cell_info[cell], block_size); };
351 default:
352 throw std::runtime_error("Unknown transformation type");
353 }
354 }

371 template <typename U>
372 std::function<void(std::span<U>, std::span<const std::uint32_t>, std::int32_t,
373 int)>
374 get_post_dof_transformation_function(doftransform ttype
375 = doftransform::standard,
376 bool scalar_element = false) const
377 {
378 if (!needs_dof_transformations())
379 {
380 // If no permutation needed, return function that does nothing
381 return [](std::span<U>, std::span<const std::uint32_t>, std::int32_t, int)
382 {
383 // Do nothing
384 };
385 }
386 else if (_sub_elements.size() != 0)
387 {
388 if (_is_mixed)
389 {
390 // Mixed element
391 std::vector<std::function<void(
392 std::span<U>, std::span<const std::uint32_t>, std::int32_t, int)>>
393 sub_element_functions;
394 for (std::size_t i = 0; i < _sub_elements.size(); ++i)
395 {
396 sub_element_functions.push_back(
397 _sub_elements[i]
398 ->template get_post_dof_transformation_function<U>(ttype));
399 }
400
401 return [this, sub_element_functions](
402 std::span<U> data, std::span<const std::uint32_t> cell_info,
403 std::int32_t cell, int block_size)
404 {
405 std::size_t offset = 0;
406 for (std::size_t e = 0; e < sub_element_functions.size(); ++e)
407 {
408 sub_element_functions[e](data.subspan(offset, data.size() - offset),
409 cell_info, cell, block_size);
410 offset += _sub_elements[e]->space_dimension();
411 }
412 };
413 }
414 else if (!scalar_element)
415 {
416 // Vector element
417 const std::function<void(std::span<U>, std::span<const std::uint32_t>,
418 std::int32_t, int)>
419 sub_function
420 = _sub_elements[0]->template dof_transformation_function<U>(
421 ttype);
422 return [this, sub_function](std::span<U> data,
423 std::span<const std::uint32_t> cell_info,
424 std::int32_t cell, int data_block_size)
425 {
426 const int ebs = block_size();
427 const std::size_t dof_count = data.size() / data_block_size;
428 for (int block = 0; block < data_block_size; ++block)
429 {
430 sub_function(data.subspan(block * dof_count, dof_count), cell_info,
431 cell, ebs);
432 }
433 };
434 }
435 }
436
437 switch (ttype)
438 {
439 case doftransform::inverse_transpose:
440 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
441 std::int32_t cell, int block_size)
442 {
443 post_apply_inverse_transpose_dof_transformation(data, cell_info[cell],
444 block_size);
445 };
446 case doftransform::transpose:
447 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
448 std::int32_t cell, int block_size) {
449 post_apply_transpose_dof_transformation(data, cell_info[cell],
450 block_size);
451 };
452 case doftransform::inverse:
453 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
454 std::int32_t cell, int block_size) {
455 post_apply_inverse_dof_transformation(data, cell_info[cell],
456 block_size);
457 };
458 case doftransform::standard:
459 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
460 std::int32_t cell, int block_size)
461 { post_apply_dof_transformation(data, cell_info[cell], block_size); };
462 default:
463 throw std::runtime_error("Unknown transformation type");
464 }
465 }
