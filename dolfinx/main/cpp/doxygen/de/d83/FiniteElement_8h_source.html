413 template <typename U>
414 std::function<void(std::span<U>, std::span<const std::uint32_t>, std::int32_t,
415 int)>
416 dof_transformation_fn(doftransform ttype, bool scalar_element = false) const
417 {
418 if (!needs_dof_transformations())
419 {
420 // If no permutation needed, return function that does nothing
421 return [](std::span<U>, std::span<const std::uint32_t>, std::int32_t, int)
422 {
423 // Do nothing
424 };
425 }
426
427 if (!_sub_elements.empty())
428 {
429 if (!_reference_value_shape) // Mixed element
430 {
431 std::vector<std::function<void(
432 std::span<U>, std::span<const std::uint32_t>, std::int32_t, int)>>
433 sub_element_fns;
434 std::vector<int> dims;
435 for (std::size_t i = 0; i < _sub_elements.size(); ++i)
436 {
437 sub_element_fns.push_back(
438 _sub_elements[i]->template dof_transformation_fn<U>(ttype));
439 dims.push_back(_sub_elements[i]->space_dimension());
440 }
441
442 return [dims, sub_element_fns](std::span<U> data,
443 std::span<const std::uint32_t> cell_info,
444 std::int32_t cell, int block_size)
445 {
446 std::size_t offset = 0;
447 for (std::size_t e = 0; e < sub_element_fns.size(); ++e)
448 {
449 const std::size_t width = dims[e] * block_size;
450 sub_element_fns[e](data.subspan(offset, width), cell_info, cell,
451 block_size);
452 offset += width;
453 }
454 };
455 }
456 else if (!scalar_element)
457 {
458 // Blocked element
459 std::function<void(std::span<U>, std::span<const std::uint32_t>,
460 std::int32_t, int)>
461 sub_fn = _sub_elements[0]->template dof_transformation_fn<U>(ttype);
462 const int ebs = _bs;
463 return [ebs, sub_fn](std::span<U> data,
464 std::span<const std::uint32_t> cell_info,
465 std::int32_t cell, int data_block_size)
466 { sub_fn(data, cell_info, cell, ebs * data_block_size); };
467 }
468 }
469
470 switch (ttype)
471 {
472 case doftransform::inverse_transpose:
473 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
474 std::int32_t cell, int block_size)
475 { Tt_inv_apply(data, cell_info[cell], block_size); };
476 case doftransform::transpose:
477 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
478 std::int32_t cell, int block_size)
479 { Tt_apply(data, cell_info[cell], block_size); };
480 case doftransform::inverse:
481 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
482 std::int32_t cell, int block_size)
483 { Tinv_apply(data, cell_info[cell], block_size); };
484 case doftransform::standard:
485 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
486 std::int32_t cell, int block_size)
487 { T_apply(data, cell_info[cell], block_size); };
488 default:
489 throw std::runtime_error("Unknown transformation type");
490 }
491 }

516 template <typename U>
517 std::function<void(std::span<U>, std::span<const std::uint32_t>, std::int32_t,
518 int)>
519 dof_transformation_right_fn(doftransform ttype,
520 bool scalar_element = false) const
521 {
522 if (!needs_dof_transformations())
523 {
524 // If no permutation needed, return function that does nothing
525 return [](std::span<U>, std::span<const std::uint32_t>, std::int32_t, int)
526 {
527 // Do nothing
528 };
529 }
530 else if (!_sub_elements.empty())
531 {
532 if (!_reference_value_shape) // Mixed element
533 {
534 std::vector<std::function<void(
535 std::span<U>, std::span<const std::uint32_t>, std::int32_t, int)>>
536 sub_element_fns;
537 for (std::size_t i = 0; i < _sub_elements.size(); ++i)
538 {
539 sub_element_fns.push_back(
540 _sub_elements[i]->template dof_transformation_right_fn<U>(ttype));
541 }
542
543 return [this, sub_element_fns](std::span<U> data,
544 std::span<const std::uint32_t> cell_info,
545 std::int32_t cell, int block_size)
546 {
547 std::size_t offset = 0;
548 for (std::size_t e = 0; e < sub_element_fns.size(); ++e)
549 {
550 sub_element_fns[e](data.subspan(offset, data.size() - offset),
551 cell_info, cell, block_size);
552 offset += _sub_elements[e]->space_dimension();
553 }
554 };
555 }
556 else if (!scalar_element)
557 {
558 // Blocked element
559 // The transformation from the left can be used here as blocked
560 // elements use xyzxyzxyz ordering, and so applying the DOF
561 // transformation from the right is equivalent to applying the DOF
562 // transformation from the left to data using xxxyyyzzz ordering
563 std::function<void(std::span<U>, std::span<const std::uint32_t>,
564 std::int32_t, int)>
565 sub_fn = _sub_elements[0]->template dof_transformation_fn<U>(ttype);
566 return [this, sub_fn](std::span<U> data,
567 std::span<const std::uint32_t> cell_info,
568 std::int32_t cell, int data_block_size)
569 {
570 const int ebs = block_size();
571 const std::size_t dof_count = data.size() / data_block_size;
572 for (int block = 0; block < data_block_size; ++block)
573 {
574 sub_fn(data.subspan(block * dof_count, dof_count), cell_info, cell,
575 ebs);
576 }
577 };
578 }
579 }
580
581 switch (ttype)
582 {
583 case doftransform::inverse_transpose:
584 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
585 std::int32_t cell, int n)
586 { Tt_inv_apply_right(data, cell_info[cell], n); };
587 case doftransform::transpose:
588 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
589 std::int32_t cell, int n)
590 { Tt_apply_right(data, cell_info[cell], n); };
591 case doftransform::inverse:
592 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
593 std::int32_t cell, int n)
594 { Tinv_apply_right(data, cell_info[cell], n); };
595 case doftransform::standard:
596 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
597 std::int32_t cell, int n)
598 { T_apply_right(data, cell_info[cell], n); };
599 default:
600 throw std::runtime_error("Unknown transformation type");
601 }
602 }
