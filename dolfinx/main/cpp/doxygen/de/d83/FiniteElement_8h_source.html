267 template <typename U>
268 std::function<void(const std::span<U>&, const std::span<const std::uint32_t>&,
269 std::int32_t, int)>
270 get_pre_dof_transformation_function(bool inverse = false,
271 bool transpose = false,
272 bool scalar_element = false) const
273 {
274 if (!needs_dof_transformations())
275 {
276 // If no permutation needed, return function that does nothing
277 return [](const std::span<U>&, const std::span<const std::uint32_t>&,
278 std::int32_t, int)
279 {
280 // Do nothing
281 };
282 }
283
284 if (_sub_elements.size() != 0)
285 {
286 if (_bs == 1)
287 {
288 // Mixed element
289 std::vector<std::function<void(const std::span<U>&,
290 const std::span<const std::uint32_t>&,
291 std::int32_t, int)>>
292 sub_element_functions;
293 std::vector<int> dims;
294 for (std::size_t i = 0; i < _sub_elements.size(); ++i)
295 {
296 sub_element_functions.push_back(
297 _sub_elements[i]->template get_pre_dof_transformation_function<U>(
298 inverse, transpose));
299 dims.push_back(_sub_elements[i]->space_dimension());
300 }
301
302 return [dims, sub_element_functions](
303 const std::span<U>& data,
304 const std::span<const std::uint32_t>& cell_info,
305 std::int32_t cell, int block_size)
306 {
307 std::size_t offset = 0;
308 for (std::size_t e = 0; e < sub_element_functions.size(); ++e)
309 {
310 const std::size_t width = dims[e] * block_size;
311 sub_element_functions[e](data.subspan(offset, width), cell_info,
312 cell, block_size);
313 offset += width;
314 }
315 };
316 }
317 else if (!scalar_element)
318 {
319 // Vector element
320 const std::function<void(const std::span<U>&,
321 const std::span<const std::uint32_t>&,
322 std::int32_t, int)>
323 sub_function
324 = _sub_elements[0]->template get_pre_dof_transformation_function<U>(
325 inverse, transpose);
326 const int ebs = _bs;
327 return
328 [ebs, sub_function](const std::span<U>& data,
329 const std::span<const std::uint32_t>& cell_info,
330 std::int32_t cell, int data_block_size)
331 { sub_function(data, cell_info, cell, ebs * data_block_size); };
332 }
333 }
334 if (transpose)
335 {
336 if (inverse)
337 {
338 return [this](const std::span<U>& data,
339 const std::span<const std::uint32_t>& cell_info,
340 std::int32_t cell, int block_size)
341 {
342 pre_apply_inverse_transpose_dof_transformation(data, cell_info[cell],
343 block_size);
344 };
345 }
346 else
347 {
348 return [this](const std::span<U>& data,
349 const std::span<const std::uint32_t>& cell_info,
350 std::int32_t cell, int block_size) {
351 pre_apply_transpose_dof_transformation(data, cell_info[cell],
352 block_size);
353 };
354 }
355 }
356 else
357 {
358 if (inverse)
359 {
360 return [this](const std::span<U>& data,
361 const std::span<const std::uint32_t>& cell_info,
362 std::int32_t cell, int block_size) {
363 pre_apply_inverse_dof_transformation(data, cell_info[cell],
364 block_size);
365 };
366 }
367 else
368 {
369 return [this](const std::span<U>& data,
370 const std::span<const std::uint32_t>& cell_info,
371 std::int32_t cell, int block_size)
372 { pre_apply_dof_transformation(data, cell_info[cell], block_size); };
373 }
374 }
375 }

395 template <typename U>
396 std::function<void(const std::span<U>&, const std::span<const std::uint32_t>&,
397 std::int32_t, int)>
398 get_post_dof_transformation_function(bool inverse = false,
399 bool transpose = false,
400 bool scalar_element = false) const
401 {
402 if (!needs_dof_transformations())
403 {
404 // If no permutation needed, return function that does nothing
405 return [](const std::span<U>&, const std::span<const std::uint32_t>&,
406 std::int32_t, int)
407 {
408 // Do nothing
409 };
410 }
411 else if (_sub_elements.size() != 0)
412 {
413 if (_bs == 1)
414 {
415 // Mixed element
416 std::vector<std::function<void(const std::span<U>&,
417 const std::span<const std::uint32_t>&,
418 std::int32_t, int)>>
419 sub_element_functions;
420 for (std::size_t i = 0; i < _sub_elements.size(); ++i)
421 {
422 sub_element_functions.push_back(
423 _sub_elements[i]
424 ->template get_post_dof_transformation_function<U>(
425 inverse, transpose));
426 }
427
428 return [this, sub_element_functions](
429 const std::span<U>& data,
430 const std::span<const std::uint32_t>& cell_info,
431 std::int32_t cell, int block_size)
432 {
433 std::size_t offset = 0;
434 for (std::size_t e = 0; e < sub_element_functions.size(); ++e)
435 {
436 sub_element_functions[e](data.subspan(offset, data.size() - offset),
437 cell_info, cell, block_size);
438 offset += _sub_elements[e]->space_dimension();
439 }
440 };
441 }
442 else if (!scalar_element)
443 {
444 // Vector element
445 const std::function<void(const std::span<U>&,
446 const std::span<const std::uint32_t>&,
447 std::int32_t, int)>
448 sub_function
449 = _sub_elements[0]->template get_pre_dof_transformation_function<U>(
450 inverse, transpose);
451 return [this,
452 sub_function](const std::span<U>& data,
453 const std::span<const std::uint32_t>& cell_info,
454 std::int32_t cell, int data_block_size)
455 {
456 const int ebs = block_size();
457 const std::size_t dof_count = data.size() / data_block_size;
458 for (int block = 0; block < data_block_size; ++block)
459 {
460 sub_function(data.subspan(block * dof_count, dof_count), cell_info,
461 cell, ebs);
462 }
463 };
464 }
465 }
466
467 if (transpose)
468 {
469 if (inverse)
470 {
471 return [this](const std::span<U>& data,
472 const std::span<const std::uint32_t>& cell_info,
473 std::int32_t cell, int block_size)
474 {
475 post_apply_inverse_transpose_dof_transformation(data, cell_info[cell],
476 block_size);
477 };
478 }
479 else
480 {
481 return [this](const std::span<U>& data,
482 const std::span<const std::uint32_t>& cell_info,
483 std::int32_t cell, int block_size) {
484 post_apply_transpose_dof_transformation(data, cell_info[cell],
485 block_size);
486 };
487 }
488 }
489 else
490 {
491 if (inverse)
492 {
493 return [this](const std::span<U>& data,
494 const std::span<const std::uint32_t>& cell_info,
495 std::int32_t cell, int block_size) {
496 post_apply_inverse_dof_transformation(data, cell_info[cell],
497 block_size);
498 };
499 }
500 else
501 {
502 return [this](const std::span<U>& data,
503 const std::span<const std::uint32_t>& cell_info,
504 std::int32_t cell, int block_size)
505 { post_apply_dof_transformation(data, cell_info[cell], block_size); };
506 }
507 }
