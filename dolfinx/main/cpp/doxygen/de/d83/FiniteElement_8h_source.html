325 template <typename U>
326 std::function<void(std::span<U>, std::span<const std::uint32_t>, std::int32_t,
327 int)>
328 dof_transformation_fn(doftransform ttype, bool scalar_element = false) const
329 {
330 if (!needs_dof_transformations())
331 {
332 // If no permutation needed, return function that does nothing
333 return [](std::span<U>, std::span<const std::uint32_t>, std::int32_t, int)
334 {
335 // Do nothing
336 };
337 }
338
339 if (!_sub_elements.empty())
340 {
341 if (!_reference_value_shape) // Mixed element
342 {
343 std::vector<std::function<void(
344 std::span<U>, std::span<const std::uint32_t>, std::int32_t, int)>>
345 sub_element_fns;
346 std::vector<int> dims;
347 for (std::size_t i = 0; i < _sub_elements.size(); ++i)
348 {
349 sub_element_fns.push_back(
350 _sub_elements[i]->template dof_transformation_fn<U>(ttype));
351 dims.push_back(_sub_elements[i]->space_dimension());
352 }
353
354 return [dims, sub_element_fns](std::span<U> data,
355 std::span<const std::uint32_t> cell_info,
356 std::int32_t cell, int block_size)
357 {
358 std::size_t offset = 0;
359 for (std::size_t e = 0; e < sub_element_fns.size(); ++e)
360 {
361 const std::size_t width = dims[e] * block_size;
362 sub_element_fns[e](data.subspan(offset, width), cell_info, cell,
363 block_size);
364 offset += width;
365 }
366 };
367 }
368 else if (!scalar_element)
369 {
370 // Blocked element
371 std::function<void(std::span<U>, std::span<const std::uint32_t>,
372 std::int32_t, int)>
373 sub_fn = _sub_elements[0]->template dof_transformation_fn<U>(ttype);
374 const int ebs = _bs;
375 return [ebs, sub_fn](std::span<U> data,
376 std::span<const std::uint32_t> cell_info,
377 std::int32_t cell, int data_block_size)
378 { sub_fn(data, cell_info, cell, ebs * data_block_size); };
379 }
380 }
381
382 switch (ttype)
383 {
384 case doftransform::inverse_transpose:
385 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
386 std::int32_t cell, int block_size)
387 { Tt_inv_apply(data, cell_info[cell], block_size); };
388 case doftransform::transpose:
389 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
390 std::int32_t cell, int block_size)
391 { Tt_apply(data, cell_info[cell], block_size); };
392 case doftransform::inverse:
393 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
394 std::int32_t cell, int block_size)
395 { Tinv_apply(data, cell_info[cell], block_size); };
396 case doftransform::standard:
397 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
398 std::int32_t cell, int block_size)
399 { T_apply(data, cell_info[cell], block_size); };
400 default:
401 throw std::runtime_error("Unknown transformation type");
402 }
403 }

428 template <typename U>
429 std::function<void(std::span<U>, std::span<const std::uint32_t>, std::int32_t,
430 int)>
431 dof_transformation_right_fn(doftransform ttype,
432 bool scalar_element = false) const
433 {
434 if (!needs_dof_transformations())
435 {
436 // If no permutation needed, return function that does nothing
437 return [](std::span<U>, std::span<const std::uint32_t>, std::int32_t, int)
438 {
439 // Do nothing
440 };
441 }
442 else if (!_sub_elements.empty())
443 {
444 if (!_reference_value_shape) // Mixed element
445 {
446 std::vector<std::function<void(
447 std::span<U>, std::span<const std::uint32_t>, std::int32_t, int)>>
448 sub_element_fns;
449 for (std::size_t i = 0; i < _sub_elements.size(); ++i)
450 {
451 sub_element_fns.push_back(
452 _sub_elements[i]->template dof_transformation_right_fn<U>(ttype));
453 }
454
455 return [this, sub_element_fns](std::span<U> data,
456 std::span<const std::uint32_t> cell_info,
457 std::int32_t cell, int block_size)
458 {
459 std::size_t offset = 0;
460 for (std::size_t e = 0; e < sub_element_fns.size(); ++e)
461 {
462 sub_element_fns[e](data.subspan(offset, data.size() - offset),
463 cell_info, cell, block_size);
464 offset += _sub_elements[e]->space_dimension();
465 }
466 };
467 }
468 else if (!scalar_element)
469 {
470 // Blocked element
471 // The transformation from the left can be used here as blocked
472 // elements use xyzxyzxyz ordering, and so applying the DOF
473 // transformation from the right is equivalent to applying the DOF
474 // transformation from the left to data using xxxyyyzzz ordering
475 std::function<void(std::span<U>, std::span<const std::uint32_t>,
476 std::int32_t, int)>
477 sub_fn = _sub_elements[0]->template dof_transformation_fn<U>(ttype);
478 return [this, sub_fn](std::span<U> data,
479 std::span<const std::uint32_t> cell_info,
480 std::int32_t cell, int data_block_size)
481 {
482 const int ebs = block_size();
483 const std::size_t dof_count = data.size() / data_block_size;
484 for (int block = 0; block < data_block_size; ++block)
485 {
486 sub_fn(data.subspan(block * dof_count, dof_count), cell_info, cell,
487 ebs);
488 }
489 };
490 }
491 }
492
493 switch (ttype)
494 {
495 case doftransform::inverse_transpose:
496 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
497 std::int32_t cell, int n)
498 { Tt_inv_apply_right(data, cell_info[cell], n); };
499 case doftransform::transpose:
500 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
501 std::int32_t cell, int n)
502 { Tt_apply_right(data, cell_info[cell], n); };
503 case doftransform::inverse:
504 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
505 std::int32_t cell, int n)
506 { Tinv_apply_right(data, cell_info[cell], n); };
507 case doftransform::standard:
508 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
509 std::int32_t cell, int n)
510 { T_apply_right(data, cell_info[cell], n); };
511 default:
512 throw std::runtime_error("Unknown transformation type");
513 }
514 }
