293 template <typename U>
294 std::function<void(std::span<U>, std::span<const std::uint32_t>, std::int32_t,
295 int)>
296 dof_transformation_fn(doftransform ttype, bool scalar_element = false) const
297 {
298 if (!needs_dof_transformations())
299 {
300 // If no permutation needed, return function that does nothing
301 return [](std::span<U>, std::span<const std::uint32_t>, std::int32_t, int)
302 {
303 // Do nothing
304 };
305 }
306
307 if (!_sub_elements.empty())
308 {
309 if (_is_mixed)
310 {
311 // Mixed element
312 std::vector<std::function<void(
313 std::span<U>, std::span<const std::uint32_t>, std::int32_t, int)>>
314 sub_element_functions;
315 std::vector<int> dims;
316 for (std::size_t i = 0; i < _sub_elements.size(); ++i)
317 {
318 sub_element_functions.push_back(
319 _sub_elements[i]->template dof_transformation_fn<U>(ttype));
320 dims.push_back(_sub_elements[i]->space_dimension());
321 }
322
323 return [dims, sub_element_functions](
324 std::span<U> data, std::span<const std::uint32_t> cell_info,
325 std::int32_t cell, int block_size)
326 {
327 std::size_t offset = 0;
328 for (std::size_t e = 0; e < sub_element_functions.size(); ++e)
329 {
330 const std::size_t width = dims[e] * block_size;
331 sub_element_functions[e](data.subspan(offset, width), cell_info,
332 cell, block_size);
333 offset += width;
334 }
335 };
336 }
337 else if (!scalar_element)
338 {
339 // Vector element
340 std::function<void(std::span<U>, std::span<const std::uint32_t>,
341 std::int32_t, int)>
342 sub_function
343 = _sub_elements[0]->template dof_transformation_fn<U>(ttype);
344 const int ebs = _bs;
345 return [ebs, sub_function](std::span<U> data,
346 std::span<const std::uint32_t> cell_info,
347 std::int32_t cell, int data_block_size)
348 { sub_function(data, cell_info, cell, ebs * data_block_size); };
349 }
350 }
351
352 switch (ttype)
353 {
354 case doftransform::inverse_transpose:
355 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
356 std::int32_t cell, int block_size)
357 { Tt_inv_apply(data, cell_info[cell], block_size); };
358 case doftransform::transpose:
359 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
360 std::int32_t cell, int block_size)
361 { Tt_apply(data, cell_info[cell], block_size); };
362 case doftransform::inverse:
363 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
364 std::int32_t cell, int block_size)
365 { Tinv_apply(data, cell_info[cell], block_size); };
366 case doftransform::standard:
367 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
368 std::int32_t cell, int block_size)
369 { T_apply(data, cell_info[cell], block_size); };
370 default:
371 throw std::runtime_error("Unknown transformation type");
372 }
373 }

398 template <typename U>
399 std::function<void(std::span<U>, std::span<const std::uint32_t>, std::int32_t,
400 int)>
401 dof_transformation_right_fn(doftransform ttype,
402 bool scalar_element = false) const
403 {
404 if (!needs_dof_transformations())
405 {
406 // If no permutation needed, return function that does nothing
407 return [](std::span<U>, std::span<const std::uint32_t>, std::int32_t, int)
408 {
409 // Do nothing
410 };
411 }
412 else if (_sub_elements.size() != 0)
413 {
414 if (_is_mixed)
415 {
416 // Mixed element
417 std::vector<std::function<void(
418 std::span<U>, std::span<const std::uint32_t>, std::int32_t, int)>>
419 sub_element_functions;
420 for (std::size_t i = 0; i < _sub_elements.size(); ++i)
421 {
422 sub_element_functions.push_back(
423 _sub_elements[i]->template dof_transformation_right_fn<U>(ttype));
424 }
425
426 return [this, sub_element_functions](
427 std::span<U> data, std::span<const std::uint32_t> cell_info,
428 std::int32_t cell, int block_size)
429 {
430 std::size_t offset = 0;
431 for (std::size_t e = 0; e < sub_element_functions.size(); ++e)
432 {
433 sub_element_functions[e](data.subspan(offset, data.size() - offset),
434 cell_info, cell, block_size);
435 offset += _sub_elements[e]->space_dimension();
436 }
437 };
438 }
439 else if (!scalar_element)
440 {
441 // Vector element
442 std::function<void(std::span<U>, std::span<const std::uint32_t>,
443 std::int32_t, int)>
444 sub_function
445 = _sub_elements[0]->template dof_transformation_fn<U>(ttype);
446 return [this, sub_function](std::span<U> data,
447 std::span<const std::uint32_t> cell_info,
448 std::int32_t cell, int data_block_size)
449 {
450 const int ebs = block_size();
451 const std::size_t dof_count = data.size() / data_block_size;
452 for (int block = 0; block < data_block_size; ++block)
453 {
454 sub_function(data.subspan(block * dof_count, dof_count), cell_info,
455 cell, ebs);
456 }
457 };
458 }
459 }
460
461 switch (ttype)
462 {
463 case doftransform::inverse_transpose:
464 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
465 std::int32_t cell, int n)
466 { Tt_inv_apply_right(data, cell_info[cell], n); };
467 case doftransform::transpose:
468 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
469 std::int32_t cell, int n)
470 { Tt_apply_right(data, cell_info[cell], n); };
471 case doftransform::inverse:
472 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
473 std::int32_t cell, int n)
474 { Tinv_apply_right(data, cell_info[cell], n); };
475 case doftransform::standard:
476 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
477 std::int32_t cell, int n)
478 { T_apply_right(data, cell_info[cell], n); };
479 default:
480 throw std::runtime_error("Unknown transformation type");
481 }
482 }
