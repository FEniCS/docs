276  template <typename T>
277  std::function<void(const xtl::span<T>&, const xtl::span<const std::uint32_t>&,
278  std::int32_t, int)>
279  get_dof_transformation_function(bool inverse = false, bool transpose = false,
280  bool scalar_element = false) const
281  {
282  if (!needs_dof_transformations())
283  {
284  // If no permutation needed, return function that does nothing
285  return [](const xtl::span<T>&, const xtl::span<const std::uint32_t>&,
286  std::int32_t, int)
287  {
288  // Do nothing
289  };
290  }
291 
292  if (_sub_elements.size() != 0)
293  {
294  if (_bs == 1)
295  {
296  // Mixed element
297  std::vector<std::function<void(const xtl::span<T>&,
298  const xtl::span<const std::uint32_t>&,
299  std::int32_t, int)>>
300  sub_element_functions;
301  std::vector<int> dims;
302  for (std::size_t i = 0; i < _sub_elements.size(); ++i)
303  {
304  sub_element_functions.push_back(
305  _sub_elements[i]->get_dof_transformation_function<T>(inverse,
306  transpose));
307  dims.push_back(_sub_elements[i]->space_dimension());
308  }
309 
310  return [dims, sub_element_functions](
311  const xtl::span<T>& data,
312  const xtl::span<const std::uint32_t>& cell_info,
313  std::int32_t cell, int block_size)
314  {
315  std::size_t offset = 0;
316  for (std::size_t e = 0; e < sub_element_functions.size(); ++e)
317  {
318  const std::size_t width = dims[e] * block_size;
319  sub_element_functions[e](data.subspan(offset, width), cell_info,
320  cell, block_size);
321  offset += width;
322  }
323  };
324  }
325  else if (!scalar_element)
326  {
327  // Vector element
328  const std::function<void(const xtl::span<T>&,
329  const xtl::span<const std::uint32_t>&,
330  std::int32_t, int)>
331  sub_function = _sub_elements[0]->get_dof_transformation_function<T>(
332  inverse, transpose);
333  const int ebs = _bs;
334  return
335  [ebs, sub_function](const xtl::span<T>& data,
336  const xtl::span<const std::uint32_t>& cell_info,
337  std::int32_t cell, int data_block_size)
338  { sub_function(data, cell_info, cell, ebs * data_block_size); };
339  }
340  }
341  if (transpose)
342  {
343  if (inverse)
344  {
345  return [this](const xtl::span<T>& data,
346  const xtl::span<const std::uint32_t>& cell_info,
347  std::int32_t cell, int block_size)
348  {
349  apply_inverse_transpose_dof_transformation(data, cell_info[cell],
350  block_size);
351  };
352  }
353  else
354  {
355  return [this](const xtl::span<T>& data,
356  const xtl::span<const std::uint32_t>& cell_info,
357  std::int32_t cell, int block_size) {
358  apply_transpose_dof_transformation(data, cell_info[cell], block_size);
359  };
360  }
361  }
362  else
363  {
364  if (inverse)
365  {
366  return [this](const xtl::span<T>& data,
367  const xtl::span<const std::uint32_t>& cell_info,
368  std::int32_t cell, int block_size) {
369  apply_inverse_dof_transformation(data, cell_info[cell], block_size);
370  };
371  }
372  else
373  {
374  return [this](const xtl::span<T>& data,
375  const xtl::span<const std::uint32_t>& cell_info,
376  std::int32_t cell, int block_size)
377  { apply_dof_transformation(data, cell_info[cell], block_size); };
378  }
379  }
380  }

400  template <typename T>
401  std::function<void(const xtl::span<T>&, const xtl::span<const std::uint32_t>&,
402  std::int32_t, int)>
403  get_dof_transformation_to_transpose_function(bool inverse = false,
404  bool transpose = false,
405  bool scalar_element
406  = false) const
407  {
408  if (!needs_dof_transformations())
409  {
410  // If no permutation needed, return function that does nothing
411  return [](const xtl::span<T>&, const xtl::span<const std::uint32_t>&,
412  std::int32_t, int)
413  {
414  // Do nothing
415  };
416  }
417  else if (_sub_elements.size() != 0)
418  {
419  if (_bs == 1)
420  {
421  // Mixed element
422  std::vector<std::function<void(const xtl::span<T>&,
423  const xtl::span<const std::uint32_t>&,
424  std::int32_t, int)>>
425  sub_element_functions;
426  for (std::size_t i = 0; i < _sub_elements.size(); ++i)
427  {
428  sub_element_functions.push_back(
429  _sub_elements[i]->get_dof_transformation_to_transpose_function<T>(
430  inverse, transpose));
431  }
432 
433  return [this, sub_element_functions](
434  const xtl::span<T>& data,
435  const xtl::span<const std::uint32_t>& cell_info,
436  std::int32_t cell, int block_size)
437  {
438  std::size_t offset = 0;
439  for (std::size_t e = 0; e < sub_element_functions.size(); ++e)
440  {
441  sub_element_functions[e](data.subspan(offset, data.size() - offset),
442  cell_info, cell, block_size);
443  offset += _sub_elements[e]->space_dimension();
444  }
445  };
446  }
447  else if (!scalar_element)
448  {
449  // Vector element
450  const std::function<void(const xtl::span<T>&,
451  const xtl::span<const std::uint32_t>&,
452  std::int32_t, int)>
453  sub_function = _sub_elements[0]->get_dof_transformation_function<T>(
454  inverse, transpose);
455  return [this,
456  sub_function](const xtl::span<T>& data,
457  const xtl::span<const std::uint32_t>& cell_info,
458  std::int32_t cell, int data_block_size)
459  {
460  const int ebs = block_size();
461  const std::size_t dof_count = data.size() / data_block_size;
462  for (int block = 0; block < data_block_size; ++block)
463  {
464  sub_function(data.subspan(block * dof_count, dof_count), cell_info,
465  cell, ebs);
466  }
467  };
468  }
469  }
470 
471  if (transpose)
472  {
473  if (inverse)
474  {
475  return [this](const xtl::span<T>& data,
476  const xtl::span<const std::uint32_t>& cell_info,
477  std::int32_t cell, int block_size)
478  {
479  apply_inverse_transpose_dof_transformation_to_transpose(
480  data, cell_info[cell], block_size);
481  };
482  }
483  else
484  {
485  return [this](const xtl::span<T>& data,
486  const xtl::span<const std::uint32_t>& cell_info,
487  std::int32_t cell, int block_size)
488  {
489  apply_transpose_dof_transformation_to_transpose(data, cell_info[cell],
490  block_size);
491  };
492  }
493  }
494  else
495  {
496  if (inverse)
497  {
498  return [this](const xtl::span<T>& data,
499  const xtl::span<const std::uint32_t>& cell_info,
500  std::int32_t cell, int block_size)
501  {
502  apply_inverse_dof_transformation_to_transpose(data, cell_info[cell],
503  block_size);
504  };
505  }
506  else
507  {
508  return [this](const xtl::span<T>& data,
509  const xtl::span<const std::uint32_t>& cell_info,
510  std::int32_t cell, int block_size) {
511  apply_dof_transformation_to_transpose(data, cell_info[cell],
512  block_size);
513  };
514  }
515  }
516  }
