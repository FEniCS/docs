271 template <typename U>
272 std::function<void(const std::span<U>&, const std::span<const std::uint32_t>&,
273 std::int32_t, int)>
274 get_pre_dof_transformation_function(doftransform ttype
275 = doftransform::standard,
276 bool scalar_element = false) const
277 {
278 if (!needs_dof_transformations())
279 {
280 // If no permutation needed, return function that does nothing
281 return [](const std::span<U>&, const std::span<const std::uint32_t>&,
282 std::int32_t, int)
283 {
284 // Do nothing
285 };
286 }
287
288 if (_sub_elements.size() != 0)
289 {
290 if (_is_mixed)
291 {
292 // Mixed element
293 std::vector<std::function<void(const std::span<U>&,
294 const std::span<const std::uint32_t>&,
295 std::int32_t, int)>>
296 sub_element_functions;
297 std::vector<int> dims;
298 for (std::size_t i = 0; i < _sub_elements.size(); ++i)
299 {
300 sub_element_functions.push_back(
301 _sub_elements[i]->template get_pre_dof_transformation_function<U>(
302 ttype));
303 dims.push_back(_sub_elements[i]->space_dimension());
304 }
305
306 return [dims, sub_element_functions](
307 const std::span<U>& data,
308 const std::span<const std::uint32_t>& cell_info,
309 std::int32_t cell, int block_size)
310 {
311 std::size_t offset = 0;
312 for (std::size_t e = 0; e < sub_element_functions.size(); ++e)
313 {
314 const std::size_t width = dims[e] * block_size;
315 sub_element_functions[e](data.subspan(offset, width), cell_info,
316 cell, block_size);
317 offset += width;
318 }
319 };
320 }
321 else if (!scalar_element)
322 {
323 // Vector element
324 const std::function<void(const std::span<U>&,
325 const std::span<const std::uint32_t>&,
326 std::int32_t, int)>
327 sub_function
328 = _sub_elements[0]->template get_pre_dof_transformation_function<U>(
329 ttype);
330 const int ebs = _bs;
331 return
332 [ebs, sub_function](const std::span<U>& data,
333 const std::span<const std::uint32_t>& cell_info,
334 std::int32_t cell, int data_block_size)
335 { sub_function(data, cell_info, cell, ebs * data_block_size); };
336 }
337 }
338 switch (ttype)
339 {
340 case doftransform::inverse_transpose:
341 return [this](const std::span<U>& data,
342 const std::span<const std::uint32_t>& cell_info,
343 std::int32_t cell, int block_size)
344 {
345 pre_apply_inverse_transpose_dof_transformation(data, cell_info[cell],
346 block_size);
347 };
348 case doftransform::transpose:
349 return [this](const std::span<U>& data,
350 const std::span<const std::uint32_t>& cell_info,
351 std::int32_t cell, int block_size) {
352 pre_apply_transpose_dof_transformation(data, cell_info[cell],
353 block_size);
354 };
355 case doftransform::inverse:
356 return [this](const std::span<U>& data,
357 const std::span<const std::uint32_t>& cell_info,
358 std::int32_t cell, int block_size) {
359 pre_apply_inverse_dof_transformation(data, cell_info[cell], block_size);
360 };
361 case doftransform::standard:
362 return [this](const std::span<U>& data,
363 const std::span<const std::uint32_t>& cell_info,
364 std::int32_t cell, int block_size)
365 { pre_apply_dof_transformation(data, cell_info[cell], block_size); };
366 default:
367 throw std::runtime_error("Unknown transformation type");
368 }
369 }
385 template <typename U>
386 std::function<void(const std::span<U>&, const std::span<const std::uint32_t>&,
387 std::int32_t, int)>
388 get_post_dof_transformation_function(doftransform ttype
389 = doftransform::standard,
390 bool scalar_element = false) const
391 {
392 if (!needs_dof_transformations())
393 {
394 // If no permutation needed, return function that does nothing
395 return [](const std::span<U>&, const std::span<const std::uint32_t>&,
396 std::int32_t, int)
397 {
398 // Do nothing
399 };
400 }
401 else if (_sub_elements.size() != 0)
402 {
403 if (_is_mixed)
404 {
405 // Mixed element
406 std::vector<std::function<void(const std::span<U>&,
407 const std::span<const std::uint32_t>&,
408 std::int32_t, int)>>
409 sub_element_functions;
410 for (std::size_t i = 0; i < _sub_elements.size(); ++i)
411 {
412 sub_element_functions.push_back(
413 _sub_elements[i]
414 ->template get_post_dof_transformation_function<U>(ttype));
415 }
416
417 return [this, sub_element_functions](
418 const std::span<U>& data,
419 const std::span<const std::uint32_t>& cell_info,
420 std::int32_t cell, int block_size)
421 {
422 std::size_t offset = 0;
423 for (std::size_t e = 0; e < sub_element_functions.size(); ++e)
424 {
425 sub_element_functions[e](data.subspan(offset, data.size() - offset),
426 cell_info, cell, block_size);
427 offset += _sub_elements[e]->space_dimension();
428 }
429 };
430 }
431 else if (!scalar_element)
432 {
433 // Vector element
434 const std::function<void(const std::span<U>&,
435 const std::span<const std::uint32_t>&,
436 std::int32_t, int)>
437 sub_function
438 = _sub_elements[0]->template get_pre_dof_transformation_function<U>(
439 ttype);
440 return [this,
441 sub_function](const std::span<U>& data,
442 const std::span<const std::uint32_t>& cell_info,
443 std::int32_t cell, int data_block_size)
444 {
445 const int ebs = block_size();
446 const std::size_t dof_count = data.size() / data_block_size;
447 for (int block = 0; block < data_block_size; ++block)
448 {
449 sub_function(data.subspan(block * dof_count, dof_count), cell_info,
450 cell, ebs);
451 }
452 };
453 }
454 }
455
456 switch (ttype)
457 {
458 case doftransform::inverse_transpose:
459 return [this](const std::span<U>& data,
460 const std::span<const std::uint32_t>& cell_info,
461 std::int32_t cell, int block_size)
462 {
463 post_apply_inverse_transpose_dof_transformation(data, cell_info[cell],
464 block_size);
465 };
466 case doftransform::transpose:
467 return [this](const std::span<U>& data,
468 const std::span<const std::uint32_t>& cell_info,
469 std::int32_t cell, int block_size) {
470 post_apply_transpose_dof_transformation(data, cell_info[cell],
471 block_size);
472 };
473 case doftransform::inverse:
474 return [this](const std::span<U>& data,
475 const std::span<const std::uint32_t>& cell_info,
476 std::int32_t cell, int block_size) {
477 post_apply_inverse_dof_transformation(data, cell_info[cell],
478 block_size);
479 };
480 case doftransform::standard:
481 return [this](const std::span<U>& data,
482 const std::span<const std::uint32_t>& cell_info,
483 std::int32_t cell, int block_size)
484 { post_apply_dof_transformation(data, cell_info[cell], block_size); };
485 default:
486 throw std::runtime_error("Unknown transformation type");
487 }
488 }
