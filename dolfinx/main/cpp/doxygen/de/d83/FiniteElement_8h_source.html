270 template <typename U>
271 std::function<void(std::span<U>, std::span<const std::uint32_t>, std::int32_t,
272 int)>
273 get_pre_dof_transformation_function(doftransform ttype
274 = doftransform::standard,
275 bool scalar_element = false) const
276 {
277 if (!needs_dof_transformations())
278 {
279 // If no permutation needed, return function that does nothing
280 return [](std::span<U>, std::span<const std::uint32_t>, std::int32_t, int)
281 {
282 // Do nothing
283 };
284 }
285
286 if (_sub_elements.size() != 0)
287 {
288 if (_is_mixed)
289 {
290 // Mixed element
291 std::vector<std::function<void(
292 std::span<U>, std::span<const std::uint32_t>, std::int32_t, int)>>
293 sub_element_functions;
294 std::vector<int> dims;
295 for (std::size_t i = 0; i < _sub_elements.size(); ++i)
296 {
297 sub_element_functions.push_back(
298 _sub_elements[i]->template get_pre_dof_transformation_function<U>(
299 ttype));
300 dims.push_back(_sub_elements[i]->space_dimension());
301 }
302
303 return [dims, sub_element_functions](
304 std::span<U> data, std::span<const std::uint32_t> cell_info,
305 std::int32_t cell, int block_size)
306 {
307 std::size_t offset = 0;
308 for (std::size_t e = 0; e < sub_element_functions.size(); ++e)
309 {
310 const std::size_t width = dims[e] * block_size;
311 sub_element_functions[e](data.subspan(offset, width), cell_info,
312 cell, block_size);
313 offset += width;
314 }
315 };
316 }
317 else if (!scalar_element)
318 {
319 // Vector element
320 const std::function<void(std::span<U>, std::span<const std::uint32_t>,
321 std::int32_t, int)>
322 sub_function
323 = _sub_elements[0]->template get_pre_dof_transformation_function<U>(
324 ttype);
325 const int ebs = _bs;
326 return [ebs, sub_function](std::span<U> data,
327 std::span<const std::uint32_t> cell_info,
328 std::int32_t cell, int data_block_size)
329 { sub_function(data, cell_info, cell, ebs * data_block_size); };
330 }
331 }
332 switch (ttype)
333 {
334 case doftransform::inverse_transpose:
335 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
336 std::int32_t cell, int block_size)
337 {
338 pre_apply_inverse_transpose_dof_transformation(data, cell_info[cell],
339 block_size);
340 };
341 case doftransform::transpose:
342 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
343 std::int32_t cell, int block_size) {
344 pre_apply_transpose_dof_transformation(data, cell_info[cell],
345 block_size);
346 };
347 case doftransform::inverse:
348 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
349 std::int32_t cell, int block_size) {
350 pre_apply_inverse_dof_transformation(data, cell_info[cell], block_size);
351 };
352 case doftransform::standard:
353 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
354 std::int32_t cell, int block_size)
355 { pre_apply_dof_transformation(data, cell_info[cell], block_size); };
356 default:
357 throw std::runtime_error("Unknown transformation type");
358 }
359 }
375 template <typename U>
376 std::function<void(std::span<U>, std::span<const std::uint32_t>, std::int32_t,
377 int)>
378 get_post_dof_transformation_function(doftransform ttype
379 = doftransform::standard,
380 bool scalar_element = false) const
381 {
382 if (!needs_dof_transformations())
383 {
384 // If no permutation needed, return function that does nothing
385 return [](std::span<U>, std::span<const std::uint32_t>, std::int32_t, int)
386 {
387 // Do nothing
388 };
389 }
390 else if (_sub_elements.size() != 0)
391 {
392 if (_is_mixed)
393 {
394 // Mixed element
395 std::vector<std::function<void(
396 std::span<U>, std::span<const std::uint32_t>, std::int32_t, int)>>
397 sub_element_functions;
398 for (std::size_t i = 0; i < _sub_elements.size(); ++i)
399 {
400 sub_element_functions.push_back(
401 _sub_elements[i]
402 ->template get_post_dof_transformation_function<U>(ttype));
403 }
404
405 return [this, sub_element_functions](
406 std::span<U> data, std::span<const std::uint32_t> cell_info,
407 std::int32_t cell, int block_size)
408 {
409 std::size_t offset = 0;
410 for (std::size_t e = 0; e < sub_element_functions.size(); ++e)
411 {
412 sub_element_functions[e](data.subspan(offset, data.size() - offset),
413 cell_info, cell, block_size);
414 offset += _sub_elements[e]->space_dimension();
415 }
416 };
417 }
418 else if (!scalar_element)
419 {
420 // Vector element
421 const std::function<void(std::span<U>, std::span<const std::uint32_t>,
422 std::int32_t, int)>
423 sub_function
424 = _sub_elements[0]->template get_pre_dof_transformation_function<U>(
425 ttype);
426 return [this, sub_function](std::span<U> data,
427 std::span<const std::uint32_t> cell_info,
428 std::int32_t cell, int data_block_size)
429 {
430 const int ebs = block_size();
431 const std::size_t dof_count = data.size() / data_block_size;
432 for (int block = 0; block < data_block_size; ++block)
433 {
434 sub_function(data.subspan(block * dof_count, dof_count), cell_info,
435 cell, ebs);
436 }
437 };
438 }
439 }
440
441 switch (ttype)
442 {
443 case doftransform::inverse_transpose:
444 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
445 std::int32_t cell, int block_size)
446 {
447 post_apply_inverse_transpose_dof_transformation(data, cell_info[cell],
448 block_size);
449 };
450 case doftransform::transpose:
451 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
452 std::int32_t cell, int block_size) {
453 post_apply_transpose_dof_transformation(data, cell_info[cell],
454 block_size);
455 };
456 case doftransform::inverse:
457 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
458 std::int32_t cell, int block_size) {
459 post_apply_inverse_dof_transformation(data, cell_info[cell],
460 block_size);
461 };
462 case doftransform::standard:
463 return [this](std::span<U> data, std::span<const std::uint32_t> cell_info,
464 std::int32_t cell, int block_size)
465 { post_apply_dof_transformation(data, cell_info[cell], block_size); };
466 default:
467 throw std::runtime_error("Unknown transformation type");
468 }
469 }
