61template <dolfinx::scalar T, std::floating_point U>
62void tabulate_expression(
63 std::span<T> values, fem::FEkernel<T> auto fn,
64 std::array<std::size_t, 2> Xshape, std::size_t value_size,
65 std::size_t num_argument_dofs,
66 MDSPAN_IMPL_STANDARD_NAMESPACE::mdspan<
67 const std::int32_t,
68 MDSPAN_IMPL_STANDARD_NAMESPACE::dextents<std::size_t, 2>>
69 x_dofmap,
70 std::span<const scalar_value_type_t<T>> x,
71 MDSPAN_IMPL_STANDARD_NAMESPACE::mdspan<
72 const T, MDSPAN_IMPL_STANDARD_NAMESPACE::dextents<std::size_t, 2>>
73 coeffs,
74 std::span<const T> constants, fem::MDSpan2 auto entities,
75 std::span<const std::uint32_t> cell_info,
76 fem::DofTransformKernel<T> auto P0)
77{
78 namespace md = MDSPAN_IMPL_STANDARD_NAMESPACE;
79 static_assert(entities.rank() == 1 or entities.rank() == 2);
80
81 // Create data structures used in evaluation
82 std::vector<U> coord_dofs(3 * x_dofmap.extent(1));
83
84 // Iterate over cells and 'assemble' into values
85 int size0 = Xshape[0] * value_size;
86 std::vector<T> values_local(size0 * num_argument_dofs, 0);
87 std::size_t offset = values_local.size();
88 for (std::size_t e = 0; e < entities.extent(0); ++e)
89 {
90 std::ranges::fill(values_local, 0);
91 if constexpr (entities.rank() == 1)
92 {
93 std::int32_t entity = entities(e);
94 auto x_dofs = md::submdspan(x_dofmap, entity, md::full_extent);
95 for (std::size_t i = 0; i < x_dofs.size(); ++i)
96 {
97 std::copy_n(std::next(x.begin(), 3 * x_dofs[i]), 3,
98 std::next(coord_dofs.begin(), 3 * i));
99 }
100 fn(values_local.data(), &coeffs(e, 0), constants.data(),
101 coord_dofs.data(), nullptr, nullptr);
102 }
103 else
104 {
105 std::int32_t entity = entities(e, 0);
106 auto x_dofs = md::submdspan(x_dofmap, entity, md::full_extent);
107 for (std::size_t i = 0; i < x_dofs.size(); ++i)
108 {
109 std::copy_n(std::next(x.begin(), 3 * x_dofs[i]), 3,
110 std::next(coord_dofs.begin(), 3 * i));
111 }
112 fn(values_local.data(), &coeffs(e, 0), constants.data(),
113 coord_dofs.data(), &entities(e, 1), nullptr);
114 }
115
116 P0(values_local, cell_info, e, size0);
117 for (std::size_t j = 0; j < values_local.size(); ++j)
118 values[e * offset + j] = values_local[j];
119 }
120}

154template <dolfinx::scalar T, std::floating_point U>
155void tabulate_expression(
156 std::span<T> values, fem::FEkernel<T> auto fn,
157 std::array<std::size_t, 2> Xshape, std::size_t value_size,
158 MDSPAN_IMPL_STANDARD_NAMESPACE::mdspan<
159 const T, MDSPAN_IMPL_STANDARD_NAMESPACE::dextents<std::size_t, 2>>
160 coeffs,
161 std::span<const T> constants, const mesh::Mesh<U>& mesh,
162 fem::MDSpan2 auto entities,
163 std::optional<
164 std::pair<std::reference_wrapper<const FiniteElement<U>>, std::size_t>>
165 element)
166{
167 namespace md = MDSPAN_IMPL_STANDARD_NAMESPACE;
168
169 std::function<void(std::span<T>, std::span<const std::uint32_t>, std::int32_t,
170 int)>
171 post_dof_transform
172 = [](std::span<T>, std::span<const std::uint32_t>, std::int32_t, int)
173 {
174 // Do nothing
175 };
176
177 std::shared_ptr<const mesh::Topology> topology = mesh.topology();
178 assert(topology);
179 std::size_t num_argument_dofs = 1;
180 std::span<const std::uint32_t> cell_info;
181 if (element)
182 {
183 num_argument_dofs = element->second;
184 if (element->first.get().needs_dof_transformations())
185 {
186 mesh.topology_mutable()->create_entity_permutations();
187 cell_info = std::span(topology->get_cell_permutation_info());
188 post_dof_transform
189 = element->first.get().template dof_transformation_right_fn<T>(
190 doftransform::transpose);
191 }
192 }
193
194 tabulate_expression<T, U>(values, fn, Xshape, value_size, num_argument_dofs,
195 mesh.geometry().dofmap(), mesh.geometry().x(),
196 coeffs, constants, entities, cell_info,
197 post_dof_transform);
198}
