25template <std::floating_point T>
26mesh::Mesh<T>
27create_refined_mesh(const mesh::Mesh<T>& mesh,
28 const graph::AdjacencyList<std::int64_t>& cell_adj,
29 std::span<const T> new_vertex_coords,
30 std::array<std::size_t, 2> xshape, bool redistribute,
31 mesh::GhostMode ghost_mode)
32{
33 if (dolfinx::MPI::size(mesh.comm()) == 1)
34 {
35 // No parallel construction necessary, i.e. redistribute also has no
36 // effect
37 return mesh::create_mesh(mesh.comm(), cell_adj.array(),
38 mesh.geometry().cmap(), new_vertex_coords, xshape,
39 ghost_mode);
40 }
41 else
42 {
43 // Let partition handle the parallel construction of the mesh
44 return partition<T>(mesh, cell_adj, new_vertex_coords, xshape, redistribute,
45 ghost_mode);
46 }
47}
48} // namespace impl

64template <std::floating_point T>
65std::tuple<mesh::Mesh<T>, std::optional<std::vector<std::int32_t>>,
66 std::optional<std::vector<std::int8_t>>>
67refine(const mesh::Mesh<T>& mesh,
68 std::optional<std::span<const std::int32_t>> edges, bool redistribute,
69 mesh::GhostMode ghost_mode = mesh::GhostMode::shared_facet,
70 Option option = Option::none)
71{
72 auto topology = mesh.topology();
73 assert(topology);
74
75 mesh::CellType cell_t = topology->cell_type();
76 if (!mesh::is_simplex(cell_t))
77 throw std::runtime_error("Refinement only defined for simplices");
78 bool oned = topology->cell_type() == mesh::CellType::interval;
79 auto [cell_adj, new_vertex_coords, xshape, parent_cell, parent_facet]
80 = oned ? interval::compute_refinement_data(mesh, edges, option)
81 : plaza::compute_refinement_data(mesh, edges, option);
82
83 mesh::Mesh<T> refined_mesh = impl::create_refined_mesh(
84 mesh, cell_adj, std::span<const T>(new_vertex_coords), xshape,
85 redistribute, ghost_mode);
86
87 // Report the number of refined cellse
88 const int D = topology->dim();
89 const std::int64_t n0 = topology->index_map(D)->size_global();
90 const std::int64_t n1 = refined_mesh.topology()->index_map(D)->size_global();
91 spdlog::info(
92 "Number of cells increased from {} to {} ({}% increase).", n0, n1,
93 100.0 * (static_cast<double>(n1) / static_cast<double>(n0) - 1.0));
94
95 return {std::move(refined_mesh), std::move(parent_cell),
96 std::move(parent_facet)};
97}
