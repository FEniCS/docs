46public:

81 std::string tag, std::shared_ptr<const mesh::Mesh<T>> mesh)

84 // Do nothing

89

153template <class T>
154adios2::Variable<T> define_variable(adios2::IO& io, std::string name,
155 const adios2::Dims& shape = adios2::Dims(),
156 const adios2::Dims& start = adios2::Dims(),
157 const adios2::Dims& count = adios2::Dims())
158{
159 if (adios2::Variable<T> v = io.InquireVariable<T>(name); v)
160 {
161 if (v.Count() != count and v.ShapeID() == adios2::ShapeID::LocalArray)
162 v.SetSelection({start, count});
163 return v;
164 }
165 else
166 return io.DefineVariable<T>(name, shape, start, count);
167}

170template <typename T>
171std::shared_ptr<const mesh::Mesh<T>>
172extract_common_mesh(const typename ADIOS2Writer<T>::U& u)
173{
174 // Extract mesh from first function
175 assert(!u.empty());
176 auto mesh = std::visit(
177 [](const auto& u) { return u->function_space()->mesh(); }, u.front());
178 assert(mesh);
179
180 // Check that all functions share the same mesh
181 for (auto& v : u)
182 {
183 std::visit(
184 [&mesh](const auto& u)
185 {
186 if (mesh != u->function_space()->mesh())
187 {
188 throw std::runtime_error(
189 "ADIOS2Writer only supports functions sharing the same mesh");
190 }
191 },
192 v);
193 }
194
195 return mesh;
196}

212template <typename T>
213void initialize_mesh_attributes(adios2::IO& io, const mesh::Mesh<T>& mesh)
214{
215 const mesh::Geometry<T>& geometry = mesh.geometry();
216 auto topology = mesh.topology();
217 assert(topology);
218
219 // Check that mesh is first order mesh
220 const int num_dofs_g = geometry.cmaps()[0].dim();
221 const int num_vertices_per_cell
222 = mesh::cell_num_entities(topology->cell_types()[0], 0);
223 if (num_dofs_g != num_vertices_per_cell)
224 throw std::runtime_error("Fides only supports lowest-order meshes.");
225
226 // NOTE: If we start using mixed element types, we can change
227 // data-model to "unstructured"
228 impl_adios2::define_attribute<std::string>(io, "Fides_Data_Model",
229 "unstructured_single");
230
231 // Define FIDES attributes pointing to ADIOS2 Variables for geometry
232 // and topology
233 impl_adios2::define_attribute<std::string>(io, "Fides_Coordinates_Variable",
234 "points");
235 impl_adios2::define_attribute<std::string>(io, "Fides_Connectivity_Variable",
236 "connectivity");
237
238 std::string cell_type = to_fides_cell(topology->cell_types()[0]);
239 impl_adios2::define_attribute<std::string>(io, "Fides_Cell_Type", cell_type);
240
241 impl_adios2::define_attribute<std::string>(io, "Fides_Time_Variable", "step");
242}

248template <typename T>
249void initialize_function_attributes(adios2::IO& io,
250 const typename ADIOS2Writer<T>::U& u)
251{
252 // Array of function (name, cell association types) for each function
253 // added to the file
254 std::vector<std::array<std::string, 2>> u_data;
255 using X = decltype(u_data);
256 for (auto& v : u)
257 {
258 auto n = std::visit(
259 impl_adios2::overload{
260 [](const std::shared_ptr<const typename ADIOS2Writer<T>::Fd32>& u)
261 -> X {
262 return {{u->name, "points"}};
263 },
264 [](const std::shared_ptr<const typename ADIOS2Writer<T>::Fd64>& u)
265 -> X {
266 return {{u->name, "points"}};
267 },
268 [](const std::shared_ptr<const typename ADIOS2Writer<T>::Fc64>& u)
269 -> X
270 {
271 return {{u->name + impl_adios2::field_ext[0], "points"},
272 {u->name + impl_adios2::field_ext[1], "points"}};
273 },
274 [](const std::shared_ptr<const typename ADIOS2Writer<T>::Fc128>& u)
275 -> X
276 {
277 return {{u->name + impl_adios2::field_ext[0], "points"},
278 {u->name + impl_adios2::field_ext[1], "points"}};
279 }},
280 v);
281 u_data.insert(u_data.end(), n.begin(), n.end());
282 }
283
284 // Write field associations to file
285 if (adios2::Attribute<std::string> assc
286 = io.InquireAttribute<std::string>("Fides_Variable_Associations");
287 !assc)
288 {
289 std::vector<std::string> u_type;
290 std::transform(u_data.cbegin(), u_data.cend(), std::back_inserter(u_type),
291 [](auto& f) { return f[1]; });
292 io.DefineAttribute<std::string>("Fides_Variable_Associations",
293 u_type.data(), u_type.size());
294 }
295
296 // Write field pointers to file
297 if (adios2::Attribute<std::string> fields
298 = io.InquireAttribute<std::string>("Fides_Variable_List");
299 !fields)
300 {
301 std::vector<std::string> names;
302 std::transform(u_data.cbegin(), u_data.cend(), std::back_inserter(names),
303 [](auto& f) { return f[0]; });
304 io.DefineAttribute<std::string>("Fides_Variable_List", names.data(),
305 names.size());
306 }
307}

311template <typename T, std::floating_point U>
312std::vector<T> pack_function_data(const fem::Function<T, U>& u)
313{
314 auto V = u.function_space();
315 assert(V);
316 auto dofmap = V->dofmap();
317 assert(dofmap);
318 auto mesh = V->mesh();
319 assert(mesh);
320 const mesh::Geometry<U>& geometry = mesh->geometry();
321 auto topology = mesh->topology();
322 assert(topology);
323
324 // The Function and the mesh must have identical element_dof_layouts
325 // (up to the block size)
326 assert(dofmap->element_dof_layout()
327 == geometry.cmaps()[0].create_dof_layout());
328
329 const int tdim = topology->dim();
330 auto cell_map = topology->index_map(tdim);
331 assert(cell_map);
332 const std::int32_t num_cells
333 = cell_map->size_local() + cell_map->num_ghosts();
334
335 auto vertex_map = topology->index_map(0);
336 assert(vertex_map);
337 const std::uint32_t num_vertices
338 = vertex_map->size_local() + vertex_map->num_ghosts();
339
340 const int rank = V->element()->value_shape().size();
341 const std::uint32_t num_components = std::pow(3, rank);
342
343 // Get dof array and pack into array (padded where appropriate)
344 const graph::AdjacencyList<std::int32_t>& dofmap_x = geometry.dofmap();
345 const int bs = dofmap->bs();
346 const auto& u_data = u.x()->array();
347 std::vector<T> data(num_vertices * num_components, 0);
348 for (std::int32_t c = 0; c < num_cells; ++c)
349 {
350 auto dofs = dofmap->cell_dofs(c);
351 auto dofs_x = dofmap_x.links(c);
352 assert(dofs.size() == dofs_x.size());
353 for (std::size_t i = 0; i < dofs.size(); ++i)
354 for (int j = 0; j < bs; ++j)
355 data[num_components * dofs_x[i] + j] = u_data[bs * dofs[i] + j];
356 }
357 return data;
358}

366template <typename T, std::floating_point U>
367void write_data(adios2::IO& io, adios2::Engine& engine,
368 const fem::Function<T, U>& u)
369{
370 // FIXME: There is an implicit assumptions that u and the mesh have
371 // the same ElementDoflayout
372 auto V = u.function_space();
373 assert(V);
374 auto dofmap = V->dofmap();
375 assert(dofmap);
376 auto mesh = V->mesh();
377 assert(mesh);
378 const int gdim = mesh->geometry().dim();
379
380 // Vectors and tensor need padding in gdim < 3
381 const int rank = V->element()->value_shape().size();
382 const bool need_padding = rank > 0 and gdim != 3 ? true : false;
383
384 // Get vertex data. If the mesh and function dofmaps are the same we
385 // can work directly with the dof array.
386 std::span<const T> data;
387 std::vector<T> _data;
388 if (mesh->geometry().dofmap() == dofmap->list() and !need_padding)
389 data = u.x()->array();
390 else
391 {
392 _data = impl_fides::pack_function_data(u);
393 data = std::span<const T>(_data);
394 }
395
396 auto vertex_map = mesh->topology()->index_map(0);
397 assert(vertex_map);
398 const std::uint32_t num_vertices
399 = vertex_map->size_local() + vertex_map->num_ghosts();
400
401 // Write each real and imaginary part of the function
402 const std::uint32_t num_components = std::pow(3, rank);
403 assert(data.size() % num_components == 0);
404 if constexpr (std::is_scalar_v<T>)
405 {
406 // ---- Real
407 adios2::Variable<T> local_output = impl_adios2::define_variable<T>(
408 io, u.name, {}, {}, {num_vertices, num_components});
409
410 // To reuse out_data, we use sync mode here
411 engine.Put<T>(local_output, data.data());
412 engine.PerformPuts();
413 }
414 else
415 {
416 // ---- Complex
417 using X = typename T::value_type;
418
419 std::vector<X> data_real(data.size()), data_imag(data.size());
420
421 adios2::Variable<X> local_output_r = impl_adios2::define_variable<X>(
422 io, u.name + impl_adios2::field_ext[0], {}, {},
423 {num_vertices, num_components});
424 std::transform(data.begin(), data.end(), data_real.begin(),
425 [](auto& x) -> X { return std::real(x); });
426 engine.Put<X>(local_output_r, data_real.data());
427
428 adios2::Variable<X> local_output_c = impl_adios2::define_variable<X>(
429 io, u.name + impl_adios2::field_ext[1], {}, {},
430 {num_vertices, num_components});
431 std::transform(data.begin(), data.end(), data_imag.begin(),
432 [](auto& x) -> X { return std::imag(x); });
433 engine.Put<X>(local_output_c, data_imag.data());
434 engine.PerformPuts();
435 }
436}

442template <typename T>
443void write_mesh(adios2::IO& io, adios2::Engine& engine,
444 const mesh::Mesh<T>& mesh)
445{
446 const mesh::Geometry<T>& geometry = mesh.geometry();
447 auto topology = mesh.topology();
448 assert(topology);
449
450 // "Put" geometry data
451 auto x_map = geometry.index_map();
452 const std::uint32_t num_vertices = x_map->size_local() + x_map->num_ghosts();
453 adios2::Variable<T> local_geometry = impl_adios2::define_variable<T>(
454 io, "points", {}, {}, {num_vertices, 3});
455 engine.Put<T>(local_geometry, geometry.x().data());
456
457 // TODO: The DOLFINx and VTK topology are the same for some cell types
458 // - no need to repack via extract_vtk_connectivity in these cases
459
460 // Get topological dimenson, number of cells and number of 'nodes' per
461 // cell, and compute 'VTK' connectivity
462 const int tdim = topology->dim();
463 const std::int32_t num_cells = topology->index_map(tdim)->size_local();
464 const int num_nodes = geometry.cmaps()[0].dim();
465 const auto [cells, shape] = io::extract_vtk_connectivity(
466 mesh.geometry().dofmap(), topology->cell_types()[0]);
467
468 // "Put" topology data in the result in the ADIOS2 file
469 adios2::Variable<std::int64_t> local_topology
470 = impl_adios2::define_variable<std::int64_t>(
471 io, "connectivity", {}, {}, {std::size_t(num_cells * num_nodes)});
472 engine.Put<std::int64_t>(local_topology, cells.data());
473
474 engine.PerformPuts();
475}

490template <std::floating_point T>
491class FidesWriter : public ADIOS2Writer<T>
492{
493public:
500 FidesWriter(MPI_Comm comm, const std::filesystem::path& filename,
501 std::shared_ptr<const mesh::Mesh<T>> mesh)
502 : ADIOS2Writer<T>(comm, filename, "Fides mesh writer", mesh),
503 _mesh_reuse_policy(FidesMeshPolicy::update)
504 {
505 assert(this->_io);
506 assert(mesh);
507 impl_fides::initialize_mesh_attributes(*this->_io, *mesh);
508 }
509
519 FidesWriter(MPI_Comm comm, const std::filesystem::path& filename,
520 const typename ADIOS2Writer<T>::U& u,
521 const FidesMeshPolicy mesh_policy = FidesMeshPolicy::update)
522 : ADIOS2Writer<T>(comm, filename, "Fides function writer",
523 impl_adios2::extract_common_mesh<T>(u), u),
524 _mesh_reuse_policy(mesh_policy)
525 {
526 if (u.empty())
527 throw std::runtime_error("FidesWriter fem::Function list is empty");
528
529 // Extract Mesh from first function
530 auto mesh = std::visit(
531 [](const auto& u) { return u->function_space()->mesh(); }, u.front());
532 assert(mesh);
533
534 // Extract element from first function
535 const fem::FiniteElement* element0 = std::visit(
536 [](const auto& e) { return e->function_space()->element().get(); },
537 u.front());
538 assert(element0);
539
540 // Check if function is mixed
541 if (element0->is_mixed())
542 throw std::runtime_error(
543 "Mixed functions are not supported by VTXWriter");
544
545 // Check if function is DG 0
546 if (element0->space_dimension() / element0->block_size() == 1)
547 {
548 throw std::runtime_error(
549 "Piecewise constants are not (yet) supported by VTXWriter");
550 }
551
552 // FIXME: is the below check adequate for detecting a
553 // Lagrange element? Check that element is Lagrange
554 if (!element0->interpolation_ident())
555 {
556 throw std::runtime_error(
557 "Only Lagrange functions are "
558 "supported. Interpolate Functions before output.");
559 }
560
561 // Check that all functions are first order Lagrange
562 auto cell_types = mesh->topology()->cell_types();
563 if (cell_types.size() > 1)
564 throw std::runtime_error("Multiple cell types in IO.");
565 int num_vertices_per_cell = mesh::cell_num_entities(cell_types.back(), 0);
566 for (auto& v : this->_u)
567 {
568 std::visit(
569 [&](const auto& u)
570 {
571 auto element = u->function_space()->element();
572 assert(element);
573 if (*element != *element0)
574 {
575 throw std::runtime_error("All functions in FidesWriter must have "
576 "the same element type");
577 }
578 auto dof_layout
579 = u->function_space()->dofmap()->element_dof_layout();
580 int num_vertex_dofs = dof_layout.num_entity_dofs(0);
581 int num_dofs = element->space_dimension() / element->block_size();
582 if (num_dofs != num_vertices_per_cell or num_vertex_dofs != 1)
583 {
584 throw std::runtime_error("Only first order Lagrange spaces are "
585 "supported by FidesWriter");
586 }
587 },
588 v);
589 }
590
591 impl_fides::initialize_mesh_attributes(*this->_io, *mesh);
592 impl_fides::initialize_function_attributes<T>(*this->_io, u);
593 }
594
595 // Copy constructor
596 FidesWriter(const FidesWriter&) = delete;
597
599 FidesWriter(FidesWriter&& file) = default;
600
602 ~FidesWriter() = default;
603
605 FidesWriter& operator=(FidesWriter&&) = default;
606
607 // Copy assignment
608 FidesWriter& operator=(const FidesWriter&) = delete;
609
612 void write(double t)
613 {
614 assert(this->_io);
615 assert(this->_engine);
616
617 this->_engine->BeginStep();
618 adios2::Variable<double> var_step
619 = impl_adios2::define_variable<double>(*this->_io, "step");
620 this->_engine->template Put<double>(var_step, t);
621
622 if (auto v
623 = this->_io->template InquireVariable<std::int64_t>("connectivity");
624 !v or _mesh_reuse_policy == FidesMeshPolicy::update)
625 {
626 impl_fides::write_mesh(*this->_io, *this->_engine, *this->_mesh);
627 }
628
629 for (auto& v : this->_u)
630 {
631 std::visit([&](const auto& u)
632 { impl_fides::write_data(*this->_io, *this->_engine, *u); },
633 v);
634 }
635
636 this->_engine->EndStep();
637 }
638
639private:
640 // Control whether the mesh is written to file once or at every time
641 // step
642 FidesMeshPolicy _mesh_reuse_policy;
643};

697template <typename T, typename X>
698void vtx_write_data(adios2::IO& io, adios2::Engine& engine,
699 const fem::Function<T, X>& u)
700{
701 // Get function data array and information about layout
702 assert(u.x());
703 std::span<const T> u_vector = u.x()->array();
704 const int rank = u.function_space()->element()->value_shape().size();
705 const std::uint32_t num_comp = std::pow(3, rank);
706 std::shared_ptr<const fem::DofMap> dofmap = u.function_space()->dofmap();
707 assert(dofmap);
708 std::shared_ptr<const common::IndexMap> index_map = dofmap->index_map;
709 assert(index_map);
710 const int index_map_bs = dofmap->index_map_bs();
711 const int dofmap_bs = dofmap->bs();
712 const std::uint32_t num_dofs
713 = index_map_bs * (index_map->size_local() + index_map->num_ghosts())
714 / dofmap_bs;
715
716 if constexpr (std::is_scalar_v<T>)
717 {
718 // ---- Real
719 std::vector<T> data(num_dofs * num_comp, 0);
720 for (std::size_t i = 0; i < num_dofs; ++i)
721 for (int j = 0; j < index_map_bs; ++j)
722 data[i * num_comp + j] = u_vector[i * index_map_bs + j];
723
724 adios2::Variable<T> output = impl_adios2::define_variable<T>(
725 io, u.name, {}, {}, {num_dofs, num_comp});
726 engine.Put<T>(output, data.data(), adios2::Mode::Sync);
727 }
728 else
729 {
730 // ---- Complex
731 using U = typename T::value_type;
732
733 std::vector<U> data(num_dofs * num_comp, 0);
734 for (std::size_t i = 0; i < num_dofs; ++i)
735 for (int j = 0; j < index_map_bs; ++j)
736 data[i * num_comp + j] = std::real(u_vector[i * index_map_bs + j]);
737
738 adios2::Variable<U> output_real = impl_adios2::define_variable<U>(
739 io, u.name + impl_adios2::field_ext[0], {}, {}, {num_dofs, num_comp});
740 engine.Put<U>(output_real, data.data(), adios2::Mode::Sync);
741
742 std::fill(data.begin(), data.end(), 0);
743 for (std::size_t i = 0; i < num_dofs; ++i)
744 for (int j = 0; j < index_map_bs; ++j)
745 data[i * num_comp + j] = std::imag(u_vector[i * index_map_bs + j]);
746 adios2::Variable<U> output_imag = impl_adios2::define_variable<U>(
747 io, u.name + impl_adios2::field_ext[1], {}, {}, {num_dofs, num_comp});
748 engine.Put<U>(output_imag, data.data(), adios2::Mode::Sync);

756template <typename T>
757void vtx_write_mesh(adios2::IO& io, adios2::Engine& engine,
758 const mesh::Mesh<T>& mesh)
759{
760 const mesh::Geometry<T>& geometry = mesh.geometry();
761 auto topology = mesh.topology();
762 assert(topology);
763
764 // "Put" geometry
765 std::shared_ptr<const common::IndexMap> x_map = geometry.index_map();
766 const std::uint32_t num_vertices = x_map->size_local() + x_map->num_ghosts();
767 adios2::Variable<T> local_geometry = impl_adios2::define_variable<T>(
768 io, "geometry", {}, {}, {num_vertices, 3});
769 engine.Put<T>(local_geometry, geometry.x().data());
770
771 // Put number of nodes. The mesh data is written with local indices,
772 // therefore we need the ghost vertices.
773 adios2::Variable<std::uint32_t> vertices
774 = impl_adios2::define_variable<std::uint32_t>(io, "NumberOfNodes",
775 {adios2::LocalValueDim});
776 engine.Put<std::uint32_t>(vertices, num_vertices);
777
778 const auto [vtkcells, shape] = io::extract_vtk_connectivity(
779 geometry.dofmap(), topology->cell_types()[0]);
780
781 // Add cell metadata
782 const int tdim = topology->dim();
783 adios2::Variable<std::uint32_t> cell_variable
784 = impl_adios2::define_variable<std::uint32_t>(io, "NumberOfCells",
785 {adios2::LocalValueDim});
786 engine.Put<std::uint32_t>(cell_variable, shape[0]);
787 adios2::Variable<std::uint32_t> celltype_variable
788 = impl_adios2::define_variable<std::uint32_t>(io, "types");
789 engine.Put<std::uint32_t>(
790 celltype_variable,
791 cells::get_vtk_cell_type(topology->cell_types()[0], tdim));
792
793 // Pack mesh 'nodes'. Output is written as [N0, v0_0,...., v0_N0, N1,
794 // v1_0,...., v1_N1,....], where N is the number of cell nodes and v0,
795 // etc, is the node index
796 std::vector<std::int64_t> cells(shape[0] * (shape[1] + 1), shape[1]);
797 for (std::size_t c = 0; c < shape[0]; ++c)
798 {
799 std::span vtkcell(vtkcells.data() + c * shape[1], shape[1]);
800 std::span cell(cells.data() + c * (shape[1] + 1), shape[1] + 1);
801 std::copy(vtkcell.begin(), vtkcell.end(), std::next(cell.begin()));
802 }
803
804 // Put topology (nodes)
805 adios2::Variable<std::int64_t> local_topology
806 = impl_adios2::define_variable<std::int64_t>(io, "connectivity", {}, {},
807 {shape[0], shape[1] + 1});
808 engine.Put<std::int64_t>(local_topology, cells.data());
809
810 // Vertex global ids and ghost markers
811 adios2::Variable<std::int64_t> orig_id
812 = impl_adios2::define_variable<std::int64_t>(io, "vtkOriginalPointIds",
813 {}, {}, {num_vertices});
814 engine.Put<std::int64_t>(orig_id, geometry.input_global_indices().data());
815
816 std::vector<std::uint8_t> x_ghost(num_vertices, 0);
817 std::fill(std::next(x_ghost.begin(), x_map->size_local()), x_ghost.end(), 1);
818 adios2::Variable<std::uint8_t> ghost
819 = impl_adios2::define_variable<std::uint8_t>(io, "vtkGhostType", {}, {},
820 {x_ghost.size()});
821 engine.Put<std::uint8_t>(ghost, x_ghost.data());
822
823 engine.PerformPuts();
824}

833template <typename T>
834void vtx_write_mesh_from_space(adios2::IO& io, adios2::Engine& engine,
835 const fem::FunctionSpace<T>& V)
836{
837 auto mesh = V.mesh();
838 assert(mesh);
839 auto topology = mesh->topology();
840 assert(topology);
841 const int tdim = topology->dim();
842
843 // Get a VTK mesh with points at the 'nodes'
844 const auto [x, xshape, x_id, x_ghost, vtk, vtkshape]
845 = io::vtk_mesh_from_space(V);
846
847 std::uint32_t num_dofs = xshape[0];
848
849 // -- Pack mesh 'nodes'. Output is written as [N0, v0_0,...., v0_N0, N1,
850 // v1_0,...., v1_N1,....], where N is the number of cell nodes and v0,
851 // etc, is the node index.
852
853 // Create vector, setting all entries to nodes per cell (vtk.shape(1))
854 std::vector<std::int64_t> cells(vtkshape[0] * (vtkshape[1] + 1), vtkshape[1]);
855
856 // Set the [v0_0,...., v0_N0, v1_0,...., v1_N1,....] data
857 for (std::size_t c = 0; c < vtkshape[0]; ++c)
858 {
859 std::span vtkcell(vtk.data() + c * vtkshape[1], vtkshape[1]);
860 std::span cell(cells.data() + c * (vtkshape[1] + 1), vtkshape[1] + 1);
861 std::copy(vtkcell.begin(), vtkcell.end(), std::next(cell.begin()));
862 }
863
864 // Define ADIOS2 variables for geometry, topology, celltypes and
865 // corresponding VTK data
866 adios2::Variable<T> local_geometry
867 = impl_adios2::define_variable<T>(io, "geometry", {}, {}, {num_dofs, 3});
868 adios2::Variable<std::int64_t> local_topology
869 = impl_adios2::define_variable<std::int64_t>(
870 io, "connectivity", {}, {}, {vtkshape[0], vtkshape[1] + 1});
871 adios2::Variable<std::uint32_t> cell_type
872 = impl_adios2::define_variable<std::uint32_t>(io, "types");
873 adios2::Variable<std::uint32_t> vertices
874 = impl_adios2::define_variable<std::uint32_t>(io, "NumberOfNodes",
875 {adios2::LocalValueDim});
876 adios2::Variable<std::uint32_t> elements
877 = impl_adios2::define_variable<std::uint32_t>(io, "NumberOfEntities",
878 {adios2::LocalValueDim});
879
880 // Write mesh information to file
881 engine.Put<std::uint32_t>(vertices, num_dofs);
882 engine.Put<std::uint32_t>(elements, vtkshape[0]);
883 engine.Put<std::uint32_t>(
884 cell_type, cells::get_vtk_cell_type(topology->cell_types()[0], tdim));
885 engine.Put<T>(local_geometry, x.data());
886 engine.Put<std::int64_t>(local_topology, cells.data());
887
888 // Node global ids
889 adios2::Variable<std::int64_t> orig_id
890 = impl_adios2::define_variable<std::int64_t>(io, "vtkOriginalPointIds",
891 {}, {}, {x_id.size()});
892 engine.Put<std::int64_t>(orig_id, x_id.data());
893 adios2::Variable<std::uint8_t> ghost
894 = impl_adios2::define_variable<std::uint8_t>(io, "vtkGhostType", {}, {},
895 {x_ghost.size()});
896 engine.Put<std::uint8_t>(ghost, x_ghost.data());
897
898 engine.PerformPuts();
899}

909{
910public:
921 VTXWriter(MPI_Comm comm, const std::filesystem::path& filename,
922 std::shared_ptr<const mesh::Mesh<T>> mesh)
923 : ADIOS2Writer<T>(comm, filename, "VTX mesh writer", mesh)
924 {
925 // Define VTK scheme attribute for mesh
926 std::string vtk_scheme = impl_vtx::create_vtk_schema({}, {}).str();

930
938 VTXWriter(MPI_Comm comm, const std::filesystem::path& filename,
939 const typename ADIOS2Writer<T>::U& u)
940 : ADIOS2Writer<T>(comm, filename, "VTX function writer",
941 impl_adios2::extract_common_mesh<T>(u), u)
942 {
943 if (u.empty())
944 throw std::runtime_error("VTXWriter fem::Function list is empty");
945
946 // Extract element from first function
947 const fem::FiniteElement* element0 = std::visit(
948 [](const auto& u) { return u->function_space()->element().get(); },
949 u.front());
950 assert(element0);
951
952 // Check if function is mixed
953 if (element0->is_mixed())
954 throw std::runtime_error(
955 "Mixed functions are not supported by VTXWriter");
956
957 // Check if function is DG 0
958 if (element0->space_dimension() / element0->block_size() == 1)
959 {
960 throw std::runtime_error(
961 "VTK does not support cell-wise fields. See "
962 "https://gitlab.kitware.com/vtk/vtk/-/issues/18458.");
963 }
964
965 // FIXME: is the below check adequate for detecting a Lagrange
966 // element?
967 // Check that element is Lagrange
968 if (!element0->interpolation_ident())
969 {
970 throw std::runtime_error(
971 "Only (discontinuous) Lagrange functions are "
972 "supported. Interpolate Functions before output.");
973 }
974
975 // Check that all functions come from same element type
976 for (auto& v : this->_u)
977 {
978 std::visit(
979 [element0](const auto& u)
980 {
981 auto element = u->function_space()->element();
982 assert(element);
983 if (*element != *element0)
984 {
985 throw std::runtime_error(
986 "All functions in VTXWriter must have the same element type");
987 }
988 },
989 v);
990 }
991
992 // Define VTK scheme attribute for set of functions
993 std::vector<std::string> names = impl_vtx::extract_function_names<T>(u);
994 std::string vtk_scheme = impl_vtx::create_vtk_schema(names, {}).str();
995 impl_adios2::define_attribute<std::string>(*this->_io, "vtk.xml",
996 vtk_scheme);
997 }

1000 VTXWriter(const VTXWriter&) = delete;

1007

1010

1016 void write(double t)
1017 {
1018 assert(this->_io);
1019 assert(this->_engine);
1020 adios2::Variable<double> var_step
1021 = impl_adios2::define_variable<double>(*this->_io, "step");
1022
1023 this->_engine->BeginStep();
1024 this->_engine->template Put<double>(var_step, t);
1025
1026 // If we have no functions write the mesh to file
1027 if (this->_u.empty())
1028 impl_vtx::vtx_write_mesh(*this->_io, *this->_engine, *this->_mesh);
1029 else
1030 {
1031 // Write a single mesh for functions as they share finite element
1032 std::visit(
1033 [&](const auto& u)
1034 {
1035 impl_vtx::vtx_write_mesh_from_space<T>(*this->_io, *this->_engine,
1036 *u->function_space());
1037 },
1038 this->_u[0]);
1039
1040 // Write function data for each function to file
1041 for (auto& v : this->_u)
1042 std::visit(
1043 [&](const auto& u)
1044 { impl_vtx::vtx_write_data(*this->_io, *this->_engine, *u); },
1045 v);
1046 }
1047
1048 this->_engine->EndStep();
1049 }
