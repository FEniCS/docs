40template <std::floating_point T>
41std::vector<T> interpolation_coords(const fem::FiniteElement& element,
42 const mesh::Geometry<T>& geometry,
43 std::span<const std::int32_t> cells)
44{
45 // Get geometry data and the element coordinate map
46 const std::size_t gdim = geometry.dim();
47 const graph::AdjacencyList<std::int32_t>& x_dofmap = geometry.dofmap();
48 std::span<const T> x_g = geometry.x();
49
50 if (geometry.cmaps().size() > 1)
51 {
52 throw std::runtime_error("Mixed topology not supported");
53 }
54
55 const CoordinateElement& cmap = geometry.cmaps()[0];
56 const std::size_t num_dofs_g = cmap.dim();
57
58 // Get the interpolation points on the reference cells
59 const auto [X, Xshape] = element.interpolation_points();
60
61 // Evaluate coordinate element basis at reference points
62 namespace stdex = std::experimental;
63 using cmdspan4_t
64 = stdex::mdspan<const double, stdex::dextents<std::size_t, 4>>;
65 std::array<std::size_t, 4> phi_shape = cmap.tabulate_shape(0, Xshape[0]);
66 std::vector<double> phi_b(
67 std::reduce(phi_shape.begin(), phi_shape.end(), 1, std::multiplies{}));
68 cmdspan4_t phi_full(phi_b.data(), phi_shape);
69 cmap.tabulate(0, X, Xshape, phi_b);
70 auto phi = stdex::submdspan(phi_full, 0, stdex::full_extent,
71 stdex::full_extent, 0);
72
73 // Push reference coordinates (X) forward to the physical coordinates
74 // (x) for each cell
75 std::vector<T> coordinate_dofs(num_dofs_g * gdim, 0);
76 std::vector<T> x(3 * (cells.size() * Xshape[0]), 0);
77 for (std::size_t c = 0; c < cells.size(); ++c)
78 {
79 // Get geometry data for current cell
80 auto x_dofs = x_dofmap.links(cells[c]);
81 for (std::size_t i = 0; i < x_dofs.size(); ++i)
82 {
83 std::copy_n(std::next(x_g.begin(), 3 * x_dofs[i]), gdim,
84 std::next(coordinate_dofs.begin(), i * gdim));
85 }
86
87 // Push forward coordinates (X -> x)
88 for (std::size_t p = 0; p < Xshape[0]; ++p)
89 {
90 for (std::size_t j = 0; j < gdim; ++j)
91 {
92 double acc = 0;
93 for (std::size_t k = 0; k < num_dofs_g; ++k)
94 acc += phi(p, k) * coordinate_dofs[k * gdim + j];
95 x[j * (cells.size() * Xshape[0]) + c * Xshape[0] + p] = acc;
96 }
97 }
98 }
99
100 return x;
101}

144template <typename T>
145void scatter_values(MPI_Comm comm, std::span<const std::int32_t> src_ranks,
146 std::span<const std::int32_t> dest_ranks,
147 std::span<const T> send_values,
148 std::array<std::size_t, 2> s_shape,
149 std::span<T> recv_values)
150{
151 const std::size_t block_size = s_shape[1];
152 assert(src_ranks.size() * block_size == send_values.size());
153 assert(recv_values.size() == dest_ranks.size() * block_size);
154
155 // Build unique set of the sorted src_ranks
156 std::vector<std::int32_t> out_ranks(src_ranks.size());
157 out_ranks.assign(src_ranks.begin(), src_ranks.end());
158 out_ranks.erase(std::unique(out_ranks.begin(), out_ranks.end()),
159 out_ranks.end());
160 out_ranks.reserve(out_ranks.size() + 1);
161
162 // Remove negative entries from dest_ranks
163 std::vector<std::int32_t> in_ranks;
164 in_ranks.reserve(dest_ranks.size());
165 std::copy_if(dest_ranks.begin(), dest_ranks.end(),
166 std::back_inserter(in_ranks),
167 [](auto rank) { return rank >= 0; });
168
169 // Create unique set of sorted in-ranks
170 std::sort(in_ranks.begin(), in_ranks.end());
171 in_ranks.erase(std::unique(in_ranks.begin(), in_ranks.end()), in_ranks.end());
172 in_ranks.reserve(in_ranks.size() + 1);
173
174 // Create neighborhood communicator
175 MPI_Comm reverse_comm;
176 MPI_Dist_graph_create_adjacent(
177 comm, in_ranks.size(), in_ranks.data(), MPI_UNWEIGHTED, out_ranks.size(),
178 out_ranks.data(), MPI_UNWEIGHTED, MPI_INFO_NULL, false, &reverse_comm);
179
180 std::vector<std::int32_t> comm_to_output;
181 std::vector<std::int32_t> recv_sizes(in_ranks.size());
182 recv_sizes.reserve(1);
183 std::vector<std::int32_t> recv_offsets(in_ranks.size() + 1, 0);
184 {
185 // Build map from parent to neighborhood communicator ranks
186 std::map<std::int32_t, std::int32_t> rank_to_neighbor;
187 for (std::size_t i = 0; i < in_ranks.size(); i++)
188 rank_to_neighbor[in_ranks[i]] = i;
189
190 // Compute receive sizes
191 std::for_each(
192 dest_ranks.begin(), dest_ranks.end(),
193 [&dest_ranks, &rank_to_neighbor, &recv_sizes, block_size](auto rank)
194 {
195 if (rank >= 0)
196 {
197 const int neighbor = rank_to_neighbor[rank];
198 recv_sizes[neighbor] += block_size;
199 }
200 });
201
202 // Compute receiving offsets
203 std::partial_sum(recv_sizes.begin(), recv_sizes.end(),
204 std::next(recv_offsets.begin(), 1));
205
206 // Compute map from receiving values to position in recv_values
207 comm_to_output.resize(recv_offsets.back() / block_size);
208 std::vector<std::int32_t> recv_counter(recv_sizes.size(), 0);
209 for (std::size_t i = 0; i < dest_ranks.size(); ++i)
210 {
211 if (const std::int32_t rank = dest_ranks[i]; rank >= 0)
212 {
213 const int neighbor = rank_to_neighbor[rank];
214 int insert_pos = recv_offsets[neighbor] + recv_counter[neighbor];
215 comm_to_output[insert_pos / block_size] = i * block_size;
216 recv_counter[neighbor] += block_size;
217 }
218 }
219 }
220
221 std::vector<std::int32_t> send_sizes(out_ranks.size());
222 send_sizes.reserve(1);
223 {
224 // Compute map from parent mpi rank to neigbor rank for outgoing data
225 std::map<std::int32_t, std::int32_t> rank_to_neighbor;
226 for (std::size_t i = 0; i < out_ranks.size(); i++)
227 rank_to_neighbor[out_ranks[i]] = i;
228
229 // Compute send sizes
230 std::for_each(src_ranks.begin(), src_ranks.end(),
231 [&rank_to_neighbor, &send_sizes, block_size](auto rank)
232 { send_sizes[rank_to_neighbor[rank]] += block_size; });
233 }
234
235 // Compute sending offsets
236 std::vector<std::int32_t> send_offsets(send_sizes.size() + 1, 0);
237 std::partial_sum(send_sizes.begin(), send_sizes.end(),
238 std::next(send_offsets.begin(), 1));
239
240 std::stringstream cc;
241 // Send values to dest ranks
242 std::vector<T> values(recv_offsets.back());
243 values.reserve(1);
244 MPI_Neighbor_alltoallv(send_values.data(), send_sizes.data(),
245 send_offsets.data(), dolfinx::MPI::mpi_type<T>(),
246 values.data(), recv_sizes.data(), recv_offsets.data(),
247 dolfinx::MPI::mpi_type<T>(), reverse_comm);
248 MPI_Comm_free(&reverse_comm);
249
250 // Insert values received from neighborhood communicator in output span
251 std::fill(recv_values.begin(), recv_values.end(), T(0));
252 for (std::size_t i = 0; i < comm_to_output.size(); i++)
253 {
254 auto vals = std::next(recv_values.begin(), comm_to_output[i]);
255 auto vals_from = std::next(values.begin(), i * block_size);
256 std::copy_n(vals_from, block_size, vals);
257 }

268template <typename U, typename V, typename T>
269void interpolation_apply(const U& Pi, const V& data, std::span<T> coeffs,
270 int bs)
271{
272 using X = typename dolfinx::scalar_value_type_t<T>;
273 static_assert(U::rank() == 2, "Must be rank 2");
274 static_assert(V::rank() == 2, "Must be rank 2");
275
276 // Compute coefficients = Pi * x (matrix-vector multiply)
277 if (bs == 1)
278 {
279 assert(data.extent(0) * data.extent(1) == Pi.extent(1));
280 for (std::size_t i = 0; i < Pi.extent(0); ++i)
281 {
282 coeffs[i] = 0.0;
283 for (std::size_t k = 0; k < data.extent(1); ++k)
284 for (std::size_t j = 0; j < data.extent(0); ++j)
285 coeffs[i]
286 += static_cast<X>(Pi(i, k * data.extent(0) + j)) * data(j, k);
287 }
288 }
289 else
290 {
291 const std::size_t cols = Pi.extent(1);
292 assert(data.extent(0) == Pi.extent(1));
293 assert(data.extent(1) == bs);
294 for (int k = 0; k < bs; ++k)
295 {
296 for (std::size_t i = 0; i < Pi.extent(0); ++i)
297 {
298 T acc = 0;
299 for (std::size_t j = 0; j < cols; ++j)
300 acc += static_cast<X>(Pi(i, j)) * data(j, k);
301 coeffs[bs * i + k] = acc;
302 }
303 }
304 }
305}

317template <typename T, std::floating_point U>
318void interpolate_same_map(Function<T, U>& u1, const Function<T, U>& u0,
319 std::span<const std::int32_t> cells)
320{
321 auto V0 = u0.function_space();
322 assert(V0);
323 auto V1 = u1.function_space();
324 assert(V1);
325 auto mesh = V0->mesh();
326 assert(mesh);
327
328 std::shared_ptr<const FiniteElement> element0 = V0->element();
329 assert(element0);
330 std::shared_ptr<const FiniteElement> element1 = V1->element();
331 assert(element1);
332
333 const int tdim = mesh->topology()->dim();
334 auto map = mesh->topology()->index_map(tdim);
335 assert(map);
336 std::span<T> u1_array = u1.x()->mutable_array();
337 std::span<const T> u0_array = u0.x()->array();
338
339 std::span<const std::uint32_t> cell_info;
340 if (element1->needs_dof_transformations()
341 or element0->needs_dof_transformations())
342 {
343 mesh->topology_mutable()->create_entity_permutations();
344 cell_info = std::span(mesh->topology()->get_cell_permutation_info());
345 }
346
347 // Get dofmaps
348 auto dofmap1 = V1->dofmap();
349 auto dofmap0 = V0->dofmap();
350
351 // Get block sizes and dof transformation operators
352 const int bs1 = dofmap1->bs();
353 const int bs0 = dofmap0->bs();
354 auto apply_dof_transformation
355 = element0->get_dof_transformation_function<T>(false, true, false);
356 auto apply_inverse_dof_transform
357 = element1->get_dof_transformation_function<T>(true, true, false);
358
359 // Create working array
360 std::vector<T> local0(element0->space_dimension());
361 std::vector<T> local1(element1->space_dimension());
362
363 // Create interpolation operator
364 const auto [i_m, im_shape]
365 = element1->create_interpolation_operator(*element0);
366
367 // Iterate over mesh and interpolate on each cell
368 using X = typename dolfinx::scalar_value_type_t<T>;
369 for (auto c : cells)
370 {
371 std::span<const std::int32_t> dofs0 = dofmap0->cell_dofs(c);
372 for (std::size_t i = 0; i < dofs0.size(); ++i)
373 for (int k = 0; k < bs0; ++k)
374 local0[bs0 * i + k] = u0_array[bs0 * dofs0[i] + k];
375
376 apply_dof_transformation(local0, cell_info, c, 1);
377
378 // FIXME: Get compile-time ranges from Basix
379 // Apply interpolation operator
380 std::fill(local1.begin(), local1.end(), 0);
381 for (std::size_t i = 0; i < im_shape[0]; ++i)
382 for (std::size_t j = 0; j < im_shape[1]; ++j)
383 local1[i] += static_cast<X>(i_m[im_shape[1] * i + j]) * local0[j];
384
385 apply_inverse_dof_transform(local1, cell_info, c, 1);
386
387 std::span<const std::int32_t> dofs1 = dofmap1->cell_dofs(c);
388 for (std::size_t i = 0; i < dofs1.size(); ++i)
389 for (int k = 0; k < bs1; ++k)
390 u1_array[bs1 * dofs1[i] + k] = local1[bs1 * i + k];
391 }
392}

403template <typename T, std::floating_point U>
404void interpolate_nonmatching_maps(Function<T, U>& u1, const Function<T, U>& u0,
405 std::span<const std::int32_t> cells)
406{
407 // Get mesh
408 auto V0 = u0.function_space();
409 assert(V0);
410 auto mesh = V0->mesh();
411 assert(mesh);
412
413 // Mesh dims
414 const int tdim = mesh->topology()->dim();
415 const int gdim = mesh->geometry().dim();
416
417 // Get elements
418 auto V1 = u1.function_space();
419 assert(V1);
420 std::shared_ptr<const FiniteElement> element0 = V0->element();
421 assert(element0);
422 std::shared_ptr<const FiniteElement> element1 = V1->element();
423 assert(element1);
424
425 std::span<const std::uint32_t> cell_info;
426 if (element1->needs_dof_transformations()
427 or element0->needs_dof_transformations())
428 {
429 mesh->topology_mutable()->create_entity_permutations();
430 cell_info = std::span(mesh->topology()->get_cell_permutation_info());
431 }
432
433 // Get dofmaps
434 auto dofmap0 = V0->dofmap();
435 auto dofmap1 = V1->dofmap();
436
437 const auto [X, Xshape] = element1->interpolation_points();
438
439 // Get block sizes and dof transformation operators
440 const int bs0 = element0->block_size();
441 const int bs1 = element1->block_size();
442 const auto apply_dof_transformation0
443 = element0->get_dof_transformation_function<double>(false, false, false);
444 const auto apply_inverse_dof_transform1
445 = element1->get_dof_transformation_function<T>(true, true, false);
446
447 // Get sizes of elements
448 const std::size_t dim0 = element0->space_dimension() / bs0;
449 const std::size_t value_size_ref0 = element0->reference_value_size() / bs0;
450 const std::size_t value_size0 = element0->value_size() / bs0;
451
452 // Get geometry data
453 if (mesh->geometry().cmaps().size() > 1)
454 throw std::runtime_error("Multiple cmaps");
455
456 const CoordinateElement& cmap = mesh->geometry().cmaps()[0];
457 const graph::AdjacencyList<std::int32_t>& x_dofmap
458 = mesh->geometry().dofmap();
459 const std::size_t num_dofs_g = cmap.dim();
460 std::span<const U> x_g = mesh->geometry().x();
461
462 namespace stdex = std::experimental;
463 using cmdspan2_t
464 = stdex::mdspan<const double, stdex::dextents<std::size_t, 2>>;
465 using cmdspan4_t
466 = stdex::mdspan<const double, stdex::dextents<std::size_t, 4>>;
467 using mdspan2_t = stdex::mdspan<double, stdex::dextents<std::size_t, 2>>;
468 using mdspan3_t = stdex::mdspan<double, stdex::dextents<std::size_t, 3>>;
469 using mdspan3T_t = stdex::mdspan<T, stdex::dextents<std::size_t, 3>>;
470
471 // Evaluate coordinate map basis at reference interpolation points
472 const std::array<std::size_t, 4> phi_shape
473 = cmap.tabulate_shape(1, Xshape[0]);
474 std::vector<double> phi_b(
475 std::reduce(phi_shape.begin(), phi_shape.end(), 1, std::multiplies{}));
476 cmdspan4_t phi(phi_b.data(), phi_shape);
477 cmap.tabulate(1, X, Xshape, phi_b);
478
479 // Evaluate v basis functions at reference interpolation points
480 const auto [_basis_derivatives_reference0, b0shape]
481 = element0->tabulate(X, Xshape, 0);
482 cmdspan4_t basis_derivatives_reference0(_basis_derivatives_reference0.data(),
483 b0shape);
484
485 // Create working arrays
486 std::vector<T> local1(element1->space_dimension());
487 std::vector<T> coeffs0(element0->space_dimension());
488
489 std::vector<double> basis0_b(Xshape[0] * dim0 * value_size0);
490 mdspan3_t basis0(basis0_b.data(), Xshape[0], dim0, value_size0);
491
492 std::vector<double> basis_reference0_b(Xshape[0] * dim0 * value_size_ref0);
493 mdspan3_t basis_reference0(basis_reference0_b.data(), Xshape[0], dim0,
494 value_size_ref0);
495
496 std::vector<T> values0_b(Xshape[0] * 1 * element1->value_size());
497 mdspan3T_t values0(values0_b.data(), Xshape[0], 1, element1->value_size());
498
499 std::vector<T> mapped_values_b(Xshape[0] * 1 * element1->value_size());
500 mdspan3T_t mapped_values0(mapped_values_b.data(), Xshape[0], 1,
501 element1->value_size());
502
503 std::vector<double> coord_dofs_b(num_dofs_g * gdim);
504 mdspan2_t coord_dofs(coord_dofs_b.data(), num_dofs_g, gdim);
505
506 std::vector<double> J_b(Xshape[0] * gdim * tdim);
507 mdspan3_t J(J_b.data(), Xshape[0], gdim, tdim);
508 std::vector<double> K_b(Xshape[0] * tdim * gdim);
509 mdspan3_t K(K_b.data(), Xshape[0], tdim, gdim);
510 std::vector<double> detJ(Xshape[0]);
511 std::vector<double> det_scratch(2 * gdim * tdim);
512
513 // Get interpolation operator
514 const auto [_Pi_1, pi_shape] = element1->interpolation_operator();
515 cmdspan2_t Pi_1(_Pi_1.data(), pi_shape);
516
517 using u_t = stdex::mdspan<double, stdex::dextents<std::size_t, 2>>;
518 using U_t = stdex::mdspan<const double, stdex::dextents<std::size_t, 2>>;
519 using J_t = stdex::mdspan<const double, stdex::dextents<std::size_t, 2>>;
520 using K_t = stdex::mdspan<const double, stdex::dextents<std::size_t, 2>>;
521 auto push_forward_fn0
522 = element0->basix_element().map_fn<u_t, U_t, J_t, K_t>();
523
524 using v_t = stdex::mdspan<const T, stdex::dextents<std::size_t, 2>>;
525 using V_t = stdex::mdspan<T, stdex::dextents<std::size_t, 2>>;
526 auto pull_back_fn1 = element1->basix_element().map_fn<V_t, v_t, K_t, J_t>();
527
528 // Iterate over mesh and interpolate on each cell
529 std::span<const T> array0 = u0.x()->array();
530 std::span<T> array1 = u1.x()->mutable_array();
531 for (auto c : cells)
532 {
533 // Get cell geometry (coordinate dofs)
534 auto x_dofs = x_dofmap.links(c);
535 for (std::size_t i = 0; i < num_dofs_g; ++i)
536 {
537 const int pos = 3 * x_dofs[i];
538 for (std::size_t j = 0; j < gdim; ++j)
539 coord_dofs(i, j) = x_g[pos + j];
540 }
541
542 // Compute Jacobians and reference points for current cell
543 std::fill(J_b.begin(), J_b.end(), 0);
544 for (std::size_t p = 0; p < Xshape[0]; ++p)
545 {
546 auto dphi = stdex::submdspan(phi, std::pair(1, tdim + 1), p,
547 stdex::full_extent, 0);
548
549 auto _J = stdex::submdspan(J, p, stdex::full_extent, stdex::full_extent);
550 cmap.compute_jacobian(dphi, coord_dofs, _J);
551 auto _K = stdex::submdspan(K, p, stdex::full_extent, stdex::full_extent);
552 cmap.compute_jacobian_inverse(_J, _K);
553 detJ[p] = cmap.compute_jacobian_determinant(_J, det_scratch);
554 }
555
556 // Copy evaluated basis on reference, apply DOF transformations, and
557 // push forward to physical element
558 for (std::size_t k0 = 0; k0 < basis_reference0.extent(0); ++k0)
559 for (std::size_t k1 = 0; k1 < basis_reference0.extent(1); ++k1)
560 for (std::size_t k2 = 0; k2 < basis_reference0.extent(2); ++k2)
561 basis_reference0(k0, k1, k2)
562 = basis_derivatives_reference0(0, k0, k1, k2);
563
564 for (std::size_t p = 0; p < Xshape[0]; ++p)
565 {
566 apply_dof_transformation0(
567 std::span(basis_reference0_b.data() + p * dim0 * value_size_ref0,
568 dim0 * value_size_ref0),
569 cell_info, c, value_size_ref0);
570 }
571
572 for (std::size_t i = 0; i < basis0.extent(0); ++i)
573 {
574 auto _u
575 = stdex::submdspan(basis0, i, stdex::full_extent, stdex::full_extent);
576 auto _U = stdex::submdspan(basis_reference0, i, stdex::full_extent,
577 stdex::full_extent);
578 auto _K = stdex::submdspan(K, i, stdex::full_extent, stdex::full_extent);
579 auto _J = stdex::submdspan(J, i, stdex::full_extent, stdex::full_extent);
580 push_forward_fn0(_u, _U, _J, detJ[i], _K);
581 }
582
583 // Copy expansion coefficients for v into local array
584 const int dof_bs0 = dofmap0->bs();
585 std::span<const std::int32_t> dofs0 = dofmap0->cell_dofs(c);
586 for (std::size_t i = 0; i < dofs0.size(); ++i)
587 for (int k = 0; k < dof_bs0; ++k)
588 coeffs0[dof_bs0 * i + k] = array0[dof_bs0 * dofs0[i] + k];
589
590 // Evaluate v at the interpolation points (physical space values)
591 using X = typename dolfinx::scalar_value_type_t<T>;
592 for (std::size_t p = 0; p < Xshape[0]; ++p)
593 {
594 for (int k = 0; k < bs0; ++k)
595 {
596 for (std::size_t j = 0; j < value_size0; ++j)
597 {
598 T acc = 0;
599 for (std::size_t i = 0; i < dim0; ++i)
600 acc += coeffs0[bs0 * i + k] * static_cast<X>(basis0(p, i, j));
601 values0(p, 0, j * bs0 + k) = acc;
602 }
603 }
604 }
605
606 // Pull back the physical values to the u reference
607 for (std::size_t i = 0; i < values0.extent(0); ++i)
608 {
609 auto _u = stdex::submdspan(values0, i, stdex::full_extent,
610 stdex::full_extent);
611 auto _U = stdex::submdspan(mapped_values0, i, stdex::full_extent,
612 stdex::full_extent);
613 auto _K = stdex::submdspan(K, i, stdex::full_extent, stdex::full_extent);
614 auto _J = stdex::submdspan(J, i, stdex::full_extent, stdex::full_extent);
615 pull_back_fn1(_U, _u, _K, 1.0 / detJ[i], _J);
616 }
617
618 auto values = stdex::submdspan(mapped_values0, stdex::full_extent, 0,
619 stdex::full_extent);
620 interpolation_apply(Pi_1, values, std::span(local1), bs1);
621 apply_inverse_dof_transform1(local1, cell_info, c, 1);
622
623 // Copy local coefficients to the correct position in u dof array
624 const int dof_bs1 = dofmap1->bs();
625 std::span<const std::int32_t> dofs1 = dofmap1->cell_dofs(c);
626 for (std::size_t i = 0; i < dofs1.size(); ++i)
627 for (int k = 0; k < dof_bs1; ++k)
628 array1[dof_bs1 * dofs1[i] + k] = local1[dof_bs1 * i + k];
629 }

632template <typename T, std::floating_point U>
633void interpolate_nonmatching_meshes(
634 Function<T, U>& u, const Function<T, U>& v,
635 std::span<const std::int32_t> cells,
636 const std::tuple<std::vector<std::int32_t>, std::vector<std::int32_t>,
637 std::vector<U>, std::vector<std::int32_t>>&
638 nmm_interpolation_data)
639{
640 int result;
641 auto mesh = u.function_space()->mesh();
642 auto mesh_v = v.function_space()->mesh();
643 MPI_Comm_compare(mesh->comm(), mesh_v->comm(), &result);
644
645 if (result == MPI_UNEQUAL)
646 throw std::runtime_error("Interpolation on different meshes is only "
647 "supported with the same communicator.");
648
649 MPI_Comm comm = mesh->comm();
650 const int tdim = mesh->topology()->dim();
651 const auto cell_map = mesh->topology()->index_map(tdim);
652
653 std::shared_ptr<const FiniteElement> element_u
654 = u.function_space()->element();
655 const std::size_t value_size = element_u->value_size();
656
657 if (std::get<0>(nmm_interpolation_data).empty())
658 {
659 throw std::runtime_error(
660 "In order to interpolate on nonmatching meshes, the user needs to "
661 "provide the necessary interpolation data. This can be computed "
662 "with fem::create_nonmatching_meshes_interpolation_data.");
663 }
664
665 const std::tuple_element_t<
666 0, std::tuple<std::vector<std::int32_t>, std::vector<std::int32_t>,
667 std::vector<U>, std::vector<std::int32_t>>>& dest_ranks
668 = std::get<0>(nmm_interpolation_data);
669 const std::tuple_element_t<
670 1, std::tuple<std::vector<std::int32_t>, std::vector<std::int32_t>,
671 std::vector<U>, std::vector<std::int32_t>>>& src_ranks
672 = std::get<1>(nmm_interpolation_data);
673 const std::tuple_element_t<
674 2, std::tuple<std::vector<std::int32_t>, std::vector<std::int32_t>,
675 std::vector<U>, std::vector<std::int32_t>>>& received_points
676 = std::get<2>(nmm_interpolation_data);
677 const std::tuple_element_t<
678 3, std::tuple<std::vector<std::int32_t>, std::vector<std::int32_t>,
679 std::vector<U>, std::vector<std::int32_t>>>&
680 evaluation_cells
681 = std::get<3>(nmm_interpolation_data);
682
683 // Evaluate the interpolating function where possible
684 std::vector<T> send_values(received_points.size() / 3 * value_size);
685 v.eval(received_points, {received_points.size() / 3, (std::size_t)3},
686 evaluation_cells, send_values,
687 {received_points.size() / 3, (std::size_t)value_size});
688
689 // Send values back to owning process
690 std::array<std::size_t, 2> v_shape = {src_ranks.size(), value_size};
691 std::vector<T> values_b(dest_ranks.size() * value_size);
692 impl::scatter_values(comm, src_ranks, dest_ranks,
693 std::span<const T>(send_values), v_shape,
694 std::span<T>(values_b));
695
696 // Transpose received data
697 namespace stdex = std::experimental;
698 stdex::mdspan<const T, stdex::dextents<std::size_t, 2>> values(
699 values_b.data(), dest_ranks.size(), value_size);
700
701 std::vector<T> valuesT_b(value_size * dest_ranks.size());
702 stdex::mdspan<T, stdex::dextents<std::size_t, 2>> valuesT(
703 valuesT_b.data(), value_size, dest_ranks.size());
704 for (std::size_t i = 0; i < values.extent(0); ++i)
705 for (std::size_t j = 0; j < values.extent(1); ++j)
706 valuesT(j, i) = values(i, j);
707
708 // Call local interpolation operator
709 fem::interpolate<T>(u, valuesT_b, {valuesT.extent(0), valuesT.extent(1)},
710 cells);
711}

715template <typename T, std::floating_point U>
716void interpolate(Function<T, U>& u, std::span<const T> f,
717 std::array<std::size_t, 2> fshape,
718 std::span<const std::int32_t> cells)
719{
720 namespace stdex = std::experimental;
721 using cmdspan2_t
722 = stdex::mdspan<const double, stdex::dextents<std::size_t, 2>>;
723 using cmdspan4_t
724 = stdex::mdspan<const double, stdex::dextents<std::size_t, 4>>;
725 using mdspan2_t = stdex::mdspan<double, stdex::dextents<std::size_t, 2>>;
726 using mdspan3_t = stdex::mdspan<double, stdex::dextents<std::size_t, 3>>;
727
728 std::shared_ptr<const FiniteElement> element = u.function_space()->element();
729 assert(element);
730 const int element_bs = element->block_size();
731 if (int num_sub = element->num_sub_elements();
732 num_sub > 0 and num_sub != element_bs)
733 {
734 throw std::runtime_error("Cannot directly interpolate a mixed space. "
735 "Interpolate into subspaces.");
736 }
737
738 if (fshape[0] != (std::size_t)element->value_size())
739 throw std::runtime_error("Interpolation data has the wrong shape/size.");
740
741 // Get mesh
742 assert(u.function_space());
743 auto mesh = u.function_space()->mesh();
744 assert(mesh);
745
746 const int gdim = mesh->geometry().dim();
747 const int tdim = mesh->topology()->dim();
748
749 std::span<const std::uint32_t> cell_info;
750 if (element->needs_dof_transformations())
751 {
752 mesh->topology_mutable()->create_entity_permutations();
753 cell_info = std::span(mesh->topology()->get_cell_permutation_info());
754 }
755
756 const std::size_t f_shape1 = f.size() / element->value_size();
757 stdex::mdspan<const T, stdex::dextents<std::size_t, 2>> _f(f.data(), fshape);
758
759 // Get dofmap
760 const auto dofmap = u.function_space()->dofmap();
761 assert(dofmap);
762 const int dofmap_bs = dofmap->bs();
763
764 // Loop over cells and compute interpolation dofs
765 const int num_scalar_dofs = element->space_dimension() / element_bs;
766 const int value_size = element->value_size() / element_bs;
767
768 std::span<T> coeffs = u.x()->mutable_array();
769 std::vector<T> _coeffs(num_scalar_dofs);
770
771 // This assumes that any element with an identity interpolation matrix
772 // is a point evaluation
773 if (element->map_ident() && element->interpolation_ident())
774 {
775 // Point evaluation element *and* the geometric map is the identity,
776 // e.g. not Piola mapped
777
778 auto apply_inv_transpose_dof_transformation
779 = element->get_dof_transformation_function<T>(true, true, true);
780
781 // Loop over cells
782 for (std::size_t c = 0; c < cells.size(); ++c)
783 {
784 const std::int32_t cell = cells[c];
785 std::span<const std::int32_t> dofs = dofmap->cell_dofs(cell);
786 for (int k = 0; k < element_bs; ++k)
787 {
788 // num_scalar_dofs is the number of interpolation points per
789 // cell in this case (interpolation matrix is identity)
790 std::copy_n(std::next(f.begin(), k * f_shape1 + c * num_scalar_dofs),
791 num_scalar_dofs, _coeffs.begin());
792 apply_inv_transpose_dof_transformation(_coeffs, cell_info, cell, 1);
793 for (int i = 0; i < num_scalar_dofs; ++i)
794 {
795 const int dof = i * element_bs + k;
796 std::div_t pos = std::div(dof, dofmap_bs);
797 coeffs[dofmap_bs * dofs[pos.quot] + pos.rem] = _coeffs[i];
798 }
799 }
800 }
801 }
802 else if (element->map_ident())
803 {
804 // Not a point evaluation, but the geometric map is the identity,
805 // e.g. not Piola mapped
806
807 const int element_vs = element->value_size() / element_bs;
808
809 if (element_vs > 1 && element_bs > 1)
810 {
811 throw std::runtime_error(
812 "Interpolation into this element not supported.");
813 }
814
815 // Get interpolation operator
816 const auto [_Pi, pi_shape] = element->interpolation_operator();
817 cmdspan2_t Pi(_Pi.data(), pi_shape);
818 const std::size_t num_interp_points = Pi.extent(1);
819 assert(Pi.extent(0) == num_scalar_dofs);
820
821 auto apply_inv_transpose_dof_transformation
822 = element->get_dof_transformation_function<T>(true, true, true);
823
824 // Loop over cells
825 std::vector<T> ref_data_b(num_interp_points);
826 stdex::mdspan<T, stdex::extents<std::size_t, stdex::dynamic_extent, 1>>
827 ref_data(ref_data_b.data(), num_interp_points, 1);
828 for (std::size_t c = 0; c < cells.size(); ++c)
829 {
830 const std::int32_t cell = cells[c];
831 std::span<const std::int32_t> dofs = dofmap->cell_dofs(cell);
832 for (int k = 0; k < element_bs; ++k)
833 {
834 for (int i = 0; i < element_vs; ++i)
835 {
836 std::copy_n(
837 std::next(f.begin(), (i + k) * f_shape1
838 + c * num_interp_points / element_vs),
839 num_interp_points / element_vs,
840 std::next(ref_data_b.begin(),
841 i * num_interp_points / element_vs));
842 }
843 impl::interpolation_apply(Pi, ref_data, std::span(_coeffs), 1);
844 apply_inv_transpose_dof_transformation(_coeffs, cell_info, cell, 1);
845 for (int i = 0; i < num_scalar_dofs; ++i)
846 {
847 const int dof = i * element_bs + k;
848 std::div_t pos = std::div(dof, dofmap_bs);
849 coeffs[dofmap_bs * dofs[pos.quot] + pos.rem] = _coeffs[i];
850 }
851 }
852 }
853 }
854 else
855 {
856 // Get the interpolation points on the reference cells
857 const auto [X, Xshape] = element->interpolation_points();
858 if (X.empty())
859 {
860 throw std::runtime_error(
861 "Interpolation into this space is not yet supported.");
862 }
863
864 if (_f.extent(1) != cells.size() * Xshape[0])
865 throw std::runtime_error("Interpolation data has the wrong shape.");
866
867 // Get coordinate map
868 if (mesh->geometry().cmaps().size() > 1)
869 throw std::runtime_error("Multiple cmaps");
870 const CoordinateElement& cmap = mesh->geometry().cmaps()[0];
871
872 // Get geometry data
873 const graph::AdjacencyList<std::int32_t>& x_dofmap
874 = mesh->geometry().dofmap();
875 const int num_dofs_g = cmap.dim();
876 std::span<const U> x_g = mesh->geometry().x();
877
878 // Create data structures for Jacobian info
879 std::vector<double> J_b(Xshape[0] * gdim * tdim);
880 mdspan3_t J(J_b.data(), Xshape[0], gdim, tdim);
881 std::vector<double> K_b(Xshape[0] * tdim * gdim);
882 mdspan3_t K(K_b.data(), Xshape[0], tdim, gdim);
883 std::vector<double> detJ(Xshape[0]);
884 std::vector<double> det_scratch(2 * gdim * tdim);
885
886 std::vector<double> coord_dofs_b(num_dofs_g * gdim);
887 mdspan2_t coord_dofs(coord_dofs_b.data(), num_dofs_g, gdim);
888
889 std::vector<T> ref_data_b(Xshape[0] * 1 * value_size);
890 stdex::mdspan<T, stdex::dextents<std::size_t, 3>> ref_data(
891 ref_data_b.data(), Xshape[0], 1, value_size);
892
893 std::vector<T> _vals_b(Xshape[0] * 1 * value_size);
894 stdex::mdspan<T, stdex::dextents<std::size_t, 3>> _vals(
895 _vals_b.data(), Xshape[0], 1, value_size);
896
897 // Tabulate 1st derivative of shape functions at interpolation
898 // coords
899 std::array<std::size_t, 4> phi_shape = cmap.tabulate_shape(1, Xshape[0]);
900 std::vector<double> phi_b(
901 std::reduce(phi_shape.begin(), phi_shape.end(), 1, std::multiplies{}));
902 cmdspan4_t phi(phi_b.data(), phi_shape);
903 cmap.tabulate(1, X, Xshape, phi_b);
904 auto dphi = stdex::submdspan(phi, std::pair(1, tdim + 1),
905 stdex::full_extent, stdex::full_extent, 0);
906
907 const std::function<void(const std::span<T>&,
908 const std::span<const std::uint32_t>&,
909 std::int32_t, int)>
910 apply_inverse_transpose_dof_transformation
911 = element->get_dof_transformation_function<T>(true, true);
912
913 // Get interpolation operator
914 const auto [_Pi, pi_shape] = element->interpolation_operator();
915 cmdspan2_t Pi(_Pi.data(), pi_shape);
916
917 namespace stdex = std::experimental;
918 using u_t = stdex::mdspan<const T, stdex::dextents<std::size_t, 2>>;
919 using U_t = stdex::mdspan<T, stdex::dextents<std::size_t, 2>>;
920 using J_t = stdex::mdspan<const double, stdex::dextents<std::size_t, 2>>;
921 using K_t = stdex::mdspan<const double, stdex::dextents<std::size_t, 2>>;
922 auto pull_back_fn = element->basix_element().map_fn<U_t, u_t, J_t, K_t>();
923
924 for (std::size_t c = 0; c < cells.size(); ++c)
925 {
926 const std::int32_t cell = cells[c];
927 auto x_dofs = x_dofmap.links(cell);
928 for (int i = 0; i < num_dofs_g; ++i)
929 {
930 const int pos = 3 * x_dofs[i];
931 for (int j = 0; j < gdim; ++j)
932 coord_dofs(i, j) = x_g[pos + j];
933 }
934
935 // Compute J, detJ and K
936 std::fill(J_b.begin(), J_b.end(), 0);
937 for (std::size_t p = 0; p < Xshape[0]; ++p)
938 {
939 auto _dphi
940 = stdex::submdspan(dphi, stdex::full_extent, p, stdex::full_extent);
941 auto _J
942 = stdex::submdspan(J, p, stdex::full_extent, stdex::full_extent);
943 cmap.compute_jacobian(_dphi, coord_dofs, _J);
944 auto _K
945 = stdex::submdspan(K, p, stdex::full_extent, stdex::full_extent);
946 cmap.compute_jacobian_inverse(_J, _K);
947 detJ[p] = cmap.compute_jacobian_determinant(_J, det_scratch);
948 }
949
950 std::span<const std::int32_t> dofs = dofmap->cell_dofs(cell);
951 for (int k = 0; k < element_bs; ++k)
952 {
953 // Extract computed expression values for element block k
954 for (int m = 0; m < value_size; ++m)
955 {
956 for (std::size_t k0 = 0; k0 < Xshape[0]; ++k0)
957 {
958 _vals(k0, 0, m)
959 = f[f_shape1 * (k * value_size + m) + c * Xshape[0] + k0];
960 }
961 }
962
963 // Get element degrees of freedom for block
964 for (std::size_t i = 0; i < Xshape[0]; ++i)
965 {
966 auto _u = stdex::submdspan(_vals, i, stdex::full_extent,
967 stdex::full_extent);
968 auto _U = stdex::submdspan(ref_data, i, stdex::full_extent,
969 stdex::full_extent);
970 auto _K
971 = stdex::submdspan(K, i, stdex::full_extent, stdex::full_extent);
972 auto _J
973 = stdex::submdspan(J, i, stdex::full_extent, stdex::full_extent);
974 pull_back_fn(_U, _u, _K, 1.0 / detJ[i], _J);
975 }
976
977 auto ref = stdex::submdspan(ref_data, stdex::full_extent, 0,
978 stdex::full_extent);
979 impl::interpolation_apply(Pi, ref, std::span(_coeffs), element_bs);
980 apply_inverse_transpose_dof_transformation(_coeffs, cell_info, cell, 1);
981
982 // Copy interpolation dofs into coefficient vector
983 assert(_coeffs.size() == num_scalar_dofs);
984 for (int i = 0; i < num_scalar_dofs; ++i)
985 {
986 const int dof = i * element_bs + k;
987 std::div_t pos = std::div(dof, dofmap_bs);
988 coeffs[dofmap_bs * dofs[pos.quot] + pos.rem] = _coeffs[i];
989 }
990 }
991 }
992 }

1055template <typename T, std::floating_point U>
1056void interpolate(
1057 Function<T, U>& u, const Function<T, U>& v,
1058 std::span<const std::int32_t> cells,
1059 const std::tuple<std::vector<std::int32_t>, std::vector<std::int32_t>,
1060 std::vector<U>, std::vector<std::int32_t>>&
1061 nmm_interpolation_data
1062 = {})
1063{
1064 assert(u.function_space());
1065 assert(v.function_space());
1066 auto mesh = u.function_space()->mesh();
1067 assert(mesh);
1068
1069 auto cell_map0 = mesh->topology()->index_map(mesh->topology()->dim());
1070 assert(cell_map0);
1071 std::size_t num_cells0 = cell_map0->size_local() + cell_map0->num_ghosts();
1072 if (u.function_space() == v.function_space() and cells.size() == num_cells0)
1073 {
1074 // Same function spaces and on whole mesh
1075 std::span<T> u1_array = u.x()->mutable_array();
1076 std::span<const T> u0_array = v.x()->array();
1077 std::copy(u0_array.begin(), u0_array.end(), u1_array.begin());
1078 }
1079 else
1080 {
1081 // Get mesh and check that functions share the same mesh
1082 if (auto mesh_v = v.function_space()->mesh(); mesh != mesh_v)
1083 {
1084 impl::interpolate_nonmatching_meshes(u, v, cells, nmm_interpolation_data);
1085 return;
1086 }
1087 else
1088 {
1089
1090 // Get elements and check value shape
1091 auto element0 = v.function_space()->element();
1092 assert(element0);
1093 auto element1 = u.function_space()->element();
1094 assert(element1);
1095 if (element0->value_shape().size() != element1->value_shape().size()
1096 or !std::equal(element0->value_shape().begin(),
1097 element0->value_shape().end(),
1098 element1->value_shape().begin()))
1099 {
1100 throw std::runtime_error(
1101 "Interpolation: elements have different value dimensions");
1102 }
1103
1104 if (element1 == element0 or *element1 == *element0)
1105 {
1106 // Same element, different dofmaps (or just a subset of cells)
1107
1108 const int tdim = mesh->topology()->dim();
1109 auto cell_map = mesh->topology()->index_map(tdim);
1110 assert(cell_map);
1111
1112 assert(element1->block_size() == element0->block_size());
1113
1114 // Get dofmaps
1115 std::shared_ptr<const DofMap> dofmap0 = v.function_space()->dofmap();
1116 assert(dofmap0);
1117 std::shared_ptr<const DofMap> dofmap1 = u.function_space()->dofmap();
1118 assert(dofmap1);
1119
1120 std::span<T> u1_array = u.x()->mutable_array();
1121 std::span<const T> u0_array = v.x()->array();
1122
1123 // Iterate over mesh and interpolate on each cell
1124 const int bs0 = dofmap0->bs();
1125 const int bs1 = dofmap1->bs();
1126 for (auto c : cells)
1127 {
1128 std::span<const std::int32_t> dofs0 = dofmap0->cell_dofs(c);
1129 std::span<const std::int32_t> dofs1 = dofmap1->cell_dofs(c);
1130 assert(bs0 * dofs0.size() == bs1 * dofs1.size());
1131 for (std::size_t i = 0; i < dofs0.size(); ++i)
1132 {
1133 for (int k = 0; k < bs0; ++k)
1134 {
1135 int index = bs0 * i + k;
1136 std::div_t dv1 = std::div(index, bs1);
1137 u1_array[bs1 * dofs1[dv1.quot] + dv1.rem]
1138 = u0_array[bs0 * dofs0[i] + k];
1139 }
1140 }
1141 }
1142 }
1143 else if (element1->map_type() == element0->map_type())
1144 {
1145 // Different elements, same basis function map type
1146 impl::interpolate_same_map(u, v, cells);
1147 }
1148 else
1149 {
1150 // Different elements with different maps for basis functions
1151 impl::interpolate_nonmatching_maps(u, v, cells);
1152 }
1153 }
1154 }
