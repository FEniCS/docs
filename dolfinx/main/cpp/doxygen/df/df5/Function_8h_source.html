110 Function collapse() const
111 {
112 // Create new collapsed FunctionSpace
113 auto [V, map] = _function_space->collapse();
114
115 // Create new vector
116 auto x = std::make_shared<la::Vector<T>>(V.dofmap()->index_map,
117 V.dofmap()->index_map_bs());
118
119 // Copy values into new vector
120 std::span<const T> x_old = _x->array();
121 std::span<T> x_new = x->mutable_array();
122 for (std::size_t i = 0; i < map.size(); ++i)
123 {
124 assert((int)i < x_new.size());
125 assert(map[i] < x_old.size());
126 x_new[i] = x_old[map[i]];
127 }
128
129 return Function(std::make_shared<FunctionSpace>(std::move(V)), x);
130 }

182 void interpolate(
183 const std::function<std::pair<std::vector<T>, std::vector<std::size_t>>(
184 std::experimental::mdspan<
185 const double,
186 std::experimental::extents<
187 std::size_t, 3, std::experimental::dynamic_extent>>)>& f,
188 std::span<const std::int32_t> cells)
189 {
190 assert(_function_space);
191 assert(_function_space->element());
192 assert(_function_space->mesh());
193 const std::vector<double> x = fem::interpolation_coords(
194 *_function_space->element(), *_function_space->mesh(), cells);
195 namespace stdex = std::experimental;
196 stdex::mdspan<const double,
197 stdex::extents<std::size_t, 3, stdex::dynamic_extent>>
198 _x(x.data(), 3, x.size() / 3);
199
200 const auto [fx, fshape] = f(_x);
201 assert(fshape.size() <= 2);
202 if (int vs = _function_space->element()->value_size();
203 vs == 1 and fshape.size() == 1)
204 {
205 // Check for scalar-valued functions
206 if (fshape.front() != x.size() / 3)
207 throw std::runtime_error("Data returned by callable has wrong length");
208 }
209 else
210 {
211 // Check for vector/tensor value
212 if (fshape.size() != 2)
213 throw std::runtime_error("Expected 2D array of data");
214
215 if (fshape[0] != vs)
216 {
217 throw std::runtime_error(
218 "Data returned by callable has wrong shape(0) size");
219 }
220 if (fshape[1] != x.size() / 3)
221 {
222 throw std::runtime_error(
223 "Data returned by callable has wrong shape(1) size");
224 }
225 }
226
227 std::array<std::size_t, 2> _fshape;
228 if (fshape.size() == 1)
229 _fshape = {1, fshape[0]};
230 else
231 _fshape = {fshape[0], fshape[1]};
232
233 fem::interpolate(*this, std::span<const T>(fx.data(), fx.size()), _fshape,
234 cells);
235 }

263 void interpolate(const Expression<T>& e, std::span<const std::int32_t> cells)
264 {
265 // Check that spaces are compatible
266 assert(_function_space);
267 assert(_function_space->element());
268 std::size_t value_size = e.value_size();
269 if (e.argument_function_space())
270 throw std::runtime_error("Cannot interpolate Expression with Argument");
271
272 if (value_size != _function_space->element()->value_size())
273 {
274 throw std::runtime_error(
275 "Function value size not equal to Expression value size");
276 }
277
278 {
279 // Compatibility check
280 auto [X0, shape0] = e.X();
281 auto [X1, shape1] = _function_space->element()->interpolation_points();
282 if (shape0 != shape1)
283 {
284 throw std::runtime_error(
285 "Function element interpolation points has different shape to "
286 "Expression interpolation points");
287 }
288 for (std::size_t i = 0; i < X0.size(); ++i)
289 {
290 if (std::abs(X0[i] - X1[i]) > 1.0e-10)
291 {
292 throw std::runtime_error("Function element interpolation points not "
293 "equal to Expression interpolation points");
294 }
295 }
296 }
297
298 namespace stdex = std::experimental;
299
300 // Array to hold evaluated Expression
301 std::size_t num_cells = cells.size();
302 std::size_t num_points = e.X().second[0];
303 std::vector<T> fdata(num_cells * num_points * value_size);
304 stdex::mdspan<const T, stdex::dextents<std::size_t, 3>> f(
305 fdata.data(), num_cells, num_points, value_size);
306
307 // Evaluate Expression at points
308 e.eval(cells, fdata, {num_cells, num_points * value_size});
309
310 // Reshape evaluated data to fit interpolate
311 // Expression returns matrix of shape (num_cells, num_points *
312 // value_size), i.e. xyzxyz ordering of dof values per cell per point.
313 // The interpolation uses xxyyzz input, ordered for all points of each
314 // cell, i.e. (value_size, num_cells*num_points)
315
316 std::vector<T> fdata1(num_cells * num_points * value_size);
317 stdex::mdspan<T, stdex::dextents<std::size_t, 3>> f1(
318 fdata1.data(), value_size, num_cells, num_points);
319 for (std::size_t i = 0; i < f.extent(0); ++i)
320 for (std::size_t j = 0; j < f.extent(1); ++j)
321 for (std::size_t k = 0; k < f.extent(2); ++k)
322 f1(k, i, j) = f(i, j, k);
323
324 // Interpolate values into appropriate space
325 fem::interpolate(*this, std::span<const T>(fdata1.data(), fdata1.size()),
326 {value_size, num_cells * num_points}, cells);
327 }

356 void eval(std::span<const double> x, std::array<std::size_t, 2> xshape,
357 std::span<const std::int32_t> cells, std::span<T> u,
358 std::array<std::size_t, 2> ushape) const
359 {
360 if (cells.empty())
361 return;
362
363 assert(x.size() == xshape[0] * xshape[1]);
364 assert(u.size() == ushape[0] * ushape[1]);
365
366 // TODO: This could be easily made more efficient by exploiting points
367 // being ordered by the cell to which they belong.
368
369 if (xshape[0] != cells.size())
370 {
371 throw std::runtime_error(
372 "Number of points and number of cells must be equal.");
373 }
374 if (xshape[0] != ushape[0])
375 {
376 throw std::runtime_error(
377 "Length of array for Function values must be the "
378 "same as the number of points.");
379 }
380
381 // Get mesh
382 assert(_function_space);
383 std::shared_ptr<const mesh::Mesh> mesh = _function_space->mesh();
384 assert(mesh);
385 const std::size_t gdim = mesh->geometry().dim();
386 const std::size_t tdim = mesh->topology().dim();
387 auto map = mesh->topology().index_map(tdim);
388
389 // Get geometry data
390 const graph::AdjacencyList<std::int32_t>& x_dofmap
391 = mesh->geometry().dofmap();
392 const std::size_t num_dofs_g = mesh->geometry().cmap().dim();
393 std::span<const double> x_g = mesh->geometry().x();
394
395 // Get coordinate map
396 const CoordinateElement& cmap = mesh->geometry().cmap();
397
398 // Get element
399 std::shared_ptr<const FiniteElement> element = _function_space->element();
400 assert(element);
401 const int bs_element = element->block_size();
402 const std::size_t reference_value_size
403 = element->reference_value_size() / bs_element;
404 const std::size_t value_size = element->value_size() / bs_element;
405 const std::size_t space_dimension = element->space_dimension() / bs_element;
406
407 // If the space has sub elements, concatenate the evaluations on the
408 // sub elements
409 const int num_sub_elements = element->num_sub_elements();
410 if (num_sub_elements > 1 and num_sub_elements != bs_element)
411 {
412 throw std::runtime_error("Function::eval is not supported for mixed "
413 "elements. Extract subspaces.");
414 }
415
416 // Create work vector for expansion coefficients
417 std::vector<T> coefficients(space_dimension * bs_element);
418
419 // Get dofmap
420 std::shared_ptr<const DofMap> dofmap = _function_space->dofmap();
421 assert(dofmap);
422 const int bs_dof = dofmap->bs();
423
424 std::span<const std::uint32_t> cell_info;
425 if (element->needs_dof_transformations())
426 {
427 mesh->topology_mutable().create_entity_permutations();
428 cell_info = std::span(mesh->topology().get_cell_permutation_info());
429 }
430
431 namespace stdex = std::experimental;
432 using cmdspan4_t
433 = stdex::mdspan<const double, stdex::dextents<std::size_t, 4>>;
434 using mdspan2_t = stdex::mdspan<double, stdex::dextents<std::size_t, 2>>;
435 using mdspan3_t = stdex::mdspan<double, stdex::dextents<std::size_t, 3>>;
436
437 std::vector<double> coord_dofs_b(num_dofs_g * gdim);
438 mdspan2_t coord_dofs(coord_dofs_b.data(), num_dofs_g, gdim);
439 std::vector<double> xp_b(1 * gdim);
440 mdspan2_t xp(xp_b.data(), 1, gdim);
441
442 // Loop over points
443 std::fill(u.data(), u.data() + u.size(), 0.0);
444 std::span<const T> _v = _x->array();
445
446 // Evaluate geometry basis at point (0, 0, 0) on the reference cell.
447 // Used in affine case.
448 std::array<std::size_t, 4> phi0_shape = cmap.tabulate_shape(1, 1);
449 std::vector<double> phi0_b(std::reduce(phi0_shape.begin(), phi0_shape.end(),
450 1, std::multiplies{}));
451 cmdspan4_t phi0(phi0_b.data(), phi0_shape);
452 cmap.tabulate(1, std::vector<double>(tdim), {1, tdim}, phi0_b);
453 auto dphi0 = stdex::submdspan(phi0, std::pair(1, tdim + 1), 0,
454 stdex::full_extent, 0);
455
456 // Data structure for evaluating geometry basis at specific points.
457 // Used in non-affine case.
458 std::array<std::size_t, 4> phi_shape = cmap.tabulate_shape(1, 1);
459 std::vector<double> phi_b(
460 std::reduce(phi_shape.begin(), phi_shape.end(), 1, std::multiplies{}));
461 cmdspan4_t phi(phi_b.data(), phi_shape);
462 auto dphi = stdex::submdspan(phi, std::pair(1, tdim + 1), 0,
463 stdex::full_extent, 0);
464
465 // Reference coordinates for each point
466 std::vector<double> Xb(xshape[0] * tdim);
467 mdspan2_t X(Xb.data(), xshape[0], tdim);
468
469 // Geometry data at each point
470 std::vector<double> J_b(xshape[0] * gdim * tdim);
471 mdspan3_t J(J_b.data(), xshape[0], gdim, tdim);
472 std::vector<double> K_b(xshape[0] * tdim * gdim);
473 mdspan3_t K(K_b.data(), xshape[0], tdim, gdim);
474 std::vector<double> detJ(xshape[0]);
475 std::vector<double> det_scratch(2 * gdim * tdim);
476
477 // Prepare geometry data in each cell
478 for (std::size_t p = 0; p < cells.size(); ++p)
479 {
480 const int cell_index = cells[p];
481
482 // Skip negative cell indices
483 if (cell_index < 0)
484 continue;
485
486 // Get cell geometry (coordinate dofs)
487 auto x_dofs = x_dofmap.links(cell_index);
488 assert(x_dofs.size() == num_dofs_g);
489 for (std::size_t i = 0; i < num_dofs_g; ++i)
490 {
491 const int pos = 3 * x_dofs[i];
492 for (std::size_t j = 0; j < gdim; ++j)
493 coord_dofs(i, j) = x_g[pos + j];
494 }
495
496 for (std::size_t j = 0; j < gdim; ++j)
497 xp(0, j) = x[p * xshape[1] + j];
498
499 auto _J = stdex::submdspan(J, p, stdex::full_extent, stdex::full_extent);
500 auto _K = stdex::submdspan(K, p, stdex::full_extent, stdex::full_extent);
501
502 std::array<double, 3> Xpb = {0, 0, 0};
503 stdex::mdspan<double,
504 stdex::extents<std::size_t, 1, stdex::dynamic_extent>>
505 Xp(Xpb.data(), 1, tdim);
506
507 // Compute reference coordinates X, and J, detJ and K
508 if (cmap.is_affine())
509 {
510 CoordinateElement::compute_jacobian(dphi0, coord_dofs, _J);
511 CoordinateElement::compute_jacobian_inverse(_J, _K);
512 std::array<double, 3> x0 = {0, 0, 0};
513 for (std::size_t i = 0; i < coord_dofs.extent(1); ++i)
514 x0[i] += coord_dofs(0, i);
515 CoordinateElement::pull_back_affine(Xp, _K, x0, xp);
516 detJ[p]
517 = CoordinateElement::compute_jacobian_determinant(_J, det_scratch);
518 }
519 else
520 {
521 // Pull-back physical point xp to reference coordinate Xp
522 cmap.pull_back_nonaffine(Xp, xp, coord_dofs);
523
524 cmap.tabulate(1, std::span(Xpb.data(), tdim), {1, tdim}, phi_b);
525 CoordinateElement::compute_jacobian(dphi, coord_dofs, _J);
526 CoordinateElement::compute_jacobian_inverse(_J, _K);
527 detJ[p]
528 = CoordinateElement::compute_jacobian_determinant(_J, det_scratch);
529 }
530
531 for (std::size_t j = 0; j < X.extent(1); ++j)
532 X(p, j) = Xpb[j];
533 }
534
535 // Prepare basis function data structures
536 std::vector<double> basis_derivatives_reference_values_b(
537 1 * xshape[0] * space_dimension * reference_value_size);
538 cmdspan4_t basis_derivatives_reference_values(
539 basis_derivatives_reference_values_b.data(), 1, xshape[0],
540 space_dimension, reference_value_size);
541 std::vector<double> basis_values_b(space_dimension * value_size);
542 mdspan2_t basis_values(basis_values_b.data(), space_dimension, value_size);
543
544 // Compute basis on reference element
545 element->tabulate(basis_derivatives_reference_values_b, Xb,
546 {X.extent(0), X.extent(1)}, 0);
547
548 using xu_t = stdex::mdspan<double, stdex::dextents<std::size_t, 2>>;
549 using xU_t = stdex::mdspan<const double, stdex::dextents<std::size_t, 2>>;
550 using xJ_t = stdex::mdspan<const double, stdex::dextents<std::size_t, 2>>;
551 using xK_t = stdex::mdspan<const double, stdex::dextents<std::size_t, 2>>;
552 auto push_forward_fn
553 = element->basix_element().map_fn<xu_t, xU_t, xJ_t, xK_t>();
554
555 auto apply_dof_transformation
556 = element->get_dof_transformation_function<double>();
557 const std::size_t num_basis_values = space_dimension * reference_value_size;
558
559 for (std::size_t p = 0; p < cells.size(); ++p)
560 {
561 const int cell_index = cells[p];
562
563 // Skip negative cell indices
564 if (cell_index < 0)
565 continue;
566
567 // Permute the reference values to account for the cell's
568 // orientation
569 apply_dof_transformation(
570 std::span(basis_derivatives_reference_values_b.data()
571 + p * num_basis_values,
572 num_basis_values),
573 cell_info, cell_index, reference_value_size);
574
575 {
576 auto _U = stdex::submdspan(basis_derivatives_reference_values, 0, p,
577 stdex::full_extent, stdex::full_extent);
578 auto _J
579 = stdex::submdspan(J, p, stdex::full_extent, stdex::full_extent);
580 auto _K
581 = stdex::submdspan(K, p, stdex::full_extent, stdex::full_extent);
582 push_forward_fn(basis_values, _U, _J, detJ[p], _K);
583 }
584
585 // Get degrees of freedom for current cell
586 std::span<const std::int32_t> dofs = dofmap->cell_dofs(cell_index);
587 for (std::size_t i = 0; i < dofs.size(); ++i)
588 for (int k = 0; k < bs_dof; ++k)
589 coefficients[bs_dof * i + k] = _v[bs_dof * dofs[i] + k];
590
591 // Compute expansion
592 for (int k = 0; k < bs_element; ++k)
593 {
594 for (std::size_t i = 0; i < space_dimension; ++i)
595 {
596 for (std::size_t j = 0; j < value_size; ++j)
597 {
598 u[p * ushape[1] + (j * bs_element + k)]
599 += coefficients[bs_element * i + k] * basis_values(i, j);
600 }
601 }
602 }
603 }
604 }
