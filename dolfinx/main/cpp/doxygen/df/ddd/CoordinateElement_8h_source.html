157 template <typename U>
158 static double
159 compute_jacobian_determinant(const U& J, std::span<typename U::value_type> w)
160 {
161 static_assert(U::rank() == 2, "Must be rank 2");
162 if (J.extent(0) == J.extent(1))
163 return math::det(J);
164 else
165 {
166 assert(w.size() >= 2 * J.extent(0) * J.extent(1));
167 using X = typename U::element_type;
168 using mdspan2_t = MDSPAN_IMPL_STANDARD_NAMESPACE::mdspan<
169 X, MDSPAN_IMPL_STANDARD_NAMESPACE::dextents<std::size_t, 2>>;
170 mdspan2_t B(w.data(), J.extent(1), J.extent(0));
171 mdspan2_t BA(w.data() + J.extent(0) * J.extent(1), B.extent(0),
172 J.extent(1));
173 for (std::size_t i = 0; i < B.extent(0); ++i)
174 for (std::size_t j = 0; j < B.extent(1); ++j)
175 B(i, j) = J(j, i);
176
177 // Zero working memory of BA
178 std::fill_n(BA.data_handle(), BA.size(), 0);
179 math::dot(B, J, BA);
180 return std::sqrt(math::det(BA));
181 }
182 }

194 template <typename U, typename V, typename W>
195 static void push_forward(U&& x, const V& cell_geometry, const W& phi)
196 {
197 for (std::size_t i = 0; i < x.extent(0); ++i)
198 for (std::size_t j = 0; j < x.extent(1); ++j)

215 template <typename U, typename V, typename W>
216 static void pull_back_affine(U&& X, const V& K, const std::array<T, 3>& x0,
217 const W& x)
218 {
219 assert(X.extent(0) == x.extent(0));
220 assert(X.extent(1) == K.extent(0));
221 assert(x.extent(1) == K.extent(1));
222 for (std::size_t i = 0; i < X.extent(0); ++i)
223 for (std::size_t j = 0; j < X.extent(1); ++j)
224 X(i, j) = 0;
225
226 // Calculate X for each point
227 for (std::size_t p = 0; p < x.extent(0); ++p)
228 for (std::size_t i = 0; i < K.extent(0); ++i)
229 for (std::size_t j = 0; j < K.extent(1); ++j)
230 X(p, i) += K(i, j) * (x(p, j) - x0[j]);
231 }
