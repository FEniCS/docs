134template <typename T>
135MeshTags<T> create_meshtags(std::shared_ptr<const Topology> topology, int dim,
136 const graph::AdjacencyList<std::int32_t>& entities,
137 std::span<const T> values)
138{
139 LOG(INFO)
140 << "Building MeshTags object from tagged entities (defined by vertices).";
141
142 // Compute the indices of the topology entities (index is set to -1 if
143 // it can't be found)
144 assert(topology);
145 const std::vector<std::int32_t> indices
146 = entities_to_index(*topology, dim, entities);
147 if (indices.size() != values.size())
148 {
149 throw std::runtime_error(
150 "Duplicate mesh entities when building MeshTags object.");
151 }
152
153 // Sort the indices and values by indices
154 auto [indices_sorted, values_sorted] = common::sort_unique(indices, values);
155
156 // Remove any entities that were not found (these have an index of -1)
157 auto it0 = std::lower_bound(indices_sorted.begin(), indices_sorted.end(), 0);
158 std::size_t pos0 = std::distance(indices_sorted.begin(), it0);
159 indices_sorted.erase(indices_sorted.begin(), it0);
160 values_sorted.erase(values_sorted.begin(),
161 std::next(values_sorted.begin(), pos0));
162
163 return MeshTags<T>(topology, dim, std::move(indices_sorted),
164 std::move(values_sorted));
165}
