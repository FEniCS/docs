50template <typename T>
51void reorder_list(std::span<T> list, std::span<const std::int32_t> nodemap)
52{
53 if (nodemap.empty())
54 return;
55
56 assert(list.size() % nodemap.size() == 0);
57 std::size_t degree = list.size() / nodemap.size();
58 const std::vector<T> orig(list.begin(), list.end());
59 for (std::size_t n = 0; n < nodemap.size(); ++n)
60 {
61 std::span links_old(orig.data() + n * degree, degree);
62 auto links_new = list.subspan(nodemap[n] * degree, degree);
63 std::ranges::copy(links_old, links_new.begin());
64 }
65}

80template <std::floating_point T>
81std::tuple<std::vector<std::int32_t>, std::vector<T>, std::vector<std::int32_t>>
82compute_vertex_coords_boundary(const mesh::Mesh<T>& mesh, int dim,
83 std::span<const std::int32_t> facets)
84{
85 auto topology = mesh.topology();
86 assert(topology);
87 const int tdim = topology->dim();
88 if (dim == tdim)
89 {
90 throw std::runtime_error(
91 "Cannot use mesh::locate_entities_boundary (boundary) for cells.");
92 }
93
94 // Build set of vertices on boundary and set of boundary entities
95 mesh.topology_mutable()->create_connectivity(tdim - 1, 0);
96 mesh.topology_mutable()->create_connectivity(tdim - 1, dim);
97 std::vector<std::int32_t> vertices, entities;
98 {
99 auto f_to_v = topology->connectivity(tdim - 1, 0);
100 assert(f_to_v);
101 auto f_to_e = topology->connectivity(tdim - 1, dim);
102 assert(f_to_e);
103 for (auto f : facets)
104 {
105 auto v = f_to_v->links(f);
106 vertices.insert(vertices.end(), v.begin(), v.end());
107 auto e = f_to_e->links(f);
108 entities.insert(entities.end(), e.begin(), e.end());
109 }
110
111 // Build vector of boundary vertices
112 {
113 std::ranges::sort(vertices);
114 auto [unique_end, range_end] = std::ranges::unique(vertices);
115 vertices.erase(unique_end, range_end);
116 }
117
118 {
119 std::ranges::sort(entities);
120 auto [unique_end, range_end] = std::ranges::unique(entities);
121 entities.erase(unique_end, range_end);
122 }
123 }
124
125 // Get geometry data
126 auto x_dofmap = mesh.geometry().dofmap();
127 std::span<const T> x_nodes = mesh.geometry().x();
128
129 // Get all vertex 'node' indices
130 mesh.topology_mutable()->create_connectivity(0, tdim);
131 mesh.topology_mutable()->create_connectivity(tdim, 0);
132 auto v_to_c = topology->connectivity(0, tdim);
133 assert(v_to_c);
134 auto c_to_v = topology->connectivity(tdim, 0);
135 assert(c_to_v);
136 std::vector<T> x_vertices(3 * vertices.size(), -1.0);
137 std::vector<std::int32_t> vertex_to_pos(v_to_c->num_nodes(), -1);
138 for (std::size_t i = 0; i < vertices.size(); ++i)
139 {
140 const std::int32_t v = vertices[i];
141
142 // Get first cell and find position
143 const std::int32_t c = v_to_c->links(v).front();
144 auto cell_vertices = c_to_v->links(c);
145 auto it = std::find(cell_vertices.begin(), cell_vertices.end(), v);
146 assert(it != cell_vertices.end());
147 const std::size_t local_pos = std::distance(cell_vertices.begin(), it);
148
149 auto dofs = MDSPAN_IMPL_STANDARD_NAMESPACE::submdspan(
150 x_dofmap, c, MDSPAN_IMPL_STANDARD_NAMESPACE::full_extent);
151 for (std::size_t j = 0; j < 3; ++j)
152 x_vertices[j * vertices.size() + i] = x_nodes[3 * dofs[local_pos] + j];
153 vertex_to_pos[v] = i;
154 }
155
156 return {std::move(entities), std::move(x_vertices), std::move(vertex_to_pos)};
157}

232template <std::floating_point T>
233std::vector<T> h(const Mesh<T>& mesh, std::span<const std::int32_t> entities,
234 int dim)
235{
236 if (entities.empty())
237 return std::vector<T>();
238 if (dim == 0)
239 return std::vector<T>(entities.size(), 0);
240
241 // Get the geometry dofs for the vertices of each entity
242 const std::vector<std::int32_t> vertex_xdofs
243 = entities_to_geometry(mesh, dim, entities, false);
244 assert(!entities.empty());
245 const std::size_t num_vertices = vertex_xdofs.size() / entities.size();
246
247 // Get the geometry coordinate
248 std::span<const T> x = mesh.geometry().x();
249
250 // Function to compute the length of (p0 - p1)
251 auto delta_norm = [](auto&& p0, auto&& p1)
252 {
253 T norm = 0;
254 for (std::size_t i = 0; i < 3; ++i)
255 norm += (p0[i] - p1[i]) * (p0[i] - p1[i]);
256 return std::sqrt(norm);
257 };
258
259 // Compute greatest distance between any to vertices
260 assert(dim > 0);
261 std::vector<T> h(entities.size(), 0);
262 for (std::size_t e = 0; e < entities.size(); ++e)
263 {
264 // Get geometry 'dof' for each vertex of entity e
265 std::span<const std::int32_t> e_vertices(
266 vertex_xdofs.data() + e * num_vertices, num_vertices);
267
268 // Compute maximum distance between any two vertices
269 for (std::size_t i = 0; i < e_vertices.size(); ++i)
270 {
271 std::span<const T, 3> p0(x.data() + 3 * e_vertices[i], 3);
272 for (std::size_t j = i + 1; j < e_vertices.size(); ++j)
273 {
274 std::span<const T, 3> p1(x.data() + 3 * e_vertices[j], 3);
275 h[e] = std::max(h[e], delta_norm(p0, p1));
276 }
277 }
278 }
279
280 return h;
281}

286template <std::floating_point T>
287std::vector<T> cell_normals(const Mesh<T>& mesh, int dim,
288 std::span<const std::int32_t> entities)
289{
290 if (entities.empty())
291 return std::vector<T>();
292
293 auto topology = mesh.topology();
294 assert(topology);
295 if (topology->cell_type() == CellType::prism and dim == 2)
296 {
297 throw std::runtime_error(
298 "Cell normal computation for prism cells not yet supported.");
299 }
300
301 const int gdim = mesh.geometry().dim();
302 const CellType type = cell_entity_type(topology->cell_type(), dim, 0);
303
304 // Find geometry nodes for topology entities
305 std::span<const T> x = mesh.geometry().x();
306 std::vector<std::int32_t> geometry_entities
307 = entities_to_geometry(mesh, dim, entities, false);
308
309 const std::size_t shape1 = geometry_entities.size() / entities.size();
310 std::vector<T> n(entities.size() * 3);
311 switch (type)
312 {
313 case CellType::interval:
314 {
315 if (gdim > 2)
316 throw std::invalid_argument("Interval cell normal undefined in 3D.");
317 for (std::size_t i = 0; i < entities.size(); ++i)
318 {
319 // Get the two vertices as points
320 std::array vertices{geometry_entities[i * shape1],
321 geometry_entities[i * shape1 + 1]};
322 std::array p = {std::span<const T, 3>(x.data() + 3 * vertices[0], 3),
323 std::span<const T, 3>(x.data() + 3 * vertices[1], 3)};
324
325 // Define normal by rotating tangent counter-clockwise
326 std::array<T, 3> t;
327 std::ranges::transform(p[1], p[0], t.begin(),
328 [](auto x, auto y) { return x - y; });
329
330 T norm = std::sqrt(t[0] * t[0] + t[1] * t[1]);
331 std::span<T, 3> ni(n.data() + 3 * i, 3);
332 ni[0] = -t[1] / norm;
333 ni[1] = t[0] / norm;
334 ni[2] = 0.0;
335 }
336 return n;
337 }
338 case CellType::triangle:
339 {
340 for (std::size_t i = 0; i < entities.size(); ++i)
341 {
342 // Get the three vertices as points
343 std::array vertices = {geometry_entities[i * shape1 + 0],
344 geometry_entities[i * shape1 + 1],
345 geometry_entities[i * shape1 + 2]};
346 std::array p = {std::span<const T, 3>(x.data() + 3 * vertices[0], 3),
347 std::span<const T, 3>(x.data() + 3 * vertices[1], 3),
348 std::span<const T, 3>(x.data() + 3 * vertices[2], 3)};
349
350 // Compute (p1 - p0) and (p2 - p0)
351 std::array<T, 3> dp1, dp2;
352 std::ranges::transform(p[1], p[0], dp1.begin(),
353 [](auto x, auto y) { return x - y; });
354 std::ranges::transform(p[2], p[0], dp2.begin(),
355 [](auto x, auto y) { return x - y; });
356
357 // Define cell normal via cross product of first two edges
358 std::array<T, 3> ni = math::cross(dp1, dp2);
359 T norm = std::sqrt(ni[0] * ni[0] + ni[1] * ni[1] + ni[2] * ni[2]);
360 std::ranges::transform(ni, std::next(n.begin(), 3 * i),
361 [norm](auto x) { return x / norm; });
362 }
363
364 return n;
365 }
366 case CellType::quadrilateral:
367 {
368 // TODO: check
369 for (std::size_t i = 0; i < entities.size(); ++i)
370 {
371 // Get the three vertices as points
372 std::array vertices = {geometry_entities[i * shape1 + 0],
373 geometry_entities[i * shape1 + 1],
374 geometry_entities[i * shape1 + 2]};
375 std::array p = {std::span<const T, 3>(x.data() + 3 * vertices[0], 3),
376 std::span<const T, 3>(x.data() + 3 * vertices[1], 3),
377 std::span<const T, 3>(x.data() + 3 * vertices[2], 3)};
378
379 // Compute (p1 - p0) and (p2 - p0)
380 std::array<T, 3> dp1, dp2;
381 std::ranges::transform(p[1], p[0], dp1.begin(),
382 [](auto x, auto y) { return x - y; });
383 std::ranges::transform(p[2], p[0], dp2.begin(),
384 [](auto x, auto y) { return x - y; });
385
386 // Define cell normal via cross product of first two edges
387 std::array<T, 3> ni = math::cross(dp1, dp2);
388 T norm = std::sqrt(ni[0] * ni[0] + ni[1] * ni[1] + ni[2] * ni[2]);
389 std::ranges::transform(ni, std::next(n.begin(), 3 * i),
390 [norm](auto x) { return x / norm; });
391 }
392
393 return n;
394 }
395 default:
396 throw std::invalid_argument(
397 "cell_normal not supported for this cell type.");
398 }

404template <std::floating_point T>
405std::vector<T> compute_midpoints(const Mesh<T>& mesh, int dim,
406 std::span<const std::int32_t> entities)
407{
408 if (entities.empty())
409 return std::vector<T>();
410
411 std::span<const T> x = mesh.geometry().x();
412
413 // Build map from entity -> geometry dof
414 const std::vector<std::int32_t> e_to_g
415 = entities_to_geometry(mesh, dim, entities, false);
416 std::size_t shape1 = e_to_g.size() / entities.size();
417
418 std::vector<T> x_mid(entities.size() * 3, 0);
419 for (std::size_t e = 0; e < entities.size(); ++e)
420 {
421 std::span<T, 3> p(x_mid.data() + 3 * e, 3);
422 std::span<const std::int32_t> rows(e_to_g.data() + e * shape1, shape1);
423 for (auto row : rows)
424 {
425 std::span<const T, 3> xg(x.data() + 3 * row, 3);
426 std::ranges::transform(p, xg, p.begin(),
427 [size = rows.size()](auto x, auto y)
428 { return x + y / size; });
429 }
430 }
431
432 return x_mid;
433}

441template <std::floating_point T>
442std::pair<std::vector<T>, std::array<std::size_t, 2>>
443compute_vertex_coords(const mesh::Mesh<T>& mesh)
444{
445 namespace md = MDSPAN_IMPL_STANDARD_NAMESPACE;
446
447 auto topology = mesh.topology();
448 assert(topology);
449 const int tdim = topology->dim();
450
451 // Create entities and connectivities
452
453 // Get all vertex 'node' indices
454 const std::int32_t num_vertices = topology->index_map(0)->size_local()
455 + topology->index_map(0)->num_ghosts();
456
457 std::vector<std::int32_t> vertex_to_node(num_vertices);
458 for (int cell_type_idx = 0,
459 num_cell_types = topology->entity_types(tdim).size();
460 cell_type_idx < num_cell_types; ++cell_type_idx)
461 {
462 auto x_dofmap = mesh.geometry().dofmap(cell_type_idx);
463 auto c_to_v = topology->connectivity({tdim, cell_type_idx}, {0, 0});
464 assert(c_to_v);
465 for (int c = 0; c < c_to_v->num_nodes(); ++c)
466 {
467 auto x_dofs = md::submdspan(x_dofmap, c, md::full_extent);
468 auto vertices = c_to_v->links(c);
469 for (std::size_t i = 0; i < vertices.size(); ++i)
470 vertex_to_node[vertices[i]] = x_dofs[i];
471 }
472 }
473
474 // Pack coordinates of vertices
475 std::span<const T> x_nodes = mesh.geometry().x();
476 std::vector<T> x_vertices(3 * vertex_to_node.size(), 0.0);
477 for (std::size_t i = 0; i < vertex_to_node.size(); ++i)
478 {
479 std::int32_t pos = 3 * vertex_to_node[i];
480 for (std::size_t j = 0; j < 3; ++j)
481 x_vertices[j * vertex_to_node.size() + i] = x_nodes[pos + j];
482 }
483
484 return {std::move(x_vertices), {3, vertex_to_node.size()}};
485}

513template <std::floating_point T, MarkerFn<T> U>
514std::vector<std::int32_t> locate_entities(const Mesh<T>& mesh, int dim,
515 U marker, int entity_type_idx)
516{
517
518 using cmdspan3x_t = MDSPAN_IMPL_STANDARD_NAMESPACE::mdspan<
519 const T,
520 MDSPAN_IMPL_STANDARD_NAMESPACE::extents<
521 std::size_t, 3, MDSPAN_IMPL_STANDARD_NAMESPACE::dynamic_extent>>;
522
523 // Run marker function on vertex coordinates
524 const auto [xdata, xshape] = impl::compute_vertex_coords(mesh);
525
526 cmdspan3x_t x(xdata.data(), xshape);
527 const std::vector<std::int8_t> marked = marker(x);
528 if (marked.size() != x.extent(1))
529 throw std::runtime_error("Length of array of markers is wrong.");
530
531 auto topology = mesh.topology();
532 assert(topology);
533 const int tdim = topology->dim();
534
535 mesh.topology_mutable()->create_entities(dim);
536 if (dim < tdim)
537 mesh.topology_mutable()->create_connectivity(dim, 0);
538
539 // Iterate over entities of dimension 'dim' to build vector of marked
540 // entities
541 auto e_to_v = topology->connectivity({dim, entity_type_idx}, {0, 0});
542 assert(e_to_v);
543 std::vector<std::int32_t> entities;
544 for (int e = 0; e < e_to_v->num_nodes(); ++e)
545 {
546 // Iterate over entity vertices
547 bool all_vertices_marked = true;
548 for (std::int32_t v : e_to_v->links(e))
549 {
550 if (!marked[v])
551 {
552 all_vertices_marked = false;
553 break;
554 }
555 }
556
557 if (all_vertices_marked)
558 entities.push_back(e);
559 }
560
561 return entities;
562}

613template <std::floating_point T, MarkerFn<T> U>
614std::vector<std::int32_t> locate_entities_boundary(const Mesh<T>& mesh, int dim,
615 U marker)
616{
617 // TODO Rewrite this function, it should be possible to simplify considerably
618 auto topology = mesh.topology();
619 assert(topology);
620 int tdim = topology->dim();
621 if (dim == tdim)
622 {
623 throw std::runtime_error(
624 "Cannot use mesh::locate_entities_boundary (boundary) for cells.");
625 }
626
627 // Compute list of boundary facets
628 mesh.topology_mutable()->create_entities(tdim - 1);
629 mesh.topology_mutable()->create_connectivity(tdim - 1, tdim);
630 std::vector<std::int32_t> boundary_facets = exterior_facet_indices(*topology);
631
632 using cmdspan3x_t = MDSPAN_IMPL_STANDARD_NAMESPACE::mdspan<
633 const T,
634 MDSPAN_IMPL_STANDARD_NAMESPACE::extents<
635 std::size_t, 3, MDSPAN_IMPL_STANDARD_NAMESPACE::dynamic_extent>>;
636
637 // Run marker function on the vertex coordinates
638 auto [facet_entities, xdata, vertex_to_pos]
639 = impl::compute_vertex_coords_boundary(mesh, dim, boundary_facets);
640 cmdspan3x_t x(xdata.data(), 3, xdata.size() / 3);
641 std::vector<std::int8_t> marked = marker(x);
642 if (marked.size() != x.extent(1))
643 throw std::runtime_error("Length of array of markers is wrong.");
644
645 // Loop over entities and check vertex markers
646 mesh.topology_mutable()->create_entities(dim);
647 auto e_to_v = topology->connectivity(dim, 0);
648 assert(e_to_v);
649 std::vector<std::int32_t> entities;
650 for (auto e : facet_entities)
651 {
652 // Iterate over entity vertices
653 bool all_vertices_marked = true;
654 for (auto v : e_to_v->links(e))
655 {
656 const std::int32_t pos = vertex_to_pos[v];
657 if (!marked[pos])
658 {
659 all_vertices_marked = false;
660 break;
661 }
662 }
663
664 // Mark facet with all vertices marked
665 if (all_vertices_marked)
666 entities.push_back(e);
667 }
668
669 return entities;
670}

690template <std::floating_point T>
691std::vector<std::int32_t>
692entities_to_geometry(const Mesh<T>& mesh, int dim,
693 std::span<const std::int32_t> entities,
694 bool permute = false)
695{
696 auto topology = mesh.topology();
697 assert(topology);
698 CellType cell_type = topology->cell_type();
699 if (cell_type == CellType::prism and dim == 2)
700 {
701 throw std::runtime_error(
702 "mesh::entities_to_geometry for prism cells not yet supported.");
703 }
704
705 const int tdim = topology->dim();
706 const Geometry<T>& geometry = mesh.geometry();
707 auto xdofs = geometry.dofmap();
708
709 // Get the DOF layout and the number of DOFs per entity
710 const fem::CoordinateElement<T>& coord_ele = geometry.cmap();
711 const fem::ElementDofLayout layout = coord_ele.create_dof_layout();
712 const std::size_t num_entity_dofs = layout.num_entity_closure_dofs(dim);
713 std::vector<std::int32_t> entity_xdofs;
714 entity_xdofs.reserve(entities.size() * num_entity_dofs);
715
716 // Get the element's closure DOFs
717 const std::vector<std::vector<std::vector<int>>>& closure_dofs_all
718 = layout.entity_closure_dofs_all();
719
720 // Special case when dim == tdim (cells)
721 if (dim == tdim)
722 {
723 for (std::size_t i = 0; i < entities.size(); ++i)
724 {
725 const std::int32_t c = entities[i];
726 // Extract degrees of freedom
727 auto x_c = MDSPAN_IMPL_STANDARD_NAMESPACE::submdspan(
728 xdofs, c, MDSPAN_IMPL_STANDARD_NAMESPACE::full_extent);
729 for (std::int32_t entity_dof : closure_dofs_all[tdim][0])
730 entity_xdofs.push_back(x_c[entity_dof]);
731 }
732 return entity_xdofs;
733 }
734
735 assert(dim != tdim);
736
737 auto e_to_c = topology->connectivity(dim, tdim);
738 if (!e_to_c)
739 {
740 throw std::runtime_error(
741 "Entity-to-cell connectivity has not been computed. Missing dims "
742 + std::to_string(dim) + "->" + std::to_string(tdim));
743 }
744
745 auto c_to_e = topology->connectivity(tdim, dim);
746 if (!c_to_e)
747 {
748 throw std::runtime_error(
749 "Cell-to-entity connectivity has not been computed. Missing dims "
750 + std::to_string(tdim) + "->" + std::to_string(dim));
751 }
752
753 // Get the cell info, which is needed to permute the closure dofs
754 std::span<const std::uint32_t> cell_info;
755 if (permute)
756 cell_info = std::span(mesh.topology()->get_cell_permutation_info());
757
758 for (std::size_t i = 0; i < entities.size(); ++i)
759 {
760 const std::int32_t e = entities[i];
761
762 // Get a cell connected to the entity
763 assert(!e_to_c->links(e).empty());
764 std::int32_t c = e_to_c->links(e).front();
765
766 // Get the local index of the entity
767 std::span<const std::int32_t> cell_entities = c_to_e->links(c);
768 auto it = std::find(cell_entities.begin(), cell_entities.end(), e);
769 assert(it != cell_entities.end());
770 std::size_t local_entity = std::distance(cell_entities.begin(), it);
771
772 std::vector<std::int32_t> closure_dofs(closure_dofs_all[dim][local_entity]);
773
774 // Cell sub-entities must be permuted so that their local
775 // orientation agrees with their global orientation
776 if (permute)
777 {
778 mesh::CellType entity_type
779 = mesh::cell_entity_type(cell_type, dim, local_entity);
780 coord_ele.permute_subentity_closure(closure_dofs, cell_info[c],
781 entity_type, local_entity);
782 }
783
784 // Extract degrees of freedom
785 auto x_c = MDSPAN_IMPL_STANDARD_NAMESPACE::submdspan(
786 xdofs, c, MDSPAN_IMPL_STANDARD_NAMESPACE::full_extent);
787 for (std::int32_t entity_dof : closure_dofs)
788 entity_xdofs.push_back(x_c[entity_dof]);
789 }
790
791 return entity_xdofs;
792}

854template <typename U>
855Mesh<typename std::remove_reference_t<typename U::value_type>> create_mesh(
856 MPI_Comm comm, MPI_Comm commt,
857 std::vector<std::span<const std::int64_t>> cells,
858 const std::vector<fem::CoordinateElement<
859 typename std::remove_reference_t<typename U::value_type>>>& elements,
860 MPI_Comm commg, const U& x, std::array<std::size_t, 2> xshape,
861 const CellPartitionFunction& partitioner)
862{
863 assert(cells.size() == elements.size());
864 std::vector<CellType> celltypes;
865 std::ranges::transform(elements, std::back_inserter(celltypes),
866 [](auto e) { return e.cell_shape(); });
867 std::vector<fem::ElementDofLayout> doflayouts;
868 std::ranges::transform(elements, std::back_inserter(doflayouts),
869 [](auto e) { return e.create_dof_layout(); });
870
871 // Note: `extract_topology` extracts topology data, i.e. just the
872 // vertices. For P1 geometry this should just be the identity
873 // operator. For other elements the filtered lists may have 'gaps',
874 // i.e. the indices might not be contiguous.
875 //
876 // `extract_topology` could be skipped for 'P1 geometry' elements
877
878 std::int32_t num_cell_types = cells.size();
879
880 // -- Partition topology across ranks of comm
881 std::vector<std::vector<std::int64_t>> cells1(num_cell_types);
882 std::vector<std::vector<std::int64_t>> original_idx1(num_cell_types);
883 std::vector<std::vector<int>> ghost_owners(num_cell_types);
884 if (partitioner)
885 {
886 spdlog::info("Using partitioner with cell data ({} cell types)",
887 num_cell_types);
888 graph::AdjacencyList<std::int32_t> dest(0);
889 if (commt != MPI_COMM_NULL)
890 {
891 int size = dolfinx::MPI::size(comm);
892 std::vector<std::vector<std::int64_t>> t(num_cell_types);
893 std::vector<std::span<const std::int64_t>> tspan(num_cell_types);
894 for (std::int32_t i = 0; i < num_cell_types; ++i)
895 {
896 t[i] = extract_topology(celltypes[i], doflayouts[i], cells[i]);
897 tspan[i] = std::span(t[i]);
898 }
899 dest = partitioner(commt, size, celltypes, tspan);
900 }
901
902 std::int32_t cell_offset = 0;
903 for (std::int32_t i = 0; i < num_cell_types; ++i)
904 {
905 std::size_t num_cell_nodes = doflayouts[i].num_dofs();
906 assert(cells[i].size() % num_cell_nodes == 0);
907 std::size_t num_cells = cells[i].size() / num_cell_nodes;
908
909 // Extract destination AdjacencyList for this cell type
910 std::vector<std::int32_t> offsets_i(
911 std::next(dest.offsets().begin(), cell_offset),
912 std::next(dest.offsets().begin(), cell_offset + num_cells + 1));
913 std::vector<std::int32_t> data_i(
914 std::next(dest.array().begin(), offsets_i.front()),
915 std::next(dest.array().begin(), offsets_i.back()));
916 std::int32_t offset_0 = offsets_i.front();
917 std::ranges::for_each(offsets_i,
918 [&offset_0](std::int32_t& j) { j -= offset_0; });
919 graph::AdjacencyList<std::int32_t> dest_i(data_i, offsets_i);
920 cell_offset += num_cells;
921
922 // Distribute cells (topology, includes higher-order 'nodes') to
923 // destination rank
924 std::vector<int> src_ranks;
925 std::tie(cells1[i], src_ranks, original_idx1[i], ghost_owners[i])
926 = graph::build::distribute(comm, cells[i],
927 {num_cells, num_cell_nodes}, dest_i);
928 spdlog::debug("Got {} cells from distribution", cells1[i].size());
929 }
930 }
931 else
932 {
933 // No partitioning, construct a global index
934 std::int64_t num_owned = 0;
935 for (std::int32_t i = 0; i < num_cell_types; ++i)
936 {
937 cells1[i] = std::vector<std::int64_t>(cells[i].begin(), cells[i].end());
938 std::int32_t num_cell_nodes = doflayouts[i].num_dofs();
939 assert(cells1[i].size() % num_cell_nodes == 0);
940 original_idx1[i].resize(cells1[i].size() / num_cell_nodes);
941 num_owned += original_idx1[i].size();
942 }
943
944 // Add on global offset
945 std::int64_t global_offset = 0;
946 MPI_Exscan(&num_owned, &global_offset, 1, MPI_INT64_T, MPI_SUM, comm);
947 for (std::int32_t i = 0; i < num_cell_types; ++i)
948 {
949 std::iota(original_idx1[i].begin(), original_idx1[i].end(),
950 global_offset);
951 global_offset += original_idx1[i].size();
952 }
953 }
954
955 // Extract cell 'topology', i.e. extract the vertices for each cell
956 // and discard any 'higher-order' nodes
957 std::vector<std::vector<std::int64_t>> cells1_v(num_cell_types);
958 for (std::int32_t i = 0; i < num_cell_types; ++i)
959 {
960 cells1_v[i] = extract_topology(celltypes[i], doflayouts[i], cells1[i]);
961 spdlog::info("Extract basic topology: {}->{}", cells1[i].size(),
962 cells1_v[i].size());
963 }
964
965 // Build local dual graph for owned cells to (i) get list of vertices
966 // on the process boundary and (ii) apply re-ordering to cells for
967 // locality
968 auto boundary_v_fn = [](const std::vector<CellType>& celltypes,
969 const std::vector<fem::ElementDofLayout>& doflayouts,
970 const std::vector<std::vector<int>>& ghost_owners,
971 std::vector<std::vector<std::int64_t>>& cells1,
972 std::vector<std::vector<std::int64_t>>& cells1_v,
973 std::vector<std::vector<std::int64_t>>& original_idx1)
974 {
975 spdlog::info("Build local dual graphs, re-order cells, and compute process "
976 "boundary vertices.");
977
978 std::vector<std::pair<std::vector<std::int64_t>, int>> facets;
979
980 // Build lists of cells (by cell type) that excludes ghosts
981 std::vector<std::span<const std::int64_t>> cells1_v_local;
982 for (std::size_t i = 0; i < celltypes.size(); ++i)
983 {
984 int num_cell_vertices = mesh::num_cell_vertices(celltypes[i]);
985 std::size_t num_owned_cells
986 = cells1_v[i].size() / num_cell_vertices - ghost_owners[i].size();
987 cells1_v_local.emplace_back(cells1_v[i].data(),
988 num_owned_cells * num_cell_vertices);
989
990 // Build local dual graph for cell type
991 auto [graph, unmatched_facets, max_v, _facet_attached_cells]
992 = build_local_dual_graph(std::vector{celltypes[i]},
993 std::vector{cells1_v_local.back()});
994
995 // Store unmatched_facets for current cell type
996 facets.emplace_back(std::move(unmatched_facets), max_v);
997
998 // Compute re-ordering of graph
999 const std::vector<std::int32_t> remap = graph::reorder_gps(graph);
1000
1001 // Update 'original' indices
1002 const std::vector<std::int64_t>& orig_idx = original_idx1[i];
1003 std::vector<std::int64_t> _original_idx(orig_idx.size());
1004 std::copy_n(orig_idx.rbegin(), ghost_owners[i].size(),
1005 _original_idx.rbegin());
1006 {
1007 for (std::size_t j = 0; j < remap.size(); ++j)
1008 _original_idx[remap[j]] = orig_idx[j];
1009 }
1010 original_idx1[i] = _original_idx;
1011
1012 // Reorder cells
1013 impl::reorder_list(
1014 std::span(cells1_v[i].data(), remap.size() * num_cell_vertices),
1015 remap);
1016 impl::reorder_list(
1017 std::span(cells1[i].data(), remap.size() * doflayouts[i].num_dofs()),
1018 remap);
1019 }
1020
1021 if (facets.size() == 1) // Optimisation for single cell type
1022 {
1023 std::vector<std::int64_t>& vertices = facets.front().first;
1024
1025 // Remove duplicated vertex indices
1026 std::ranges::sort(vertices);
1027 auto [unique_end, range_end] = std::ranges::unique(vertices);
1028 vertices.erase(unique_end, range_end);
1029
1030 // Remove -1 if it appears as first entity. This can happen in
1031 // mixed topology meshes where '-1' is used to pad facet data when
1032 // cells facets have differing numbers of vertices.
1033 if (!vertices.empty() and vertices.front() == -1)
1034 vertices.erase(vertices.begin());
1035
1036 return vertices;
1037 }
1038 else
1039 {
1040 // Pack 'unmatched' facets for all cell types into single array
1041 // (facets0)
1042 std::vector<std::int64_t> facets0;
1043 facets0.reserve(std::accumulate(facets.begin(), facets.end(),
1044 std::size_t(0), [](std::size_t x, auto& y)
1045 { return x + y.first.size(); }));
1046 int max_v = std::ranges::max_element(facets, [](auto& a, auto& b)
1047 { return a.second < b.second; })
1048 ->second;
1049 for (const auto& [v_data, num_v] : facets)
1050 {
1051 for (auto it = v_data.begin(); it != v_data.end(); it += num_v)
1052 {
1053 facets0.insert(facets0.end(), it, std::next(it, num_v));
1054 facets0.insert(facets0.end(), max_v - num_v, -1);
1055 }
1056 }
1057
1058 // Compute row permutation
1059 const std::vector<std::int32_t> perm = dolfinx::sort_by_perm(
1060 std::span<const std::int64_t>(facets0), max_v);
1061
1062 // For facets in facets0 that appear only once, store the facet
1063 // vertices
1064 std::vector<std::int64_t> vertices;
1065 auto it = perm.begin();
1066 while (it != perm.end())
1067 {
1068 // Find iterator to next facet different from f and trim any -1
1069 // padding
1070 std::span _f(facets0.data() + (*it) * max_v, max_v);
1071 auto end = std::find_if(_f.rbegin(), _f.rend(),
1072 [](auto a) { return a >= 0; });
1073 auto f = _f.first(std::distance(end, _f.rend()));
1074
1075 auto it1 = std::find_if_not(
1076 it, perm.end(),
1077 [f, max_v, it0 = facets0.begin()](auto p) -> bool
1078 {
1079 return std::equal(f.begin(), f.end(), std::next(it0, p * max_v));
1080 });
1081
1082 // If no repeated facet found, insert f vertices
1083 if (std::distance(it, it1) == 1)
1084 vertices.insert(vertices.end(), f.begin(), f.end());
1085 else if (std::distance(it, it1) > 2)
1086 throw std::runtime_error("More than two matching facets found.");
1087
1088 // Advance iterator
1089 it = it1;
1090 }
1091
1092 // Remove duplicate indices
1093 std::ranges::sort(vertices);
1094 auto [unique_end, range_end] = std::ranges::unique(vertices);
1095 vertices.erase(unique_end, range_end);
1096
1097 return vertices;
1098 }
1099 };
1100
1101 const std::vector<std::int64_t> boundary_v = boundary_v_fn(
1102 celltypes, doflayouts, ghost_owners, cells1, cells1_v, original_idx1);
1103
1104 spdlog::debug("Got {} boundary vertices", boundary_v.size());
1105
1106 // Create Topology
1107 std::vector<std::span<const std::int64_t>> cells1_v_span;
1108 std::ranges::transform(cells1_v, std::back_inserter(cells1_v_span),
1109 [](auto& c) { return std::span(c); });
1110 std::vector<std::span<const std::int64_t>> original_idx1_span;
1111 std::ranges::transform(original_idx1, std::back_inserter(original_idx1_span),
1112 [](auto& c) { return std::span(c); });
1113 std::vector<std::span<const int>> ghost_owners_span;
1114 std::ranges::transform(ghost_owners, std::back_inserter(ghost_owners_span),
1115 [](auto& c) { return std::span(c); });
1116 Topology topology
1117 = create_topology(comm, celltypes, cells1_v_span, original_idx1_span,
1118 ghost_owners_span, boundary_v);
1119
1120 // Create connectivities required higher-order geometries for creating
1121 // a Geometry object
1122 for (int i = 0; i < num_cell_types; ++i)
1123 {
1124 for (int e = 1; e < topology.dim(); ++e)
1125 {
1126 if (doflayouts[i].num_entity_dofs(e) > 0)
1127 topology.create_entities(e);
1128 }
1129
1130 if (elements[i].needs_dof_permutations())
1131 topology.create_entity_permutations();
1132 }
1133
1134 // Build list of unique (global) node indices from cells1 and
1135 // distribute coordinate data
1136 std::vector<std::int64_t> nodes1, nodes2;
1137 for (std::vector<std::int64_t>& c : cells1)
1138 nodes1.insert(nodes1.end(), c.begin(), c.end());
1139 for (std::vector<std::int64_t>& c : cells1)
1140 nodes2.insert(nodes2.end(), c.begin(), c.end());
1141
1142 dolfinx::radix_sort(nodes1);
1143 auto [unique_end, range_end] = std::ranges::unique(nodes1);
1144 nodes1.erase(unique_end, range_end);
1145
1146 std::vector coords
1147 = dolfinx::MPI::distribute_data(comm, nodes1, commg, x, xshape[1]);
1148
1149 // Create geometry object
1150 Geometry geometry
1151 = create_geometry(topology, elements, nodes1, nodes2, coords, xshape[1]);
1152
1153 return Mesh(comm, std::make_shared<Topology>(std::move(topology)),
1154 std::move(geometry));
1155}

1250template <std::floating_point T>
1251std::pair<Geometry<T>, std::vector<int32_t>>
1252create_subgeometry(const Mesh<T>& mesh, int dim,
1253 std::span<const std::int32_t> subentity_to_entity)
1254{
1255 const Geometry<T>& geometry = mesh.geometry();
1256
1257 // Get the geometry dofs in the sub-geometry based on the entities in
1258 // sub-geometry
1259 const fem::ElementDofLayout layout = geometry.cmap().create_dof_layout();
1260
1261 std::vector<std::int32_t> x_indices
1262 = entities_to_geometry(mesh, dim, subentity_to_entity, true);
1263
1264 std::vector<std::int32_t> sub_x_dofs = x_indices;
1265 std::ranges::sort(sub_x_dofs);
1266 auto [unique_end, range_end] = std::ranges::unique(sub_x_dofs);
1267 sub_x_dofs.erase(unique_end, range_end);
1268
1269 // Get the sub-geometry dofs owned by this process
1270 auto x_index_map = geometry.index_map();
1271 assert(x_index_map);
1272
1273 std::shared_ptr<common::IndexMap> sub_x_dof_index_map;
1274 std::vector<std::int32_t> subx_to_x_dofmap;
1275 {
1276 auto [map, new_to_old] = common::create_sub_index_map(
1277 *x_index_map, sub_x_dofs, common::IndexMapOrder::any, true);
1278 sub_x_dof_index_map = std::make_shared<common::IndexMap>(std::move(map));
1279 subx_to_x_dofmap = std::move(new_to_old);
1280 }
1281
1282 // Create sub-geometry coordinates
1283 std::span<const T> x = geometry.x();
1284 std::int32_t sub_num_x_dofs = subx_to_x_dofmap.size();
1285 std::vector<T> sub_x(3 * sub_num_x_dofs);
1286 for (std::int32_t i = 0; i < sub_num_x_dofs; ++i)
1287 {
1288 std::copy_n(std::next(x.begin(), 3 * subx_to_x_dofmap[i]), 3,
1289 std::next(sub_x.begin(), 3 * i));
1290 }
1291
1292 // Create geometry to sub-geometry map
1293 std::vector<std::int32_t> x_to_subx_dof_map(
1294 x_index_map->size_local() + x_index_map->num_ghosts(), -1);
1295 for (std::size_t i = 0; i < subx_to_x_dofmap.size(); ++i)
1296 x_to_subx_dof_map[subx_to_x_dofmap[i]] = i;
1297
1298 // Create sub-geometry dofmap
1299 std::vector<std::int32_t> sub_x_dofmap;
1300 sub_x_dofmap.reserve(x_indices.size());
1301 std::ranges::transform(x_indices, std::back_inserter(sub_x_dofmap),
1302 [&x_to_subx_dof_map](auto x_dof)
1303 {
1304 assert(x_to_subx_dof_map[x_dof] != -1);
1305 return x_to_subx_dof_map[x_dof];
1306 });
1307
1308 // Create sub-geometry coordinate element
1309 CellType sub_coord_cell
1310 = cell_entity_type(geometry.cmap().cell_shape(), dim, 0);
1311 // Special handling if point meshes, as they only support constant basis
1312 // functions
1313 int degree = geometry.cmap().degree();
1314 if (sub_coord_cell == CellType::point)
1315 degree = 0;
1316 fem::CoordinateElement<T> sub_cmap(sub_coord_cell, degree,
1317 geometry.cmap().variant());
1318
1319 // Sub-geometry input_global_indices
1320 const std::vector<std::int64_t>& igi = geometry.input_global_indices();
1321 std::vector<std::int64_t> sub_igi;
1322 sub_igi.reserve(subx_to_x_dofmap.size());
1323 std::ranges::transform(subx_to_x_dofmap, std::back_inserter(sub_igi),
1324 [&igi](auto sub_x_dof) { return igi[sub_x_dof]; });
1325
1326 // Create geometry
1327 return {Geometry(
1328 sub_x_dof_index_map,
1329 std::vector<std::vector<std::int32_t>>{std::move(sub_x_dofmap)},
1330 {sub_cmap}, std::move(sub_x), geometry.dim(), std::move(sub_igi)),
1331 std::move(subx_to_x_dofmap)};
1332}
