44template <typename T>
45void reorder_list(std::span<T> list, std::span<const std::int32_t> nodemap)
46{
47 if (nodemap.empty())
48 return;
49
50 assert(list.size() % nodemap.size() == 0);
51 int degree = list.size() / nodemap.size();
52 const std::vector<T> orig(list.begin(), list.end());
53 for (std::size_t n = 0; n < nodemap.size(); ++n)
54 {
55 auto links_old = std::span(orig.data() + n * degree, degree);
56 auto links_new = list.subspan(nodemap[n] * degree, degree);
57 std::copy(links_old.begin(), links_old.end(), links_new.begin());
58 }
59}

74template <std::floating_point T>
75std::tuple<std::vector<std::int32_t>, std::vector<T>, std::vector<std::int32_t>>
76compute_vertex_coords_boundary(const mesh::Mesh<T>& mesh, int dim,
77 std::span<const std::int32_t> facets)
78{
79 auto topology = mesh.topology();
80 assert(topology);
81 const int tdim = topology->dim();
82 if (dim == tdim)
83 {
84 throw std::runtime_error(
85 "Cannot use mesh::locate_entities_boundary (boundary) for cells.");
86 }
87
88 // Build set of vertices on boundary and set of boundary entities
89 mesh.topology_mutable()->create_connectivity(tdim - 1, 0);
90 mesh.topology_mutable()->create_connectivity(tdim - 1, dim);
91 std::vector<std::int32_t> vertices, entities;
92 {
93 auto f_to_v = topology->connectivity(tdim - 1, 0);
94 assert(f_to_v);
95 auto f_to_e = topology->connectivity(tdim - 1, dim);
96 assert(f_to_e);
97 for (auto f : facets)
98 {
99 auto v = f_to_v->links(f);
100 vertices.insert(vertices.end(), v.begin(), v.end());
101 auto e = f_to_e->links(f);
102 entities.insert(entities.end(), e.begin(), e.end());
103 }
104
105 // Build vector of boundary vertices
106 std::sort(vertices.begin(), vertices.end());
107 vertices.erase(std::unique(vertices.begin(), vertices.end()),
108 vertices.end());
109 std::sort(entities.begin(), entities.end());
110 entities.erase(std::unique(entities.begin(), entities.end()),
111 entities.end());
112 }
113
114 // Get geometry data
115 auto x_dofmap = mesh.geometry().dofmap();
116 std::span<const T> x_nodes = mesh.geometry().x();
117
118 // Get all vertex 'node' indices
119 mesh.topology_mutable()->create_connectivity(0, tdim);
120 mesh.topology_mutable()->create_connectivity(tdim, 0);
121 auto v_to_c = topology->connectivity(0, tdim);
122 assert(v_to_c);
123 auto c_to_v = topology->connectivity(tdim, 0);
124 assert(c_to_v);
125 std::vector<T> x_vertices(3 * vertices.size(), -1.0);
126 std::vector<std::int32_t> vertex_to_pos(v_to_c->num_nodes(), -1);
127 for (std::size_t i = 0; i < vertices.size(); ++i)
128 {
129 const std::int32_t v = vertices[i];
130
131 // Get first cell and find position
132 const int c = v_to_c->links(v).front();
133 auto cell_vertices = c_to_v->links(c);
134 auto it = std::find(cell_vertices.begin(), cell_vertices.end(), v);
135 assert(it != cell_vertices.end());
136 const int local_pos = std::distance(cell_vertices.begin(), it);
137
138 auto dofs = MDSPAN_IMPL_STANDARD_NAMESPACE::submdspan(
139 x_dofmap, c, MDSPAN_IMPL_STANDARD_NAMESPACE::full_extent);
140 for (std::size_t j = 0; j < 3; ++j)
141 x_vertices[j * vertices.size() + i] = x_nodes[3 * dofs[local_pos] + j];
142 vertex_to_pos[v] = i;
143 }
144
145 return {std::move(entities), std::move(x_vertices), std::move(vertex_to_pos)};
146}

204template <std::floating_point T>
205std::vector<T> h(const Mesh<T>& mesh, std::span<const std::int32_t> entities,
206 int dim)
207{
208 if (entities.empty())
209 return std::vector<T>();
210 if (dim == 0)
211 return std::vector<T>(entities.size(), 0);
212
213 // Get the geometry dofs for the vertices of each entity
214 const std::vector<std::int32_t> vertex_xdofs
215 = entities_to_geometry(mesh, dim, entities, false);
216 assert(!entities.empty());
217 const std::size_t num_vertices = vertex_xdofs.size() / entities.size();
218
219 // Get the geometry coordinate
220 std::span<const T> x = mesh.geometry().x();
221
222 // Function to compute the length of (p0 - p1)
223 auto delta_norm = [](auto&& p0, auto&& p1)
224 {
225 T norm = 0;
226 for (std::size_t i = 0; i < 3; ++i)
227 norm += (p0[i] - p1[i]) * (p0[i] - p1[i]);
228 return std::sqrt(norm);
229 };
230
231 // Compute greatest distance between any to vertices
232 assert(dim > 0);
233 std::vector<T> h(entities.size(), 0);
234 for (std::size_t e = 0; e < entities.size(); ++e)
235 {
236 // Get geometry 'dof' for each vertex of entity e
237 std::span<const std::int32_t> e_vertices(
238 vertex_xdofs.data() + e * num_vertices, num_vertices);
239
240 // Compute maximum distance between any two vertices
241 for (std::size_t i = 0; i < e_vertices.size(); ++i)
242 {
243 std::span<const T, 3> p0(x.data() + 3 * e_vertices[i], 3);
244 for (std::size_t j = i + 1; j < e_vertices.size(); ++j)
245 {
246 std::span<const T, 3> p1(x.data() + 3 * e_vertices[j], 3);
247 h[e] = std::max(h[e], delta_norm(p0, p1));
248 }
249 }
250 }
251
252 return h;
253}

258template <std::floating_point T>
259std::vector<T> cell_normals(const Mesh<T>& mesh, int dim,
260 std::span<const std::int32_t> entities)
261{
262 auto topology = mesh.topology();
263 assert(topology);
264
265 if (entities.empty())
266 return std::vector<T>();
267
268 if (topology->cell_type() == CellType::prism and dim == 2)
269 throw std::runtime_error("More work needed for prism cell");
270
271 const int gdim = mesh.geometry().dim();
272 const CellType type = cell_entity_type(topology->cell_type(), dim, 0);
273
274 // Find geometry nodes for topology entities
275 std::span<const T> x = mesh.geometry().x();
276 std::vector<std::int32_t> geometry_entities
277 = entities_to_geometry(mesh, dim, entities, false);
278
279 const std::size_t shape1 = geometry_entities.size() / entities.size();
280 std::vector<T> n(entities.size() * 3);
281 switch (type)
282 {
283 case CellType::interval:
284 {
285 if (gdim > 2)
286 throw std::invalid_argument("Interval cell normal undefined in 3D");
287 for (std::size_t i = 0; i < entities.size(); ++i)
288 {
289 // Get the two vertices as points
290 std::array vertices{geometry_entities[i * shape1],
291 geometry_entities[i * shape1 + 1]};
292 std::array p = {std::span<const T, 3>(x.data() + 3 * vertices[0], 3),
293 std::span<const T, 3>(x.data() + 3 * vertices[1], 3)};
294
295 // Define normal by rotating tangent counter-clockwise
296 std::array<T, 3> t;
297 std::transform(p[1].begin(), p[1].end(), p[0].begin(), t.begin(),
298 [](auto x, auto y) { return x - y; });
299
300 T norm = std::sqrt(t[0] * t[0] + t[1] * t[1]);
301 std::span<T, 3> ni(n.data() + 3 * i, 3);
302 ni[0] = -t[1] / norm;
303 ni[1] = t[0] / norm;
304 ni[2] = 0.0;
305 }
306 return n;
307 }
308 case CellType::triangle:
309 {
310 for (std::size_t i = 0; i < entities.size(); ++i)
311 {
312 // Get the three vertices as points
313 std::array vertices = {geometry_entities[i * shape1 + 0],
314 geometry_entities[i * shape1 + 1],
315 geometry_entities[i * shape1 + 2]};
316 std::array p = {std::span<const T, 3>(x.data() + 3 * vertices[0], 3),
317 std::span<const T, 3>(x.data() + 3 * vertices[1], 3),
318 std::span<const T, 3>(x.data() + 3 * vertices[2], 3)};
319
320 // Compute (p1 - p0) and (p2 - p0)
321 std::array<T, 3> dp1, dp2;
322 std::transform(p[1].begin(), p[1].end(), p[0].begin(), dp1.begin(),
323 [](auto x, auto y) { return x - y; });
324 std::transform(p[2].begin(), p[2].end(), p[0].begin(), dp2.begin(),
325 [](auto x, auto y) { return x - y; });
326
327 // Define cell normal via cross product of first two edges
328 std::array<T, 3> ni = math::cross(dp1, dp2);
329 T norm = std::sqrt(ni[0] * ni[0] + ni[1] * ni[1] + ni[2] * ni[2]);
330 std::transform(ni.begin(), ni.end(), std::next(n.begin(), 3 * i),
331 [norm](auto x) { return x / norm; });
332 }
333
334 return n;
335 }
336 case CellType::quadrilateral:
337 {
338 // TODO: check
339 for (std::size_t i = 0; i < entities.size(); ++i)
340 {
341 // Get the three vertices as points
342 std::array vertices = {geometry_entities[i * shape1 + 0],
343 geometry_entities[i * shape1 + 1],
344 geometry_entities[i * shape1 + 2]};
345 std::array p = {std::span<const T, 3>(x.data() + 3 * vertices[0], 3),
346 std::span<const T, 3>(x.data() + 3 * vertices[1], 3),
347 std::span<const T, 3>(x.data() + 3 * vertices[2], 3)};
348
349 // Compute (p1 - p0) and (p2 - p0)
350 std::array<T, 3> dp1, dp2;
351 std::transform(p[1].begin(), p[1].end(), p[0].begin(), dp1.begin(),
352 [](auto x, auto y) { return x - y; });
353 std::transform(p[2].begin(), p[2].end(), p[0].begin(), dp2.begin(),
354 [](auto x, auto y) { return x - y; });
355
356 // Define cell normal via cross product of first two edges
357 std::array<T, 3> ni = math::cross(dp1, dp2);
358 T norm = std::sqrt(ni[0] * ni[0] + ni[1] * ni[1] + ni[2] * ni[2]);
359 std::transform(ni.begin(), ni.end(), std::next(n.begin(), 3 * i),
360 [norm](auto x) { return x / norm; });
361 }
362
363 return n;
364 }
365 default:
366 throw std::invalid_argument(
367 "cell_normal not supported for this cell type.");
368 }

374template <std::floating_point T>
375std::vector<T> compute_midpoints(const Mesh<T>& mesh, int dim,
376 std::span<const std::int32_t> entities)
377{
378 if (entities.empty())
379 return std::vector<T>();
380
381 std::span<const T> x = mesh.geometry().x();
382
383 // Build map from entity -> geometry dof
384 // FIXME: This assumes a linear geometry.
385 const std::vector<std::int32_t> e_to_g
386 = entities_to_geometry(mesh, dim, entities, false);
387 std::size_t shape1 = e_to_g.size() / entities.size();
388
389 std::vector<T> x_mid(entities.size() * 3, 0);
390 for (std::size_t e = 0; e < entities.size(); ++e)
391 {
392 std::span<T, 3> p(x_mid.data() + 3 * e, 3);
393 std::span<const std::int32_t> rows(e_to_g.data() + e * shape1, shape1);
394 for (auto row : rows)
395 {
396 std::span<const T, 3> xg(x.data() + 3 * row, 3);
397 std::transform(p.begin(), p.end(), xg.begin(), p.begin(),
398 [size = rows.size()](auto x, auto y)
399 { return x + y / size; });
400 }
401 }
402
403 return x_mid;
404}

412template <std::floating_point T>
413std::pair<std::vector<T>, std::array<std::size_t, 2>>
414compute_vertex_coords(const mesh::Mesh<T>& mesh)
415{
416 auto topology = mesh.topology();
417 assert(topology);
418 const int tdim = topology->dim();
419
420 // Create entities and connectivities
421 mesh.topology_mutable()->create_connectivity(tdim, 0);
422
423 // Get all vertex 'node' indices
424 auto x_dofmap = mesh.geometry().dofmap();
425 const std::int32_t num_vertices = topology->index_map(0)->size_local()
426 + topology->index_map(0)->num_ghosts();
427 auto c_to_v = topology->connectivity(tdim, 0);
428 assert(c_to_v);
429 std::vector<std::int32_t> vertex_to_node(num_vertices);
430 for (int c = 0; c < c_to_v->num_nodes(); ++c)
431 {
432 auto x_dofs = MDSPAN_IMPL_STANDARD_NAMESPACE::submdspan(
433 x_dofmap, c, MDSPAN_IMPL_STANDARD_NAMESPACE::full_extent);
434 auto vertices = c_to_v->links(c);
435 for (std::size_t i = 0; i < vertices.size(); ++i)
436 vertex_to_node[vertices[i]] = x_dofs[i];
437 }
438
439 // Pack coordinates of vertices
440 std::span<const T> x_nodes = mesh.geometry().x();
441 std::vector<T> x_vertices(3 * vertex_to_node.size(), 0.0);
442 for (std::size_t i = 0; i < vertex_to_node.size(); ++i)
443 {
444 const int pos = 3 * vertex_to_node[i];
445 for (std::size_t j = 0; j < 3; ++j)
446 x_vertices[j * vertex_to_node.size() + i] = x_nodes[pos + j];
447 }
448
449 return {std::move(x_vertices), {3, vertex_to_node.size()}};
450}

476template <std::floating_point T, MarkerFn<T> U>
477std::vector<std::int32_t> locate_entities(const Mesh<T>& mesh, int dim,
478 U marker)
479{
480 using cmdspan3x_t = MDSPAN_IMPL_STANDARD_NAMESPACE::mdspan<
481 const T,
482 MDSPAN_IMPL_STANDARD_NAMESPACE::extents<
483 std::size_t, 3, MDSPAN_IMPL_STANDARD_NAMESPACE::dynamic_extent>>;
484
485 // Run marker function on vertex coordinates
486 const auto [xdata, xshape] = impl::compute_vertex_coords(mesh);
487 cmdspan3x_t x(xdata.data(), xshape);
488 const std::vector<std::int8_t> marked = marker(x);
489 if (marked.size() != x.extent(1))
490 throw std::runtime_error("Length of array of markers is wrong.");
491
492 auto topology = mesh.topology();
493 assert(topology);
494 const int tdim = topology->dim();
495
496 mesh.topology_mutable()->create_entities(dim);
497 mesh.topology_mutable()->create_connectivity(tdim, 0);
498 if (dim < tdim)
499 mesh.topology_mutable()->create_connectivity(dim, 0);
500
501 // Iterate over entities of dimension 'dim' to build vector of marked
502 // entities
503 auto e_to_v = topology->connectivity(dim, 0);
504 assert(e_to_v);
505 std::vector<std::int32_t> entities;
506 for (int e = 0; e < e_to_v->num_nodes(); ++e)
507 {
508 // Iterate over entity vertices
509 bool all_vertices_marked = true;
510 for (std::int32_t v : e_to_v->links(e))
511 {
512 if (!marked[v])
513 {
514 all_vertices_marked = false;
515 break;
516 }
517 }
518
519 if (all_vertices_marked)
520 entities.push_back(e);
521 }
522
523 return entities;
524}

549template <std::floating_point T, MarkerFn<T> U>
550std::vector<std::int32_t> locate_entities_boundary(const Mesh<T>& mesh, int dim,
551 U marker)
552{
553 auto topology = mesh.topology();
554 assert(topology);
555 const int tdim = topology->dim();
556 if (dim == tdim)
557 {
558 throw std::runtime_error(
559 "Cannot use mesh::locate_entities_boundary (boundary) for cells.");
560 }
561
562 // Compute list of boundary facets
563 mesh.topology_mutable()->create_entities(tdim - 1);
564 mesh.topology_mutable()->create_connectivity(tdim - 1, tdim);
565 const std::vector<std::int32_t> boundary_facets
566 = exterior_facet_indices(*topology);
567
568 using cmdspan3x_t = MDSPAN_IMPL_STANDARD_NAMESPACE::mdspan<
569 const T,
570 MDSPAN_IMPL_STANDARD_NAMESPACE::extents<
571 std::size_t, 3, MDSPAN_IMPL_STANDARD_NAMESPACE::dynamic_extent>>;
572
573 // Run marker function on the vertex coordinates
574 const auto [facet_entities, xdata, vertex_to_pos]
575 = impl::compute_vertex_coords_boundary(mesh, dim, boundary_facets);
576 cmdspan3x_t x(xdata.data(), 3, xdata.size() / 3);
577 const std::vector<std::int8_t> marked = marker(x);
578 if (marked.size() != x.extent(1))
579 throw std::runtime_error("Length of array of markers is wrong.");
580
581 // Loop over entities and check vertex markers
582 mesh.topology_mutable()->create_entities(dim);
583 auto e_to_v = topology->connectivity(dim, 0);
584 assert(e_to_v);
585 std::vector<std::int32_t> entities;
586 for (auto e : facet_entities)
587 {
588 // Iterate over entity vertices
589 bool all_vertices_marked = true;
590 for (auto v : e_to_v->links(e))
591 {
592 const std::int32_t pos = vertex_to_pos[v];
593 if (!marked[pos])
594 {
595 all_vertices_marked = false;
596 break;
597 }
598 }
599
600 // Mark facet with all vertices marked
601 if (all_vertices_marked)
602 entities.push_back(e);
603 }
604
605 return entities;
606}

626template <std::floating_point T>
627std::vector<std::int32_t>
628entities_to_geometry(const Mesh<T>& mesh, int dim,
629 std::span<const std::int32_t> entities,
630 bool permute = false)
631{
632 auto topology = mesh.topology();
633 assert(topology);
634 CellType cell_type = topology->cell_type();
635 if (cell_type == CellType::prism and dim == 2)
636 throw std::runtime_error("More work needed for prism cells");
637
638 const int tdim = topology->dim();
639 const Geometry<T>& geometry = mesh.geometry();
640 auto xdofs = geometry.dofmap();
641
642 // Get the DOF layout and the number of DOFs per entity
643 const fem::CoordinateElement<T>& coord_ele = geometry.cmap();
644 const fem::ElementDofLayout layout = coord_ele.create_dof_layout();
645 const std::size_t num_entity_dofs = layout.num_entity_closure_dofs(dim);
646 std::vector<std::int32_t> entity_xdofs;
647 entity_xdofs.reserve(entities.size() * num_entity_dofs);
648
649 // Get the element's closure DOFs
650 const std::vector<std::vector<std::vector<int>>>& closure_dofs_all
651 = layout.entity_closure_dofs_all();
652
653 // Special case when dim == tdim (cells)
654 if (dim == tdim)
655 {
656 for (std::size_t i = 0; i < entities.size(); ++i)
657 {
658 const std::int32_t c = entities[i];
659 // Extract degrees of freedom
660 auto x_c = MDSPAN_IMPL_STANDARD_NAMESPACE::submdspan(
661 xdofs, c, MDSPAN_IMPL_STANDARD_NAMESPACE::full_extent);
662 for (std::int32_t entity_dof : closure_dofs_all[tdim][0])
663 entity_xdofs.push_back(x_c[entity_dof]);
664 }
665 return entity_xdofs;
666 }
667
668 assert(dim != tdim);
669
670 auto e_to_c = topology->connectivity(dim, tdim);
671 if (!e_to_c)
672 {
673 throw std::runtime_error(
674 "Entity-to-cell connectivity has not been computed. Missing dims "
675 + std::to_string(dim) + "->" + std::to_string(tdim));
676 }
677
678 auto c_to_e = topology->connectivity(tdim, dim);
679 if (!c_to_e)
680 {
681 throw std::runtime_error(
682 "Cell-to-entity connectivity has not been computed. Missing dims "
683 + std::to_string(tdim) + "->" + std::to_string(dim));
684 }
685
686 // Get the cell info, which is needed to permute the closure dofs
687 std::span<const std::uint32_t> cell_info;
688 if (permute)
689 cell_info = std::span(mesh.topology()->get_cell_permutation_info());
690
691 for (std::size_t i = 0; i < entities.size(); ++i)
692 {
693 const std::int32_t e = entities[i];
694
695 // Get a cell connected to the entity
696 assert(!e_to_c->links(e).empty());
697 std::int32_t c = e_to_c->links(e).front();
698
699 // Get the local index of the entity
700 std::span<const std::int32_t> cell_entities = c_to_e->links(c);
701 auto it = std::find(cell_entities.begin(), cell_entities.end(), e);
702 assert(it != cell_entities.end());
703 std::size_t local_entity = std::distance(cell_entities.begin(), it);
704
705 std::vector<std::int32_t> closure_dofs(closure_dofs_all[dim][local_entity]);
706
707 // Cell sub-entities must be permuted so that their local orientation agrees
708 // with their global orientation
709 if (permute)
710 {
711 mesh::CellType entity_type
712 = mesh::cell_entity_type(cell_type, dim, local_entity);
713 coord_ele.permute_subentity_closure(closure_dofs, cell_info[c],
714 entity_type, local_entity);
715 }
716
717 // Extract degrees of freedom
718 auto x_c = MDSPAN_IMPL_STANDARD_NAMESPACE::submdspan(
719 xdofs, c, MDSPAN_IMPL_STANDARD_NAMESPACE::full_extent);
720 for (std::int32_t entity_dof : closure_dofs)
721 entity_xdofs.push_back(x_c[entity_dof]);
722 }
723
724 return entity_xdofs;
725}

777template <typename U>
778Mesh<typename std::remove_reference_t<typename U::value_type>> create_mesh(
779 MPI_Comm comm, MPI_Comm commt, std::span<const std::int64_t> cells,
780 const fem::CoordinateElement<
781 typename std::remove_reference_t<typename U::value_type>>& element,
782 MPI_Comm commg, const U& x, std::array<std::size_t, 2> xshape,
783 const CellPartitionFunction& partitioner)
784{
785 CellType celltype = element.cell_shape();
786 const fem::ElementDofLayout doflayout = element.create_dof_layout();
787
788 const int num_cell_vertices = mesh::num_cell_vertices(element.cell_shape());
789 std::size_t num_cell_nodes = doflayout.num_dofs();
790
791 // Note: `extract_topology` extracts topology data, i.e. just the
792 // vertices. For P1 geometry this should just be the identity
793 // operator. For other elements the filtered lists may have 'gaps',
794 // i.e. the indices might not be contiguous.
795 //
796 // `extract_topology` could be skipped for 'P1 geometry' elements
797
798 // -- Partition topology across ranks of comm
799 std::vector<std::int64_t> cells1;
800 std::vector<std::int64_t> original_idx1;
801 std::vector<int> ghost_owners;
802 if (partitioner)
803 {
804 spdlog::info("Using partitioner with {} cell data", cells.size());
805 graph::AdjacencyList<std::int32_t> dest(0);
806 if (commt != MPI_COMM_NULL)
807 {
808 int size = dolfinx::MPI::size(comm);
809 auto t = extract_topology(element.cell_shape(), doflayout, cells);
810 dest = partitioner(commt, size, {celltype}, {t});
811 }
812
813 // Distribute cells (topology, includes higher-order 'nodes') to
814 // destination rank
815 assert(cells.size() % num_cell_nodes == 0);
816 std::size_t num_cells = cells.size() / num_cell_nodes;
817 std::tie(cells1, original_idx1, ghost_owners) = graph::build::distribute(
818 comm, cells, {num_cells, num_cell_nodes}, dest);
819 spdlog::debug("Got {} cells from distribution", cells1.size());
820 }
821 else
822 {
823 cells1 = std::vector<std::int64_t>(cells.begin(), cells.end());
824 assert(cells1.size() % num_cell_nodes == 0);
825 std::int64_t offset = 0;
826 std::int64_t num_owned = cells1.size() / num_cell_nodes;
827 MPI_Exscan(&num_owned, &offset, 1, MPI_INT64_T, MPI_SUM, comm);
828 original_idx1.resize(num_owned);
829 std::iota(original_idx1.begin(), original_idx1.end(), offset);
830 }
831
832 // Extract cell 'topology', i.e. extract the vertices for each cell
833 // and discard any 'higher-order' nodes
834 std::vector<std::int64_t> cells1_v
835 = extract_topology(celltype, doflayout, cells1);
836 spdlog::info("Extract basic topology: {}->{}", cells1.size(),
837 cells1_v.size());
838
839 // Build local dual graph for owned cells to (i) get list of vertices
840 // on the process boundary and (ii) apply re-ordering to cells for
841 // locality
842 std::vector<std::int64_t> boundary_v;
843 {
844 std::int32_t num_owned_cells
845 = cells1_v.size() / num_cell_vertices - ghost_owners.size();
846 std::vector<std::int32_t> cell_offsets(num_owned_cells + 1, 0);
847 for (std::size_t i = 1; i < cell_offsets.size(); ++i)
848 cell_offsets[i] = cell_offsets[i - 1] + num_cell_vertices;
849 spdlog::info("Build local dual graph");
850 auto [graph, unmatched_facets, max_v, facet_attached_cells]
851 = build_local_dual_graph(
852 std::vector{celltype},
853 {std::span(cells1_v.data(), num_owned_cells * num_cell_vertices)});
854 const std::vector<int> remap = graph::reorder_gps(graph);
855
856 // Create re-ordered cell lists (leaves ghosts unchanged)
857 std::vector<std::int64_t> _original_idx(original_idx1.size());
858 for (std::size_t i = 0; i < remap.size(); ++i)
859 _original_idx[remap[i]] = original_idx1[i];
860 std::copy_n(std::next(original_idx1.cbegin(), num_owned_cells),
861 ghost_owners.size(),
862 std::next(_original_idx.begin(), num_owned_cells));
863 impl::reorder_list(
864 std::span(cells1_v.data(), remap.size() * num_cell_vertices), remap);
865 impl::reorder_list(std::span(cells1.data(), remap.size() * num_cell_nodes),
866 remap);
867 original_idx1 = _original_idx;
868
869 // Boundary vertices are marked as 'unknown'
870 boundary_v = unmatched_facets;
871 std::sort(boundary_v.begin(), boundary_v.end());
872 boundary_v.erase(std::unique(boundary_v.begin(), boundary_v.end()),
873 boundary_v.end());
874
875 // Remove -1 if it occurs in boundary vertices (may occur in mixed
876 // topology)
877 if (!boundary_v.empty() > 0 and boundary_v[0] == -1)
878 boundary_v.erase(boundary_v.begin());
879 }
880
881 // Create Topology
882 Topology topology = create_topology(comm, cells1_v, original_idx1,
883 ghost_owners, celltype, boundary_v);
884
885 // Create connectivities required higher-order geometries for creating
886 // a Geometry object
887 for (int e = 1; e < topology.dim(); ++e)
888 if (doflayout.num_entity_dofs(e) > 0)
889 topology.create_entities(e);
890 if (element.needs_dof_permutations())
891 topology.create_entity_permutations();
892
893 // Build list of unique (global) node indices from cells1 and
894 // distribute coordinate data
895 std::vector<std::int64_t> nodes1 = cells1;
896 dolfinx::radix_sort(std::span(nodes1));
897 nodes1.erase(std::unique(nodes1.begin(), nodes1.end()), nodes1.end());
898 std::vector coords
899 = dolfinx::MPI::distribute_data(comm, nodes1, commg, x, xshape[1]);
900
901 // Create geometry object
902 Geometry geometry
903 = create_geometry(topology, element, nodes1, cells1, coords, xshape[1]);
904
905 return Mesh(comm, std::make_shared<Topology>(std::move(topology)),
906 std::move(geometry));
907}

943template <typename U>
944Mesh<typename std::remove_reference_t<typename U::value_type>> create_mesh(
945 MPI_Comm comm, MPI_Comm commt,
946 const std::vector<std::span<const std::int64_t>>& cells,
947 const std::vector<fem::CoordinateElement<
948 typename std::remove_reference_t<typename U::value_type>>>& elements,
949 MPI_Comm commg, const U& x, std::array<std::size_t, 2> xshape,
950 const CellPartitionFunction& partitioner)
951{
952 assert(cells.size() == elements.size());
953 std::int32_t num_cell_types = cells.size();
954 std::vector<CellType> celltypes;
955 std::transform(elements.cbegin(), elements.cend(),
956 std::back_inserter(celltypes),
957 [](auto e) { return e.cell_shape(); });
958 std::vector<fem::ElementDofLayout> doflayouts;
959 std::transform(elements.cbegin(), elements.cend(),
960 std::back_inserter(doflayouts),
961 [](auto e) { return e.create_dof_layout(); });
962
963 // Note: `extract_topology` extracts topology data, i.e. just the
964 // vertices. For P1 geometry this should just be the identity
965 // operator. For other elements the filtered lists may have 'gaps',
966 // i.e. the indices might not be contiguous.
967 //
968 // `extract_topology` could be skipped for 'P1 geometry' elements
969
970 // -- Partition topology across ranks of comm
971 std::vector<std::vector<std::int64_t>> cells1(num_cell_types);
972 std::vector<std::vector<std::int64_t>> original_idx1(num_cell_types);
973 std::vector<std::vector<int>> ghost_owners(num_cell_types);
974 if (partitioner)
975 {
976 spdlog::info("Using partitioner with cell data ({} cell types)",
977 num_cell_types);
978 graph::AdjacencyList<std::int32_t> dest(0);
979 if (commt != MPI_COMM_NULL)
980 {
981 int size = dolfinx::MPI::size(comm);
982 std::vector<std::vector<std::int64_t>> t(num_cell_types);
983 std::vector<std::span<const std::int64_t>> tspan(num_cell_types);
984 for (std::int32_t i = 0; i < num_cell_types; ++i)
985 {
986 t[i] = extract_topology(celltypes[i], doflayouts[i], cells[i]);
987 tspan[i] = std::span(t[i]);
988 }
989 dest = partitioner(commt, size, celltypes, tspan);
990 }
991
992 std::int32_t cell_offset = 0;
993 for (std::int32_t i = 0; i < num_cell_types; ++i)
994 {
995 std::size_t num_cell_nodes = doflayouts[i].num_dofs();
996 assert(cells[i].size() % num_cell_nodes == 0);
997 std::size_t num_cells = cells[i].size() / num_cell_nodes;
998
999 // Extract destination AdjacencyList for this cell type
1000 std::vector<std::int32_t> offsets_i(
1001 std::next(dest.offsets().begin(), cell_offset),
1002 std::next(dest.offsets().begin(), cell_offset + num_cells + 1));
1003 std::vector<std::int32_t> data_i(
1004 std::next(dest.array().begin(), offsets_i.front()),
1005 std::next(dest.array().begin(), offsets_i.back()));
1006 std::int32_t offset_0 = offsets_i.front();
1007 std::for_each(offsets_i.begin(), offsets_i.end(),
1008 [&offset_0](std::int32_t& j) { j -= offset_0; });
1009 graph::AdjacencyList<std::int32_t> dest_i(data_i, offsets_i);
1010 cell_offset += num_cells;
1011
1012 // Distribute cells (topology, includes higher-order 'nodes') to
1013 // destination rank
1014 std::tie(cells1[i], original_idx1[i], ghost_owners[i])
1015 = graph::build::distribute(comm, cells[i],
1016 {num_cells, num_cell_nodes}, dest_i);
1017 spdlog::debug("Got {} cells from distribution", cells1[i].size());
1018 }
1019 }
1020 else
1021 {
1022 // No partitioning, construct a global index
1023 std::int64_t num_owned = 0;
1024 for (std::int32_t i = 0; i < num_cell_types; ++i)
1025 {
1026 cells1[i] = std::vector<std::int64_t>(cells[i].begin(), cells[i].end());
1027 std::int32_t num_cell_nodes = doflayouts[i].num_dofs();
1028 assert(cells1[i].size() % num_cell_nodes == 0);
1029 original_idx1[i].resize(cells1[i].size() / num_cell_nodes);
1030 num_owned += original_idx1[i].size();
1031 }
1032 // Add on global offset
1033 std::int64_t global_offset = 0;
1034 MPI_Exscan(&num_owned, &global_offset, 1, MPI_INT64_T, MPI_SUM, comm);
1035 for (std::int32_t i = 0; i < num_cell_types; ++i)
1036 {
1037 std::iota(original_idx1[i].begin(), original_idx1[i].end(),
1038 global_offset);
1039 global_offset += original_idx1[i].size();
1040 }
1041 }
1042
1043 // Extract cell 'topology', i.e. extract the vertices for each cell
1044 // and discard any 'higher-order' nodes
1045 std::vector<std::vector<std::int64_t>> cells1_v(num_cell_types);
1046 for (std::int32_t i = 0; i < num_cell_types; ++i)
1047 {
1048 cells1_v[i] = extract_topology(celltypes[i], doflayouts[i], cells1[i]);
1049 spdlog::info("Extract basic topology: {}->{}", cells1[i].size(),
1050 cells1_v[i].size());
1051 }
1052
1053 // Build local dual graph for owned cells to (i) get list of vertices
1054 // on the process boundary and (ii) apply re-ordering to cells for
1055 // locality
1056 std::vector<std::int64_t> boundary_v;
1057 {
1058 std::vector<std::span<const std::int64_t>> cells1_v_local_cells;
1059
1060 for (std::int32_t i = 0; i < num_cell_types; ++i)
1061 {
1062 std::int32_t num_cell_vertices = mesh::num_cell_vertices(celltypes[i]);
1063 std::int32_t num_owned_cells
1064 = cells1_v[i].size() / num_cell_vertices - ghost_owners[i].size();
1065 cells1_v_local_cells.push_back(
1066 std::span(cells1_v[i].data(), num_owned_cells * num_cell_vertices));
1067 }
1068 spdlog::info("Build local dual graph");
1069 auto [graph, unmatched_facets, max_v, facet_attached_cells]
1070 = build_local_dual_graph(celltypes, cells1_v_local_cells);
1071
1072 // TODO: in the original create_mesh(), cell reordering is done here.
1073 // This needs reworking for mixed cell topology
1074
1075 // Boundary vertices are marked as 'unknown'
1076 boundary_v = unmatched_facets;
1077 std::sort(boundary_v.begin(), boundary_v.end());
1078 boundary_v.erase(std::unique(boundary_v.begin(), boundary_v.end()),
1079 boundary_v.end());
1080
1081 // Remove -1 if it occurs in boundary vertices (may occur in mixed
1082 // topology)
1083 if (!boundary_v.empty() > 0 and boundary_v[0] == -1)
1084 boundary_v.erase(boundary_v.begin());
1085 }
1086
1087 spdlog::debug("Got {} boundary vertices", boundary_v.size());
1088
1089 // Create Topology
1090
1091 std::vector<std::span<const std::int64_t>> cells1_v_span;
1092 std::transform(cells1_v.cbegin(), cells1_v.cend(),
1093 std::back_inserter(cells1_v_span),
1094 [](auto& c) { return std::span(c); });
1095 std::vector<std::span<const std::int64_t>> original_idx1_span;
1096 std::transform(original_idx1.cbegin(), original_idx1.cend(),
1097 std::back_inserter(original_idx1_span),
1098 [](auto& c) { return std::span(c); });
1099 std::vector<std::span<const int>> ghost_owners_span;
1100 std::transform(ghost_owners.cbegin(), ghost_owners.cend(),
1101 std::back_inserter(ghost_owners_span),
1102 [](auto& c) { return std::span(c); });
1103
1104 Topology topology
1105 = create_topology(comm, celltypes, cells1_v_span, original_idx1_span,
1106 ghost_owners_span, boundary_v);
1107
1108 // Create connectivities required higher-order geometries for creating
1109 // a Geometry object
1110 for (int i = 0; i < num_cell_types; ++i)
1111 {
1112 for (int e = 1; e < topology.dim(); ++e)
1113 if (doflayouts[i].num_entity_dofs(e) > 0)
1114 topology.create_entities(e);
1115 if (elements[i].needs_dof_permutations())
1116 topology.create_entity_permutations();
1117 }
1118
1119 // Build list of unique (global) node indices from cells1 and
1120 // distribute coordinate data
1121 std::vector<std::int64_t> nodes1, nodes2;
1122 for (std::vector<std::int64_t>& c : cells1)
1123 nodes1.insert(nodes1.end(), c.begin(), c.end());
1124 for (std::vector<std::int64_t>& c : cells1)
1125 nodes2.insert(nodes2.end(), c.begin(), c.end());
1126
1127 dolfinx::radix_sort(std::span(nodes1));
1128 nodes1.erase(std::unique(nodes1.begin(), nodes1.end()), nodes1.end());
1129 std::vector coords
1130 = dolfinx::MPI::distribute_data(comm, nodes1, commg, x, xshape[1]);
1131
1132 // Create geometry object
1133 Geometry geometry
1134 = create_geometry(topology, elements, nodes1, nodes2, coords, xshape[1]);
1135
1136 return Mesh(comm, std::make_shared<Topology>(std::move(topology)),
1137 std::move(geometry));
1138}

1185template <std::floating_point T>
1186std::pair<Geometry<T>, std::vector<int32_t>>
1187create_subgeometry(const Mesh<T>& mesh, int dim,
1188 std::span<const std::int32_t> subentity_to_entity)
1189{
1190 const Geometry<T>& geometry = mesh.geometry();
1191
1192 // Get the geometry dofs in the sub-geometry based on the entities in
1193 // sub-geometry
1194 const fem::ElementDofLayout layout = geometry.cmap().create_dof_layout();
1195
1196 std::vector<std::int32_t> x_indices
1197 = entities_to_geometry(mesh, dim, subentity_to_entity, true);
1198
1199 std::vector<std::int32_t> sub_x_dofs = x_indices;
1200 std::sort(sub_x_dofs.begin(), sub_x_dofs.end());
1201 sub_x_dofs.erase(std::unique(sub_x_dofs.begin(), sub_x_dofs.end()),
1202 sub_x_dofs.end());
1203
1204 // Get the sub-geometry dofs owned by this process
1205 auto x_index_map = geometry.index_map();
1206 assert(x_index_map);
1207
1208 std::shared_ptr<common::IndexMap> sub_x_dof_index_map;
1209 std::vector<std::int32_t> subx_to_x_dofmap;
1210 {
1211 auto [map, new_to_old] = common::create_sub_index_map(
1212 *x_index_map, sub_x_dofs, common::IndexMapOrder::any, true);
1213 sub_x_dof_index_map = std::make_shared<common::IndexMap>(std::move(map));
1214 subx_to_x_dofmap = std::move(new_to_old);
1215 }
1216
1217 // Create sub-geometry coordinates
1218 std::span<const T> x = geometry.x();
1219 std::int32_t sub_num_x_dofs = subx_to_x_dofmap.size();
1220 std::vector<T> sub_x(3 * sub_num_x_dofs);
1221 for (int i = 0; i < sub_num_x_dofs; ++i)
1222 {
1223 std::copy_n(std::next(x.begin(), 3 * subx_to_x_dofmap[i]), 3,
1224 std::next(sub_x.begin(), 3 * i));
1225 }
1226
1227 // Create geometry to sub-geometry map
1228 std::vector<std::int32_t> x_to_subx_dof_map(
1229 x_index_map->size_local() + x_index_map->num_ghosts(), -1);
1230 for (std::size_t i = 0; i < subx_to_x_dofmap.size(); ++i)
1231 x_to_subx_dof_map[subx_to_x_dofmap[i]] = i;
1232
1233 // Create sub-geometry dofmap
1234 std::vector<std::int32_t> sub_x_dofmap;
1235 sub_x_dofmap.reserve(x_indices.size());
1236 std::transform(x_indices.cbegin(), x_indices.cend(),
1237 std::back_inserter(sub_x_dofmap),
1238 [&x_to_subx_dof_map](auto x_dof)
1239 {
1240 assert(x_to_subx_dof_map[x_dof] != -1);
1241 return x_to_subx_dof_map[x_dof];
1242 });
1243
1244 // Create sub-geometry coordinate element
1245 CellType sub_coord_cell
1246 = cell_entity_type(geometry.cmap().cell_shape(), dim, 0);
1247 fem::CoordinateElement<T> sub_cmap(sub_coord_cell, geometry.cmap().degree(),
1248 geometry.cmap().variant());
1249
1250 // Sub-geometry input_global_indices
1251 const std::vector<std::int64_t>& igi = geometry.input_global_indices();
1252 std::vector<std::int64_t> sub_igi;
1253 sub_igi.reserve(subx_to_x_dofmap.size());
1254 std::transform(subx_to_x_dofmap.begin(), subx_to_x_dofmap.end(),
1255 std::back_inserter(sub_igi),
1256 [&igi](std::int32_t sub_x_dof) { return igi[sub_x_dof]; });
1257
1258 // Create geometry
1259 return {Geometry(sub_x_dof_index_map, std::move(sub_x_dofmap), {sub_cmap},
1260 std::move(sub_x), geometry.dim(), std::move(sub_igi)),
1261 std::move(subx_to_x_dofmap)};
1262}
