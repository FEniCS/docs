50template <typename T>
51void reorder_list(std::span<T> list, std::span<const std::int32_t> nodemap)
52{
53 if (nodemap.empty())
54 return;
55
56 assert(list.size() % nodemap.size() == 0);
57 std::size_t degree = list.size() / nodemap.size();
58 const std::vector<T> orig(list.begin(), list.end());
59 for (std::size_t n = 0; n < nodemap.size(); ++n)
60 {
61 std::span links_old(orig.data() + n * degree, degree);
62 auto links_new = list.subspan(nodemap[n] * degree, degree);
63 std::ranges::copy(links_old, links_new.begin());
64 }
65}

80template <std::floating_point T>
81std::tuple<std::vector<std::int32_t>, std::vector<T>, std::vector<std::int32_t>>
82compute_vertex_coords_boundary(const mesh::Mesh<T>& mesh, int dim,
83 std::span<const std::int32_t> facets)
84{
85 auto topology = mesh.topology();
86 assert(topology);
87 const int tdim = topology->dim();
88 if (dim == tdim)
89 {
90 throw std::runtime_error(
91 "Cannot use mesh::locate_entities_boundary (boundary) for cells.");
92 }
93
94 // Build set of vertices on boundary and set of boundary entities
95 mesh.topology_mutable()->create_connectivity(tdim - 1, 0);
96 mesh.topology_mutable()->create_connectivity(tdim - 1, dim);
97 std::vector<std::int32_t> vertices, entities;
98 {
99 auto f_to_v = topology->connectivity(tdim - 1, 0);
100 assert(f_to_v);
101 auto f_to_e = topology->connectivity(tdim - 1, dim);
102 assert(f_to_e);
103 for (auto f : facets)
104 {
105 auto v = f_to_v->links(f);
106 vertices.insert(vertices.end(), v.begin(), v.end());
107 auto e = f_to_e->links(f);
108 entities.insert(entities.end(), e.begin(), e.end());
109 }
110
111 // Build vector of boundary vertices
112 {
113 std::ranges::sort(vertices);
114 auto [unique_end, range_end] = std::ranges::unique(vertices);
115 vertices.erase(unique_end, range_end);
116 }
117
118 {
119 std::ranges::sort(entities);
120 auto [unique_end, range_end] = std::ranges::unique(entities);
121 entities.erase(unique_end, range_end);
122 }
123 }
124
125 // Get geometry data
126 auto x_dofmap = mesh.geometry().dofmap();
127 std::span<const T> x_nodes = mesh.geometry().x();
128
129 // Get all vertex 'node' indices
130 mesh.topology_mutable()->create_connectivity(0, tdim);
131 mesh.topology_mutable()->create_connectivity(tdim, 0);
132 auto v_to_c = topology->connectivity(0, tdim);
133 assert(v_to_c);
134 auto c_to_v = topology->connectivity(tdim, 0);
135 assert(c_to_v);
136 std::vector<T> x_vertices(3 * vertices.size(), -1.0);
137 std::vector<std::int32_t> vertex_to_pos(v_to_c->num_nodes(), -1);
138 for (std::size_t i = 0; i < vertices.size(); ++i)
139 {
140 const std::int32_t v = vertices[i];
141
142 // Get first cell and find position
143 const std::int32_t c = v_to_c->links(v).front();
144 auto cell_vertices = c_to_v->links(c);
145 auto it = std::find(cell_vertices.begin(), cell_vertices.end(), v);
146 assert(it != cell_vertices.end());
147 const std::size_t local_pos = std::distance(cell_vertices.begin(), it);
148
149 auto dofs = md::submdspan(x_dofmap, c, md::full_extent);
150 for (std::size_t j = 0; j < 3; ++j)
151 x_vertices[j * vertices.size() + i] = x_nodes[3 * dofs[local_pos] + j];
152 vertex_to_pos[v] = i;
153 }
154
155 return {std::move(entities), std::move(x_vertices), std::move(vertex_to_pos)};
156}

223inline auto create_boundary_vertices_fn(const CellReorderFunction& reorder_fn)
224{
236 return [&](const std::vector<CellType>& celltypes,
237 const std::vector<fem::ElementDofLayout>& doflayouts,
238 const std::vector<std::vector<int>>& ghost_owners,
239 std::vector<std::vector<std::int64_t>>& cells,
240 std::vector<std::vector<std::int64_t>>& cells_v,
241 std::vector<std::vector<std::int64_t>>& original_idx)
242 -> std::vector<std::int64_t>
243 {
244 // Build local dual graph for owned cells to (i) get list of vertices
245 // on the process boundary and (ii) apply re-ordering to cells for
246 // locality
247
248 spdlog::info("Build local dual graphs, re-order cells, and compute process "
249 "boundary vertices.");
250
251 std::vector<std::pair<std::vector<std::int64_t>, int>> facets;
252
253 // Build lists of cells (by cell type) that excludes ghosts
254 std::vector<std::span<const std::int64_t>> cells1_v_local;
255 for (std::size_t i = 0; i < celltypes.size(); ++i)
256 {
257 int num_cell_vertices = mesh::num_cell_vertices(celltypes[i]);
258 std::size_t num_owned_cells
259 = cells_v[i].size() / num_cell_vertices - ghost_owners[i].size();
260 cells1_v_local.emplace_back(cells_v[i].data(),
261 num_owned_cells * num_cell_vertices);
262
263 // Build local dual graph for cell type
264 auto [graph, unmatched_facets, max_v, _facet_attached_cells]
265 = build_local_dual_graph(std::vector{celltypes[i]},
266 std::vector{cells1_v_local.back()});
267
268 // Store unmatched_facets for current cell type
269 facets.emplace_back(std::move(unmatched_facets), max_v);
270
271 // Compute re-ordering of graph
272 const std::vector<std::int32_t> remap = reorder_fn(graph);
273
274 // Update 'original' indices
275 const std::vector<std::int64_t>& orig_idx = original_idx[i];
276 std::vector<std::int64_t> _original_idx(orig_idx.size());
277 std::copy_n(orig_idx.rbegin(), ghost_owners[i].size(),
278 _original_idx.rbegin());
279 {
280 for (std::size_t j = 0; j < remap.size(); ++j)
281 _original_idx[remap[j]] = orig_idx[j];
282 }
283 original_idx[i] = _original_idx;
284
285 // Reorder cells
286 impl::reorder_list(
287 std::span(cells_v[i].data(), remap.size() * num_cell_vertices),
288 remap);
289 impl::reorder_list(
290 std::span(cells[i].data(), remap.size() * doflayouts[i].num_dofs()),
291 remap);
292 }
293
294 if (facets.size() == 1) // Optimisation for single cell type
295 {
296 std::vector<std::int64_t>& vertices = facets.front().first;
297
298 // Remove duplicated vertex indices
299 std::ranges::sort(vertices);
300 auto [unique_end, range_end] = std::ranges::unique(vertices);
301 vertices.erase(unique_end, range_end);
302
303 // Remove -1 if it appears as first entity. This can happen in
304 // mixed topology meshes where '-1' is used to pad facet data when
305 // cells facets have differing numbers of vertices.
306 if (!vertices.empty() and vertices.front() == -1)
307 vertices.erase(vertices.begin());
308
309 return vertices;
310 }
311 else
312 {
313 // Pack 'unmatched' facets for all cell types into single array
314 // (facets0)
315 std::vector<std::int64_t> facets0;
316 facets0.reserve(std::accumulate(facets.begin(), facets.end(),
317 std::size_t(0), [](std::size_t x, auto& y)
318 { return x + y.first.size(); }));
319 int max_v = std::ranges::max_element(facets, [](auto& a, auto& b)
320 { return a.second < b.second; })
321 ->second;
322 for (const auto& [v_data, num_v] : facets)
323 {
324 for (auto it = v_data.begin(); it != v_data.end(); it += num_v)
325 {
326 facets0.insert(facets0.end(), it, std::next(it, num_v));
327 facets0.insert(facets0.end(), max_v - num_v, -1);
328 }
329 }
330
331 // Compute row permutation
332 const std::vector<std::int32_t> perm = dolfinx::sort_by_perm(
333 std::span<const std::int64_t>(facets0), max_v);
334
335 // For facets in facets0 that appear only once, store the facet
336 // vertices
337 std::vector<std::int64_t> vertices;
338 // TODO: allocate memory for vertices
339 auto it = perm.begin();
340 while (it != perm.end())
341 {
342 // Find iterator to next facet different from f and trim any -1
343 // padding
344 std::span _f(facets0.data() + (*it) * max_v, max_v);
345 auto end = std::find_if(_f.rbegin(), _f.rend(),
346 [](auto a) { return a >= 0; });
347 auto f = _f.first(std::distance(end, _f.rend()));
348
349 auto it1 = std::find_if_not(
350 it, perm.end(),
351 [f, max_v, it0 = facets0.begin()](auto p) -> bool
352 {
353 return std::equal(f.begin(), f.end(), std::next(it0, p * max_v));
354 });
355
356 // If no repeated facet found, insert f vertices
357 if (std::distance(it, it1) == 1)
358 vertices.insert(vertices.end(), f.begin(), f.end());
359 else if (std::distance(it, it1) > 2)
360 throw std::runtime_error("More than two matching facets found.");
361
362 // Advance iterator
363 it = it1;
364 }
365
366 // Remove duplicate indices
367 std::ranges::sort(vertices);
368 auto [unique_end, range_end] = std::ranges::unique(vertices);
369 vertices.erase(unique_end, range_end);
370
371 return vertices;
372 }
373 };

397template <std::floating_point T>
398std::vector<T> h(const Mesh<T>& mesh, std::span<const std::int32_t> entities,
399 int dim)
400{
401 if (entities.empty())
402 return std::vector<T>();
403 if (dim == 0)
404 return std::vector<T>(entities.size(), 0);
405
406 // Get the geometry dofs for the vertices of each entity
407 const auto [vertex_xdofs, xdof_shape]
408 = entities_to_geometry(mesh, dim, entities, false);
409
410 // Get the geometry coordinate
411 std::span<const T> x = mesh.geometry().x();
412
413 // Function to compute the length of (p0 - p1)
414 auto delta_norm = [](auto&& p0, auto&& p1)
415 {
416 T norm = 0;
417 for (std::size_t i = 0; i < 3; ++i)
418 norm += (p0[i] - p1[i]) * (p0[i] - p1[i]);
419 return std::sqrt(norm);
420 };
421
422 // Compute greatest distance between any to vertices
423 assert(dim > 0);
424 std::vector<T> h(entities.size(), 0);
425 for (std::size_t e = 0; e < entities.size(); ++e)
426 {
427 // Get geometry 'dof' for each vertex of entity e
428 std::span<const std::int32_t> e_vertices(
429 vertex_xdofs.data() + e * xdof_shape[1], xdof_shape[1]);
430
431 // Compute maximum distance between any two vertices
432 for (std::size_t i = 0; i < e_vertices.size(); ++i)
433 {
434 std::span<const T, 3> p0(x.data() + 3 * e_vertices[i], 3);
435 for (std::size_t j = i + 1; j < e_vertices.size(); ++j)
436 {
437 std::span<const T, 3> p1(x.data() + 3 * e_vertices[j], 3);
438 h[e] = std::max(h[e], delta_norm(p0, p1));
439 }
440 }
441 }
442
443 return h;
444}

449template <std::floating_point T>
450std::vector<T> cell_normals(const Mesh<T>& mesh, int dim,
451 std::span<const std::int32_t> entities)
452{
453 if (entities.empty())
454 return std::vector<T>();
455
456 auto topology = mesh.topology();
457 assert(topology);
458 if (topology->cell_type() == CellType::prism and dim == 2)
459 {
460 throw std::runtime_error(
461 "Cell normal computation for prism cells not yet supported.");
462 }
463
464 const int gdim = mesh.geometry().dim();
465 const CellType type = cell_entity_type(topology->cell_type(), dim, 0);
466
467 // Find geometry nodes for topology entities
468 std::span<const T> x = mesh.geometry().x();
469 const auto [geometry_entities, eshape]
470 = entities_to_geometry(mesh, dim, entities, false);
471
472 std::vector<T> n(entities.size() * 3);
473 switch (type)
474 {
475 case CellType::interval:
476 {
477 if (gdim > 2)
478 throw std::invalid_argument("Interval cell normal undefined in 3D.");
479 for (std::size_t i = 0; i < entities.size(); ++i)
480 {
481 // Get the two vertices as points
482 std::array vertices{geometry_entities[i * eshape[1]],
483 geometry_entities[i * eshape[1] + 1]};
484 std::array p = {std::span<const T, 3>(x.data() + 3 * vertices[0], 3),
485 std::span<const T, 3>(x.data() + 3 * vertices[1], 3)};
486
487 // Define normal by rotating tangent counter-clockwise
488 std::array<T, 3> t;
489 std::ranges::transform(p[1], p[0], t.begin(),
490 [](auto x, auto y) { return x - y; });
491
492 T norm = std::sqrt(t[0] * t[0] + t[1] * t[1]);
493 std::span<T, 3> ni(n.data() + 3 * i, 3);
494 ni[0] = -t[1] / norm;
495 ni[1] = t[0] / norm;
496 ni[2] = 0.0;
497 }
498 return n;
499 }
500 case CellType::triangle:
501 {
502 for (std::size_t i = 0; i < entities.size(); ++i)
503 {
504 // Get the three vertices as points
505 std::array vertices = {geometry_entities[i * eshape[1] + 0],
506 geometry_entities[i * eshape[1] + 1],
507 geometry_entities[i * eshape[1] + 2]};
508 std::array p = {std::span<const T, 3>(x.data() + 3 * vertices[0], 3),
509 std::span<const T, 3>(x.data() + 3 * vertices[1], 3),
510 std::span<const T, 3>(x.data() + 3 * vertices[2], 3)};
511
512 // Compute (p1 - p0) and (p2 - p0)
513 std::array<T, 3> dp1, dp2;
514 std::ranges::transform(p[1], p[0], dp1.begin(),
515 [](auto x, auto y) { return x - y; });
516 std::ranges::transform(p[2], p[0], dp2.begin(),
517 [](auto x, auto y) { return x - y; });
518
519 // Define cell normal via cross product of first two edges
520 std::array<T, 3> ni = math::cross(dp1, dp2);
521 T norm = std::sqrt(ni[0] * ni[0] + ni[1] * ni[1] + ni[2] * ni[2]);
522 std::ranges::transform(ni, std::next(n.begin(), 3 * i),
523 [norm](auto x) { return x / norm; });
524 }
525
526 return n;
527 }
528 case CellType::quadrilateral:
529 {
530 // TODO: check
531 for (std::size_t i = 0; i < entities.size(); ++i)
532 {
533 // Get the three vertices as points
534 std::array vertices = {geometry_entities[i * eshape[1] + 0],
535 geometry_entities[i * eshape[1] + 1],
536 geometry_entities[i * eshape[1] + 2]};
537 std::array p = {std::span<const T, 3>(x.data() + 3 * vertices[0], 3),
538 std::span<const T, 3>(x.data() + 3 * vertices[1], 3),
539 std::span<const T, 3>(x.data() + 3 * vertices[2], 3)};
540
541 // Compute (p1 - p0) and (p2 - p0)
542 std::array<T, 3> dp1, dp2;
543 std::ranges::transform(p[1], p[0], dp1.begin(),
544 [](auto x, auto y) { return x - y; });
545 std::ranges::transform(p[2], p[0], dp2.begin(),
546 [](auto x, auto y) { return x - y; });
547
548 // Define cell normal via cross product of first two edges
549 std::array<T, 3> ni = math::cross(dp1, dp2);
550 T norm = std::sqrt(ni[0] * ni[0] + ni[1] * ni[1] + ni[2] * ni[2]);
551 std::ranges::transform(ni, std::next(n.begin(), 3 * i),
552 [norm](auto x) { return x / norm; });
553 }
554
555 return n;
556 }
557 default:
558 throw std::invalid_argument(
559 "cell_normal not supported for this cell type.");
560 }

566template <std::floating_point T>
567std::vector<T> compute_midpoints(const Mesh<T>& mesh, int dim,
568 std::span<const std::int32_t> entities)
569{
570 if (entities.empty())
571 return std::vector<T>();
572
573 std::span<const T> x = mesh.geometry().x();
574
575 // Build map from entity -> geometry dof
576 const auto [e_to_g, eshape]
577 = entities_to_geometry(mesh, dim, entities, false);
578
579 std::vector<T> x_mid(entities.size() * 3, 0);
580 for (std::size_t e = 0; e < entities.size(); ++e)
581 {
582 std::span<T, 3> p(x_mid.data() + 3 * e, 3);
583 std::span<const std::int32_t> rows(e_to_g.data() + e * eshape[1],
584 eshape[1]);
585 for (auto row : rows)
586 {
587 std::span<const T, 3> xg(x.data() + 3 * row, 3);
588 std::ranges::transform(p, xg, p.begin(),
589 [size = rows.size()](auto x, auto y)
590 { return x + y / size; });
591 }
592 }
593
594 return x_mid;
595}

603template <std::floating_point T>
604std::pair<std::vector<T>, std::array<std::size_t, 2>>
605compute_vertex_coords(const mesh::Mesh<T>& mesh)
606{
607 auto topology = mesh.topology();
608 assert(topology);
609 const int tdim = topology->dim();
610
611 // Create entities and connectivities
612
613 // Get all vertex 'node' indices
614 const std::int32_t num_vertices = topology->index_map(0)->size_local()
615 + topology->index_map(0)->num_ghosts();
616
617 std::vector<std::int32_t> vertex_to_node(num_vertices);
618 for (int cell_type_idx = 0,
619 num_cell_types = topology->entity_types(tdim).size();
620 cell_type_idx < num_cell_types; ++cell_type_idx)
621 {
622 auto x_dofmap = mesh.geometry().dofmap(cell_type_idx);
623 auto c_to_v = topology->connectivity({tdim, cell_type_idx}, {0, 0});
624 assert(c_to_v);
625 for (int c = 0; c < c_to_v->num_nodes(); ++c)
626 {
627 auto x_dofs = md::submdspan(x_dofmap, c, md::full_extent);
628 auto vertices = c_to_v->links(c);
629 for (std::size_t i = 0; i < vertices.size(); ++i)
630 vertex_to_node[vertices[i]] = x_dofs[i];
631 }
632 }
633
634 // Pack coordinates of vertices
635 std::span<const T> x_nodes = mesh.geometry().x();
636 std::vector<T> x_vertices(3 * vertex_to_node.size(), 0.0);
637 for (std::size_t i = 0; i < vertex_to_node.size(); ++i)
638 {
639 std::int32_t pos = 3 * vertex_to_node[i];
640 for (std::size_t j = 0; j < 3; ++j)
641 x_vertices[j * vertex_to_node.size() + i] = x_nodes[pos + j];
642 }
643
644 return {std::move(x_vertices), {3, vertex_to_node.size()}};
645}

650template <typename Fn, typename T>
651concept MarkerFn = std::is_invocable_r<
652 std::vector<std::int8_t>, Fn,
653 md::mdspan<const T,
654 md::extents<std::size_t, 3, md::dynamic_extent>>>::value;

671template <std::floating_point T, MarkerFn<T> U>
672std::vector<std::int32_t> locate_entities(const Mesh<T>& mesh, int dim,
673 U marker, int entity_type_idx)
674{
675
676 using cmdspan3x_t
677 = md::mdspan<const T, md::extents<std::size_t, 3, md::dynamic_extent>>;
678
679 // Run marker function on vertex coordinates
680 const auto [xdata, xshape] = impl::compute_vertex_coords(mesh);
681
682 cmdspan3x_t x(xdata.data(), xshape);
683 const std::vector<std::int8_t> marked = marker(x);
684 if (marked.size() != x.extent(1))
685 throw std::runtime_error("Length of array of markers is wrong.");
686
687 auto topology = mesh.topology();
688 assert(topology);
689 const int tdim = topology->dim();
690
691 mesh.topology_mutable()->create_entities(dim);
692 if (dim < tdim)
693 mesh.topology_mutable()->create_connectivity(dim, 0);
694
695 // Iterate over entities of dimension 'dim' to build vector of marked
696 // entities
697 auto e_to_v = topology->connectivity({dim, entity_type_idx}, {0, 0});
698 assert(e_to_v);
699 std::vector<std::int32_t> entities;
700 for (int e = 0; e < e_to_v->num_nodes(); ++e)
701 {
702 // Iterate over entity vertices
703 bool all_vertices_marked = true;
704 for (std::int32_t v : e_to_v->links(e))
705 {
706 if (!marked[v])
707 {
708 all_vertices_marked = false;
709 break;
710 }
711 }
712
713 if (all_vertices_marked)
714 entities.push_back(e);
715 }
716
717 return entities;
718}

769template <std::floating_point T, MarkerFn<T> U>
770std::vector<std::int32_t> locate_entities_boundary(const Mesh<T>& mesh, int dim,
771 U marker)
772{
773 // TODO Rewrite this function, it should be possible to simplify considerably
774 auto topology = mesh.topology();
775 assert(topology);
776 int tdim = topology->dim();
777 if (dim == tdim)
778 {
779 throw std::runtime_error(
780 "Cannot use mesh::locate_entities_boundary (boundary) for cells.");
781 }
782
783 // Compute list of boundary facets
784 mesh.topology_mutable()->create_entities(tdim - 1);
785 mesh.topology_mutable()->create_connectivity(tdim - 1, tdim);
786 std::vector<std::int32_t> boundary_facets = exterior_facet_indices(*topology);
787
788 using cmdspan3x_t
789 = md::mdspan<const T, md::extents<std::size_t, 3, md::dynamic_extent>>;
790
791 // Run marker function on the vertex coordinates
792 auto [facet_entities, xdata, vertex_to_pos]
793 = impl::compute_vertex_coords_boundary(mesh, dim, boundary_facets);
794 cmdspan3x_t x(xdata.data(), 3, xdata.size() / 3);
795 std::vector<std::int8_t> marked = marker(x);
796 if (marked.size() != x.extent(1))
797 throw std::runtime_error("Length of array of markers is wrong.");
798
799 // Loop over entities and check vertex markers
800 mesh.topology_mutable()->create_entities(dim);
801 auto e_to_v = topology->connectivity(dim, 0);
802 assert(e_to_v);
803 std::vector<std::int32_t> entities;
804 for (auto e : facet_entities)
805 {
806 // Iterate over entity vertices
807 bool all_vertices_marked = true;
808 for (auto v : e_to_v->links(e))
809 {
810 const std::int32_t pos = vertex_to_pos[v];
811 if (!marked[pos])
812 {
813 all_vertices_marked = false;
814 break;
815 }
816 }
817
818 // Mark facet with all vertices marked
819 if (all_vertices_marked)
820 entities.push_back(e);
821 }
822
823 return entities;
824}

844template <std::floating_point T>
845std::pair<std::vector<std::int32_t>, std::array<std::size_t, 2>>
846entities_to_geometry(const Mesh<T>& mesh, int dim,
847 std::span<const std::int32_t> entities,
848 bool permute = false)
849{
850 auto topology = mesh.topology();
851 assert(topology);
852 CellType cell_type = topology->cell_type();
853 if (cell_type == CellType::prism and dim == 2)
854 {
855 throw std::runtime_error(
856 "mesh::entities_to_geometry for prism cells not yet supported.");
857 }
858
859 const int tdim = topology->dim();
860 const Geometry<T>& geometry = mesh.geometry();
861 auto xdofs = geometry.dofmap();
862
863 // Get the DOF layout and the number of DOFs per entity
864 const fem::CoordinateElement<T>& coord_ele = geometry.cmap();
865 const fem::ElementDofLayout layout = coord_ele.create_dof_layout();
866 const std::size_t num_entity_dofs = layout.num_entity_closure_dofs(dim);
867 std::vector<std::int32_t> entity_xdofs;
868 entity_xdofs.reserve(entities.size() * num_entity_dofs);
869 std::array<std::size_t, 2> eshape{entities.size(), num_entity_dofs};
870
871 // Get the element's closure DOFs
872 const std::vector<std::vector<std::vector<int>>>& closure_dofs_all
873 = layout.entity_closure_dofs_all();
874
875 // Special case when dim == tdim (cells)
876 if (dim == tdim)
877 {
878 for (std::int32_t c : entities)
879 {
880 // Extract degrees of freedom
881 auto x_c = md::submdspan(xdofs, c, md::full_extent);
882 for (std::int32_t entity_dof : closure_dofs_all[tdim][0])
883 entity_xdofs.push_back(x_c[entity_dof]);
884 }
885
886 return {std::move(entity_xdofs), eshape};
887 }
888
889 assert(dim != tdim);
890
891 auto e_to_c = topology->connectivity(dim, tdim);
892 if (!e_to_c)
893 {
894 throw std::runtime_error(
895 "Entity-to-cell connectivity has not been computed. Missing dims "
896 + std::to_string(dim) + "->" + std::to_string(tdim));
897 }
898
899 auto c_to_e = topology->connectivity(tdim, dim);
900 if (!c_to_e)
901 {
902 throw std::runtime_error(
903 "Cell-to-entity connectivity has not been computed. Missing dims "
904 + std::to_string(tdim) + "->" + std::to_string(dim));
905 }
906
907 // Get the cell info, which is needed to permute the closure dofs
908 std::span<const std::uint32_t> cell_info;
909 if (permute)
910 cell_info = std::span(mesh.topology()->get_cell_permutation_info());
911
912 for (std::int32_t e : entities)
913 {
914 // Get a cell connected to the entity
915 assert(!e_to_c->links(e).empty());
916 std::int32_t c = e_to_c->links(e).front();
917
918 // Get the local index of the entity
919 std::span<const std::int32_t> cell_entities = c_to_e->links(c);
920 auto it = std::find(cell_entities.begin(), cell_entities.end(), e);
921 assert(it != cell_entities.end());
922 std::size_t local_entity = std::distance(cell_entities.begin(), it);
923
924 // Cell sub-entities must be permuted so that their local
925 // orientation agrees with their global orientation
926 std::vector<std::int32_t> closure_dofs(closure_dofs_all[dim][local_entity]);
927 if (permute)
928 {
929 mesh::CellType entity_type
930 = mesh::cell_entity_type(cell_type, dim, local_entity);
931 coord_ele.permute_subentity_closure(closure_dofs, cell_info[c],
932 entity_type, local_entity);
933 }
934
935 // Extract degrees of freedom
936 auto x_c = md::submdspan(xdofs, c, md::full_extent);
937 for (std::int32_t entity_dof : closure_dofs)
938 entity_xdofs.push_back(x_c[entity_dof]);
939 }
940
941 return {std::move(entity_xdofs), eshape};
942}

1006template <typename U>
1007Mesh<typename std::remove_reference_t<typename U::value_type>> create_mesh(
1008 MPI_Comm comm, MPI_Comm commt,
1009 std::vector<std::span<const std::int64_t>> cells,
1010 const std::vector<fem::CoordinateElement<
1011 typename std::remove_reference_t<typename U::value_type>>>& elements,
1012 MPI_Comm commg, const U& x, std::array<std::size_t, 2> xshape,
1013 const CellPartitionFunction& partitioner,
1014 const CellReorderFunction& reorder_fn = graph::reorder_gps)
1015{
1016 assert(cells.size() == elements.size());
1017 std::vector<CellType> celltypes;
1018 std::ranges::transform(elements, std::back_inserter(celltypes),
1019 [](auto e) { return e.cell_shape(); });
1020 std::vector<fem::ElementDofLayout> doflayouts;
1021 std::ranges::transform(elements, std::back_inserter(doflayouts),
1022 [](auto e) { return e.create_dof_layout(); });
1023
1024 // Note: `extract_topology` extracts topology data, i.e. just the
1025 // vertices. For P1 geometry this should just be the identity
1026 // operator. For other elements the filtered lists may have 'gaps',
1027 // i.e. the indices might not be contiguous.
1028 //
1029 // `extract_topology` could be skipped for 'P1 geometry' elements
1030
1031 std::int32_t num_cell_types = cells.size();
1032
1033 // -- Partition topology across ranks of comm
1034 std::vector<std::vector<std::int64_t>> cells1(num_cell_types);
1035 std::vector<std::vector<std::int64_t>> original_idx1(num_cell_types);
1036 std::vector<std::vector<int>> ghost_owners(num_cell_types);
1037 if (partitioner)
1038 {
1039 spdlog::info("Using partitioner with cell data ({} cell types)",
1040 num_cell_types);
1041 graph::AdjacencyList<std::int32_t> dest(0);
1042 if (commt != MPI_COMM_NULL)
1043 {
1044 int size = dolfinx::MPI::size(comm);
1045 std::vector<std::vector<std::int64_t>> t(num_cell_types);
1046 std::vector<std::span<const std::int64_t>> tspan(num_cell_types);
1047 for (std::int32_t i = 0; i < num_cell_types; ++i)
1048 {
1049 t[i] = extract_topology(celltypes[i], doflayouts[i], cells[i]);
1050 tspan[i] = std::span(t[i]);
1051 }
1052 dest = partitioner(commt, size, celltypes, tspan);
1053 }
1054
1055 std::int32_t cell_offset = 0;
1056 for (std::int32_t i = 0; i < num_cell_types; ++i)
1057 {
1058 std::size_t num_cell_nodes = doflayouts[i].num_dofs();
1059 assert(cells[i].size() % num_cell_nodes == 0);
1060 std::size_t num_cells = cells[i].size() / num_cell_nodes;
1061
1062 // Extract destination AdjacencyList for this cell type
1063 std::vector<std::int32_t> offsets_i(
1064 std::next(dest.offsets().begin(), cell_offset),
1065 std::next(dest.offsets().begin(), cell_offset + num_cells + 1));
1066 std::vector<std::int32_t> data_i(
1067 std::next(dest.array().begin(), offsets_i.front()),
1068 std::next(dest.array().begin(), offsets_i.back()));
1069 std::int32_t offset_0 = offsets_i.front();
1070 std::ranges::for_each(offsets_i,
1071 [&offset_0](std::int32_t& j) { j -= offset_0; });
1072 graph::AdjacencyList<std::int32_t> dest_i(data_i, offsets_i);
1073 cell_offset += num_cells;
1074
1075 // Distribute cells (topology, includes higher-order 'nodes') to
1076 // destination rank
1077 std::vector<int> src_ranks;
1078 std::tie(cells1[i], src_ranks, original_idx1[i], ghost_owners[i])
1079 = graph::build::distribute(comm, cells[i],
1080 {num_cells, num_cell_nodes}, dest_i);
1081 spdlog::debug("Got {} cells from distribution", cells1[i].size());
1082 }
1083 }
1084 else
1085 {
1086 // No partitioning, construct a global index
1087 std::int64_t num_owned = 0;
1088 for (std::int32_t i = 0; i < num_cell_types; ++i)
1089 {
1090 cells1[i] = std::vector<std::int64_t>(cells[i].begin(), cells[i].end());
1091 std::int32_t num_cell_nodes = doflayouts[i].num_dofs();
1092 assert(cells1[i].size() % num_cell_nodes == 0);
1093 original_idx1[i].resize(cells1[i].size() / num_cell_nodes);
1094 num_owned += original_idx1[i].size();
1095 }
1096
1097 // Add on global offset
1098 std::int64_t global_offset = 0;
1099 MPI_Exscan(&num_owned, &global_offset, 1, MPI_INT64_T, MPI_SUM, comm);
1100 for (std::int32_t i = 0; i < num_cell_types; ++i)
1101 {
1102 std::iota(original_idx1[i].begin(), original_idx1[i].end(),
1103 global_offset);
1104 global_offset += original_idx1[i].size();
1105 }
1106 }
1107
1108 // Extract cell 'topology', i.e. extract the vertices for each cell
1109 // and discard any 'higher-order' nodes
1110 std::vector<std::vector<std::int64_t>> cells1_v(num_cell_types);
1111 for (std::int32_t i = 0; i < num_cell_types; ++i)
1112 {
1113 cells1_v[i] = extract_topology(celltypes[i], doflayouts[i], cells1[i]);
1114 spdlog::info("Extract basic topology: {}->{}", cells1[i].size(),
1115 cells1_v[i].size());
1116 }
1117
1118 auto boundary_v_fn = create_boundary_vertices_fn(reorder_fn);
1119 const std::vector<std::int64_t> boundary_v = boundary_v_fn(
1120 celltypes, doflayouts, ghost_owners, cells1, cells1_v, original_idx1);
1121
1122 spdlog::debug("Got {} boundary vertices", boundary_v.size());
1123
1124 // Create Topology
1125 std::vector<std::span<const std::int64_t>> cells1_v_span;
1126 std::ranges::transform(cells1_v, std::back_inserter(cells1_v_span),
1127 [](auto& c) { return std::span(c); });
1128 std::vector<std::span<const std::int64_t>> original_idx1_span;
1129 std::ranges::transform(original_idx1, std::back_inserter(original_idx1_span),
1130 [](auto& c) { return std::span(c); });
1131 std::vector<std::span<const int>> ghost_owners_span;
1132 std::ranges::transform(ghost_owners, std::back_inserter(ghost_owners_span),
1133 [](auto& c) { return std::span(c); });
1134 Topology topology
1135 = create_topology(comm, celltypes, cells1_v_span, original_idx1_span,
1136 ghost_owners_span, boundary_v);
1137
1138 // Create connectivities required higher-order geometries for creating
1139 // a Geometry object
1140 for (int i = 0; i < num_cell_types; ++i)
1141 {
1142 for (int e = 1; e < topology.dim(); ++e)
1143 {
1144 if (doflayouts[i].num_entity_dofs(e) > 0)
1145 topology.create_entities(e);
1146 }
1147
1148 if (elements[i].needs_dof_permutations())
1149 topology.create_entity_permutations();
1150 }
1151
1152 // Build list of unique (global) node indices from cells1 and
1153 // distribute coordinate data
1154 std::vector<std::int64_t> nodes1, nodes2;
1155 for (std::vector<std::int64_t>& c : cells1)
1156 nodes1.insert(nodes1.end(), c.begin(), c.end());
1157 for (std::vector<std::int64_t>& c : cells1)
1158 nodes2.insert(nodes2.end(), c.begin(), c.end());
1159
1160 dolfinx::radix_sort(nodes1);
1161 auto [unique_end, range_end] = std::ranges::unique(nodes1);
1162 nodes1.erase(unique_end, range_end);
1163
1164 std::vector coords
1165 = dolfinx::MPI::distribute_data(comm, nodes1, commg, x, xshape[1]);
1166
1167 // Create geometry object
1168 Geometry geometry
1169 = create_geometry(topology, elements, nodes1, nodes2, coords, xshape[1]);
1170
1171 return Mesh(comm, std::make_shared<Topology>(std::move(topology)),
1172 std::move(geometry));
1173}

1271template <std::floating_point T>
1272std::pair<Geometry<T>, std::vector<int32_t>>
1273create_subgeometry(const Mesh<T>& mesh, int dim,
1274 std::span<const std::int32_t> subentity_to_entity)
1275{
1276 const Geometry<T>& geometry = mesh.geometry();
1277
1278 // Get the geometry dofs in the sub-geometry based on the entities in
1279 // sub-geometry
1280 const fem::ElementDofLayout layout = geometry.cmap().create_dof_layout();
1281
1282 const std::vector<std::int32_t> x_indices
1283 = entities_to_geometry(mesh, dim, subentity_to_entity, true).first;
1284
1285 std::vector<std::int32_t> sub_x_dofs = x_indices;
1286 std::ranges::sort(sub_x_dofs);
1287 auto [unique_end, range_end] = std::ranges::unique(sub_x_dofs);
1288 sub_x_dofs.erase(unique_end, range_end);
1289
1290 // Get the sub-geometry dofs owned by this process
1291 auto x_index_map = geometry.index_map();
1292 assert(x_index_map);
1293
1294 std::shared_ptr<common::IndexMap> sub_x_dof_index_map;
1295 std::vector<std::int32_t> subx_to_x_dofmap;
1296 {
1297 auto [map, new_to_old] = common::create_sub_index_map(
1298 *x_index_map, sub_x_dofs, common::IndexMapOrder::any, true);
1299 sub_x_dof_index_map = std::make_shared<common::IndexMap>(std::move(map));
1300 subx_to_x_dofmap = std::move(new_to_old);
1301 }
1302
1303 // Create sub-geometry coordinates
1304 std::span<const T> x = geometry.x();
1305 std::int32_t sub_num_x_dofs = subx_to_x_dofmap.size();
1306 std::vector<T> sub_x(3 * sub_num_x_dofs);
1307 for (std::int32_t i = 0; i < sub_num_x_dofs; ++i)
1308 {
1309 std::copy_n(std::next(x.begin(), 3 * subx_to_x_dofmap[i]), 3,
1310 std::next(sub_x.begin(), 3 * i));
1311 }
1312
1313 // Create geometry to sub-geometry map
1314 std::vector<std::int32_t> x_to_subx_dof_map(
1315 x_index_map->size_local() + x_index_map->num_ghosts(), -1);
1316 for (std::size_t i = 0; i < subx_to_x_dofmap.size(); ++i)
1317 x_to_subx_dof_map[subx_to_x_dofmap[i]] = i;
1318
1319 // Create sub-geometry dofmap
1320 std::vector<std::int32_t> sub_x_dofmap;
1321 sub_x_dofmap.reserve(x_indices.size());
1322 std::ranges::transform(x_indices, std::back_inserter(sub_x_dofmap),
1323 [&x_to_subx_dof_map](auto x_dof)
1324 {
1325 assert(x_to_subx_dof_map[x_dof] != -1);
1326 return x_to_subx_dof_map[x_dof];
1327 });
1328
1329 // Sub-geometry coordinate element
1330 CellType sub_xcell = cell_entity_type(geometry.cmap().cell_shape(), dim, 0);
1331
1332 // Special handling of point meshes, as they only support constant
1333 // basis functions
1334 int degree = (sub_xcell == CellType::point) ? 0 : geometry.cmap().degree();
1335 fem::CoordinateElement<T> sub_cmap(sub_xcell, degree,
1336 geometry.cmap().variant());
1337
1338 // Sub-geometry input_global_indices
1339 const std::vector<std::int64_t>& igi = geometry.input_global_indices();
1340 std::vector<std::int64_t> sub_igi;
1341 sub_igi.reserve(subx_to_x_dofmap.size());
1342 std::ranges::transform(subx_to_x_dofmap, std::back_inserter(sub_igi),
1343 [&igi](auto sub_x_dof) { return igi[sub_x_dof]; });
1344
1345 // Create geometry
1346 return {Geometry(
1347 sub_x_dof_index_map,
1348 std::vector<std::vector<std::int32_t>>{std::move(sub_x_dofmap)},
1349 {sub_cmap}, std::move(sub_x), geometry.dim(), std::move(sub_igi)),
1350 std::move(subx_to_x_dofmap)};
1351}
