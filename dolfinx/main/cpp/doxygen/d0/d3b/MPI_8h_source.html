151 constexpr std::array<std::int64_t, 2> local_range(int rank, std::int64_t N,
152  int size)
153 {
154  assert(rank >= 0);
155  assert(N >= 0);
156  assert(size > 0);
157 
158  // Compute number of items per rank and remainder
159  const std::int64_t n = N / size;
160  const std::int64_t r = N % size;
161 
162  // Compute local range
163  if (rank < r)
164  return {{rank * (n + 1), rank * (n + 1) + n + 1}};
165  else
166  return {{rank * n + r, rank * n + r + n}};
167 }

174 constexpr int index_owner(int size, std::size_t index, std::size_t N)
175 {
176  assert(index < N);
177 
178  // Compute number of items per rank and remainder
179  const std::size_t n = N / size;
180  const std::size_t r = N % size;
181 
182  // First r ranks own n + 1 indices
183  if (index < r * (n + 1))
184  return index / (n + 1);
185 
186  // Remaining ranks own n indices
187  return r + (index - r * (n + 1)) / n;
188 }

196 template <typename T>
197 constexpr MPI_Datatype mpi_type()
198 {
199  if constexpr (std::is_same<T, float>::value)
200  return MPI_FLOAT;
201  else if constexpr (std::is_same<T, double>::value)
202  return MPI_DOUBLE;
203  else if constexpr (std::is_same<T, std::complex<double>>::value)
204  return MPI_C_DOUBLE_COMPLEX;
205  else if constexpr (std::is_same<T, std::complex<float>>::value)
206  return MPI_C_FLOAT_COMPLEX;
207  else if constexpr (std::is_same<T, short int>::value)
208  return MPI_SHORT;
209  else if constexpr (std::is_same<T, int>::value)
210  return MPI_INT;
211  else if constexpr (std::is_same<T, unsigned int>::value)
212  return MPI_UNSIGNED;
213  else if constexpr (std::is_same<T, long int>::value)
214  return MPI_LONG;
215  else if constexpr (std::is_same<T, unsigned long>::value)
216  return MPI_UNSIGNED_LONG;
217  else if constexpr (std::is_same<T, long long>::value)
218  return MPI_LONG_LONG;
219  else if constexpr (std::is_same<T, unsigned long long>::value)
220  return MPI_UNSIGNED_LONG_LONG;
221  else if constexpr (std::is_same<T, bool>::value)
222  return MPI_C_BOOL;
223  else if constexpr (std::is_same<T, std::int8_t>::value)
224  return MPI_INT8_T;
225  else
226  // Issue compile time error
227  static_assert(!std::is_same<T, T>::value);
228 }

231 template <typename T>
232 graph::AdjacencyList<T> all_to_all(MPI_Comm comm,
233  const graph::AdjacencyList<T>& send_data)
234 {
235  const std::vector<std::int32_t>& send_offsets = send_data.offsets();
236  const std::vector<T>& values_in = send_data.array();
237 
238  const int comm_size = dolfinx::MPI::size(comm);
239  assert(send_data.num_nodes() == comm_size);
240 
241  // Data size per destination rank
242  std::vector<int> send_size(comm_size);
243  std::adjacent_difference(std::next(send_offsets.begin()), send_offsets.end(),
244  send_size.begin());
245 
246  // Get received data sizes from each rank
247  std::vector<int> recv_size(comm_size);
248  MPI_Alltoall(send_size.data(), 1, mpi_type<int>(), recv_size.data(), 1,
249  mpi_type<int>(), comm);
250 
251  // Compute receive offset
252  std::vector<std::int32_t> recv_offset(comm_size + 1, 0);
253  std::partial_sum(recv_size.begin(), recv_size.end(),
254  std::next(recv_offset.begin()));
255 
256  // Send/receive data
257  std::vector<T> recv_values(recv_offset.back());
258  MPI_Alltoallv(values_in.data(), send_size.data(), send_offsets.data(),
259  mpi_type<T>(), recv_values.data(), recv_size.data(),
260  recv_offset.data(), mpi_type<T>(), comm);
261 
262  return graph::AdjacencyList<T>(std::move(recv_values),
263  std::move(recv_offset));
264 }

266 template <typename T>
267 graph::AdjacencyList<T>
268 neighbor_all_to_all(MPI_Comm comm, const graph::AdjacencyList<T>& send_data)
269 {
270  // Get neighbor processes
271  int indegree(-1), outdegree(-2), weighted(-1);
272  MPI_Dist_graph_neighbors_count(comm, &indegree, &outdegree, &weighted);
273 
274  // Allocate memory (add '1' to handle empty case as OpenMPI fails for
275  // null pointers
276  std::vector<int> send_sizes(outdegree + 1, 0);
277  std::vector<int> recv_sizes(indegree + 1);
278  std::adjacent_difference(std::next(send_data.offsets().begin()),
279  send_data.offsets().end(), send_sizes.begin());
280  // Get receive sizes
281  MPI_Neighbor_alltoall(send_sizes.data(), 1, MPI_INT, recv_sizes.data(), 1,
282  MPI_INT, comm);
283  send_sizes.pop_back();
284  recv_sizes.pop_back();
285 
286  // Work out recv offsets
287  std::vector<int> recv_offsets(indegree + 1);
288  recv_offsets[0] = 0;
289  std::partial_sum(recv_sizes.begin(), recv_sizes.end(),
290  std::next(recv_offsets.begin(), 1));
291 
292  std::vector<T> recv_data(recv_offsets[recv_offsets.size() - 1]);
293  MPI_Neighbor_alltoallv(
294  send_data.array().data(), send_sizes.data(), send_data.offsets().data(),
295  dolfinx::MPI::mpi_type<T>(), recv_data.data(), recv_sizes.data(),
296  recv_offsets.data(), dolfinx::MPI::mpi_type<T>(), comm);
297 
298  return graph::AdjacencyList<T>(std::move(recv_data), std::move(recv_offsets));
299 }
