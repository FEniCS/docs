58template <std::floating_point T>
59Mat create_matrix_block(
60 const std::vector<std::vector<const Form<PetscScalar, T>*>>& a,
61 const std::string& type = std::string())
62{
63 // Extract and check row/column ranges
64 std::array<std::vector<std::shared_ptr<const FunctionSpace<T>>>, 2> V
65 = fem::common_function_spaces(extract_function_spaces(a));
66 std::array<std::vector<int>, 2> bs_dofs;
67 for (std::size_t i = 0; i < 2; ++i)
68 {
69 for (auto& _V : V[i])
70 bs_dofs[i].push_back(_V->dofmap()->bs());
71 }
72
73 std::shared_ptr mesh = V[0][0]->mesh();
74 assert(mesh);
75 const int tdim = mesh->topology()->dim();
76
77 // Build sparsity pattern for each block
78 std::vector<std::vector<std::unique_ptr<la::SparsityPattern>>> patterns(
79 V[0].size());
80 for (std::size_t row = 0; row < V[0].size(); ++row)
81 {
82 for (std::size_t col = 0; col < V[1].size(); ++col)
83 {
84 const std::array<std::shared_ptr<const common::IndexMap>, 2> index_maps
85 = {{V[0][row]->dofmap()->index_map, V[1][col]->dofmap()->index_map}};
86 const std::array bs = {V[0][row]->dofmap()->index_map_bs(),
87 V[1][col]->dofmap()->index_map_bs()};
88 if (const Form<PetscScalar, T>* form = a[row][col]; form)
89 {
90 // Create sparsity pattern for block
91 patterns[row].push_back(std::make_unique<la::SparsityPattern>(
92 mesh->comm(), index_maps, bs));
93
94 // Build sparsity pattern for block
95 assert(V[0][row]->dofmap());
96 assert(V[1][col]->dofmap());
97 std::array<const std::reference_wrapper<const DofMap>, 2> dofmaps{
98 *V[0][row]->dofmap(), *V[1][col]->dofmap()};
99 assert(patterns[row].back());
100 auto& sp = patterns[row].back();
101 assert(sp);
102 if (form->num_integrals(IntegralType::cell) > 0)
103 sparsitybuild::cells(*sp, *mesh->topology(), dofmaps);
104 if (form->num_integrals(IntegralType::interior_facet) > 0)
105 {
106 mesh->topology_mutable()->create_entities(tdim - 1);
107 sparsitybuild::interior_facets(*sp, *mesh->topology(), dofmaps);
108 }
109 if (form->num_integrals(IntegralType::exterior_facet) > 0)
110 {
111 mesh->topology_mutable()->create_entities(tdim - 1);
112 sparsitybuild::exterior_facets(*sp, *mesh->topology(), dofmaps);
113 }
114 }
115 else
116 patterns[row].push_back(nullptr);
117 }
118 }
119
120 // Compute offsets for the fields
121 std::array<std::vector<std::pair<
122 std::reference_wrapper<const common::IndexMap>, int>>,
123 2>
124 maps;
125 for (std::size_t d = 0; d < 2; ++d)
126 {
127 for (auto space : V[d])
128 {
129 maps[d].emplace_back(*space->dofmap()->index_map,
130 space->dofmap()->index_map_bs());
131 }
132 }
133
134 // Create merged sparsity pattern
135 std::vector<std::vector<const la::SparsityPattern*>> p(V[0].size());
136 for (std::size_t row = 0; row < V[0].size(); ++row)
137 for (std::size_t col = 0; col < V[1].size(); ++col)
138 p[row].push_back(patterns[row][col].get());
139
140 la::SparsityPattern pattern(mesh->comm(), p, maps, bs_dofs);
141 pattern.assemble();
142
143 // FIXME: Add option to pass customised local-to-global map to PETSc
144 // Mat constructor
145
146 // Initialise matrix
147 Mat A = la::petsc::create_matrix(mesh->comm(), pattern, type);
148
149 // Create row and column local-to-global maps (field0, field1, field2,
150 // etc), i.e. ghosts of field0 appear before owned indices of field1
151 std::array<std::vector<PetscInt>, 2> _maps;
152 for (int d = 0; d < 2; ++d)
153 {
154 // FIXME: Index map concatenation has already been computed inside
155 // the SparsityPattern constructor, but we also need it here to
156 // build the PETSc local-to-global map. Compute outside and pass
157 // into SparsityPattern constructor.
158
159 // FIXME: avoid concatenating the same maps twice in case that V[0]
160 // == V[1].
161
162 // Concatenate the block index map in the row and column directions
163 auto [rank_offset, local_offset, ghosts, _]
164 = common::stack_index_maps(maps[d]);
165 for (std::size_t f = 0; f < maps[d].size(); ++f)
166 {
167 const common::IndexMap& map = maps[d][f].first.get();
168 const int bs = maps[d][f].second;
169 const std::int32_t size_local = bs * map.size_local();
170 const std::vector global = map.global_indices();
171 for (std::int32_t i = 0; i < size_local; ++i)
172 _maps[d].push_back(i + rank_offset + local_offset[f]);
173 for (std::size_t i = size_local; i < bs * global.size(); ++i)
174 _maps[d].push_back(ghosts[f][i - size_local]);
175 }
176 }
177
178 // Create PETSc local-to-global map/index sets and attach to matrix
179 ISLocalToGlobalMapping petsc_local_to_global0;
180 ISLocalToGlobalMappingCreate(MPI_COMM_SELF, 1, _maps[0].size(),
181 _maps[0].data(), PETSC_COPY_VALUES,
182 &petsc_local_to_global0);
183 if (V[0] == V[1])
184 {
185 MatSetLocalToGlobalMapping(A, petsc_local_to_global0,
186 petsc_local_to_global0);
187 ISLocalToGlobalMappingDestroy(&petsc_local_to_global0);
188 }
189 else
190 {
191
192 ISLocalToGlobalMapping petsc_local_to_global1;
193 ISLocalToGlobalMappingCreate(MPI_COMM_SELF, 1, _maps[1].size(),
194 _maps[1].data(), PETSC_COPY_VALUES,
195 &petsc_local_to_global1);
196 MatSetLocalToGlobalMapping(A, petsc_local_to_global0,
197 petsc_local_to_global1);
198 ISLocalToGlobalMappingDestroy(&petsc_local_to_global0);
199 ISLocalToGlobalMappingDestroy(&petsc_local_to_global1);
200 }
201
202 return A;
203}

208template <std::floating_point T>
209Mat create_matrix_nest(
210 const std::vector<std::vector<const Form<PetscScalar, T>*>>& a,
211 const std::vector<std::vector<std::string>>& types)
212{
213 // Extract and check row/column ranges
214 auto V = fem::common_function_spaces(extract_function_spaces(a));
215
216 std::vector<std::vector<std::string>> _types(
217 a.size(), std::vector<std::string>(a.front().size()));
218 if (!types.empty())
219 _types = types;
220
221 // Loop over each form and create matrix
222 const int rows = a.size();
223 const int cols = a.front().size();
224 std::vector<Mat> mats(rows * cols, nullptr);
225 for (int i = 0; i < rows; ++i)
226 {
227 for (int j = 0; j < cols; ++j)
228 {
229 if (const Form<PetscScalar, T>* form = a[i][j]; form)
230 mats[i * cols + j] = create_matrix(*form, _types[i][j]);
231 }
232 }
233
234 // Initialise block (MatNest) matrix
235 Mat A;
236 MatCreate(V[0][0]->mesh()->comm(), &A);
237 MatSetType(A, MATNEST);
238 MatNestSetSubMats(A, rows, nullptr, cols, nullptr, mats.data());
239 MatSetUp(A);
240
241 // De-reference Mat objects
242 for (std::size_t i = 0; i < mats.size(); ++i)
243 {
244 if (mats[i])
245 MatDestroy(&mats[i]);
246 }
247
248 return A;
249}

341template <std::floating_point T>
342void apply_lifting(
343 Vec b, const std::vector<std::shared_ptr<const Form<PetscScalar, T>>>& a,
344 const std::vector<std::span<const PetscScalar>>& constants,
345 const std::vector<std::map<std::pair<IntegralType, int>,
346 std::pair<std::span<const PetscScalar>, int>>>&
347 coeffs,
348 const std::vector<
349 std::vector<std::shared_ptr<const DirichletBC<PetscScalar, T>>>>& bcs1,
350 const std::vector<Vec>& x0, PetscScalar scale)
351{
352 Vec b_local;
353 VecGhostGetLocalForm(b, &b_local);
354 PetscInt n = 0;
355 VecGetSize(b_local, &n);
356 PetscScalar* array = nullptr;
357 VecGetArray(b_local, &array);
358 std::span<PetscScalar> _b(array, n);
359
360 if (x0.empty())
361 fem::apply_lifting(_b, a, constants, coeffs, bcs1, {}, scale);
362 else
363 {
364 std::vector<std::span<const PetscScalar>> x0_ref;
365 std::vector<Vec> x0_local(a.size());
366 std::vector<const PetscScalar*> x0_array(a.size());
367 for (std::size_t i = 0; i < a.size(); ++i)
368 {
369 assert(x0[i]);
370 VecGhostGetLocalForm(x0[i], &x0_local[i]);
371 PetscInt n = 0;
372 VecGetSize(x0_local[i], &n);
373 VecGetArrayRead(x0_local[i], &x0_array[i]);
374 x0_ref.emplace_back(x0_array[i], n);
375 }
376
377 std::vector x0_tmp(x0_ref.begin(), x0_ref.end());
378 fem::apply_lifting(_b, a, constants, coeffs, bcs1, x0_tmp, scale);
379
380 for (std::size_t i = 0; i < x0_local.size(); ++i)
381 {
382 VecRestoreArrayRead(x0_local[i], &x0_array[i]);
383 VecGhostRestoreLocalForm(x0[i], &x0_local[i]);
384 }
385 }
386
387 VecRestoreArray(b_local, &array);
388 VecGhostRestoreLocalForm(b, &b_local);
389}
