50template <typename T>
51void discrete_gradient(mesh::Topology& topology,
52 std::pair<std::reference_wrapper<const FiniteElement>,
53 std::reference_wrapper<const DofMap>>
54 V0,
55 std::pair<std::reference_wrapper<const FiniteElement>,
56 std::reference_wrapper<const DofMap>>
57 V1,
58 auto&& mat_set)
59{
60 const FiniteElement& e0 = V0.first.get();
61 const DofMap& dofmap0 = V0.second.get();
62 const FiniteElement& e1 = V1.first.get();
63 const DofMap& dofmap1 = V1.second.get();
64
65 namespace stdex = std::experimental;
66 using cmdspan2_t
67 = stdex::mdspan<const double, stdex::dextents<std::size_t, 2>>;
68 using mdspan2_t = stdex::mdspan<double, stdex::dextents<std::size_t, 2>>;
69 using cmdspan4_t
70 = stdex::mdspan<const double, stdex::dextents<std::size_t, 4>>;
71
72 // Check elements
73 if (e0.map_type() != basix::maps::type::identity)
74 throw std::runtime_error("Wrong finite element space for V0.");
75 if (e0.block_size() != 1)
76 throw std::runtime_error("Block size is greater than 1 for V0.");
77 if (e0.reference_value_size() != 1)
78 throw std::runtime_error("Wrong value size for V0.");
79
80 if (e1.map_type() != basix::maps::type::covariantPiola)
81 throw std::runtime_error("Wrong finite element space for V1.");
82 if (e1.block_size() != 1)
83 throw std::runtime_error("Block size is greater than 1 for V1.");
84
85 // Get V0 (H(curl)) space interpolation points
86 const auto [X, Xshape] = e1.interpolation_points();
87
88 // Tabulate first order derivatives of Lagrange space at H(curl)
89 // interpolation points
90 const int ndofs0 = e0.space_dimension();
91 const int tdim = topology.dim();
92 std::vector<double> phi0_b((tdim + 1) * Xshape[0] * ndofs0 * 1);
93 cmdspan4_t phi0(phi0_b.data(), tdim + 1, Xshape[0], ndofs0, 1);
94 e0.tabulate(phi0_b, X, Xshape, 1);
95
96 // Reshape lagrange basis derivatives as a matrix of shape (tdim *
97 // num_points, num_dofs_per_cell)
98 cmdspan2_t dphi_reshaped(
99 phi0_b.data() + phi0.extent(3) * phi0.extent(2) * phi0.extent(1),
100 tdim * phi0.extent(1), phi0.extent(2));
101
102 // Get inverse DOF transform function
103 auto apply_inverse_dof_transform
104 = e1.get_dof_transformation_function<T>(true, true, false);
105
106 // Generate cell permutations
107 topology.create_entity_permutations();
108 const std::vector<std::uint32_t>& cell_info
109 = topology.get_cell_permutation_info();
110
111 // Create element kernel function
112
113 // Build the element interpolation matrix
114 std::vector<T> Ab(e1.space_dimension() * ndofs0);
115 {
116 stdex::mdspan<T, stdex::dextents<std::size_t, 2>> A(
117 Ab.data(), e1.space_dimension(), ndofs0);
118 const auto [Pi, shape] = e1.interpolation_operator();
119 cmdspan2_t _Pi(Pi.data(), shape);
120 math::dot(_Pi, dphi_reshaped, A);
121 }
122
123 // Insert local interpolation matrix for each cell
124 auto cell_map = topology.index_map(tdim);
125 assert(cell_map);
126 std::int32_t num_cells = cell_map->size_local();
127 std::vector<T> Ae(Ab.size());
128 for (std::int32_t c = 0; c < num_cells; ++c)
129 {
130 std::copy(Ab.cbegin(), Ab.cend(), Ae.begin());
131 apply_inverse_dof_transform(Ae, cell_info, c, ndofs0);
132 mat_set(dofmap1.cell_dofs(c), dofmap0.cell_dofs(c), Ae);
133 }
134}

151template <typename T, std::floating_point U>
152void interpolation_matrix(const FunctionSpace<U>& V0,
153 const FunctionSpace<U>& V1, auto&& mat_set)
154{
155 // Get mesh
156 auto mesh = V0.mesh();
157 assert(mesh);
158
159 // Mesh dims
160 const int tdim = mesh->topology().dim();
161 const int gdim = mesh->geometry().dim();
162
163 // Get elements
164 std::shared_ptr<const FiniteElement> e0 = V0.element();
165 assert(e0);
166 std::shared_ptr<const FiniteElement> e1 = V1.element();
167 assert(e1);
168
169 std::span<const std::uint32_t> cell_info;
170 if (e1->needs_dof_transformations() or e0->needs_dof_transformations())
171 {
172 mesh->topology_mutable().create_entity_permutations();
173 cell_info = std::span(mesh->topology().get_cell_permutation_info());
174 }
175
176 // Get dofmaps
177 auto dofmap0 = V0.dofmap();
178 assert(dofmap0);
179 auto dofmap1 = V1.dofmap();
180 assert(dofmap1);
181
182 // Get block sizes and dof transformation operators
183 const int bs0 = e0->block_size();
184 const int bs1 = e1->block_size();
185 const auto apply_dof_transformation0
186 = e0->get_dof_transformation_function<double>(false, false, false);
187 const auto apply_inverse_dof_transform1
188 = e1->get_dof_transformation_function<T>(true, true, false);
189
190 // Get sizes of elements
191 const std::size_t space_dim0 = e0->space_dimension();
192 const std::size_t space_dim1 = e1->space_dimension();
193 const std::size_t dim0 = space_dim0 / bs0;
194 const std::size_t value_size_ref0 = e0->reference_value_size() / bs0;
195 const std::size_t value_size0 = e0->value_size() / bs0;
196 const std::size_t value_size1 = e1->value_size() / bs1;
197
198 // Get geometry data
199 const CoordinateElement& cmap = mesh->geometry().cmap();
200 const graph::AdjacencyList<std::int32_t>& x_dofmap
201 = mesh->geometry().dofmap();
202 const std::size_t num_dofs_g = cmap.dim();
203 std::span<const U> x_g = mesh->geometry().x();
204
205 namespace stdex = std::experimental;
206 using mdspan2_t = stdex::mdspan<double, stdex::dextents<std::size_t, 2>>;
207 using cmdspan2_t
208 = stdex::mdspan<const double, stdex::dextents<std::size_t, 2>>;
209 using cmdspan3_t
210 = stdex::mdspan<const double, stdex::dextents<std::size_t, 3>>;
211 using cmdspan4_t
212 = stdex::mdspan<const double, stdex::dextents<std::size_t, 4>>;
213 using mdspan3_t = stdex::mdspan<double, stdex::dextents<std::size_t, 3>>;
214
215 // Evaluate coordinate map basis at reference interpolation points
216 const auto [X, Xshape] = e1->interpolation_points();
217 std::array<std::size_t, 4> phi_shape = cmap.tabulate_shape(1, Xshape[0]);
218 std::vector<double> phi_b(
219 std::reduce(phi_shape.begin(), phi_shape.end(), 1, std::multiplies{}));
220 cmdspan4_t phi(phi_b.data(), phi_shape);
221 cmap.tabulate(1, X, Xshape, phi_b);
222
223 // Evaluate V0 basis functions at reference interpolation points for V1
224 std::vector<double> basis_derivatives_reference0_b(Xshape[0] * dim0
225 * value_size_ref0);
226 cmdspan4_t basis_derivatives_reference0(basis_derivatives_reference0_b.data(),
227 1, Xshape[0], dim0, value_size_ref0);
228 e0->tabulate(basis_derivatives_reference0_b, X, Xshape, 0);
229
230 // Clamp values
231 std::transform(basis_derivatives_reference0_b.begin(),
232 basis_derivatives_reference0_b.end(),
233 basis_derivatives_reference0_b.begin(),
234 [atol = 1e-14](auto x)
235 { return std::abs(x) < atol ? 0.0 : x; });
236
237 // Create working arrays
238 std::vector<double> basis_reference0_b(Xshape[0] * dim0 * value_size_ref0);
239 mdspan3_t basis_reference0(basis_reference0_b.data(), Xshape[0], dim0,
240 value_size_ref0);
241 std::vector<double> J_b(Xshape[0] * gdim * tdim);
242 mdspan3_t J(J_b.data(), Xshape[0], gdim, tdim);
243 std::vector<double> K_b(Xshape[0] * tdim * gdim);
244 mdspan3_t K(K_b.data(), Xshape[0], tdim, gdim);
245 std::vector<double> detJ(Xshape[0]);
246 std::vector<double> det_scratch(2 * tdim * gdim);
247
248 // Get the interpolation operator (matrix) `Pi` that maps a function
249 // evaluated at the interpolation points to the element degrees of
250 // freedom, i.e. dofs = Pi f_x
251 const auto [_Pi_1, pi_shape] = e1->interpolation_operator();
252 cmdspan2_t Pi_1(_Pi_1.data(), pi_shape);
253
254 bool interpolation_ident = e1->interpolation_ident();
255
256 namespace stdex = std::experimental;
257 using u_t = stdex::mdspan<double, stdex::dextents<std::size_t, 2>>;
258 using U_t = stdex::mdspan<const double, stdex::dextents<std::size_t, 2>>;
259 using J_t = stdex::mdspan<const double, stdex::dextents<std::size_t, 2>>;
260 using K_t = stdex::mdspan<const double, stdex::dextents<std::size_t, 2>>;
261 auto push_forward_fn0 = e0->basix_element().map_fn<u_t, U_t, J_t, K_t>();
262
263 // Basis values of Lagrange space unrolled for block size
264 // (num_quadrature_points, Lagrange dof, value_size)
265 std::vector<double> basis_values_b(Xshape[0] * bs0 * dim0 * e1->value_size());
266 mdspan3_t basis_values(basis_values_b.data(), Xshape[0], bs0 * dim0,
267 e1->value_size());
268 std::vector<double> mapped_values_b(Xshape[0] * bs0 * dim0
269 * e1->value_size());
270 mdspan3_t mapped_values(mapped_values_b.data(), Xshape[0], bs0 * dim0,
271 e1->value_size());
272
273 auto pull_back_fn1 = e1->basix_element().map_fn<u_t, U_t, K_t, J_t>();
274
275 std::vector<double> coord_dofs_b(num_dofs_g * gdim);
276 mdspan2_t coord_dofs(coord_dofs_b.data(), num_dofs_g, gdim);
277 std::vector<double> basis0_b(Xshape[0] * dim0 * value_size0);
278 mdspan3_t basis0(basis0_b.data(), Xshape[0], dim0, value_size0);
279
280 // Buffers
281 std::vector<T> Ab(space_dim0 * space_dim1);
282 std::vector<T> local1(space_dim1);
283
284 // Iterate over mesh and interpolate on each cell
285 auto cell_map = mesh->topology().index_map(tdim);
286 assert(cell_map);
287 std::int32_t num_cells = cell_map->size_local();
288 for (std::int32_t c = 0; c < num_cells; ++c)
289 {
290 // Get cell geometry (coordinate dofs)
291 auto x_dofs = x_dofmap.links(c);
292 for (std::size_t i = 0; i < x_dofs.size(); ++i)
293 {
294 for (std::size_t j = 0; j < gdim; ++j)
295 coord_dofs(i, j) = x_g[3 * x_dofs[i] + j];
296 }
297
298 // Compute Jacobians and reference points for current cell
299 std::fill(J_b.begin(), J_b.end(), 0);
300 for (std::size_t p = 0; p < Xshape[0]; ++p)
301 {
302 auto dphi = stdex::submdspan(phi, std::pair(1, tdim + 1), p,
303 stdex::full_extent, 0);
304 auto _J = stdex::submdspan(J, p, stdex::full_extent, stdex::full_extent);
305 cmap.compute_jacobian(dphi, coord_dofs, _J);
306 auto _K = stdex::submdspan(K, p, stdex::full_extent, stdex::full_extent);
307 cmap.compute_jacobian_inverse(_J, _K);
308 detJ[p] = cmap.compute_jacobian_determinant(_J, det_scratch);
309 }
310
311 // Copy evaluated basis on reference, apply DOF transformations, and
312 // push forward to physical element
313 for (std::size_t k0 = 0; k0 < basis_reference0.extent(0); ++k0)
314 for (std::size_t k1 = 0; k1 < basis_reference0.extent(1); ++k1)
315 for (std::size_t k2 = 0; k2 < basis_reference0.extent(2); ++k2)
316 basis_reference0(k0, k1, k2)
317 = basis_derivatives_reference0(0, k0, k1, k2);
318 for (std::size_t p = 0; p < Xshape[0]; ++p)
319 {
320 apply_dof_transformation0(
321 std::span(basis_reference0.data_handle() + p * dim0 * value_size_ref0,
322 dim0 * value_size_ref0),
323 cell_info, c, value_size_ref0);
324 }
325
326 for (std::size_t p = 0; p < basis0.extent(0); ++p)
327 {
328 auto _u
329 = stdex::submdspan(basis0, p, stdex::full_extent, stdex::full_extent);
330 auto _U = stdex::submdspan(basis_reference0, p, stdex::full_extent,
331 stdex::full_extent);
332 auto _K = stdex::submdspan(K, p, stdex::full_extent, stdex::full_extent);
333 auto _J = stdex::submdspan(J, p, stdex::full_extent, stdex::full_extent);
334 push_forward_fn0(_u, _U, _J, detJ[p], _K);
335 }
336
337 // Unroll basis function for input space for block size
338 for (std::size_t p = 0; p < Xshape[0]; ++p)
339 for (std::size_t i = 0; i < dim0; ++i)
340 for (std::size_t j = 0; j < value_size0; ++j)
341 for (int k = 0; k < bs0; ++k)
342 basis_values(p, i * bs0 + k, j * bs0 + k) = basis0(p, i, j);
343
344 // Pull back the physical values to the reference of output space
345 for (std::size_t p = 0; p < basis_values.extent(0); ++p)
346 {
347 auto _u = stdex::submdspan(basis_values, p, stdex::full_extent,
348 stdex::full_extent);
349 auto _U = stdex::submdspan(mapped_values, p, stdex::full_extent,
350 stdex::full_extent);
351 auto _K = stdex::submdspan(K, p, stdex::full_extent, stdex::full_extent);
352 auto _J = stdex::submdspan(J, p, stdex::full_extent, stdex::full_extent);
353 pull_back_fn1(_U, _u, _K, 1.0 / detJ[p], _J);
354 }
355
356 // Apply interpolation matrix to basis values of V0 at the
357 // interpolation points of V1
358 if (interpolation_ident)
359 {
360 stdex::mdspan<T, stdex::dextents<std::size_t, 3>> A(
361 Ab.data(), Xshape[0], e1->value_size(), space_dim0);
362 for (std::size_t i = 0; i < mapped_values.extent(0); ++i)
363 for (std::size_t j = 0; j < mapped_values.extent(1); ++j)
364 for (std::size_t k = 0; k < mapped_values.extent(2); ++k)
365 A(i, k, j) = mapped_values(i, j, k);
366 }
367 else
368 {
369 for (std::size_t i = 0; i < mapped_values.extent(1); ++i)
370 {
371 auto values = stdex::submdspan(mapped_values, stdex::full_extent, i,
372 stdex::full_extent);
373 impl::interpolation_apply(Pi_1, values, std::span(local1), bs1);
374 for (std::size_t j = 0; j < local1.size(); j++)
375 Ab[space_dim0 * j + i] = local1[j];
376 }
377 }
378
379 apply_inverse_dof_transform1(Ab, cell_info, c, space_dim0);
380 mat_set(dofmap1->cell_dofs(c), dofmap0->cell_dofs(c), Ab);
381 }
382}
