48 template <typename T, typename U>
49 void assemble_discrete_gradient(const fem::FunctionSpace& V0,
50  const fem::FunctionSpace& V1, U&& mat_set)
51 {
52  // Get mesh
53  std::shared_ptr<const mesh::Mesh> mesh = V1.mesh();
54  assert(mesh);
55 
56  // Check spaces
57  std::shared_ptr<const FiniteElement> e0 = V0.element();
58  assert(e0);
59  if (e0->map_type() != basix::maps::type::identity)
60  throw std::runtime_error("Wrong finite element space for V0.");
61  if (e0->block_size() != 1)
62  throw std::runtime_error("Block size is greather than 1 for V0.");
63  if (e0->reference_value_size() != 1)
64  throw std::runtime_error("Wrong value size for V0.");
65 
66  std::shared_ptr<const FiniteElement> e1 = V1.element();
67  assert(e1);
68  if (e1->map_type() != basix::maps::type::covariantPiola)
69  throw std::runtime_error("Wrong finite element space for V1.");
70  if (e1->block_size() != 1)
71  throw std::runtime_error("Block size is greather than 1 for V1.");
72 
73  // Get V0 (H(curl)) space interpolation points
74  const xt::xtensor<double, 2> X = e1->interpolation_points();
75 
76  // Tabulate first order derivatives of Lagrange space at H(curl)
77  // interpolation points
78  const int ndofs0 = e0->space_dimension();
79  const int tdim = mesh->topology().dim();
80  xt::xtensor<double, 4> phi0
81  = xt::empty<double>({tdim + 1, int(X.shape(0)), ndofs0, 1});
82  e0->tabulate(phi0, X, 1);
83 
84  // Reshape lagrange basis derivatives as a matrix of shape (tdim *
85  // num_points, num_dofs_per_cell)
86  auto dphi0 = xt::view(phi0, xt::xrange(std::size_t(1), phi0.shape(0)),
87  xt::all(), xt::all(), 0);
88  auto dphi_reshaped
89  = xt::reshape_view(dphi0, {tdim * phi0.shape(1), phi0.shape(2)});
90 
91  // Get inverse DOF transform function
92  auto apply_inverse_dof_transform
93  = e1->get_dof_transformation_function<T>(true, true, false);
94 
95  // Generate cell permutations
96  mesh->topology_mutable().create_entity_permutations();
97  const std::vector<std::uint32_t>& cell_info
98  = mesh->topology().get_cell_permutation_info();
99 
100  // Create element kernel function
101  std::shared_ptr<const DofMap> dofmap0 = V0.dofmap();
102  assert(dofmap0);
103  std::shared_ptr<const DofMap> dofmap1 = V1.dofmap();
104  assert(dofmap1);
105 
106  // Build the element interpolation matrix
107  std::vector<T> A(e1->space_dimension() * ndofs0);
108  {
109  auto _A = xt::adapt(A, std::vector<int>{e1->space_dimension(), ndofs0});
110  const xt::xtensor<double, 2> Pi = e1->interpolation_operator();
111  math::dot(Pi, dphi_reshaped, _A);
112  }
113 
114  // Insert local interpolation matrix for each cell
115  auto cell_map = mesh->topology().index_map(tdim);
116  assert(cell_map);
117  std::int32_t num_cells = cell_map->size_local();
118  std::vector<T> Ae(A.size());
119  for (std::int32_t c = 0; c < num_cells; ++c)
120  {
121  std::copy(A.cbegin(), A.cend(), Ae.begin());
122  apply_inverse_dof_transform(Ae, cell_info, c, ndofs0);
123  mat_set(dofmap1->cell_dofs(c), dofmap0->cell_dofs(c), Ae);
124  }
125 }
