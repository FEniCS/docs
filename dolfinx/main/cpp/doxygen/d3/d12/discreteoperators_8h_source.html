85template <std::floating_point T, dolfinx::scalar U = T>
86void discrete_curl(const FunctionSpace<T>& V0, const FunctionSpace<T>& V1,
87 la::MatSet<U> auto&& mat_set)
88{
89 // Get mesh
90 auto mesh = V0.mesh();
91 assert(mesh);
92 assert(V1.mesh());
93 if (mesh != V1.mesh())
94 throw std::runtime_error("Meshes must be the same.");
95
96 if (mesh->geometry().dim() != 3)
97 throw std::runtime_error("Geometric must be equal to 3..");
98 if (mesh->geometry().dim() != mesh->topology()->dim())
99 {
100 throw std::runtime_error(
101 "Geometric and topological dimensions must be equal.");
102 }
103 constexpr int gdim = 3;
104
105 // Get elements
106 std::shared_ptr<const FiniteElement<T>> e0 = V0.element();
107 assert(e0);
108 if (e0->map_type() != basix::maps::type::covariantPiola)
109 {
110 throw std::runtime_error(
111 "Finite element for parent space must be covariant Piola.");
112 }
113
114 std::shared_ptr<const FiniteElement<T>> e1 = V1.element();
115 assert(e1);
116 if (e1->map_type() != basix::maps::type::contravariantPiola)
117 {
118 throw std::runtime_error(
119 "Finite element for target space must be contracovariant Piola.");
120 }
121
122 // Get cell orientation information
123 std::span<const std::uint32_t> cell_info;
124 if (e1->needs_dof_transformations() or e0->needs_dof_transformations())
125 {
126 mesh->topology_mutable()->create_entity_permutations();
127 cell_info = std::span(mesh->topology()->get_cell_permutation_info());
128 }
129
130 // Get dofmaps
131 auto dofmap0 = V0.dofmap();
132 assert(dofmap0);
133 auto dofmap1 = V1.dofmap();
134 assert(dofmap1);
135
136 // Get dof transformation operators
137 auto apply_dof_transformation0
138 = e0->template dof_transformation_fn<T>(doftransform::standard, false);
139 auto apply_inverse_dof_transform1 = e1->template dof_transformation_fn<U>(
140 doftransform::inverse_transpose, false);
141
142 // Get sizes of elements
143 const std::size_t space_dim0 = e0->space_dimension();
144 const std::size_t space_dim1 = e1->space_dimension();
145 if (e0->reference_value_size() != 3)
146 throw std::runtime_error("Value size for parent space should be 3.");
147 if (e1->reference_value_size() != 3)
148 throw std::runtime_error("Value size for target space should be 3.");
149
150 // Get the V1 reference interpolation points
151 const auto [X, Xshape] = e1->interpolation_points();
152
153 // Get/compute geometry map and evaluate at interpolation points
154 const CoordinateElement<T>& cmap = mesh->geometry().cmap();
155 auto x_dofmap = mesh->geometry().dofmap();
156 const std::size_t num_dofs_g = cmap.dim();
157 std::span<const T> x_g = mesh->geometry().x();
158 std::array<std::size_t, 4> Phi_g_shape = cmap.tabulate_shape(1, Xshape[0]);
159 std::vector<T> Phi_g_b(std::reduce(Phi_g_shape.begin(), Phi_g_shape.end(), 1,
160 std::multiplies{}));
161 // (derivative, point index, phi, component)
162 md::mdspan<const T, md::extents<std::size_t, gdim + 1, md::dynamic_extent,
163 md::dynamic_extent, 1>>
164 Phi_g(Phi_g_b.data(), Phi_g_shape);
165 cmap.tabulate(1, X, Xshape, Phi_g_b);
166
167 // Geometry data structures
168 std::vector<T> coord_dofs_b(num_dofs_g * gdim);
169 md::mdspan<T, md::extents<std::size_t, md::dynamic_extent, 3>> coord_dofs(
170 coord_dofs_b.data(), num_dofs_g, gdim);
171 std::vector<T> J_b(Xshape[0] * gdim * gdim);
172 md::mdspan<T, md::extents<std::size_t, md::dynamic_extent, 3, 3>> J(
173 J_b.data(), Xshape[0], gdim, gdim);
174 std::vector<T> K_b(Xshape[0] * gdim * gdim);
175 md::mdspan<T, typename decltype(J)::extents_type> K(K_b.data(), J.extents());
176 std::vector<T> detJ(Xshape[0]);
177 std::vector<T> det_scratch(2 * gdim * gdim);
178
179 // Evaluate V0 basis function derivatives at reference interpolation
180 // points for V1, (deriv, pt_idx, phi (dof), comp)
181 const auto [Phi0_b, Phi0_shape] = e0->tabulate(X, Xshape, 1);
182 md::mdspan<const T, md::extents<std::size_t, 4, md::dynamic_extent,
183 md::dynamic_extent, 3>>
184 Phi0(Phi0_b.data(), Phi0_shape);
185
186 // Create working arrays, (point, phi (dof), deriv, comp)
187 md::extents<std::size_t, md::dynamic_extent, md::dynamic_extent, 3, 3>
188 dPhi0_ext(Phi0.extent(1), Phi0.extent(2), Phi0.extent(0) - 1,
189 Phi0.extent(3));
190 std::vector<T> dPhi0_b(dPhi0_ext.extent(0) * dPhi0_ext.extent(1)
191 * dPhi0_ext.extent(2) * dPhi0_ext.extent(3));
192 md::mdspan<T, decltype(dPhi0_ext)> dPhi0(dPhi0_b.data(), dPhi0_ext);
193
194 // Get the interpolation operator (matrix) Pi that maps a function
195 // evaluated at the interpolation points to the V1 element degrees of
196 // freedom, i.e. dofs = Pi f_x
197 const auto [Pi1_b, pi_shape] = e1->interpolation_operator();
198 md::mdspan<const T, md::dextents<std::size_t, 2>> Pi_1(Pi1_b.data(),
199 pi_shape);
200
201 // curl data structure, (pt_idx, phi (dof), comp)
202 std::vector<T> curl_b(dPhi0.extent(0) * dPhi0.extent(1) * dPhi0.extent(3));
203 md::mdspan<
204 T, md::extents<std::size_t, md::dynamic_extent, md::dynamic_extent, 3>>
205 curl(curl_b.data(), dPhi0.extent(0), dPhi0.extent(1), dPhi0.extent(3));
206
207 std::vector<U> Ab(space_dim0 * space_dim1);
208 std::vector<U> local1(space_dim1);
209
210 // Iterate over mesh and interpolate on each cell
211 assert(mesh->topology()->index_map(gdim));
212 for (std::int32_t c = 0; c < mesh->topology()->index_map(gdim)->size_local();
213 ++c)
214 {
215 // Get cell geometry (coordinate dofs)
216 auto x_dofs = md::submdspan(x_dofmap, c, md::full_extent);
217 for (std::size_t i = 0; i < x_dofs.size(); ++i)
218 for (std::size_t j = 0; j < gdim; ++j)
219 coord_dofs(i, j) = x_g[3 * x_dofs[i] + j];
220
221 // Compute Jacobians at reference points for current cell
222 std::ranges::fill(J_b, 0);
223 for (std::size_t p = 0; p < Xshape[0]; ++p)
224 {
225 auto dPhi_g
226 = md::submdspan(Phi_g, std::pair(1, gdim + 1), p, md::full_extent, 0);
227 auto _J = md::submdspan(J, p, md::full_extent, md::full_extent);
228 cmap.compute_jacobian(dPhi_g, coord_dofs, _J);
229 auto _K = md::submdspan(K, p, md::full_extent, md::full_extent);
230 cmap.compute_jacobian_inverse(_J, _K);
231 detJ[p] = cmap.compute_jacobian_determinant(_J, det_scratch);
232 }
233
234 // TODO: re-order loops and/or re-pack Phi0 to allow a simple flat
235 // copy?
236
237 // Copy (d)Phi0 (on reference) and apply DOF transformation
238 // Phi0: (deriv, pt_idx, phi (dof), comp)
239 // dPhi0: (pt_idx, phi (dov), deriv, comp)
240 for (std::size_t p = 0; p < Phi0.extent(1); ++p) // point
241 for (std::size_t phi = 0; phi < Phi0.extent(2); ++phi) // phi_i
242 for (std::size_t d = 0; d < Phi0.extent(3); ++d) // Comp. of phi
243 for (std::size_t dx = 0; dx < dPhi0.extent(2); ++dx) // dx
244 dPhi0(p, phi, dx, d) = Phi0(dx + 1, p, phi, d);
245
246 for (std::size_t p = 0; p < dPhi0.extent(0); ++p) // point
247 {
248 // Size: num_phi * num_derivs * num_components
249 std::size_t size = dPhi0.extent(1) * dPhi0.extent(2) * dPhi0.extent(3);
250 std::size_t offset = p * size; // Offset for point p
251
252 // Shape: (num_phi , (value_size * num_derivs))
253 apply_dof_transformation0(std::span(dPhi0.data_handle() + offset, size),
254 cell_info, c,
255 dPhi0.extent(2) * dPhi0.extent(3));
256 }
257
258 // Compute curl
259 // dPhi0: (pt_idx, phi_idx, deriv, comp)
260 // curl: (pt_idx, phi_idx, comp)
261 for (std::size_t p = 0; p < curl.extent(0); ++p) // point
262 {
263 for (std::size_t i = 0; i < curl.extent(1); ++i) // phi_i
264 {
265 curl(p, i, 0) = dPhi0(p, i, 1, 2) - dPhi0(p, i, 2, 1);
266 curl(p, i, 1) = dPhi0(p, i, 2, 0) - dPhi0(p, i, 0, 2);
267 curl(p, i, 2) = dPhi0(p, i, 0, 1) - dPhi0(p, i, 1, 0);
268 }
269 }
270
271 // Apply interpolation matrix to basis derivatives values of V0 at
272 // the interpolation points of V1
273 for (std::size_t i = 0; i < curl.extent(1); ++i)
274 {
275 auto values = md::submdspan(curl, md::full_extent, i, md::full_extent);
276 impl::interpolation_apply(Pi_1, values, std::span(local1), 1);
277 for (std::size_t j = 0; j < local1.size(); ++j)
278 Ab[space_dim0 * j + i] = local1[j];
279 }
280
281 apply_inverse_dof_transform1(Ab, cell_info, c, space_dim0);
282 mat_set(dofmap1->cell_dofs(c), dofmap0->cell_dofs(c), Ab);
283 }
284}

312template <dolfinx::scalar T,
313 std::floating_point U = dolfinx::scalar_value_t<T>>
314void discrete_gradient(mesh::Topology& topology,
315 std::pair<std::reference_wrapper<const FiniteElement<U>>,
316 std::reference_wrapper<const DofMap>>
317 V0,
318 std::pair<std::reference_wrapper<const FiniteElement<U>>,
319 std::reference_wrapper<const DofMap>>
320 V1,
321 auto&& mat_set)
322{
323 auto& e0 = V0.first.get();
324 const DofMap& dofmap0 = V0.second.get();
325 auto& e1 = V1.first.get();
326 const DofMap& dofmap1 = V1.second.get();
327
328 using cmdspan2_t = md::mdspan<const U, md::dextents<std::size_t, 2>>;
329 using mdspan2_t = md::mdspan<U, md::dextents<std::size_t, 2>>;
330 using cmdspan4_t = md::mdspan<const U, md::dextents<std::size_t, 4>>;
331
332 // Check elements
333 if (e0.map_type() != basix::maps::type::identity)
334 throw std::runtime_error("Wrong finite element space for V0.");
335 if (e0.block_size() != 1)
336 throw std::runtime_error("Block size is greater than 1 for V0.");
337 if (e0.reference_value_size() != 1)
338 throw std::runtime_error("Wrong value size for V0.");
339
340 if (e1.map_type() != basix::maps::type::covariantPiola)
341 throw std::runtime_error("Wrong finite element space for V1.");
342 if (e1.block_size() != 1)
343 throw std::runtime_error("Block size is greater than 1 for V1.");
344
345 // Get V1 (H(curl)) space interpolation points
346 const auto [X, Xshape] = e1.interpolation_points();
347
348 // Tabulate first derivatives of Lagrange space at V1 interpolation
349 // points
350 const int ndofs0 = e0.space_dimension();
351 const int tdim = topology.dim();
352 std::vector<U> phi0_b((tdim + 1) * Xshape[0] * ndofs0 * 1);
353 cmdspan4_t phi0(phi0_b.data(), tdim + 1, Xshape[0], ndofs0, 1);
354 e0.tabulate(phi0_b, X, Xshape, 1);
355
356 // Reshape Lagrange basis derivatives as a matrix of shape (tdim *
357 // num_points, num_dofs_per_cell)
358 cmdspan2_t dphi_reshaped(
359 phi0_b.data() + phi0.extent(3) * phi0.extent(2) * phi0.extent(1),
360 tdim * phi0.extent(1), phi0.extent(2));
361
362 // Get inverse DOF transform function
363 auto apply_inverse_dof_transform = e1.template dof_transformation_fn<T>(
364 doftransform::inverse_transpose, false);
365
366 // Generate cell permutations
367 topology.create_entity_permutations();
368 const std::vector<std::uint32_t>& cell_info
369 = topology.get_cell_permutation_info();
370
371 // Create element kernel function
372
373 // Build the element interpolation matrix
374 std::vector<T> Ab(e1.space_dimension() * ndofs0);
375 {
376 md::mdspan<T, md::dextents<std::size_t, 2>> A(Ab.data(),
377 e1.space_dimension(), ndofs0);
378 const auto [Pi, shape] = e1.interpolation_operator();
379 cmdspan2_t _Pi(Pi.data(), shape);
380 math::dot(_Pi, dphi_reshaped, A);
381 }
382
383 // Insert local interpolation matrix for each cell
384 auto cell_map = topology.index_map(tdim);
385 assert(cell_map);
386 std::int32_t num_cells = cell_map->size_local();
387 std::vector<T> Ae(Ab.size());
388 for (std::int32_t c = 0; c < num_cells; ++c)
389 {
390 std::ranges::copy(Ab, Ae.begin());
391 apply_inverse_dof_transform(Ae, cell_info, c, ndofs0);
392 mat_set(dofmap1.cell_dofs(c), dofmap0.cell_dofs(c), Ae);
393 }
394}

411template <dolfinx::scalar T, std::floating_point U>
412void interpolation_matrix(const FunctionSpace<U>& V0,
413 const FunctionSpace<U>& V1, auto&& mat_set)
414{
415 // Get mesh
416 auto mesh = V0.mesh();
417 assert(mesh);
418
419 // Mesh dims
420 const int tdim = mesh->topology()->dim();
421 const int gdim = mesh->geometry().dim();
422
423 // Get elements
424 std::shared_ptr<const FiniteElement<U>> e0 = V0.element();
425 assert(e0);
426 std::shared_ptr<const FiniteElement<U>> e1 = V1.element();
427 assert(e1);
428
429 std::span<const std::uint32_t> cell_info;
430 if (e1->needs_dof_transformations() or e0->needs_dof_transformations())
431 {
432 mesh->topology_mutable()->create_entity_permutations();
433 cell_info = std::span(mesh->topology()->get_cell_permutation_info());
434 }
435
436 // Get dofmaps
437 auto dofmap0 = V0.dofmap();
438 assert(dofmap0);
439 auto dofmap1 = V1.dofmap();
440 assert(dofmap1);
441
442 // Get block sizes and dof transformation operators
443 const int bs0 = e0->block_size();
444 const int bs1 = e1->block_size();
445 auto apply_dof_transformation0
446 = e0->template dof_transformation_fn<U>(doftransform::standard, false);
447 auto apply_inverse_dof_transform1 = e1->template dof_transformation_fn<T>(
448 doftransform::inverse_transpose, false);
449
450 // Get sizes of elements
451 const std::size_t space_dim0 = e0->space_dimension();
452 const std::size_t space_dim1 = e1->space_dimension();
453 const std::size_t dim0 = space_dim0 / bs0;
454 const std::size_t value_size_ref0 = e0->reference_value_size();
455 const std::size_t value_size0 = V0.element()->reference_value_size();
456 const std::size_t value_size1 = V1.element()->reference_value_size();
457
458 // Get geometry data
459 const CoordinateElement<U>& cmap = mesh->geometry().cmap();
460 auto x_dofmap = mesh->geometry().dofmap();
461 const std::size_t num_dofs_g = cmap.dim();
462 std::span<const U> x_g = mesh->geometry().x();
463
464 using mdspan2_t = md::mdspan<U, md::dextents<std::size_t, 2>>;
465 using cmdspan2_t = md::mdspan<const U, md::dextents<std::size_t, 2>>;
466 using cmdspan3_t = md::mdspan<const U, md::dextents<std::size_t, 3>>;
467 using cmdspan4_t = md::mdspan<const U, md::dextents<std::size_t, 4>>;
468 using mdspan3_t = md::mdspan<U, md::dextents<std::size_t, 3>>;
469
470 // Evaluate coordinate map basis at reference interpolation points
471 const auto [X, Xshape] = e1->interpolation_points();
472 std::array<std::size_t, 4> phi_shape = cmap.tabulate_shape(1, Xshape[0]);
473 std::vector<U> phi_b(
474 std::reduce(phi_shape.begin(), phi_shape.end(), 1, std::multiplies{}));
475 cmdspan4_t phi(phi_b.data(), phi_shape);
476 cmap.tabulate(1, X, Xshape, phi_b);
477
478 // Evaluate V0 basis functions at reference interpolation points for V1
479 std::vector<U> basis_derivatives_reference0_b(Xshape[0] * dim0
480 * value_size_ref0);
481 cmdspan4_t basis_derivatives_reference0(basis_derivatives_reference0_b.data(),
482 1, Xshape[0], dim0, value_size_ref0);
483 e0->tabulate(basis_derivatives_reference0_b, X, Xshape, 0);
484
485 // Clamp values
486 std::ranges::transform(
487 basis_derivatives_reference0_b, basis_derivatives_reference0_b.begin(),
488 [atol = 1e-14](auto x) { return std::abs(x) < atol ? 0.0 : x; });
489
490 // Create working arrays
491 std::vector<U> basis_reference0_b(Xshape[0] * dim0 * value_size_ref0);
492 mdspan3_t basis_reference0(basis_reference0_b.data(), Xshape[0], dim0,
493 value_size_ref0);
494 std::vector<U> J_b(Xshape[0] * gdim * tdim);
495 mdspan3_t J(J_b.data(), Xshape[0], gdim, tdim);
496 std::vector<U> K_b(Xshape[0] * tdim * gdim);
497 mdspan3_t K(K_b.data(), Xshape[0], tdim, gdim);
498 std::vector<U> detJ(Xshape[0]);
499 std::vector<U> det_scratch(2 * tdim * gdim);
500
501 // Get the interpolation operator (matrix) `Pi` that maps a function
502 // evaluated at the interpolation points to the element degrees of
503 // freedom, i.e. dofs = Pi f_x
504 const auto [_Pi_1, pi_shape] = e1->interpolation_operator();
505 cmdspan2_t Pi_1(_Pi_1.data(), pi_shape);
506
507 bool interpolation_ident = e1->interpolation_ident();
508
509 using u_t = md::mdspan<U, md::dextents<std::size_t, 2>>;
510 using U_t = md::mdspan<const U, md::dextents<std::size_t, 2>>;
511 using J_t = md::mdspan<const U, md::dextents<std::size_t, 2>>;
512 using K_t = md::mdspan<const U, md::dextents<std::size_t, 2>>;
513 auto push_forward_fn0
514 = e0->basix_element().template map_fn<u_t, U_t, J_t, K_t>();
515
516 // Basis values of Lagrange space unrolled for block size
517 // (num_quadrature_points, Lagrange dof, value_size)
518 std::vector<U> basis_values_b(Xshape[0] * bs0 * dim0
519 * V1.element()->value_size());
520 mdspan3_t basis_values(basis_values_b.data(), Xshape[0], bs0 * dim0,
521 V1.element()->value_size());
522 std::vector<U> mapped_values_b(Xshape[0] * bs0 * dim0
523 * V1.element()->value_size());
524 mdspan3_t mapped_values(mapped_values_b.data(), Xshape[0], bs0 * dim0,
525 V1.element()->value_size());
526
527 auto pull_back_fn1
528 = e1->basix_element().template map_fn<u_t, U_t, K_t, J_t>();
529
530 std::vector<U> coord_dofs_b(num_dofs_g * gdim);
531 mdspan2_t coord_dofs(coord_dofs_b.data(), num_dofs_g, gdim);
532 std::vector<U> basis0_b(Xshape[0] * dim0 * value_size0);
533 mdspan3_t basis0(basis0_b.data(), Xshape[0], dim0, value_size0);
534
535 // Buffers
536 std::vector<T> Ab(space_dim0 * space_dim1);
537 std::vector<T> local1(space_dim1);
538
539 // Iterate over mesh and interpolate on each cell
540 auto cell_map = mesh->topology()->index_map(tdim);
541 assert(cell_map);
542 std::int32_t num_cells = cell_map->size_local();
543 for (std::int32_t c = 0; c < num_cells; ++c)
544 {
545 // Get cell geometry (coordinate dofs)
546 auto x_dofs = md::submdspan(x_dofmap, c, md::full_extent);
547 for (std::size_t i = 0; i < x_dofs.size(); ++i)
548 {
549 for (std::size_t j = 0; j < gdim; ++j)
550 coord_dofs(i, j) = x_g[3 * x_dofs[i] + j];
551 }
552
553 // Compute Jacobians and reference points for current cell
554 std::ranges::fill(J_b, 0);
555 for (std::size_t p = 0; p < Xshape[0]; ++p)
556 {
557 auto dphi
558 = md::submdspan(phi, std::pair(1, tdim + 1), p, md::full_extent, 0);
559 auto _J = md::submdspan(J, p, md::full_extent, md::full_extent);
560 cmap.compute_jacobian(dphi, coord_dofs, _J);
561 auto _K = md::submdspan(K, p, md::full_extent, md::full_extent);
562 cmap.compute_jacobian_inverse(_J, _K);
563 detJ[p] = cmap.compute_jacobian_determinant(_J, det_scratch);
564 }
565
566 // Copy evaluated basis on reference, apply DOF transformations, and
567 // push forward to physical element
568 for (std::size_t k0 = 0; k0 < basis_reference0.extent(0); ++k0)
569 for (std::size_t k1 = 0; k1 < basis_reference0.extent(1); ++k1)
570 for (std::size_t k2 = 0; k2 < basis_reference0.extent(2); ++k2)
571 basis_reference0(k0, k1, k2)
572 = basis_derivatives_reference0(0, k0, k1, k2);
573 for (std::size_t p = 0; p < Xshape[0]; ++p)
574 {
575 apply_dof_transformation0(
576 std::span(basis_reference0.data_handle() + p * dim0 * value_size_ref0,
577 dim0 * value_size_ref0),
578 cell_info, c, value_size_ref0);
579 }
580
581 for (std::size_t p = 0; p < basis0.extent(0); ++p)
582 {
583 auto _u = md::submdspan(basis0, p, md::full_extent, md::full_extent);
584 auto _U = md::submdspan(basis_reference0, p, md::full_extent,
585 md::full_extent);
586 auto _K = md::submdspan(K, p, md::full_extent, md::full_extent);
587 auto _J = md::submdspan(J, p, md::full_extent, md::full_extent);
588 push_forward_fn0(_u, _U, _J, detJ[p], _K);
589 }
590
591 // Unroll basis function for input space for block size
592 for (std::size_t p = 0; p < Xshape[0]; ++p)
593 for (std::size_t i = 0; i < dim0; ++i)
594 for (std::size_t j = 0; j < value_size0; ++j)
595 for (int k = 0; k < bs0; ++k)
596 basis_values(p, i * bs0 + k, j * bs0 + k) = basis0(p, i, j);
597
598 // Pull back the physical values to the reference of output space
599 for (std::size_t p = 0; p < basis_values.extent(0); ++p)
600 {
601 auto _u
602 = md::submdspan(basis_values, p, md::full_extent, md::full_extent);
603 auto _U
604 = md::submdspan(mapped_values, p, md::full_extent, md::full_extent);
605 auto _K = md::submdspan(K, p, md::full_extent, md::full_extent);
606 auto _J = md::submdspan(J, p, md::full_extent, md::full_extent);
607 pull_back_fn1(_U, _u, _K, 1.0 / detJ[p], _J);
608 }
609
610 // Apply interpolation matrix to basis values of V0 at the
611 // interpolation points of V1
612 if (interpolation_ident)
613 {
614 md::mdspan<T, md::dextents<std::size_t, 3>> A(
615 Ab.data(), Xshape[0], V1.element()->value_size(), space_dim0);
616 for (std::size_t i = 0; i < mapped_values.extent(0); ++i)
617 for (std::size_t j = 0; j < mapped_values.extent(1); ++j)
618 for (std::size_t k = 0; k < mapped_values.extent(2); ++k)
619 A(i, k, j) = mapped_values(i, j, k);
620 }
621 else
622 {
623 for (std::size_t i = 0; i < mapped_values.extent(1); ++i)
624 {
625 auto values
626 = md::submdspan(mapped_values, md::full_extent, i, md::full_extent);
627 impl::interpolation_apply(Pi_1, values, std::span(local1), bs1);
628 for (std::size_t j = 0; j < local1.size(); j++)
629 Ab[space_dim0 * j + i] = local1[j];
630 }
631 }
632
633 apply_inverse_dof_transform1(Ab, cell_info, c, space_dim0);
634 mat_set(dofmap1->cell_dofs(c), dofmap0->cell_dofs(c), Ab);
635 }
636}
