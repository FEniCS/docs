152  template <typename T>
153  void scatter_fwd(const std::span<const T>& local_data,
154  std::span<T> remote_data) const
155  {
156  MPI_Request request;
157  std::vector<T> local_buffer(local_buffer_size(), 0);
158  std::vector<T> remote_buffer(remote_buffer_size(), 0);
159  auto pack_fn = [](const auto& in, const auto& idx, auto& out)
160  {
161  for (std::size_t i = 0; i < idx.size(); ++i)
162  out[i] = in[idx[i]];
163  };
164  scatter_fwd_begin(local_data, std::span<T>(local_buffer),
165  std::span<T>(remote_buffer), pack_fn, request);
166 
167  auto unpack_fn = [](const auto& in, const auto& idx, auto& out, auto op)
168  {
169  for (std::size_t i = 0; i < idx.size(); ++i)
170  out[idx[i]] = op(out[idx[i]], in[i]);
171  };

173  scatter_fwd_end(std::span<const T>(remote_buffer), remote_data, unpack_fn,
174  request);

299  template <typename T, typename BinaryOp>
300  void scatter_rev(std::span<T> local_data,
301  const std::span<const T>& remote_data, BinaryOp op)
302  {
303  std::vector<T> local_buffer(local_buffer_size(), 0);
304  std::vector<T> remote_buffer(remote_buffer_size(), 0);
305  auto pack_fn = [](const auto& in, const auto& idx, auto& out)
306  {
307  for (std::size_t i = 0; i < idx.size(); ++i)
308  out[i] = in[idx[i]];
309  };
310  auto unpack_fn = [](const auto& in, const auto& idx, auto& out, auto op)
311  {
312  for (std::size_t i = 0; i < idx.size(); ++i)
313  out[idx[i]] = op(out[idx[i]], in[i]);
314  };
315  MPI_Request request;
316  scatter_rev_begin(remote_data, std::span<T>(remote_buffer),
317  std::span<T>(local_buffer), pack_fn, request);
318  scatter_rev_end(std::span<const T>(local_buffer), local_data, unpack_fn, op,
319  request);
320  }
