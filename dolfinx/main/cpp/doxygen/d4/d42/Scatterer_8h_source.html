55 template <typename T>
56 void scatter_fwd_begin(std::span<const T> send_buffer,
57 std::span<T> recv_buffer, MPI_Request& request) const
58 {
59 // Return early if there are no incoming or outgoing edges
60 if (_sizes_local.empty() and _sizes_remote.empty())
61 return;
62
63 int err = MPI_Ineighbor_alltoallv(
64 send_buffer.data(), _sizes_local.data(), _displs_local.data(),
65 MPI::mpi_type<T>(), recv_buffer.data(), _sizes_remote.data(),
66 _displs_remote.data(), MPI::mpi_type<T>(), _comm0.comm(), &request);
67 dolfinx::MPI::check_error(_comm0.comm(), err);
68 }

151 template <typename T>
152 void scatter_fwd(std::span<const T> local_data,
153 std::span<T> remote_data) const
154 {
155 MPI_Request request;
156 std::vector<T> local_buffer(local_buffer_size(), 0);
157 std::vector<T> remote_buffer(remote_buffer_size(), 0);
158 auto pack_fn = [](const auto& in, const auto& idx, auto& out)
159 {
160 for (std::size_t i = 0; i < idx.size(); ++i)
161 out[i] = in[idx[i]];
162 };
163 scatter_fwd_begin(local_data, std::span<T>(local_buffer),
164 std::span<T>(remote_buffer), pack_fn, request);
165
166 auto unpack_fn = [](const auto& in, const auto& idx, auto& out, auto op)
167 {
168 for (std::size_t i = 0; i < idx.size(); ++i)
169 out[idx[i]] = op(out[idx[i]], in[i]);
170 };
171
172 scatter_fwd_end(std::span<const T>(remote_buffer), remote_data, unpack_fn,
173 request);
174 }

282 template <typename T, typename Functor, typename BinaryOp>
283 void scatter_rev_end(std::span<const T> local_buffer, std::span<T> local_data,
284 Functor unpack_fn, BinaryOp op, MPI_Request& request)

286 assert(local_buffer.size() == _local_inds.size());

289 < std::int32_t(local_data.size()));
290 scatter_rev_end(request);
291 unpack_fn(local_buffer, _local_inds, local_data, op);

296 template <typename T, typename BinaryOp>
297 void scatter_rev(std::span<T> local_data, std::span<const T> remote_data,
298 BinaryOp op)
299 {
300 std::vector<T> local_buffer(local_buffer_size(), 0);
301 std::vector<T> remote_buffer(remote_buffer_size(), 0);
302 auto pack_fn = [](const auto& in, const auto& idx, auto& out)
303 {
304 for (std::size_t i = 0; i < idx.size(); ++i)
305 out[i] = in[idx[i]];
306 };
307 auto unpack_fn = [](const auto& in, const auto& idx, auto& out, auto op)
308 {
309 for (std::size_t i = 0; i < idx.size(); ++i)
310 out[idx[i]] = op(out[idx[i]], in[i]);
311 };
312 MPI_Request request;
313 scatter_rev_begin(remote_data, std::span<T>(remote_buffer),
314 std::span<T>(local_buffer), pack_fn, request);
315 scatter_rev_end(std::span<const T>(local_buffer), local_data, unpack_fn, op,
316 request);
317 }

347 // Communicator where the source ranks own the indices in the callers
348 // halo, and the destination ranks 'ghost' indices owned by the
349 // caller. I.e.,
350 // - in-edges (src) are from ranks that own my ghosts
351 // - out-edges (dest) go to ranks that 'ghost' my owned indices
352 dolfinx::MPI::Comm _comm0;
353
354 // Communicator where the source ranks have ghost indices that are
355 // owned by the caller, and the destination ranks are the owners of
356 // indices in the callers halo region. I.e.,
357 // - in-edges (src) are from ranks that 'ghost' my owned indices
358 // - out-edges (dest) are to the owning ranks of my ghost indices
359 dolfinx::MPI::Comm _comm1;
360
361 // Permutation indices used to pack and unpack ghost data (remote)
362 std::vector<std::int32_t> _remote_inds;
363
364 // Number of remote indices (ghosts) for each neighbor process
365 std::vector<int> _sizes_remote;
366
367 // Displacements of remote data for mpi scatter and gather
368 std::vector<int> _displs_remote;
369
370 // Permutation indices used to pack and unpack local shared data
371 // (owned indices that are shared with other processes). Indices are
372 // grouped by neighbor process.
373 std::vector<std::int32_t> _local_inds;
374
375 // Number of local shared indices per neighbor process
376 std::vector<int> _sizes_local;
377
378 // Displacements of local data for mpi scatter and gather
