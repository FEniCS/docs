29template <typename T>
30std::map<std::pair<IntegralType, int>, std::pair<std::span<const T>, int>>
31make_coefficients_span(const std::map<std::pair<IntegralType, int>,
32 std::pair<std::vector<T>, int>>& coeffs)
33{
34 using Key = typename std::remove_reference_t<decltype(coeffs)>::key_type;
35 std::map<Key, std::pair<std::span<const T>, int>> c;
36 std::transform(coeffs.cbegin(), coeffs.cend(), std::inserter(c, c.end()),
37 [](auto& e) -> typename decltype(c)::value_type {
38 return {e.first, {e.second.first, e.second.second}};
39 });
40 return c;
41}

56template <typename T>
57T assemble_scalar(
58 const Form<T>& M, std::span<const T> constants,
59 const std::map<std::pair<IntegralType, int>,
60 std::pair<std::span<const T>, int>>& coefficients)
61{
62 std::shared_ptr<const mesh::Mesh> mesh = M.mesh();
63 assert(mesh);
64 if constexpr (std::is_same_v<double, impl::scalar_value_type_t<T>>)
65 return impl::assemble_scalar(M, mesh->geometry(), constants, coefficients);
66 else
67 {
68 return impl::assemble_scalar(
69 M, mesh->geometry().astype<impl::scalar_value_type_t<T>>(), constants,
70 coefficients);
71 }
72}

142template <typename T>
143void apply_lifting(
144 std::span<T> b, const std::vector<std::shared_ptr<const Form<T>>>& a,
145 const std::vector<std::span<const T>>& constants,
146 const std::vector<std::map<std::pair<IntegralType, int>,
147 std::pair<std::span<const T>, int>>>& coeffs,
148 const std::vector<std::vector<std::shared_ptr<const DirichletBC<T>>>>& bcs1,
149 const std::vector<std::span<const T>>& x0, T scale)
150{
151 std::shared_ptr<const mesh::Mesh> mesh;
152 for (auto& a_i : a)
153 {
154 if (a_i and !mesh)
155 mesh = a_i->mesh();
156 if (a_i and mesh and a_i->mesh() != mesh)
157 throw std::runtime_error("Mismatch between meshes.");
158 }
159
160 if (!mesh)
161 throw std::runtime_error("Unable to extract a mesh.");
162
163 if constexpr (std::is_same_v<double, impl::scalar_value_type_t<T>>)
164 {
165 impl::apply_lifting<T>(b, a, mesh->geometry(), constants, coeffs, bcs1, x0,
166 scale);
167 }
168 else
169 {
170 impl::apply_lifting<T>(
171 b, a, mesh->geometry().astype<impl::scalar_value_type_t<T>>(),
172 constants, coeffs, bcs1, x0, scale);
173 }
174}

188template <typename T>
189void apply_lifting(
190 std::span<T> b, const std::vector<std::shared_ptr<const Form<T>>>& a,
191 const std::vector<std::vector<std::shared_ptr<const DirichletBC<T>>>>& bcs1,
192 const std::vector<std::span<const T>>& x0, T scale)
193{
194 std::vector<
195 std::map<std::pair<IntegralType, int>, std::pair<std::vector<T>, int>>>
196 coeffs;
197 std::vector<std::vector<T>> constants;
198 for (auto _a : a)
199 {
200 if (_a)
201 {
202 auto coefficients = allocate_coefficient_storage(*_a);
203 pack_coefficients(*_a, coefficients);
204 coeffs.push_back(coefficients);
205 constants.push_back(pack_constants(*_a));
206 }
207 else
208 {
209 coeffs.emplace_back();
210 constants.emplace_back();
211 }
212 }
213
214 std::vector<std::span<const T>> _constants(constants.begin(),
215 constants.end());
216 std::vector<std::map<std::pair<IntegralType, int>,
217 std::pair<std::span<const T>, int>>>
218 _coeffs;
219 std::transform(coeffs.cbegin(), coeffs.cend(), std::back_inserter(_coeffs),
220 [](auto& c) { return make_coefficients_span(c); });
221
222 apply_lifting(b, a, _constants, _coeffs, bcs1, x0, scale);
223}

239template <typename T>
240void assemble_matrix(
241 auto mat_add, const Form<T>& a, std::span<const T> constants,
242 const std::map<std::pair<IntegralType, int>,
243 std::pair<std::span<const T>, int>>& coefficients,
244 std::span<const std::int8_t> dof_marker0,
245 std::span<const std::int8_t> dof_marker1)
246
247{
248 std::shared_ptr<const mesh::Mesh> mesh = a.mesh();
249 assert(mesh);
250 if constexpr (std::is_same_v<double, impl::scalar_value_type_t<T>>)

253 dof_marker0, dof_marker1);
254 }
255 else
256 {
257 impl::assemble_matrix(

259 constants, coefficients, dof_marker0, dof_marker1);
260 }

270template <typename T>
271void assemble_matrix(
272 auto mat_add, const Form<T>& a, std::span<const T> constants,
273 const std::map<std::pair<IntegralType, int>,
274 std::pair<std::span<const T>, int>>& coefficients,
275 const std::vector<std::shared_ptr<const DirichletBC<T>>>& bcs)
276{
277 // Index maps for dof ranges
278 auto map0 = a.function_spaces().at(0)->dofmap()->index_map;
279 auto map1 = a.function_spaces().at(1)->dofmap()->index_map;
280 auto bs0 = a.function_spaces().at(0)->dofmap()->index_map_bs();
281 auto bs1 = a.function_spaces().at(1)->dofmap()->index_map_bs();
282
283 // Build dof markers
284 std::vector<std::int8_t> dof_marker0, dof_marker1;
285 assert(map0);
286 std::int32_t dim0 = bs0 * (map0->size_local() + map0->num_ghosts());
287 assert(map1);
288 std::int32_t dim1 = bs1 * (map1->size_local() + map1->num_ghosts());
289 for (std::size_t k = 0; k < bcs.size(); ++k)
290 {
291 assert(bcs[k]);
292 assert(bcs[k]->function_space());
293 if (a.function_spaces().at(0)->contains(*bcs[k]->function_space()))
294 {
295 dof_marker0.resize(dim0, false);
296 bcs[k]->mark_dofs(dof_marker0);
297 }
298
299 if (a.function_spaces().at(1)->contains(*bcs[k]->function_space()))
300 {
301 dof_marker1.resize(dim1, false);
302 bcs[k]->mark_dofs(dof_marker1);
303 }
304 }
305
306 // Assemble
307 assemble_matrix(mat_add, a, constants, coefficients, dof_marker0,
308 dof_marker1);
309}

371 T diagonal = 1.0)
372{
373 for (std::size_t i = 0; i < rows.size(); ++i)

396template <typename T>
397void set_diagonal(auto set_fn, const FunctionSpace& V,
398 const std::vector<std::shared_ptr<const DirichletBC<T>>>& bcs,
399 T diagonal = 1.0)
400{
401 for (const auto& bc : bcs)
402 {
403 assert(bc);
404 if (V.contains(*bc->function_space()))
405 {
406 const auto [dofs, range] = bc->dof_indices();
407 set_diagonal(set_fn, dofs.first(range), diagonal);
408 }
409 }
410}
