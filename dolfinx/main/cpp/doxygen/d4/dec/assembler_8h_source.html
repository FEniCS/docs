32template <typename T>
33std::map<std::pair<IntegralType, int>, std::pair<std::span<const T>, int>>
34make_coefficients_span(const std::map<std::pair<IntegralType, int>,
35 std::pair<std::vector<T>, int>>& coeffs)
36{
37 using Key = typename std::remove_reference_t<decltype(coeffs)>::key_type;
38 std::map<Key, std::pair<std::span<const T>, int>> c;
39 std::transform(coeffs.cbegin(), coeffs.cend(), std::inserter(c, c.end()),
40 [](auto& e) -> typename decltype(c)::value_type {
41 return {e.first, {e.second.first, e.second.second}};
42 });
43 return c;
44}

59template <typename T, std::floating_point U>
60T assemble_scalar(
61 const Form<T, U>& M, std::span<const T> constants,
62 const std::map<std::pair<IntegralType, int>,
63 std::pair<std::span<const T>, int>>& coefficients)
64{

66 assert(mesh);
67 if constexpr (std::is_same_v<U, scalar_value_type_t<T>>)
68 return impl::assemble_scalar(M, mesh->geometry(), constants, coefficients);

70 {
71 return impl::assemble_scalar(
72 M, mesh->geometry().template astype<scalar_value_type_t<T>>(),
73 constants, coefficients);
74 }
75}

145template <typename T, std::floating_point U>
146void apply_lifting(
147 std::span<T> b, const std::vector<std::shared_ptr<const Form<T, U>>>& a,
148 const std::vector<std::span<const T>>& constants,
149 const std::vector<std::map<std::pair<IntegralType, int>,
150 std::pair<std::span<const T>, int>>>& coeffs,
151 const std::vector<std::vector<std::shared_ptr<const DirichletBC<T, U>>>>&
152 bcs1,
153 const std::vector<std::span<const T>>& x0, T scale)
154{
155 std::shared_ptr<const mesh::Mesh<U>> mesh;
156 for (auto& a_i : a)
157 {
158 if (a_i and !mesh)
159 mesh = a_i->mesh();
160 if (a_i and mesh and a_i->mesh() != mesh)
161 throw std::runtime_error("Mismatch between meshes.");
162 }
163
164 if (!mesh)
165 throw std::runtime_error("Unable to extract a mesh.");
166
167 if constexpr (std::is_same_v<U, scalar_value_type_t<T>>)
168 {
169 impl::apply_lifting<T>(b, a, mesh->geometry(), constants, coeffs, bcs1, x0,
170 scale);
171 }
172 else
173 {
174 impl::apply_lifting<T>(
175 b, a, mesh->geometry().template astype<scalar_value_type_t<T>>(),
176 constants, coeffs, bcs1, x0, scale);
177 }
178}

192template <typename T, std::floating_point U>
193void apply_lifting(
194 std::span<T> b, const std::vector<std::shared_ptr<const Form<T, U>>>& a,
195 const std::vector<std::vector<std::shared_ptr<const DirichletBC<T, U>>>>&
196 bcs1,
197 const std::vector<std::span<const T>>& x0, T scale)
198{
199 std::vector<
200 std::map<std::pair<IntegralType, int>, std::pair<std::vector<T>, int>>>
201 coeffs;
202 std::vector<std::vector<T>> constants;
203 for (auto _a : a)
204 {
205 if (_a)
206 {
207 auto coefficients = allocate_coefficient_storage(*_a);
208 pack_coefficients(*_a, coefficients);
209 coeffs.push_back(coefficients);
210 constants.push_back(pack_constants(*_a));
211 }
212 else
213 {
214 coeffs.emplace_back();
215 constants.emplace_back();
216 }
217 }
218
219 std::vector<std::span<const T>> _constants(constants.begin(),
220 constants.end());
221 std::vector<std::map<std::pair<IntegralType, int>,
222 std::pair<std::span<const T>, int>>>
223 _coeffs;
224 std::transform(coeffs.cbegin(), coeffs.cend(), std::back_inserter(_coeffs),
225 [](auto& c) { return make_coefficients_span(c); });
226
227 apply_lifting(b, a, _constants, _coeffs, bcs1, x0, scale);
228}

247 std::span<const T> constants,
248 const std::map<std::pair<IntegralType, int>,
249 std::pair<std::span<const T>, int>>& coefficients,

276template <typename T, std::floating_point U>
277void assemble_matrix(
278 auto mat_add, const Form<T, U>& a, std::span<const T> constants,
279 const std::map<std::pair<IntegralType, int>,
280 std::pair<std::span<const T>, int>>& coefficients,
281 const std::vector<std::shared_ptr<const DirichletBC<T, U>>>& bcs)
282{
283 // Index maps for dof ranges
284 auto map0 = a.function_spaces().at(0)->dofmap()->index_map;
285 auto map1 = a.function_spaces().at(1)->dofmap()->index_map;
286 auto bs0 = a.function_spaces().at(0)->dofmap()->index_map_bs();
287 auto bs1 = a.function_spaces().at(1)->dofmap()->index_map_bs();
288
289 // Build dof markers
290 std::vector<std::int8_t> dof_marker0, dof_marker1;
291 assert(map0);
292 std::int32_t dim0 = bs0 * (map0->size_local() + map0->num_ghosts());
293 assert(map1);
294 std::int32_t dim1 = bs1 * (map1->size_local() + map1->num_ghosts());
295 for (std::size_t k = 0; k < bcs.size(); ++k)
296 {
297 assert(bcs[k]);
298 assert(bcs[k]->function_space());
299 if (a.function_spaces().at(0)->contains(*bcs[k]->function_space()))
300 {
301 dof_marker0.resize(dim0, false);
302 bcs[k]->mark_dofs(dof_marker0);
303 }
304
305 if (a.function_spaces().at(1)->contains(*bcs[k]->function_space()))
306 {
307 dof_marker1.resize(dim1, false);
308 bcs[k]->mark_dofs(dof_marker1);
309 }
310 }
311
312 // Assemble
313 assemble_matrix(mat_add, a, constants, coefficients, dof_marker0,
314 dof_marker1);
315}

375template <typename T>
376void set_diagonal(auto set_fn, std::span<const std::int32_t> rows,
377 T diagonal = 1.0)
378{
379 for (std::size_t i = 0; i < rows.size(); ++i)

402template <typename T, std::floating_point U>
403void set_diagonal(
404 auto set_fn, const FunctionSpace<U>& V,
405 const std::vector<std::shared_ptr<const DirichletBC<T, U>>>& bcs,
406 T diagonal = 1.0)
407{
408 for (const auto& bc : bcs)
409 {
410 assert(bc);
411 if (V.contains(*bc->function_space()))
412 {
413 const auto [dofs, range] = bc->dof_indices();
414 set_diagonal(set_fn, dofs.first(range), diagonal);
415 }
416 }
