96template <dolfinx::scalar T, std::floating_point U>
97void tabulate_expression(std::span<T> values, const fem::Expression<T, U>& e,
98 const mesh::Mesh<U>& mesh, fem::MDSpan2 auto entities)
99{
100 namespace md = MDSPAN_IMPL_STANDARD_NAMESPACE;
101
102 std::optional<
103 std::pair<std::reference_wrapper<const FiniteElement<U>>, std::size_t>>
104 element = std::nullopt;
105 if (auto V = e.argument_space(); V)
106 {
107 std::size_t num_argument_dofs
108 = V->dofmap()->element_dof_layout().num_dofs() * V->dofmap()->bs();
109 assert(V->element());
110 element = {std::cref(*V->element()), num_argument_dofs};
111 }
112
113 std::vector<int> coffsets = e.coefficient_offsets();
114 const std::vector<std::shared_ptr<const Function<T, U>>>& coefficients
115 = e.coefficients();
116 std::vector<T> coeffs(entities.extent(0) * coffsets.back());
117 int cstride = coffsets.back();
118 {
119 std::vector<std::reference_wrapper<const Function<T, U>>> c;
120 std::ranges::transform(coefficients, std::back_inserter(c),
121 [](auto c) -> const Function<T, U>& { return *c; });
122 fem::pack_coefficients(c, coffsets, entities, std::span(coeffs));
123 }
124 std::vector<T> constants = fem::pack_constants(e);
125
126 tabulate_expression<T, U>(
127 values, e, md::mdspan(coeffs.data(), entities.size(), cstride),
128 std::span<const T>(constants), mesh, entities, element);
129}

134template <dolfinx::scalar T>
135std::map<std::pair<IntegralType, int>, std::pair<std::span<const T>, int>>
136make_coefficients_span(const std::map<std::pair<IntegralType, int>,
137 std::pair<std::vector<T>, int>>& coeffs)
138{
139 using Key = typename std::remove_reference_t<decltype(coeffs)>::key_type;
140 std::map<Key, std::pair<std::span<const T>, int>> c;
141 std::ranges::transform(
142 coeffs, std::inserter(c, c.end()),
143 [](auto& e) -> typename decltype(c)::value_type
144 { return {e.first, {e.second.first, e.second.second}}; });
145 return c;
146}

161template <dolfinx::scalar T, std::floating_point U>
162T assemble_scalar(
163 const Form<T, U>& M, std::span<const T> constants,
164 const std::map<std::pair<IntegralType, int>,
165 std::pair<std::span<const T>, int>>& coefficients)
166{
167 namespace md = MDSPAN_IMPL_STANDARD_NAMESPACE;
168 using mdspanx3_t
169 = md::mdspan<const scalar_value_type_t<T>,
170 md::extents<std::size_t, md::dynamic_extent, 3>>;
171
172 std::shared_ptr<const mesh::Mesh<U>> mesh = M.mesh();
173 assert(mesh);
174 std::span x = mesh->geometry().x();
175 if constexpr (std::is_same_v<U, scalar_value_type_t<T>>)
176 {
177 return impl::assemble_scalar(M, mesh->geometry().dofmap(),
178 mdspanx3_t(x.data(), x.size() / 3, 3),
179 constants, coefficients);
180 }
181 else
182 {
183 std::vector<scalar_value_type_t<T>> _x(x.begin(), x.end());
184 return impl::assemble_scalar(M, mesh->geometry().dofmap(),
185 mdspanx3_t(_x.data(), _x.size() / 3, 3),
186 constants, coefficients);
187 }
188}

198T assemble_scalar(const Form<T, U>& M)

260template <dolfinx::scalar T, std::floating_point U>
261void apply_lifting(

290template <dolfinx::scalar T, std::floating_point U>
291void apply_lifting(
292 std::span<T> b,
293 std::vector<std::optional<std::reference_wrapper<const Form<T, U>>>> a,
294 const std::vector<
295 std::vector<std::reference_wrapper<const DirichletBC<T, U>>>>& bcs1,
296 const std::vector<std::span<const T>>& x0, T alpha)
297{
298 std::vector<
299 std::map<std::pair<IntegralType, int>, std::pair<std::vector<T>, int>>>
300 coeffs;
301 std::vector<std::vector<T>> constants;
302 for (auto _a : a)
303 {
304 if (_a)
305 {
306 auto coefficients = allocate_coefficient_storage(_a->get());
307 pack_coefficients(_a->get(), coefficients);
308 coeffs.push_back(coefficients);
309 constants.push_back(pack_constants(_a->get()));
310 }
311 else
312 {
313 coeffs.emplace_back();
314 constants.emplace_back();
315 }
316 }
317
318 std::vector<std::span<const T>> _constants(constants.begin(),
319 constants.end());
320 std::vector<std::map<std::pair<IntegralType, int>,
321 std::pair<std::span<const T>, int>>>
322 _coeffs;
323 std::ranges::transform(coeffs, std::back_inserter(_coeffs),
324 [](auto& c) { return make_coefficients_span(c); });
325
326 apply_lifting(b, a, _constants, _coeffs, bcs1, x0, alpha);
327}

343template <dolfinx::scalar T, std::floating_point U>
344void assemble_matrix(
345 la::MatSet<T> auto mat_add, const Form<T, U>& a,
346 std::span<const T> constants,
347 const std::map<std::pair<IntegralType, int>,
348 std::pair<std::span<const T>, int>>& coefficients,
349 std::span<const std::int8_t> dof_marker0,
350 std::span<const std::int8_t> dof_marker1)
351
352{
353 namespace md = MDSPAN_IMPL_STANDARD_NAMESPACE;
354 using mdspanx3_t
355 = md::mdspan<const scalar_value_type_t<T>,
356 md::extents<std::size_t, md::dynamic_extent, 3>>;
357
358 std::shared_ptr<const mesh::Mesh<U>> mesh = a.mesh();
359 assert(mesh);
360 std::span x = mesh->geometry().x();
361 if constexpr (std::is_same_v<U, scalar_value_type_t<T>>)
362 {
363 impl::assemble_matrix(mat_add, a, mdspanx3_t(x.data(), x.size() / 3, 3),
364 constants, coefficients, dof_marker0, dof_marker1);
365 }
366 else
367 {
368 std::vector<scalar_value_type_t<T>> _x(x.begin(), x.end());
369 impl::assemble_matrix(mat_add, a, mdspanx3_t(_x.data(), _x.size() / 3, 3),
370 constants, coefficients, dof_marker0, dof_marker1);
371 }

381template <dolfinx::scalar T, std::floating_point U>
382void assemble_matrix(
383 auto mat_add, const Form<T, U>& a, std::span<const T> constants,
384 const std::map<std::pair<IntegralType, int>,
385 std::pair<std::span<const T>, int>>& coefficients,
386 const std::vector<std::reference_wrapper<const DirichletBC<T, U>>>& bcs)
387{
388 // Index maps for dof ranges
389 // NOTE: For mixed-topology meshes, there will be multiple DOF maps,
390 // but the index maps are the same.
391 auto map0 = a.function_spaces().at(0)->dofmaps(0)->index_map;
392 auto map1 = a.function_spaces().at(1)->dofmaps(0)->index_map;
393 auto bs0 = a.function_spaces().at(0)->dofmaps(0)->index_map_bs();
394 auto bs1 = a.function_spaces().at(1)->dofmaps(0)->index_map_bs();
395
396 // Build dof markers
397 std::vector<std::int8_t> dof_marker0, dof_marker1;
398 assert(map0);
399 std::int32_t dim0 = bs0 * (map0->size_local() + map0->num_ghosts());
400 assert(map1);
401 std::int32_t dim1 = bs1 * (map1->size_local() + map1->num_ghosts());
402 for (std::size_t k = 0; k < bcs.size(); ++k)
403 {
404 assert(bcs[k].get().function_space());
405 if (a.function_spaces().at(0)->contains(*bcs[k].get().function_space()))
406 {
407 dof_marker0.resize(dim0, false);
408 bcs[k].get().mark_dofs(dof_marker0);
409 }
410
411 if (a.function_spaces().at(1)->contains(*bcs[k].get().function_space()))
412 {
413 dof_marker1.resize(dim1, false);
414 bcs[k].get().mark_dofs(dof_marker1);
415 }
416 }
417
418 // Assemble
419 assemble_matrix(mat_add, a, constants, coefficients, dof_marker0,
420 dof_marker1);
421}
