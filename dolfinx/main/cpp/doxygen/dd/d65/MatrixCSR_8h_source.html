294template <typename U, typename V, typename W, typename X, typename Y>
295void impl::set_csr(U&& data, const V& cols, const W& row_ptr, const X& x,
296 const Y& xrows, const Y& xcols,
297 [[maybe_unused]] typename Y::value_type local_size)
298{
299 assert(x.size() == xrows.size() * xcols.size());
300 for (std::size_t r = 0; r < xrows.size(); ++r)
301 {
302 // Row index and current data row
303 auto row = xrows[r];
304 using T = typename X::value_type;
305 const T* xr = x.data() + r * xcols.size();
306
307#ifndef NDEBUG
308 if (row >= local_size)
309 throw std::runtime_error("Local row out of range");
310#endif
311
312 // Columns indices for row
313 auto cit0 = std::next(cols.begin(), row_ptr[row]);
314 auto cit1 = std::next(cols.begin(), row_ptr[row + 1]);
315 for (std::size_t c = 0; c < xcols.size(); ++c)
316 {
317 // Find position of column index
318 auto it = std::lower_bound(cit0, cit1, xcols[c]);
319 assert(it != cit1);
320 std::size_t d = std::distance(cols.begin(), it);
321 assert(d < data.size());
322 data[d] = xr[c];
323 }
324 }
325}

327template <typename U, typename V, typename W, typename X, typename Y>
328void impl::add_csr(U&& data, const V& cols, const W& row_ptr, const X& x,
329 const Y& xrows, const Y& xcols)
330{
331 assert(x.size() == xrows.size() * xcols.size());
332 for (std::size_t r = 0; r < xrows.size(); ++r)
333 {
334 // Row index and current data row
335 auto row = xrows[r];
336 using T = typename X::value_type;
337 const T* xr = x.data() + r * xcols.size();
338
339#ifndef NDEBUG
340 if (row >= (int)row_ptr.size())
341 throw std::runtime_error("Local row out of range");
342#endif
343
344 // Columns indices for row
345 auto cit0 = std::next(cols.begin(), row_ptr[row]);
346 auto cit1 = std::next(cols.begin(), row_ptr[row + 1]);
347 for (std::size_t c = 0; c < xcols.size(); ++c)
348 {
349 // Find position of column index
350 auto it = std::lower_bound(cit0, cit1, xcols[c]);
351 assert(it != cit1);
352 std::size_t d = std::distance(cols.begin(), it);
353 assert(d < data.size());
354 data[d] += xr[c];
355 }
356 }
357}

359template <class U, class V, class W, class X>
360MatrixCSR<U, V, W, X>::MatrixCSR(const SparsityPattern& p)
361 : _index_maps({p.index_map(0),
362 std::make_shared<common::IndexMap>(p.column_index_map())}),
363 _bs({p.block_size(0), p.block_size(1)}), _data(p.num_nonzeros(), 0),
364 _cols(p.graph().first.begin(), p.graph().first.end()),
365 _row_ptr(p.graph().second.begin(), p.graph().second.end()),
366 _comm(MPI_COMM_NULL)
367{
368 // TODO: handle block sizes
369 if (_bs[0] > 1 or _bs[1] > 1)
370 throw std::runtime_error("Block size not yet supported");
371
372 // Compute off-diagonal offset for each row
373 std::span<const std::int32_t> num_diag_nnz = p.off_diagonal_offsets();
374 _off_diagonal_offset.reserve(num_diag_nnz.size());
375 std::transform(num_diag_nnz.begin(), num_diag_nnz.end(), _row_ptr.begin(),
376 std::back_inserter(_off_diagonal_offset), std::plus{});
377
378 // Some short-hand
379 const std::array local_size
380 = {_index_maps[0]->size_local(), _index_maps[1]->size_local()};
381 const std::array local_range
382 = {_index_maps[0]->local_range(), _index_maps[1]->local_range()};
383 const std::vector<std::int64_t>& ghosts1 = _index_maps[1]->ghosts();
384
385 const std::vector<std::int64_t>& ghosts0 = _index_maps[0]->ghosts();
386 const std::vector<int>& src_ranks = _index_maps[0]->src();
387 const std::vector<int>& dest_ranks = _index_maps[0]->dest();
388
389 // Create neighbourhood communicator (owner <- ghost)
390 MPI_Comm comm;
391 MPI_Dist_graph_create_adjacent(_index_maps[0]->comm(), dest_ranks.size(),
392 dest_ranks.data(), MPI_UNWEIGHTED,
393 src_ranks.size(), src_ranks.data(),
394 MPI_UNWEIGHTED, MPI_INFO_NULL, false, &comm);
395 _comm = dolfinx::MPI::Comm(comm, false);
396
397 // Build map from ghost row index position to owning (neighborhood)
398 // rank
399 _ghost_row_to_rank.reserve(_index_maps[0]->owners().size());
400 for (int r : _index_maps[0]->owners())
401 {
402 auto it = std::lower_bound(src_ranks.begin(), src_ranks.end(), r);
403 assert(it != src_ranks.end() and *it == r);
404 int pos = std::distance(src_ranks.begin(), it);
405 _ghost_row_to_rank.push_back(pos);
406 }
407
408 // Compute size of data to send to each neighbor
409 std::vector<std::int32_t> data_per_proc(src_ranks.size(), 0);
410 for (std::size_t i = 0; i < _ghost_row_to_rank.size(); ++i)
411 {
412 assert(_ghost_row_to_rank[i] < data_per_proc.size());
413 std::size_t pos = local_size[0] + i;
414 data_per_proc[_ghost_row_to_rank[i]] += _row_ptr[pos + 1] - _row_ptr[pos];
415 }
416
417 // Compute send displacements
418 _val_send_disp.resize(src_ranks.size() + 1, 0);
419 std::partial_sum(data_per_proc.begin(), data_per_proc.end(),
420 std::next(_val_send_disp.begin()));
421
422 // For each ghost row, pack and send indices to neighborhood
423 std::vector<std::int64_t> ghost_index_data(2 * _val_send_disp.back());
424 {
425 std::vector<int> insert_pos = _val_send_disp;
426 for (std::size_t i = 0; i < _ghost_row_to_rank.size(); ++i)
427 {
428 const int rank = _ghost_row_to_rank[i];
429 int row_id = local_size[0] + i;
430 for (int j = _row_ptr[row_id]; j < _row_ptr[row_id + 1]; ++j)
431 {
432 // Get position in send buffer
433 const std::int32_t idx_pos = 2 * insert_pos[rank];
434
435 // Pack send data (row, col) as global indices
436 ghost_index_data[idx_pos] = ghosts0[i];
437 if (std::int32_t col_local = _cols[j]; col_local < local_size[1])
438 ghost_index_data[idx_pos + 1] = col_local + local_range[1][0];
439 else
440 ghost_index_data[idx_pos + 1] = ghosts1[col_local - local_size[1]];
441
442 insert_pos[rank] += 1;
443 }
444 }
445 }
446
447 // Communicate data with neighborhood
448 std::vector<std::int64_t> ghost_index_array;
449 std::vector<int> recv_disp;
450 {
451 std::vector<int> send_sizes;
452 std::transform(data_per_proc.begin(), data_per_proc.end(),
453 std::back_inserter(send_sizes),
454 [](auto x) { return 2 * x; });
455
456 std::vector<int> recv_sizes(dest_ranks.size());
457 send_sizes.reserve(1);
458 recv_sizes.reserve(1);
459 MPI_Neighbor_alltoall(send_sizes.data(), 1, MPI_INT, recv_sizes.data(), 1,
460 MPI_INT, _comm.comm());
461
462 // Build send/recv displacement
463 std::vector<int> send_disp = {0};
464 std::partial_sum(send_sizes.begin(), send_sizes.end(),
465 std::back_inserter(send_disp));
466 recv_disp = {0};
467 std::partial_sum(recv_sizes.begin(), recv_sizes.end(),
468 std::back_inserter(recv_disp));
469
470 ghost_index_array.resize(recv_disp.back());
471 MPI_Neighbor_alltoallv(ghost_index_data.data(), send_sizes.data(),
472 send_disp.data(), MPI_INT64_T,
473 ghost_index_array.data(), recv_sizes.data(),
474 recv_disp.data(), MPI_INT64_T, _comm.comm());
475 }
476
477 // Store receive displacements for future use, when transferring
478 // data values
479 _val_recv_disp.resize(recv_disp.size());
480 std::transform(recv_disp.begin(), recv_disp.end(), _val_recv_disp.begin(),
481 [](int d) { return d / 2; });
482
483 // Global-to-local map for ghost columns
484 std::vector<std::pair<std::int64_t, std::int32_t>> global_to_local;
485 global_to_local.reserve(ghosts1.size());
486 std::int32_t local_i = local_size[1];
487 for (std::int64_t idx : ghosts1)
488 global_to_local.push_back({idx, global_to_local.size() + local_size[1]});
489 std::sort(global_to_local.begin(), global_to_local.end());
490
491 // Compute location in which data for each index should be stored
492 // when received
493 for (std::size_t i = 0; i < ghost_index_array.size(); i += 2)
494 {
495 // Row must be on this process
496 const std::int32_t local_row = ghost_index_array[i] - local_range[0][0];
497 assert(local_row >= 0 and local_row < local_size[0]);
498
499 // Column may be owned or unowned
500 std::int32_t local_col = ghost_index_array[i + 1] - local_range[1][0];
501 if (local_col < 0 or local_col >= local_size[1])
502 {
503 auto it = std::lower_bound(global_to_local.begin(), global_to_local.end(),
504 std::pair(ghost_index_array[i + 1], -1),
505 [](auto& a, auto& b)
506 { return a.first < b.first; });
507 assert(it != global_to_local.end()
508 and it->first == ghost_index_array[i + 1]);
509 local_col = it->second;
510 }
511 auto cit0 = std::next(_cols.begin(), _row_ptr[local_row]);
512 auto cit1 = std::next(_cols.begin(), _row_ptr[local_row + 1]);
513
514 // Find position of column index and insert data
515 auto cit = std::lower_bound(cit0, cit1, local_col);
516 assert(cit != cit1);
517 assert(*cit == local_col);
518 std::size_t d = std::distance(_cols.begin(), cit);
519 _unpack_pos.push_back(d);
520 }
521}

523template <typename U, typename V, typename W, typename X>
524std::vector<typename MatrixCSR<U, V, W, X>::value_type>
525MatrixCSR<U, V, W, X>::to_dense() const
526{
527 const std::size_t nrows = num_all_rows();
528 const std::size_t ncols
529 = _index_maps[1]->size_local() + _index_maps[1]->num_ghosts();
530 std::vector<value_type> A(nrows * ncols);
531 for (std::size_t r = 0; r < nrows; ++r)
532 for (std::int32_t j = _row_ptr[r]; j < _row_ptr[r + 1]; ++j)
533 A[r * ncols + _cols[j]] = _data[j];
534
535 return A;
536}

538template <typename U, typename V, typename W, typename X>
539void MatrixCSR<U, V, W, X>::finalize_begin()
540{
541 const std::int32_t local_size0 = _index_maps[0]->size_local();
542 const std::int32_t num_ghosts0 = _index_maps[0]->num_ghosts();
543
544 // For each ghost row, pack and send values to send to neighborhood
545 std::vector<int> insert_pos = _val_send_disp;
546 std::vector<value_type> ghost_value_data(_val_send_disp.back());
547 for (int i = 0; i < num_ghosts0; ++i)
548 {
549 const int rank = _ghost_row_to_rank[i];
550
551 // Get position in send buffer to place data to send to this
552 // neighbour
553 const std::int32_t val_pos = insert_pos[rank];
554 std::copy(std::next(_data.data(), _row_ptr[local_size0 + i]),
555 std::next(_data.data(), _row_ptr[local_size0 + i + 1]),
556 std::next(ghost_value_data.begin(), val_pos));
557 insert_pos[rank]
558 += _row_ptr[local_size0 + i + 1] - _row_ptr[local_size0 + i];
559 }
560
561 _ghost_value_data_in.resize(_val_recv_disp.back());
562
563 // Compute data sizes for send and receive from displacements
564 std::vector<int> val_send_count(_val_send_disp.size() - 1);
565 std::adjacent_difference(std::next(_val_send_disp.begin()),
566 _val_send_disp.end(), val_send_count.begin());
567
568 std::vector<int> val_recv_count(_val_recv_disp.size() - 1);
569 std::adjacent_difference(std::next(_val_recv_disp.begin()),
570 _val_recv_disp.end(), val_recv_count.begin());
571
572 int status = MPI_Ineighbor_alltoallv(
573 ghost_value_data.data(), val_send_count.data(), _val_send_disp.data(),
574 dolfinx::MPI::mpi_type<value_type>(), _ghost_value_data_in.data(),
575 val_recv_count.data(), _val_recv_disp.data(),
576 dolfinx::MPI::mpi_type<value_type>(), _comm.comm(), &_request);
577 assert(status == MPI_SUCCESS);
578}
579//-----------------------------------------------------------------------------
580template <typename U, typename V, typename W, typename X>
581void MatrixCSR<U, V, W, X>::finalize_end()
582{
583 int status = MPI_Wait(&_request, MPI_STATUS_IGNORE);
584 assert(status == MPI_SUCCESS);
585
586 // Add to local rows
587 assert(_ghost_value_data_in.size() == _unpack_pos.size());
588 for (std::size_t i = 0; i < _ghost_value_data_in.size(); ++i)
589 _data[_unpack_pos[i]] += _ghost_value_data_in[i];
590
591 // Set ghost row data to zero
592 const std::int32_t local_size0 = _index_maps[0]->size_local();
593 std::fill(std::next(_data.begin(), _row_ptr[local_size0]), _data.end(), 0);
594}
595//-----------------------------------------------------------------------------
596template <typename U, typename V, typename W, typename X>
597double MatrixCSR<U, V, W, X>::norm_squared() const
598{
599 const std::size_t num_owned_rows = _index_maps[0]->size_local();
600 assert(num_owned_rows < _row_ptr.size());
601 double norm_sq_local = std::accumulate(
602 _data.cbegin(), std::next(_data.cbegin(), _row_ptr[num_owned_rows]),
603 double(0), [](auto norm, value_type y) { return norm + std::norm(y); });
604 double norm_sq;
605 MPI_Allreduce(&norm_sq_local, &norm_sq, 1, MPI_DOUBLE, MPI_SUM, _comm.comm());
606 return norm_sq;
607}
