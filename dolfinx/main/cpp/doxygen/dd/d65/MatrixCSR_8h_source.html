36 template <typename U, typename V, typename W, typename X>
37 void set_csr(U&& data, const V& cols, const V& row_ptr, const W& x,
38  const X& xrows, const X& xcols,
39  [[maybe_unused]] typename X::value_type local_size)
40 {
41  assert(x.size() == xrows.size() * xcols.size());
42  for (std::size_t r = 0; r < xrows.size(); ++r)
43  {
44  // Row index and current data row
45  auto row = xrows[r];
46  using T = typename W::value_type;
47  const T* xr = x.data() + r * xcols.size();
48 
49 #ifndef NDEBUG
50  if (row >= local_size)
51  throw std::runtime_error("Local row out of range");
52 #endif
53 
54  // Columns indices for row
55  auto cit0 = std::next(cols.begin(), row_ptr[row]);
56  auto cit1 = std::next(cols.begin(), row_ptr[row + 1]);
57  for (std::size_t c = 0; c < xcols.size(); ++c)
58  {
59  // Find position of column index
60  auto it = std::lower_bound(cit0, cit1, xcols[c]);
61  assert(it != cit1);
62  std::size_t d = std::distance(cols.begin(), it);
63  assert(d < data.size());
64  data[d] = xr[c];
65  }
66  }
67 }

79 void add_csr(U&& data, const V& cols, const V& row_ptr, const W& x,
80  const X& xrows, const X& xcols)
81 {
82  assert(x.size() == xrows.size() * xcols.size());
83  for (std::size_t r = 0; r < xrows.size(); ++r)
84  {
85  // Row index and current data row
86  auto row = xrows[r];
87  using T = typename W::value_type;
88  const T* xr = x.data() + r * xcols.size();
89 
90 #ifndef NDEBUG
91  if (row >= (int)row_ptr.size())
92  throw std::runtime_error("Local row out of range");
93 #endif
94 
95  // Columns indices for row
96  auto cit0 = std::next(cols.begin(), row_ptr[row]);
97  auto cit1 = std::next(cols.begin(), row_ptr[row + 1]);
98  for (std::size_t c = 0; c < xcols.size(); ++c)
99  {
100  // Find position of column index
101  auto it = std::lower_bound(cit0, cit1, xcols[c]);
102  assert(it != cit1);
103  std::size_t d = std::distance(cols.begin(), it);
104  assert(d < data.size());
105  data[d] += xr[c];
106  }
107  }

170  MatrixCSR(const SparsityPattern& p, const Allocator& alloc = Allocator())
171  : _index_maps({p.index_map(0),
172  std::make_shared<common::IndexMap>(p.column_index_map())}),
173  _bs({p.block_size(0), p.block_size(1)}),
174  _data(p.num_nonzeros(), 0, alloc),
175  _cols(p.graph().array().begin(), p.graph().array().end()),
176  _row_ptr(p.graph().offsets().begin(), p.graph().offsets().end()),
177  _comm(MPI_COMM_NULL)
178  {
179  // TODO: handle block sizes
180  if (_bs[0] > 1 or _bs[1] > 1)
181  throw std::runtime_error("Block size not yet supported");
182 
183  // Compute off-diagonal offset for each row
184  xtl::span<const std::int32_t> num_diag_nnz = p.off_diagonal_offset();
185  _off_diagonal_offset.reserve(num_diag_nnz.size());
186  std::transform(num_diag_nnz.begin(), num_diag_nnz.end(), _row_ptr.begin(),
187  std::back_inserter(_off_diagonal_offset),
188  std::plus<std::int32_t>());
189 
190  // Some short-hand
191  const std::array local_size
192  = {_index_maps[0]->size_local(), _index_maps[1]->size_local()};
193  const std::array local_range
194  = {_index_maps[0]->local_range(), _index_maps[1]->local_range()};
195  const std::vector<std::int64_t>& ghosts1 = _index_maps[1]->ghosts();
196 
197  const std::vector<std::int64_t>& ghosts0 = _index_maps[0]->ghosts();
198  const std::vector<int>& src_ranks = _index_maps[0]->src();
199  const std::vector<int>& dest_ranks = _index_maps[0]->dest();
200 
201  // Create neigbourhood communicator (owner <- ghost)
202  MPI_Comm comm;
203  MPI_Dist_graph_create_adjacent(_index_maps[0]->comm(), dest_ranks.size(),
204  dest_ranks.data(), MPI_UNWEIGHTED,
205  src_ranks.size(), src_ranks.data(),
206  MPI_UNWEIGHTED, MPI_INFO_NULL, false, &comm);
207  _comm = dolfinx::MPI::Comm(comm, false);
208 
209  // Build map from ghost row index position to owning (neighborhood)
210  // rank
211  _ghost_row_to_rank.reserve(_index_maps[0]->owners().size());
212  for (int r : _index_maps[0]->owners())
213  {
214  auto it = std::lower_bound(src_ranks.begin(), src_ranks.end(), r);
215  assert(it != src_ranks.end() and *it == r);
216  int pos = std::distance(src_ranks.begin(), it);
217  _ghost_row_to_rank.push_back(pos);
218  }
219 
220  // Compute size of data to send to each neighbor
221  std::vector<std::int32_t> data_per_proc(src_ranks.size(), 0);
222  for (std::size_t i = 0; i < _ghost_row_to_rank.size(); ++i)
223  {
224  assert(_ghost_row_to_rank[i] < data_per_proc.size());
225  std::size_t pos = local_size[0] + i;
226  data_per_proc[_ghost_row_to_rank[i]] += _row_ptr[pos + 1] - _row_ptr[pos];
227  }
228 
229  // Compute send displacements
230  _val_send_disp.resize(src_ranks.size() + 1, 0);
231  std::partial_sum(data_per_proc.begin(), data_per_proc.end(),
232  std::next(_val_send_disp.begin()));
233 
234  // For each ghost row, pack and send indices to neighborhood
235  std::vector<std::int64_t> ghost_index_data(2 * _val_send_disp.back());
236  {
237  std::vector<int> insert_pos = _val_send_disp;
238  for (std::size_t i = 0; i < _ghost_row_to_rank.size(); ++i)
239  {
240  const int rank = _ghost_row_to_rank[i];
241  int row_id = local_size[0] + i;
242  for (int j = _row_ptr[row_id]; j < _row_ptr[row_id + 1]; ++j)
243  {
244  // Get position in send buffer
245  const std::int32_t idx_pos = 2 * insert_pos[rank];
246 
247  // Pack send data (row, col) as global indices
248  ghost_index_data[idx_pos] = ghosts0[i];
249  if (std::int32_t col_local = _cols[j]; col_local < local_size[1])
250  ghost_index_data[idx_pos + 1] = col_local + local_range[1][0];
251  else
252  ghost_index_data[idx_pos + 1] = ghosts1[col_local - local_size[1]];
253 
254  insert_pos[rank] += 1;
255  }
256  }
257  }
258 
259  // Communicate data with neighborhood
260  std::vector<std::int64_t> ghost_index_array;
261  std::vector<int> recv_disp;
262  {
263  std::vector<int> send_sizes;
264  std::transform(data_per_proc.begin(), data_per_proc.end(),
265  std::back_inserter(send_sizes),
266  [](auto x) { return 2 * x; });
267 
268  std::vector<int> recv_sizes(dest_ranks.size());
269  send_sizes.reserve(1);
270  recv_sizes.reserve(1);
271  MPI_Neighbor_alltoall(send_sizes.data(), 1, MPI_INT, recv_sizes.data(), 1,
272  MPI_INT, _comm.comm());
273 
274  // Build send/recv displacement
275  std::vector<int> send_disp = {0};
276  std::partial_sum(send_sizes.begin(), send_sizes.end(),
277  std::back_inserter(send_disp));
278  recv_disp = {0};
279  std::partial_sum(recv_sizes.begin(), recv_sizes.end(),
280  std::back_inserter(recv_disp));
281 
282  ghost_index_array.resize(recv_disp.back());
283  MPI_Neighbor_alltoallv(ghost_index_data.data(), send_sizes.data(),
284  send_disp.data(), MPI_INT64_T,
285  ghost_index_array.data(), recv_sizes.data(),
286  recv_disp.data(), MPI_INT64_T, _comm.comm());
287  }
288 
289  // Store receive displacements for future use, when transferring
290  // data values
291  _val_recv_disp.resize(recv_disp.size());
292  std::transform(recv_disp.begin(), recv_disp.end(), _val_recv_disp.begin(),
293  [](int d) { return d / 2; });
294 
295  // Global-to-local map for ghost columns
296  std::vector<std::pair<std::int64_t, std::int32_t>> global_to_local;
297  global_to_local.reserve(ghosts1.size());
298  std::int32_t local_i = local_size[1];
299  for (std::int64_t idx : ghosts1)
300  global_to_local.push_back({idx, global_to_local.size() + local_size[1]});
301  std::sort(global_to_local.begin(), global_to_local.end());
302 
303  // Compute location in which data for each index should be stored
304  // when received
305  for (std::size_t i = 0; i < ghost_index_array.size(); i += 2)
306  {
307  // Row must be on this process
308  const std::int32_t local_row = ghost_index_array[i] - local_range[0][0];
309  assert(local_row >= 0 and local_row < local_size[0]);
310 
311  // Column may be owned or unowned
312  std::int32_t local_col = ghost_index_array[i + 1] - local_range[1][0];
313  if (local_col < 0 or local_col >= local_size[1])
314  {
315  const auto it = std::lower_bound(
316  global_to_local.begin(), global_to_local.end(),
317  std::pair(ghost_index_array[i + 1], -1),
318  [](auto& a, auto& b) { return a.first < b.first; });
319  assert(it != global_to_local.end()
320  and it->first == ghost_index_array[i + 1]);
321  local_col = it->second;
322  }
323  auto cit0 = std::next(_cols.begin(), _row_ptr[local_row]);
324  auto cit1 = std::next(_cols.begin(), _row_ptr[local_row + 1]);
325 
326  // Find position of column index and insert data
327  auto cit = std::lower_bound(cit0, cit1, local_col);
328  assert(cit != cit1);
329  assert(*cit == local_col);
330  std::size_t d = std::distance(_cols.begin(), cit);
331  _unpack_pos.push_back(d);
332  }
333  }

425  void finalize_begin()
426  {
427  const std::int32_t local_size0 = _index_maps[0]->size_local();
428  const std::int32_t num_ghosts0 = _index_maps[0]->num_ghosts();
429 
430  // For each ghost row, pack and send values to send to neighborhood
431  std::vector<int> insert_pos = _val_send_disp;
432  std::vector<T> ghost_value_data(_val_send_disp.back());
433  for (int i = 0; i < num_ghosts0; ++i)
434  {
435  const int rank = _ghost_row_to_rank[i];
436 
437  // Get position in send buffer to place data to send to this
438  // neighbour
439  const std::int32_t val_pos = insert_pos[rank];
440  std::copy(std::next(_data.data(), _row_ptr[local_size0 + i]),
441  std::next(_data.data(), _row_ptr[local_size0 + i + 1]),
442  std::next(ghost_value_data.begin(), val_pos));
443  insert_pos[rank]
444  += _row_ptr[local_size0 + i + 1] - _row_ptr[local_size0 + i];
445  }
446 
447  _ghost_value_data_in.resize(_val_recv_disp.back());
448 
449  // Compute data sizes for send and receive from displacements
450  std::vector<int> val_send_count(_val_send_disp.size() - 1);
451  std::adjacent_difference(std::next(_val_send_disp.begin()),
452  _val_send_disp.end(), val_send_count.begin());
453 
454  std::vector<int> val_recv_count(_val_recv_disp.size() - 1);
455  std::adjacent_difference(std::next(_val_recv_disp.begin()),
456  _val_recv_disp.end(), val_recv_count.begin());
457 
458  int status = MPI_Ineighbor_alltoallv(
459  ghost_value_data.data(), val_send_count.data(), _val_send_disp.data(),
460  dolfinx::MPI::mpi_type<T>(), _ghost_value_data_in.data(),
461  val_recv_count.data(), _val_recv_disp.data(),
462  dolfinx::MPI::mpi_type<T>(), _comm.comm(), &_request);
463  assert(status == MPI_SUCCESS);
464  }
465 
471  void finalize_end()
472  {
473  int status = MPI_Wait(&_request, MPI_STATUS_IGNORE);
474  assert(status == MPI_SUCCESS);
475 
476  // Add to local rows
477  assert(_ghost_value_data_in.size() == _unpack_pos.size());
478  for (std::size_t i = 0; i < _ghost_value_data_in.size(); ++i)
479  _data[_unpack_pos[i]] += _ghost_value_data_in[i];
480 
481  // Set ghost row data to zero
482  const std::int32_t local_size0 = _index_maps[0]->size_local();
483  std::fill(std::next(_data.begin(), _row_ptr[local_size0]), _data.end(), 0);
484  }

487  double norm_squared() const
488  {
489  const std::size_t num_owned_rows = _index_maps[0]->size_local();
490  assert(num_owned_rows < _row_ptr.size());
491 
492  const double norm_sq_local = std::accumulate(
493  _data.cbegin(), std::next(_data.cbegin(), _row_ptr[num_owned_rows]),
494  double(0), [](double norm, T y) { return norm + std::norm(y); });
495  double norm_sq;
496  MPI_Allreduce(&norm_sq_local, &norm_sq, 1, MPI_DOUBLE, MPI_SUM,
497  _comm.comm());
498  return norm_sq;
499  }
