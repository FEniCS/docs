50template <std::floating_point T>
51std::vector<T> shortest_vector(const mesh::Mesh<T>& mesh, int dim,
52 std::span<const std::int32_t> entities,
53 std::span<const T> points)
54{
55 const int tdim = mesh.topology()->dim();
56 const mesh::Geometry<T>& geometry = mesh.geometry();
57
58 std::span<const T> geom_dofs = geometry.x();
59 auto x_dofmap = geometry.dofmap();
60 std::vector<T> shortest_vectors;
61 shortest_vectors.reserve(3 * entities.size());
62 if (dim == tdim)
63 {
64 for (std::size_t e = 0; e < entities.size(); e++)
65 {
66 // Check that we have sent in valid entities, i.e. that they exist in the
67 // local dofmap. One gets a cryptical memory segfault if entities is -1
68 assert(entities[e] >= 0);
69 auto dofs = MDSPAN_IMPL_STANDARD_NAMESPACE::submdspan(
70 x_dofmap, entities[e], MDSPAN_IMPL_STANDARD_NAMESPACE::full_extent);
71 std::vector<T> nodes(3 * dofs.size());
72 for (std::size_t i = 0; i < dofs.size(); ++i)
73 {
74 const std::int32_t pos = 3 * dofs[i];
75 for (std::size_t j = 0; j < 3; ++j)
76 nodes[3 * i + j] = geom_dofs[pos + j];
77 }
78
79 std::array<T, 3> d
80 = compute_distance_gjk<T>(points.subspan(3 * e, 3), nodes);
81 shortest_vectors.insert(shortest_vectors.end(), d.begin(), d.end());
82 }
83 }
84 else
85 {
86 mesh.topology_mutable()->create_connectivity(dim, tdim);
87 mesh.topology_mutable()->create_connectivity(tdim, dim);
88 auto e_to_c = mesh.topology()->connectivity(dim, tdim);
89 assert(e_to_c);
90 auto c_to_e = mesh.topology_mutable()->connectivity(tdim, dim);
91 assert(c_to_e);
92 for (std::size_t e = 0; e < entities.size(); e++)
93 {
94 const std::int32_t index = entities[e];
95
96 // Find attached cell
97 assert(e_to_c->num_links(index) > 0);
98 const std::int32_t c = e_to_c->links(index)[0];
99
100 // Find local number of entity wrt cell
101 auto cell_entities = c_to_e->links(c);
102 auto it0 = std::find(cell_entities.begin(), cell_entities.end(), index);
103 assert(it0 != cell_entities.end());
104 const int local_cell_entity = std::distance(cell_entities.begin(), it0);
105
106 // Tabulate geometry dofs for the entity
107 auto dofs = MDSPAN_IMPL_STANDARD_NAMESPACE::submdspan(
108 x_dofmap, c, MDSPAN_IMPL_STANDARD_NAMESPACE::full_extent);
109 const std::vector<int> entity_dofs
110 = geometry.cmap().create_dof_layout().entity_closure_dofs(
111 dim, local_cell_entity);
112 std::vector<T> nodes(3 * entity_dofs.size());
113 for (std::size_t i = 0; i < entity_dofs.size(); i++)
114 {
115 const std::int32_t pos = 3 * dofs[entity_dofs[i]];
116 for (std::size_t j = 0; j < 3; ++j)
117 nodes[3 * i + j] = geom_dofs[pos + j];
118 }
119
120 std::array<T, 3> d
121 = compute_distance_gjk<T>(points.subspan(3 * e, 3), nodes);
122 shortest_vectors.insert(shortest_vectors.end(), d.begin(), d.end());
123 }
124 }
125
126 return shortest_vectors;
127}

135template <std::floating_point T>
136T compute_squared_distance_bbox(std::span<const T, 6> b,
137 std::span<const T, 3> x)
138{
139 auto b0 = b.template subspan<0, 3>();
140 auto b1 = b.template subspan<3, 3>();
141 return std::transform_reduce(x.begin(), x.end(), b0.begin(), 0.0,
142 std::plus<>{},
143 [](auto x, auto b)
144 {
145 auto dx = x - b;
146 return dx > 0 ? 0 : dx * dx;
147 })
148 + std::transform_reduce(x.begin(), x.end(), b1.begin(), 0.0,
149 std::plus<>{},
150 [](auto x, auto b)
151 {
152 auto dx = x - b;
153 return dx < 0 ? 0 : dx * dx;
154 });
155}

172template <std::floating_point T>
173std::vector<T> squared_distance(const mesh::Mesh<T>& mesh, int dim,
174 std::span<const std::int32_t> entities,
175 std::span<const T> points)
176{
177 std::vector<T> v = shortest_vector(mesh, dim, entities, points);
178 std::vector<T> d(v.size() / 3, 0);
179 for (std::size_t i = 0; i < d.size(); ++i)
180 for (std::size_t j = 0; j < 3; ++j)
181 d[i] += v[3 * i + j] * v[3 * i + j];
182 return d;
183}

188constexpr bool is_leaf(std::array<int, 2> bbox)
189{
190 // Leaf nodes are marked by setting child_0 equal to child_1
191 return bbox[0] == bbox[1];
192}

198template <std::floating_point T>
199constexpr bool point_in_bbox(const std::array<T, 6>& b, std::span<const T, 3> x)
200{
201 constexpr T rtol = 1e-14;
202 bool in = true;
203 for (std::size_t i = 0; i < 3; i++)
204 {
205 T eps = rtol * (b[i + 3] - b[i]);
206 in &= x[i] >= (b[i] - eps);
207 in &= x[i] <= (b[i + 3] + eps);
208 }
209
210 return in;
211}

216template <std::floating_point T>
217constexpr bool bbox_in_bbox(std::span<const T, 6> a, std::span<const T, 6> b)
218{
219 constexpr T rtol = 1e-14;
220 auto a0 = a.template subspan<0, 3>();
221 auto a1 = a.template subspan<3, 3>();
222 auto b0 = b.template subspan<0, 3>();
223 auto b1 = b.template subspan<3, 3>();
224
225 bool in = true;
226 for (std::size_t i = 0; i < 3; i++)
227 {
228 T eps = rtol * (b1[i] - b0[i]);
229 in &= a1[i] >= (b0[i] - eps);
230 in &= a0[i] <= (b1[i] + eps);
231 }
232
233 return in;
234}

237template <std::floating_point T>
238std::pair<std::int32_t, T>
239_compute_closest_entity(const geometry::BoundingBoxTree<T>& tree,
240 std::span<const T, 3> point, std::int32_t node,
241 const mesh::Mesh<T>& mesh, std::int32_t closest_entity,
242 T R2)
243{
244 // Get children of current bounding box node (child_1 denotes entity
245 // index for leaves)
246 const std::array<int, 2> bbox = tree.bbox(node);
247 T r2;
248 if (is_leaf(bbox))
249 {
250 // If point cloud tree the exact distance is easy to compute
251 if (tree.tdim() == 0)
252 {
253 std::array<T, 6> diff = tree.get_bbox(node);
254 for (std::size_t k = 0; k < 3; ++k)
255 diff[k] -= point[k];
256 r2 = diff[0] * diff[0] + diff[1] * diff[1] + diff[2] * diff[2];
257 }
258 else
259 {
260 r2 = compute_squared_distance_bbox<T>(tree.get_bbox(node), point);
261
262 // If bounding box closer than previous closest entity, use gjk to
263 // obtain exact distance to the convex hull of the entity
264 if (r2 <= R2)
265 {
266 r2 = squared_distance<T>(mesh, tree.tdim(),
267 std::span(std::next(bbox.begin(), 1), 1),
268 point)
269 .front();
270 }
271 }
272
273 // If entity is closer than best result so far, return it
274 if (r2 <= R2)
275 {
276 closest_entity = bbox.back();
277 R2 = r2;
278 }
279
280 return {closest_entity, R2};
281 }
282 else
283 {
284 // If bounding box is outside radius, then don't search further
285 r2 = compute_squared_distance_bbox<T>(tree.get_bbox(node), point);
286 if (r2 > R2)
287 return {closest_entity, R2};
288
289 // Check both children
290 // We use R2 (as opposed to r2), as a bounding box can be closer
291 // than the actual entity
292 std::pair<int, T> p0 = _compute_closest_entity(tree, point, bbox.front(),
293 mesh, closest_entity, R2);
294 std::pair<int, T> p1 = _compute_closest_entity(tree, point, bbox.back(),
295 mesh, p0.first, p0.second);
296 return p1;
297 }

305template <std::floating_point T>
306void _compute_collisions_point(const geometry::BoundingBoxTree<T>& tree,
307 std::span<const T, 3> p,
308 std::vector<std::int32_t>& entities)
309{
310 std::deque<std::int32_t> stack;
311 std::int32_t next = tree.num_bboxes() - 1;
312 while (next != -1)
313 {
314 const std::array<int, 2> bbox = tree.bbox(next);
315 if (is_leaf(bbox) and point_in_bbox(tree.get_bbox(next), p))
316 {
317 // If box is a leaf node then add it to the list of colliding
318 // entities
319 entities.push_back(bbox[1]);
320 next = -1;
321 }
322 else
323 {
324 // Check whether the point collides with child nodes (left and
325 // right)
326 bool left = point_in_bbox(tree.get_bbox(bbox[0]), p);
327 bool right = point_in_bbox(tree.get_bbox(bbox[1]), p);
328 if (left and right)
329 {
330 // If the point collides with both child nodes, add the right
331 // node to the stack (for later visiting) and continue the tree
332 // traversal with the left subtree
333 stack.push_back(bbox[1]);
334 next = bbox[0];
335 }
336 else if (left)
337 {
338 // Traverse the current node's left subtree
339 next = bbox[0];
340 }
341 else if (right)
342 {
343 // Traverse the current node's right subtree
344 next = bbox[1];
345 }
346 else
347 next = -1;
348 }
349
350 // If tree traversal reaches a dead end (box is a leaf node or no
351 // collision detected), check the stack for deferred subtrees
352 if (next == -1 and !stack.empty())
353 {
354 next = stack.back();
355 stack.pop_back();
356 }
357 }
358}

360// Compute collisions with tree (recursive)
361template <std::floating_point T>
362void _compute_collisions_tree(const geometry::BoundingBoxTree<T>& A,
363 const geometry::BoundingBoxTree<T>& B,
364 std::int32_t node_A, std::int32_t node_B,
365 std::vector<std::int32_t>& entities)
366{
367 // If bounding boxes don't collide, then don't search further
368 if (!bbox_in_bbox<T>(A.get_bbox(node_A), B.get_bbox(node_B)))
369 return;
370
371 // Get bounding boxes for current nodes
372 const std::array<std::int32_t, 2> bbox_A = A.bbox(node_A);
373 const std::array<std::int32_t, 2> bbox_B = B.bbox(node_B);
374
375 // Check whether we've reached a leaf in A or B
376 const bool is_leaf_A = is_leaf(bbox_A);
377 const bool is_leaf_B = is_leaf(bbox_B);
378 if (is_leaf_A and is_leaf_B)
379 {
380 // If both boxes are leaves (which we know collide), then add them
381 // child_1 denotes entity for leaves
382 entities.push_back(bbox_A[1]);
383 entities.push_back(bbox_B[1]);
384 }
385 else if (is_leaf_A)
386 {
387 // If we reached the leaf in A, then descend B
388 _compute_collisions_tree(A, B, node_A, bbox_B[0], entities);
389 _compute_collisions_tree(A, B, node_A, bbox_B[1], entities);
390 }
391 else if (is_leaf_B)
392 {
393 // If we reached the leaf in B, then descend A
394 _compute_collisions_tree(A, B, bbox_A[0], node_B, entities);
395 _compute_collisions_tree(A, B, bbox_A[1], node_B, entities);
396 }
397 else if (node_A > node_B)
398 {
399 // At this point, we know neither is a leaf so descend the largest
400 // tree first. Note that nodes are added in reverse order with the
401 // top bounding box at the end so the largest tree (the one with the
402 // the most boxes left to traverse) has the largest node number.
403 _compute_collisions_tree(A, B, bbox_A[0], node_B, entities);
404 _compute_collisions_tree(A, B, bbox_A[1], node_B, entities);
405 }
406 else
407 {
408 _compute_collisions_tree(A, B, node_A, bbox_B[0], entities);
409 _compute_collisions_tree(A, B, node_A, bbox_B[1], entities);
410 }
411
412 // Note that cases above can be collected in fewer cases but this way
413 // the logic is easier to follow.
414}

424template <std::floating_point T>
425BoundingBoxTree<T> create_midpoint_tree(const mesh::Mesh<T>& mesh, int tdim,
426 std::span<const std::int32_t> entities)
427{
428 LOG(INFO) << "Building point search tree to accelerate distance queries for "
429 "a given topological dimension and subset of entities.";
430
431 const std::vector<T> midpoints
432 = mesh::compute_midpoints(mesh, tdim, entities);
433 std::vector<std::pair<std::array<T, 3>, std::int32_t>> points(
434 entities.size());
435 for (std::size_t i = 0; i < points.size(); ++i)
436 {
437 for (std::size_t j = 0; j < 3; ++j)
438 points[i].first[j] = midpoints[3 * i + j];
439 points[i].second = entities[i];
440 }
441
442 // Build tree
443 return BoundingBoxTree(points);
444}

451template <std::floating_point T>
452std::vector<std::int32_t> compute_collisions(const BoundingBoxTree<T>& tree0,
453 const BoundingBoxTree<T>& tree1)
454{
455 // Call recursive find function
456 std::vector<std::int32_t> entities;
457 if (tree0.num_bboxes() > 0 and tree1.num_bboxes() > 0)
458 {
459 impl::_compute_collisions_tree(tree0, tree1, tree0.num_bboxes() - 1,
460 tree1.num_bboxes() - 1, entities);
461 }
462
463 return entities;
464}

476template <std::floating_point T>
477graph::AdjacencyList<std::int32_t>
478compute_collisions(const BoundingBoxTree<T>& tree, std::span<const T> points)
479{
480 if (tree.num_bboxes() > 0)
481 {
482 std::vector<std::int32_t> entities, offsets(points.size() / 3 + 1, 0);
483 entities.reserve(points.size() / 3);
484 for (std::size_t p = 0; p < points.size() / 3; ++p)
485 {
486 impl::_compute_collisions_point(
487 tree, std::span<const T, 3>(points.data() + 3 * p, 3), entities);
488 offsets[p + 1] = entities.size();
489 }
490
491 return graph::AdjacencyList(std::move(entities), std::move(offsets));
492 }
493 else
494 {
495 return graph::AdjacencyList(
496 std::vector<std::int32_t>(),
497 std::vector<std::int32_t>(points.size() / 3 + 1, 0));
498 }

518template <std::floating_point T>
519std::int32_t compute_first_colliding_cell(const mesh::Mesh<T>& mesh,
520 std::span<const std::int32_t> cells,
521 std::array<T, 3> point, T tol)
522{
523 if (cells.empty())
524 return -1;
525 else
526 {
527 const mesh::Geometry<T>& geometry = mesh.geometry();
528 std::span<const T> geom_dofs = geometry.x();
529 auto x_dofmap = geometry.dofmap();
530 const std::size_t num_nodes = x_dofmap.extent(1);
531 std::vector<T> coordinate_dofs(num_nodes * 3);
532 for (auto cell : cells)
533 {
534 auto dofs = MDSPAN_IMPL_STANDARD_NAMESPACE::submdspan(
535 x_dofmap, cell, MDSPAN_IMPL_STANDARD_NAMESPACE::full_extent);
536 for (std::size_t i = 0; i < num_nodes; ++i)
537 {
538 std::copy_n(std::next(geom_dofs.begin(), 3 * dofs[i]), 3,
539 std::next(coordinate_dofs.begin(), 3 * i));
540 }
541
542 std::array<T, 3> shortest_vector
543 = compute_distance_gjk<T>(point, coordinate_dofs);
544 T d2 = std::reduce(shortest_vector.begin(), shortest_vector.end(), T(0),
545 [](auto d, auto e) { return d + e * e; });
546 if (d2 < tol)
547 return cell;
548 }
549
550 return -1;
551 }

566template <std::floating_point T>
567std::vector<std::int32_t>
568compute_closest_entity(const BoundingBoxTree<T>& tree,
569 const BoundingBoxTree<T>& midpoint_tree,
570 const mesh::Mesh<T>& mesh, std::span<const T> points)
571{
572 if (tree.num_bboxes() == 0)
573 return std::vector<std::int32_t>(points.size() / 3, -1);
574
575 std::vector<std::int32_t> entities;
576 entities.reserve(points.size() / 3);
577 for (std::size_t i = 0; i < points.size() / 3; ++i)
578 {
579 // Use midpoint tree to find initial closest entity to the point.
580 // Start by using a leaf node as the initial guess for the input
581 // entity
582 std::array<int, 2> leaf0 = midpoint_tree.bbox(0);
583 assert(impl::is_leaf(leaf0));
584 std::array<T, 6> diff = midpoint_tree.get_bbox(0);
585 for (std::size_t k = 0; k < 3; ++k)
586 diff[k] -= points[3 * i + k];
587 T R2 = diff[0] * diff[0] + diff[1] * diff[1] + diff[2] * diff[2];
588
589 // Use a recursive search through the bounding box tree
590 // to find determine the entity with the closest midpoint.
591 // As the midpoint tree only consist of points, the distance
592 // queries are lightweight.
593 const auto [m_index, m_distance2] = impl::_compute_closest_entity(
594 midpoint_tree, std::span<const T, 3>(points.data() + 3 * i, 3),
595 midpoint_tree.num_bboxes() - 1, mesh, leaf0[0], R2);
596
597 // Use a recursives search through the bounding box tree to
598 // determine which entity is actually closest.
599 // Uses the entity with the closest midpoint as initial guess, and
600 // the distance from the midpoint to the point of interest as the
601 // initial search radius.
602 const auto [index, distance2] = impl::_compute_closest_entity(
603 tree, std::span<const T, 3>(points.data() + 3 * i, 3),
604 tree.num_bboxes() - 1, mesh, m_index, m_distance2);
605
606 entities.push_back(index);
607 }
608
609 return entities;
610}

627template <std::floating_point T>
628graph::AdjacencyList<std::int32_t> compute_colliding_cells(
629 const mesh::Mesh<T>& mesh,
630 const graph::AdjacencyList<std::int32_t>& candidate_cells,
631 std::span<const T> points)
632{
633 std::vector<std::int32_t> offsets = {0};
634 offsets.reserve(candidate_cells.num_nodes() + 1);
635 std::vector<std::int32_t> colliding_cells;
636 constexpr T eps2 = 1e-12;
637 const int tdim = mesh.topology()->dim();
638 for (std::int32_t i = 0; i < candidate_cells.num_nodes(); i++)
639 {
640 auto cells = candidate_cells.links(i);
641 std::vector<T> _point(3 * cells.size());
642 for (std::size_t j = 0; j < cells.size(); ++j)
643 for (std::size_t k = 0; k < 3; ++k)
644 _point[3 * j + k] = points[3 * i + k];
645
646 std::vector distances_sq = squared_distance<T>(mesh, tdim, cells, _point);
647 for (std::size_t j = 0; j < cells.size(); j++)
648 if (distances_sq[j] < eps2)
649 colliding_cells.push_back(cells[j]);
650
651 offsets.push_back(colliding_cells.size());
652 }
653
654 return graph::AdjacencyList(std::move(colliding_cells), std::move(offsets));
655}

685template <std::floating_point T>
686PointOwnershipData<T> determine_point_ownership(const mesh::Mesh<T>& mesh,
687 std::span<const T> points,
688 T padding)
689{
690 MPI_Comm comm = mesh.comm();
691
692 // Create a global bounding-box tree to find candidate processes with
693 // cells that could collide with the points
694 const int tdim = mesh.topology()->dim();
695 auto cell_map = mesh.topology()->index_map(tdim);
696 const std::int32_t num_cells = cell_map->size_local();
697 // NOTE: Should we send the cells in as input?
698 std::vector<std::int32_t> cells(num_cells, 0);
699 std::iota(cells.begin(), cells.end(), 0);
700 BoundingBoxTree bb(mesh, tdim, cells, padding);
701 BoundingBoxTree global_bbtree = bb.create_global_tree(comm);
702
703 // Compute collisions:
704 // For each point in `points` get the processes it should be sent to
705 graph::AdjacencyList collisions = compute_collisions(global_bbtree, points);
706
707 // Get unique list of outgoing ranks
708 std::vector<std::int32_t> out_ranks = collisions.array();
709 std::sort(out_ranks.begin(), out_ranks.end());
710 out_ranks.erase(std::unique(out_ranks.begin(), out_ranks.end()),
711 out_ranks.end());
712 // Compute incoming edges (source processes)
713 std::vector in_ranks = dolfinx::MPI::compute_graph_edges_nbx(comm, out_ranks);
714 std::sort(in_ranks.begin(), in_ranks.end());
715
716 // Create neighborhood communicator in forward direction
717 MPI_Comm forward_comm;
718 MPI_Dist_graph_create_adjacent(
719 comm, in_ranks.size(), in_ranks.data(), MPI_UNWEIGHTED, out_ranks.size(),
720 out_ranks.data(), MPI_UNWEIGHTED, MPI_INFO_NULL, false, &forward_comm);
721
722 // Compute map from global mpi rank to neighbor rank, "collisions"
723 // uses global rank
724 std::map<std::int32_t, std::int32_t> rank_to_neighbor;
725 for (std::size_t i = 0; i < out_ranks.size(); i++)
726 rank_to_neighbor[out_ranks[i]] = i;
727
728 // Count the number of points to send per neighbor process
729 std::vector<std::int32_t> send_sizes(out_ranks.size());
730 for (std::size_t i = 0; i < points.size() / 3; ++i)
731 for (auto p : collisions.links(i))
732 send_sizes[rank_to_neighbor[p]] += 3;
733
734 // Compute receive sizes
735 std::vector<std::int32_t> recv_sizes(in_ranks.size());
736 send_sizes.reserve(1);
737 recv_sizes.reserve(1);
738 MPI_Request sizes_request;
739 MPI_Ineighbor_alltoall(send_sizes.data(), 1, MPI_INT, recv_sizes.data(), 1,
740 MPI_INT, forward_comm, &sizes_request);
741
742 // Compute sending offsets
743 std::vector<std::int32_t> send_offsets(send_sizes.size() + 1, 0);
744 std::partial_sum(send_sizes.begin(), send_sizes.end(),
745 std::next(send_offsets.begin(), 1));
746
747 // Pack data to send and store unpack map
748 std::vector<T> send_data(send_offsets.back());
749 std::vector<std::int32_t> counter(send_sizes.size(), 0);
750 // unpack map: [index in adj list][pos in x]
751 std::vector<std::int32_t> unpack_map(send_offsets.back() / 3);
752 for (std::size_t i = 0; i < points.size(); i += 3)
753 {
754 for (auto p : collisions.links(i / 3))
755 {
756 int neighbor = rank_to_neighbor[p];
757 int pos = send_offsets[neighbor] + counter[neighbor];
758 auto it = std::next(send_data.begin(), pos);
759 std::copy_n(std::next(points.begin(), i), 3, it);
760 unpack_map[pos / 3] = i / 3;
761 counter[neighbor] += 3;
762 }
763 }
764
765 MPI_Wait(&sizes_request, MPI_STATUS_IGNORE);
766 std::vector<std::int32_t> recv_offsets(in_ranks.size() + 1, 0);
767 std::partial_sum(recv_sizes.begin(), recv_sizes.end(),
768 std::next(recv_offsets.begin(), 1));
769
770 std::vector<T> received_points((std::size_t)recv_offsets.back());
771 MPI_Neighbor_alltoallv(
772 send_data.data(), send_sizes.data(), send_offsets.data(),
773 dolfinx::MPI::mpi_type<T>(), received_points.data(), recv_sizes.data(),
774 recv_offsets.data(), dolfinx::MPI::mpi_type<T>(), forward_comm);
775
776 // Get mesh geometry for closest entity
777 const mesh::Geometry<T>& geometry = mesh.geometry();
778 std::span<const T> geom_dofs = geometry.x();
779 auto x_dofmap = geometry.dofmap();
780
781 // Compute candidate cells for collisions (and extrapolation)
782 const graph::AdjacencyList<std::int32_t> candidate_collisions
783 = compute_collisions(bb, std::span<const T>(received_points.data(),
784 received_points.size()));
785
786 // Each process checks which points collide with a cell on the process
787 const int rank = dolfinx::MPI::rank(comm);
788 std::vector<std::int32_t> cell_indicator(received_points.size() / 3);
789 std::vector<std::int32_t> closest_cells(received_points.size() / 3);
790 for (std::size_t p = 0; p < received_points.size(); p += 3)
791 {
792 std::array<T, 3> point;
793 std::copy_n(std::next(received_points.begin(), p), 3, point.begin());
794 // Find first colliding cell among the cells with colliding bounding boxes
795 const int colliding_cell = geometry::compute_first_colliding_cell(
796 mesh, candidate_collisions.links(p / 3), point,
797 10 * std::numeric_limits<T>::epsilon());
798 // If a collding cell is found, store the rank of the current process
799 // which will be sent back to the owner of the point
800 cell_indicator[p / 3] = (colliding_cell >= 0) ? rank : -1;
801 // Store the cell index for lookup once the owning processes has determined
802 // the ownership of the point
803 closest_cells[p / 3] = colliding_cell;
804 }
805
806 // Create neighborhood communicator in the reverse direction: send
807 // back col to requesting processes
808 MPI_Comm reverse_comm;
809 MPI_Dist_graph_create_adjacent(
810 comm, out_ranks.size(), out_ranks.data(), MPI_UNWEIGHTED, in_ranks.size(),
811 in_ranks.data(), MPI_UNWEIGHTED, MPI_INFO_NULL, false, &reverse_comm);
812
813 // Reuse sizes and offsets from first communication set
814 // but divide by three
815 {
816 auto rescale = [](auto& x)
817 {
818 std::transform(x.cbegin(), x.cend(), x.begin(),
819 [](auto e) { return (e / 3); });
820 };
821 rescale(recv_sizes);
822 rescale(recv_offsets);
823 rescale(send_sizes);
824 rescale(send_offsets);
825
826 // The communication is reversed, so swap recv to send offsets
827 std::swap(recv_sizes, send_sizes);
828 std::swap(recv_offsets, send_offsets);
829 }
830
831 std::vector<std::int32_t> recv_ranks(recv_offsets.back());
832 MPI_Neighbor_alltoallv(cell_indicator.data(), send_sizes.data(),
833 send_offsets.data(), MPI_INT32_T, recv_ranks.data(),
834 recv_sizes.data(), recv_offsets.data(), MPI_INT32_T,
835 reverse_comm);
836
837 std::vector<int> point_owners(points.size() / 3, -1);
838 for (std::size_t i = 0; i < unpack_map.size(); i++)
839 {
840 const std::int32_t pos = unpack_map[i];
841 // Only insert new owner if no owner has previously been found
842 if (recv_ranks[i] >= 0 && point_owners[pos] == -1)
843 point_owners[pos] = recv_ranks[i];
844 }
845
846 // Create extrapolation marker for those points already sent to other
847 // process
848 std::vector<std::uint8_t> send_extrapolate(recv_offsets.back());
849 for (std::int32_t i = 0; i < recv_offsets.back(); i++)
850 {
851 const std::int32_t pos = unpack_map[i];
852 send_extrapolate[i] = point_owners[pos] == -1;
853 }
854
855 // Swap communication direction, to send extrapolation marker to other
856 // processes
857 std::swap(send_sizes, recv_sizes);
858 std::swap(send_offsets, recv_offsets);
859 std::vector<std::uint8_t> dest_extrapolate(recv_offsets.back());
860 MPI_Neighbor_alltoallv(send_extrapolate.data(), send_sizes.data(),
861 send_offsets.data(), MPI_UINT8_T,
862 dest_extrapolate.data(), recv_sizes.data(),
863 recv_offsets.data(), MPI_UINT8_T, forward_comm);
864
865 std::vector<T> squared_distances(received_points.size() / 3, -1);
866
867 for (std::size_t i = 0; i < dest_extrapolate.size(); i++)
868 {
869 if (dest_extrapolate[i] == 1)
870 {
871 assert(closest_cells[i] == -1);
872 std::array<T, 3> point;
873 std::copy_n(std::next(received_points.begin(), 3 * i), 3, point.begin());
874
875 // Find shortest distance among cells with colldiing bounding box
876 T shortest_distance = std::numeric_limits<T>::max();
877 std::int32_t closest_cell = -1;
878 for (auto cell : candidate_collisions.links(i))
879 {
880 auto dofs = MDSPAN_IMPL_STANDARD_NAMESPACE::submdspan(
881 x_dofmap, cell, MDSPAN_IMPL_STANDARD_NAMESPACE::full_extent);
882 std::vector<T> nodes(3 * dofs.size());
883 for (std::size_t j = 0; j < dofs.size(); ++j)
884 {
885 const int pos = 3 * dofs[j];
886 for (std::size_t k = 0; k < 3; ++k)
887 nodes[3 * j + k] = geom_dofs[pos + k];
888 }
889 const std::array<T, 3> d = compute_distance_gjk<T>(
890 std::span<const T>(point.data(), point.size()), nodes);
891 if (T current_distance = d[0] * d[0] + d[1] * d[1] + d[2] * d[2];
892 current_distance < shortest_distance)
893 {
894 shortest_distance = current_distance;
895 closest_cell = cell;
896 }
897 }
898 closest_cells[i] = closest_cell;
899 squared_distances[i] = shortest_distance;
900 }
901 }
902
903 std::swap(recv_sizes, send_sizes);
904 std::swap(recv_offsets, send_offsets);
905
906 // Get distances from closest entity of points that were on the other process
907 std::vector<T> recv_distances(recv_offsets.back());
908 MPI_Neighbor_alltoallv(
909 squared_distances.data(), send_sizes.data(), send_offsets.data(),
910 dolfinx::MPI::mpi_type<T>(), recv_distances.data(), recv_sizes.data(),
911 recv_offsets.data(), dolfinx::MPI::mpi_type<T>(), reverse_comm);
912
913 // Update point ownership with extrapolation information
914 std::vector<T> closest_distance(point_owners.size(),
915 std::numeric_limits<T>::max());
916 for (std::size_t i = 0; i < out_ranks.size(); i++)
917 {
918 for (std::int32_t j = recv_offsets[i]; j < recv_offsets[i + 1]; j++)
919 {
920 const std::int32_t pos = unpack_map[j];
921 auto current_dist = recv_distances[j];
922 // Update if closer than previous guess and was found
923 if (auto d = closest_distance[pos];
924 (current_dist > 0) and (current_dist < d))
925 {
926 point_owners[pos] = out_ranks[i];
927 closest_distance[pos] = current_dist;
928 }
929 }
930 }
931
932 // Communication is reversed again to send dest ranks to all processes
933 std::swap(send_sizes, recv_sizes);
934 std::swap(send_offsets, recv_offsets);
935
936 // Pack ownership data
937 std::vector<std::int32_t> send_owners(send_offsets.back());
938 std::fill(counter.begin(), counter.end(), 0);
939 for (std::size_t i = 0; i < points.size() / 3; ++i)
940 {
941 for (auto p : collisions.links(i))
942 {
943 int neighbor = rank_to_neighbor[p];
944 send_owners[send_offsets[neighbor] + counter[neighbor]++]
945 = point_owners[i];
946 }
947 }
948
949 // Send ownership info
950 std::vector<std::int32_t> dest_ranks(recv_offsets.back());
951 MPI_Neighbor_alltoallv(send_owners.data(), send_sizes.data(),
952 send_offsets.data(), MPI_INT32_T, dest_ranks.data(),
953 recv_sizes.data(), recv_offsets.data(), MPI_INT32_T,
954 forward_comm);
955
956 // Unpack dest ranks if point owner is this rank
957 std::vector<int> owned_recv_ranks;
958 owned_recv_ranks.reserve(recv_offsets.back());
959 std::vector<T> owned_recv_points;
960 std::vector<std::int32_t> owned_recv_cells;
961 for (std::size_t i = 0; i < in_ranks.size(); i++)
962 {
963 for (std::int32_t j = recv_offsets[i]; j < recv_offsets[i + 1]; j++)
964 {
965 if (rank == dest_ranks[j])
966 {
967 owned_recv_ranks.push_back(in_ranks[i]);
968 owned_recv_points.insert(
969 owned_recv_points.end(), std::next(received_points.cbegin(), 3 * j),
970 std::next(received_points.cbegin(), 3 * (j + 1)));
971 owned_recv_cells.push_back(closest_cells[j]);
972 }
973 }
974 }
975
976 MPI_Comm_free(&forward_comm);
977 MPI_Comm_free(&reverse_comm);
978 return PointOwnershipData<T>{.src_owner = std::move(point_owners),
979 .dest_owners = std::move(owned_recv_ranks),
980 .dest_points = std::move(owned_recv_points),
981 .dest_cells = std::move(owned_recv_cells)};
982}
