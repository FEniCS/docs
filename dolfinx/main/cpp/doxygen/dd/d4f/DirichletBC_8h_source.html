136  std::size_t num_owned(const FunctionSpace& V,
137  std::span<const std::int32_t> dofs)
138  {
139  int bs = V.dofmap()->index_map_bs();
140  std::int32_t map_size = V.dofmap()->index_map->size_local();
141  std::int32_t owned_size = bs * map_size;
142  auto it = std::lower_bound(dofs.begin(), dofs.end(), owned_size);
143  return std::distance(dofs.begin(), it);
144  }

147  std::vector<std::int32_t> unroll_dofs(std::span<const std::int32_t> dofs,
148  int bs)
149  {
150  std::vector<std::int32_t> dofs_unrolled(bs * dofs.size());
151  for (std::size_t i = 0; i < dofs.size(); ++i)
152  for (int k = 0; k < bs; ++k)
153  dofs_unrolled[bs * i + k] = bs * dofs[i] + k;
154  return dofs_unrolled;
155  }

201  template <typename U>
202  DirichletBC(std::shared_ptr<const Constant<T>> g, U&& dofs,
203  std::shared_ptr<const FunctionSpace> V)
204  : _function_space(V), _g(g), _dofs0(std::forward<U>(dofs)),
205  _owned_indices0(num_owned(*V, _dofs0))
206  {
207  assert(g);
208  assert(V);
209  if (g->shape.size() != V->element()->value_shape().size())
210  {
211  throw std::runtime_error(
212  "Rank mis-match between Constant and function space in DirichletBC");
213  }
214 
215  if (g->value.size() != _function_space->dofmap()->bs())
216  {
217  throw std::runtime_error(
218  "Creating a DirichletBC using a Constant is not supported when the "
219  "Constant size is not equal to the block size of the constrained "
220  "(sub-)space. Use a fem::Function to create the fem::DirichletBC.");
221  }
222 
223  if (!V->element()->interpolation_ident())
224  {
225  throw std::runtime_error(
226  "Constant can be used only with point-evaluation elements");
227  }
228 
229  // Unroll _dofs0 if dofmap block size > 1
230  if (const int bs = V->dofmap()->bs(); bs > 1)
231  {
232  _owned_indices0 *= bs;
233  _dofs0 = unroll_dofs(_dofs0, bs);
234  }
235  }

353  void set(std::span<T> x, double scale = 1.0) const
354  {
355  if (std::holds_alternative<std::shared_ptr<const Function<T>>>(_g))
356  {
357  auto g = std::get<std::shared_ptr<const Function<T>>>(_g);
358  assert(g);
359  std::span<const T> values = g->x()->array();
360  auto dofs1_g = _dofs1_g.empty() ? std::span(_dofs0) : std::span(_dofs1_g);
361  std::int32_t x_size = x.size();
362  for (std::size_t i = 0; i < _dofs0.size(); ++i)
363  {
364  if (_dofs0[i] < x_size)
365  {
366  assert(dofs1_g[i] < (std::int32_t)values.size());
367  x[_dofs0[i]] = scale * values[dofs1_g[i]];
368  }
369  }
370  }
371  else if (std::holds_alternative<std::shared_ptr<const Constant<T>>>(_g))
372  {
373  auto g = std::get<std::shared_ptr<const Constant<T>>>(_g);
374  std::vector<T> value = g->value;
375  int bs = _function_space->dofmap()->bs();
376  std::int32_t x_size = x.size();
377  std::for_each(_dofs0.cbegin(), _dofs0.cend(),
378  [x_size, bs, scale, &value, &x](auto dof)
379  {
380  if (dof < x_size)
381  x[dof] = scale * value[dof % bs];
382  });
383  }
384  }

390  void set(std::span<T> x, const std::span<const T>& x0,
391  double scale = 1.0) const
392  {
393  if (std::holds_alternative<std::shared_ptr<const Function<T>>>(_g))
394  {
395  auto g = std::get<std::shared_ptr<const Function<T>>>(_g);
396  assert(g);
397  std::span<const T> values = g->x()->array();
398  assert(x.size() <= x0.size());
399  auto dofs1_g = _dofs1_g.empty() ? std::span(_dofs0) : std::span(_dofs1_g);
400  std::int32_t x_size = x.size();
401  for (std::size_t i = 0; i < _dofs0.size(); ++i)
402  {
403  if (_dofs0[i] < x_size)
404  {
405  assert(dofs1_g[i] < (std::int32_t)values.size());
406  x[_dofs0[i]] = scale * (values[dofs1_g[i]] - x0[_dofs0[i]]);
407  }
408  }
409  }
410  else if (std::holds_alternative<std::shared_ptr<const Constant<T>>>(_g))
411  {
412  auto g = std::get<std::shared_ptr<const Constant<T>>>(_g);
413  const std::vector<T>& value = g->value;
414  std::int32_t bs = _function_space->dofmap()->bs();
415  std::for_each(_dofs0.cbegin(), _dofs0.cend(),
416  [&x, &x0, &value, scale, bs](auto dof)
417  {
418  if (dof < (std::int32_t)x.size())
419  x[dof] = scale * (value[dof % bs] - x0[dof]);
420  });
421  }
422  }

441  values[_dofs0[i]] = g_values[dofs1_g[i]];

444  {

446  assert(g);

448  const std::int32_t bs = _function_space->dofmap()->bs();
449  for (std::size_t i = 0; i < _dofs0.size(); ++i)
450  values[_dofs0[i]] = g_value[_dofs0[i] % bs];
