63template <int num_cells>
64std::array<std::int32_t, 2 * num_cells>
65get_cell_facet_pairs(std::int32_t f, const std::span<const std::int32_t>& cells,
66 const graph::AdjacencyList<std::int32_t>& c_to_f)
67{
68 // Loop over cells sharing facet
69 assert(cells.size() == num_cells);
70 std::array<std::int32_t, 2 * num_cells> cell_local_facet_pairs;
71 for (int c = 0; c < num_cells; ++c)
72 {
73 // Get local index of facet with respect to the cell
74 std::int32_t cell = cells[c];
75 auto cell_facets = c_to_f.links(cell);
76 auto facet_it = std::find(cell_facets.begin(), cell_facets.end(), f);
77 assert(facet_it != cell_facets.end());
78 int local_f = std::distance(cell_facets.begin(), facet_it);
79 cell_local_facet_pairs[2 * c] = cell;
80 cell_local_facet_pairs[2 * c + 1] = local_f;
81 }
82
83 return cell_local_facet_pairs;
84}

154template <typename T>
155la::SparsityPattern create_sparsity_pattern(const Form<T>& a)
156{
157 if (a.rank() != 2)
158 {
159 throw std::runtime_error(
160 "Cannot create sparsity pattern. Form is not a bilinear form");
161 }
162
163 // Get dof maps and mesh
164 std::array<std::reference_wrapper<const DofMap>, 2> dofmaps{
165 *a.function_spaces().at(0)->dofmap(),
166 *a.function_spaces().at(1)->dofmap()};
167 std::shared_ptr mesh = a.mesh();
168 assert(mesh);
169
170 const std::set<IntegralType> types = a.integral_types();
171 if (types.find(IntegralType::interior_facet) != types.end()
172 or types.find(IntegralType::exterior_facet) != types.end())
173 {
174 // FIXME: cleanup these calls? Some of the happen internally again.
175 const int tdim = mesh->topology().dim();
176 mesh->topology_mutable().create_entities(tdim - 1);
177 mesh->topology_mutable().create_connectivity(tdim - 1, tdim);
178 }
179
180 common::Timer t0("Build sparsity");
181
182 // Get common::IndexMaps for each dimension
183 const std::array index_maps{dofmaps[0].get().index_map,
184 dofmaps[1].get().index_map};
185 const std::array bs
186 = {dofmaps[0].get().index_map_bs(), dofmaps[1].get().index_map_bs()};
187
188 // Create and build sparsity pattern
189 la::SparsityPattern pattern(mesh->comm(), index_maps, bs);
190 for (auto type : types)
191 {
192 std::vector<int> ids = a.integral_ids(type);
193 switch (type)
194 {
195 case IntegralType::cell:
196 for (int id : ids)
197 {
198 const std::vector<std::int32_t>& cells = a.cell_domains(id);
199 sparsitybuild::cells(pattern, cells, {{dofmaps[0], dofmaps[1]}});
200 }
201 break;
202 case IntegralType::interior_facet:
203 for (int id : ids)
204 {
205 const std::vector<std::int32_t>& facets = a.interior_facet_domains(id);
206 sparsitybuild::interior_facets(pattern, facets,
207 {{dofmaps[0], dofmaps[1]}});
208 }
209 break;
210 case IntegralType::exterior_facet:
211 for (int id : ids)
212 {
213 const std::vector<std::int32_t>& facets = a.exterior_facet_domains(id);
214 sparsitybuild::exterior_facets(pattern, facets,
215 {{dofmaps[0], dofmaps[1]}});
216 }
217 break;
218 default:
219 throw std::runtime_error("Unsupported integral type");
220 }
221 }
222
223 t0.stop();
224
225 return pattern;
226}

266template <typename T>
267Form<T> create_form(
268 const ufcx_form& ufcx_form,
269 const std::vector<std::shared_ptr<const FunctionSpace>>& spaces,
270 const std::vector<std::shared_ptr<const Function<T>>>& coefficients,
271 const std::vector<std::shared_ptr<const Constant<T>>>& constants,
272 const std::map<IntegralType, const mesh::MeshTags<int>*>& subdomains,
273 std::shared_ptr<const mesh::Mesh> mesh = nullptr)
274{
275 if (ufcx_form.rank != (int)spaces.size())
276 throw std::runtime_error("Wrong number of argument spaces for Form.");
277 if (ufcx_form.num_coefficients != (int)coefficients.size())
278 {
279 throw std::runtime_error(
280 "Mismatch between number of expected and provided Form coefficients.");
281 }
282 if (ufcx_form.num_constants != (int)constants.size())
283 {
284 throw std::runtime_error(
285 "Mismatch between number of expected and provided Form constants.");
286 }
287
288 // Check argument function spaces
289#ifndef NDEBUG
290 for (std::size_t i = 0; i < spaces.size(); ++i)
291 {
292 assert(spaces[i]->element());
293 ufcx_finite_element* ufcx_element = ufcx_form.finite_elements[i];
294 assert(ufcx_element);
295 if (std::string(ufcx_element->signature)
296 != spaces[i]->element()->signature())
297 {
298 throw std::runtime_error(
299 "Cannot create form. Wrong type of function space for argument.");
300 }
301 }
302#endif
303
304 // Extract mesh from FunctionSpace, and check they are the same
305 if (!mesh and !spaces.empty())
306 mesh = spaces[0]->mesh();
307 for (auto& V : spaces)
308 {
309 if (mesh != V->mesh())
310 throw std::runtime_error("Incompatible mesh");
311 }
312 if (!mesh)
313 throw std::runtime_error("No mesh could be associated with the Form.");
314
315 const mesh::Topology& topology = mesh->topology();
316 const int tdim = topology.dim();
317
318 // Create facets, if required
319 if (ufcx_form.num_integrals(exterior_facet) > 0
320 or ufcx_form.num_integrals(interior_facet) > 0)
321 {
322 mesh->topology_mutable().create_entities(tdim - 1);
323 mesh->topology_mutable().create_connectivity(tdim - 1, tdim);
324 mesh->topology_mutable().create_connectivity(tdim, tdim - 1);
325 }
326
327 // Get list of integral IDs, and load tabulate tensor into memory for
328 // each
329 using kern = std::function<void(
330 T*, const T*, const T*,
331 const typename impl::scalar_value_type<T>::value_type*, const int*,
332 const std::uint8_t*)>;
333 std::map<IntegralType,
334 std::vector<std::tuple<int, kern, std::vector<std::int32_t>>>>
335 integral_data;
336
337 bool needs_facet_permutations = false;
338
339 // Attach cell kernels
340 {
341 std::span<const int> ids(ufcx_form.integral_ids(cell),
342 ufcx_form.num_integrals(cell));
343 auto itg = integral_data.insert({IntegralType::cell, {}});
344 auto sd = subdomains.find(IntegralType::cell);
345 for (int i = 0; i < ufcx_form.num_integrals(cell); ++i)
346 {
347 const int id = ids[i];
348 ufcx_integral* integral = ufcx_form.integrals(cell)[i];
349 assert(integral);
350
351 kern k = nullptr;
352 if constexpr (std::is_same_v<T, float>)
353 k = integral->tabulate_tensor_float32;
354 else if constexpr (std::is_same_v<T, std::complex<float>>)
355 {
356 k = reinterpret_cast<void (*)(
357 T*, const T*, const T*,
358 const typename impl::scalar_value_type<T>::value_type*, const int*,
359 const unsigned char*)>(integral->tabulate_tensor_complex64);
360 }
361 else if constexpr (std::is_same_v<T, double>)
362 k = integral->tabulate_tensor_float64;
363 else if constexpr (std::is_same_v<T, std::complex<double>>)
364 {
365 k = reinterpret_cast<void (*)(
366 T*, const T*, const T*,
367 const typename impl::scalar_value_type<T>::value_type*, const int*,
368 const unsigned char*)>(integral->tabulate_tensor_complex128);
369 }
370 assert(k);
371
372 // Build list of entities to assembler over
373 std::vector<std::int32_t> e;
374 if (id == -1)
375 {
376 // Default kernel, operates on all (owned) cells
377 assert(topology.index_map(tdim));
378 e.resize(topology.index_map(tdim)->size_local(), 0);
379 std::iota(e.begin(), e.end(), 0);
380 }
381 else if (sd != subdomains.end() and sd->second)
382 {
383 assert(topology.index_map(tdim));
384 std::span<const std::int32_t> entities = sd->second->indices();
385 std::span<const int> values = sd->second->values();
386 auto it0 = entities.begin();
387 auto it1 = std::lower_bound(it0, entities.end(),
388 topology.index_map(tdim)->size_local());
389 entities = entities.first(std::distance(it0, it1));
390 for (std::size_t j = 0; j < entities.size(); ++j)
391 if (values[j] == id)
392 e.push_back(entities[j]);
393 }
394
395 itg.first->second.emplace_back(id, k, std::move(e));
396 if (integral->needs_facet_permutations)
397 needs_facet_permutations = true;
398 }
399 }
400
401 // Attach exterior facet kernels
402 {
403 std::span<const int> ids(ufcx_form.integral_ids(exterior_facet),
404 ufcx_form.num_integrals(exterior_facet));
405 auto itg = integral_data.insert({IntegralType::exterior_facet, {}});
406 auto sd = subdomains.find(IntegralType::exterior_facet);
407 for (int i = 0; i < ufcx_form.num_integrals(exterior_facet); ++i)
408 {
409 const int id = ids[i];
410 ufcx_integral* integral = ufcx_form.integrals(exterior_facet)[i];
411 assert(integral);
412
413 kern k = nullptr;
414 if constexpr (std::is_same_v<T, float>)
415 k = integral->tabulate_tensor_float32;
416 else if constexpr (std::is_same_v<T, std::complex<float>>)
417 {
418 k = reinterpret_cast<void (*)(
419 T*, const T*, const T*,
420 const typename impl::scalar_value_type<T>::value_type*, const int*,
421 const unsigned char*)>(integral->tabulate_tensor_complex64);
422 }
423 else if constexpr (std::is_same_v<T, double>)
424 k = integral->tabulate_tensor_float64;
425 else if constexpr (std::is_same_v<T, std::complex<double>>)
426 {
427 k = reinterpret_cast<void (*)(
428 T*, const T*, const T*,
429 const typename impl::scalar_value_type<T>::value_type*, const int*,
430 const unsigned char*)>(integral->tabulate_tensor_complex128);
431 }
432 assert(k);
433
434 // Build list of entities to assembler over
435 std::vector<std::int32_t> e;
436 const std::vector bfacets = mesh::exterior_facet_indices(topology);
437 auto f_to_c = topology.connectivity(tdim - 1, tdim);
438 assert(f_to_c);
439 auto c_to_f = topology.connectivity(tdim, tdim - 1);
440 assert(c_to_f);
441 if (id == -1)
442 {
443 // Default kernel, operates on all (owned) exterior facets
444 e.reserve(2 * bfacets.size());
445 for (std::int32_t f : bfacets)
446 {
447 // There will only be one pair for an exterior facet integral
448 auto pair
449 = impl::get_cell_facet_pairs<1>(f, f_to_c->links(f), *c_to_f);
450 e.insert(e.end(), pair.begin(), pair.end());
451 }
452 }
453 else if (sd != subdomains.end() and sd->second)
454 {
455 // Create list of tagged boundary facets
456 std::span<const std::int32_t> entities = sd->second->indices();
457 std::span<const int> values = sd->second->values();
458 std::vector<std::int32_t> facets;
459 std::set_intersection(entities.begin(), entities.end(), bfacets.begin(),
460 bfacets.end(), std::back_inserter(facets));
461 for (auto f : facets)
462 {
463 auto index_it = std::lower_bound(entities.begin(), entities.end(), f);
464 assert(index_it != entities.end() and *index_it == f);
465 std::size_t pos = std::distance(entities.begin(), index_it);
466 if (values[pos] == id)
467 {
468 auto facet
469 = impl::get_cell_facet_pairs<1>(f, f_to_c->links(f), *c_to_f);
470 e.insert(e.end(), facet.begin(), facet.end());
471 }
472 }
473 }
474
475 itg.first->second.emplace_back(id, k, std::move(e));
476 if (integral->needs_facet_permutations)
477 needs_facet_permutations = true;
478 }
479 }
480
481 // Attach interior facet kernels
482 {
483 std::span<const int> ids(ufcx_form.integral_ids(interior_facet),
484 ufcx_form.num_integrals(interior_facet));
485 auto itg = integral_data.insert({IntegralType::interior_facet, {}});
486 auto sd = subdomains.find(IntegralType::interior_facet);
487 for (int i = 0; i < ufcx_form.num_integrals(interior_facet); ++i)
488 {
489 const int id = ids[i];
490 ufcx_integral* integral = ufcx_form.integrals(interior_facet)[i];
491 assert(integral);
492
493 kern k = nullptr;
494 if constexpr (std::is_same_v<T, float>)
495 k = integral->tabulate_tensor_float32;
496 else if constexpr (std::is_same_v<T, std::complex<float>>)
497 {
498 k = reinterpret_cast<void (*)(
499 T*, const T*, const T*,
500 const typename impl::scalar_value_type<T>::value_type*, const int*,
501 const unsigned char*)>(integral->tabulate_tensor_complex64);
502 }
503 else if constexpr (std::is_same_v<T, double>)
504 k = integral->tabulate_tensor_float64;
505 else if constexpr (std::is_same_v<T, std::complex<double>>)
506 {
507 k = reinterpret_cast<void (*)(
508 T*, const T*, const T*,
509 const typename impl::scalar_value_type<T>::value_type*, const int*,
510 const unsigned char*)>(integral->tabulate_tensor_complex128);
511 }
512 assert(k);
513
514 // Build list of entities to assembler over
515 std::vector<std::int32_t> e;
516 auto f_to_c = topology.connectivity(tdim - 1, tdim);
517 assert(f_to_c);
518 auto c_to_f = topology.connectivity(tdim, tdim - 1);
519 assert(c_to_f);
520 if (id == -1)
521 {
522 // Default kernel, operates on all (owned) interior facets
523 assert(topology.index_map(tdim - 1));
524 std::int32_t num_facets = topology.index_map(tdim - 1)->size_local();
525 e.reserve(4 * num_facets);
526 for (std::int32_t f = 0; f < num_facets; ++f)
527 {
528 if (f_to_c->num_links(f) == 2)
529 {
530 auto pairs
531 = impl::get_cell_facet_pairs<2>(f, f_to_c->links(f), *c_to_f);
532 e.insert(e.end(), pairs.begin(), pairs.end());
533 }
534 }
535 }
536 else if (sd != subdomains.end() and sd->second)
537 {
538 std::span<const std::int32_t> entities = sd->second->indices();
539 std::span<const int> values = sd->second->values();
540 for (std::size_t j = 0; j < entities.size(); ++j)
541 {
542 const std::int32_t f = entities[j];
543 if (f_to_c->num_links(f) == 2 and values[j] == id)
544 {
545 // Get the facet as a pair of (cell, local facet) pairs, one
546 // for each cell
547 auto facets
548 = impl::get_cell_facet_pairs<2>(f, f_to_c->links(f), *c_to_f);
549 e.insert(e.end(), facets.begin(), facets.end());
550 }
551 }
552 }
553
554 itg.first->second.emplace_back(id, k, std::move(e));
555 if (integral->needs_facet_permutations)
556 needs_facet_permutations = true;
557 }
558 }
559
560 return Form<T>(spaces, integral_data, coefficients, constants,
561 needs_facet_permutations, mesh);
562}

573template <typename T>
574Form<T> create_form(
575 const ufcx_form& ufcx_form,
576 const std::vector<std::shared_ptr<const FunctionSpace>>& spaces,
577 const std::map<std::string, std::shared_ptr<const Function<T>>>&
578 coefficients,
579 const std::map<std::string, std::shared_ptr<const Constant<T>>>& constants,
580 const std::map<IntegralType, const mesh::MeshTags<int>*>& subdomains,
581 std::shared_ptr<const mesh::Mesh> mesh = nullptr)
582{
583 // Place coefficients in appropriate order
584 std::vector<std::shared_ptr<const Function<T>>> coeff_map;
585 for (const std::string& name : get_coefficient_names(ufcx_form))
586 {
587 if (auto it = coefficients.find(name); it != coefficients.end())
588 coeff_map.push_back(it->second);
589 else
590 {
591 throw std::runtime_error("Form coefficient \"" + name
592 + "\" not provided.");
593 }
594 }
595
596 // Place constants in appropriate order
597 std::vector<std::shared_ptr<const Constant<T>>> const_map;
598 for (const std::string& name : get_constant_names(ufcx_form))
599 {
600 if (auto it = constants.find(name); it != constants.end())
601 const_map.push_back(it->second);
602 else
603 throw std::runtime_error("Form constant \"" + name + "\" not provided.");
604 }
605
606 return create_form(ufcx_form, spaces, coeff_map, const_map, subdomains, mesh);
607}

695 cell_info = std::span(mesh->topology().get_cell_permutation_info());
696 }
697
698 return cell_info;
699}

702template <typename T, int _bs>
703void pack(std::span<T> coeffs, std::int32_t cell, int bs, std::span<const T> v,
704 std::span<const std::uint32_t> cell_info, const DofMap& dofmap,
705 auto transform)
706{
707 auto dofs = dofmap.cell_dofs(cell);
708 for (std::size_t i = 0; i < dofs.size(); ++i)
709 {
710 if constexpr (_bs < 0)
711 {
712 const int pos_c = bs * i;
713 const int pos_v = bs * dofs[i];
714 for (int k = 0; k < bs; ++k)
715 coeffs[pos_c + k] = v[pos_v + k];
716 }
717 else
718 {
719 const int pos_c = _bs * i;
720 const int pos_v = _bs * dofs[i];
721 for (int k = 0; k < _bs; ++k)
722 coeffs[pos_c + k] = v[pos_v + k];
723 }
724 }
725
726 transform(coeffs, cell_info, cell, 1);
727}

759{
760 // Read data from coefficient Function u
761 std::span<const T> v = u.x()->array();
762 const DofMap& dofmap = *u.function_space()->dofmap();
763 std::shared_ptr<const FiniteElement> element = u.function_space()->element();
764 assert(element);
765 int space_dim = element->space_dimension();
766 const auto transformation
767 = element->get_dof_transformation_function<T>(false, true);
768
769 const int bs = dofmap.bs();
770 switch (bs)
771 {
772 case 1:
773 for (std::size_t e = 0; e < entities.size(); e += estride)
774 {
775 auto entity = entities.subspan(e, estride);
776 std::int32_t cell = fetch_cells(entity);
777 auto cell_coeff = c.subspan((e / estride) * cstride + offset, space_dim);
778 pack<T, 1>(cell_coeff, cell, bs, v, cell_info, dofmap, transformation);
779 }
780 break;
781 case 2:
782 for (std::size_t e = 0; e < entities.size(); e += estride)
783 {
784 auto entity = entities.subspan(e, estride);
785 std::int32_t cell = fetch_cells(entity);
786 auto cell_coeff = c.subspan((e / estride) * cstride + offset, space_dim);
787 pack<T, 2>(cell_coeff, cell, bs, v, cell_info, dofmap, transformation);
788 }
789 break;
790 case 3:
791 for (std::size_t e = 0; e < entities.size(); e += estride)
792 {
793 auto entity = entities.subspan(e, estride);
794 std::int32_t cell = fetch_cells(entity);
795 auto cell_coeff = c.subspan(e / estride * cstride + offset, space_dim);
796 pack<T, 3>(cell_coeff, cell, bs, v, cell_info, dofmap, transformation);
797 }
798 break;
799 default:
800 for (std::size_t e = 0; e < entities.size(); e += estride)
801 {
802 auto entity = entities.subspan(e, estride);
803 std::int32_t cell = fetch_cells(entity);
804 auto cell_coeff = c.subspan((e / estride) * cstride + offset, space_dim);
805 pack<T, -1>(cell_coeff, cell, bs, v, cell_info, dofmap, transformation);
806 }
807 break;
808 }
809}

819template <typename T>
820std::pair<std::vector<T>, int>
821allocate_coefficient_storage(const Form<T>& form, IntegralType integral_type,
822 int id)
823{
824 // Get form coefficient offsets and dofmaps
825 const std::vector<std::shared_ptr<const Function<T>>>& coefficients
826 = form.coefficients();
827 const std::vector<int> offsets = form.coefficient_offsets();
828
829 std::size_t num_entities = 0;
830 int cstride = 0;
831 if (!coefficients.empty())
832 {
833 cstride = offsets.back();
834 switch (integral_type)
835 {
836 case IntegralType::cell:
837 num_entities = form.cell_domains(id).size();
838 break;
839 case IntegralType::exterior_facet:
840 num_entities = form.exterior_facet_domains(id).size() / 2;
841 break;
842 case IntegralType::interior_facet:
843 num_entities = form.interior_facet_domains(id).size() / 2;
844 break;
845 default:
846 throw std::runtime_error("Could not allocate coefficient data. "
847 "Integral type not supported.");
848 }
849 }
850
851 return {std::vector<T>(num_entities * cstride), cstride};
852}

883template <typename T>
884void pack_coefficients(const Form<T>& form, IntegralType integral_type, int id,
885 std::span<T> c, int cstride)
886{
887 // Get form coefficient offsets and dofmaps
888 const std::vector<std::shared_ptr<const Function<T>>>& coefficients
889 = form.coefficients();
890 const std::vector<int> offsets = form.coefficient_offsets();
891
892 if (!coefficients.empty())
893 {
894 std::span<const std::uint32_t> cell_info
895 = impl::get_cell_orientation_info(coefficients);
896
897 switch (integral_type)
898 {
899 case IntegralType::cell:
900 {
901 auto fetch_cell = [](auto entity) { return entity.front(); };
902 const std::vector<std::int32_t>& cells = form.cell_domains(id);
903
904 // Iterate over coefficients
905 for (std::size_t coeff = 0; coeff < coefficients.size(); ++coeff)
906 {
907 impl::pack_coefficient_entity(c, cstride, *coefficients[coeff],
908 cell_info, cells, 1, fetch_cell,
909 offsets[coeff]);
910 }
911 break;
912 }
913 case IntegralType::exterior_facet:
914 {
915 const std::vector<std::int32_t>& facets = form.exterior_facet_domains(id);
916
917 // Function to fetch cell index from exterior facet entity
918 auto fetch_cell = [](auto entity) { return entity.front(); };
919
920 // Iterate over coefficients
921 for (std::size_t coeff = 0; coeff < coefficients.size(); ++coeff)
922 {
923 impl::pack_coefficient_entity(c, cstride, *coefficients[coeff],
924 cell_info, facets, 2, fetch_cell,
925 offsets[coeff]);
926 }
927 break;
928 }
929 case IntegralType::interior_facet:
930 {
931 const std::vector<std::int32_t>& facets = form.interior_facet_domains(id);
932
933 // Functions to fetch cell indices from interior facet entity
934 auto fetch_cell0 = [](auto entity) { return entity[0]; };
935 auto fetch_cell1 = [](auto entity) { return entity[2]; };
936
937 // Iterate over coefficients
938 for (std::size_t coeff = 0; coeff < coefficients.size(); ++coeff)
939 {
940 // Pack coefficient ['+']
941 impl::pack_coefficient_entity(c, 2 * cstride, *coefficients[coeff],
942 cell_info, facets, 4, fetch_cell0,
943 2 * offsets[coeff]);
944 // Pack coefficient ['-']
945 impl::pack_coefficient_entity(c, 2 * cstride, *coefficients[coeff],
946 cell_info, facets, 4, fetch_cell1,
947 offsets[coeff] + offsets[coeff + 1]);
948 }
949 break;
950 }
951 default:
952 throw std::runtime_error(
953 "Could not pack coefficient. Integral type not supported.");
954 }
955 }

959template <typename T>
960Expression<T> create_expression(
961 const ufcx_expression& expression,
962 const std::vector<std::shared_ptr<const Function<T>>>& coefficients,
963 const std::vector<std::shared_ptr<const Constant<T>>>& constants,
964 std::shared_ptr<const mesh::Mesh> mesh = nullptr,
965 std::shared_ptr<const FunctionSpace> argument_function_space = nullptr)
966{
967 if (expression.rank > 0 and !argument_function_space)
968 {
969 throw std::runtime_error("Expression has Argument but no Argument "
970 "function space was provided.");
971 }
972
973 const std::size_t size
974 = expression.num_points * expression.topological_dimension;
975 std::span<const double> X(expression.points, size);
976 std::array<std::size_t, 2> Xshape
977 = {static_cast<std::size_t>(expression.num_points),
978 static_cast<std::size_t>(expression.topological_dimension)};
979
980 std::vector<int> value_shape;
981 for (int i = 0; i < expression.num_components; ++i)
982 value_shape.push_back(expression.value_shape[i]);
983
984 std::function<void(T*, const T*, const T*,
985 const typename impl::scalar_value_type<T>::value_type*,
986 const int*, const std::uint8_t*)>
987 tabulate_tensor = nullptr;
988 if constexpr (std::is_same_v<T, float>)
989 tabulate_tensor = expression.tabulate_tensor_float32;
990 else if constexpr (std::is_same_v<T, std::complex<float>>)
991 {
992 tabulate_tensor = reinterpret_cast<void (*)(
993 T*, const T*, const T*,
994 const typename impl::scalar_value_type<T>::value_type*, const int*,
995 const unsigned char*)>(expression.tabulate_tensor_complex64);
996 }
997 else if constexpr (std::is_same_v<T, double>)
998 tabulate_tensor = expression.tabulate_tensor_float64;
999 else if constexpr (std::is_same_v<T, std::complex<double>>)
1000 {
1001 tabulate_tensor = reinterpret_cast<void (*)(
1002 T*, const T*, const T*,
1003 const typename impl::scalar_value_type<T>::value_type*, const int*,
1004 const unsigned char*)>(expression.tabulate_tensor_complex128);
1005 }
1006 else
1007 throw std::runtime_error("Type not supported.");
1008
1009 assert(tabulate_tensor);
1010 return Expression(coefficients, constants, X, Xshape, tabulate_tensor,
1011 value_shape, mesh, argument_function_space);
1012}

1016template <typename T>
1017Expression<T> create_expression(
1018 const ufcx_expression& expression,
1019 const std::map<std::string, std::shared_ptr<const Function<T>>>&
1020 coefficients,
1021 const std::map<std::string, std::shared_ptr<const Constant<T>>>& constants,

1024{
1025 // Place coefficients in appropriate order

1027 std::vector<std::string> coefficient_names;
1028 for (int i = 0; i < expression.num_coefficients; ++i)

1030

1053 {

1113template <typename U>
1114std::vector<typename U::scalar_type> pack_constants(const U& u)
1115{
1116 using T = typename U::scalar_type;
1117 const std::vector<std::shared_ptr<const Constant<T>>>& constants
1118 = u.constants();
1119
1120 // Calculate size of array needed to store packed constants
1121 std::int32_t size = std::accumulate(constants.cbegin(), constants.cend(), 0,
1122 [](std::int32_t sum, auto& constant)
1123 { return sum + constant->value.size(); });
1124
1125 // Pack constants
1126 std::vector<T> constant_values(size);
1127 std::int32_t offset = 0;
1128 for (auto& constant : constants)
1129 {
1130 const std::vector<T>& value = constant->value;
1131 std::copy(value.begin(), value.end(),
1132 std::next(constant_values.begin(), offset));
1133 offset += value.size();
1134 }
1135
1136 return constant_values;
1137}
