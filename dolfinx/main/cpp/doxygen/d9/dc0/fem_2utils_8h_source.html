62template <int num_cells>
63std::array<std::int32_t, 2 * num_cells>
64get_cell_facet_pairs(std::int32_t f, std::span<const std::int32_t> cells,
65 const graph::AdjacencyList<std::int32_t>& c_to_f)
66{
67 // Loop over cells sharing facet
68 assert(cells.size() == num_cells);
69 std::array<std::int32_t, 2 * num_cells> cell_local_facet_pairs;
70 for (int c = 0; c < num_cells; ++c)
71 {
72 // Get local index of facet with respect to the cell
73 std::int32_t cell = cells[c];
74 auto cell_facets = c_to_f.links(cell);
75 auto facet_it = std::find(cell_facets.begin(), cell_facets.end(), f);
76 assert(facet_it != cell_facets.end());
77 int local_f = std::distance(cell_facets.begin(), facet_it);
78 cell_local_facet_pairs[2 * c] = cell;
79 cell_local_facet_pairs[2 * c + 1] = local_f;
80 }
81
82 return cell_local_facet_pairs;
83}

132 a.size(),
133 std::vector<std::array<std::shared_ptr<const FunctionSpace<U>>, 2>>(

135 for (std::size_t i = 0; i < a.size(); ++i)

141 }
142 }
143 return spaces;

151template <dolfinx::scalar T, std::floating_point U>
152la::SparsityPattern create_sparsity_pattern(const Form<T, U>& a)
153{
154 std::shared_ptr mesh = a.mesh();
155 assert(mesh);
156
157 // Get index maps and block sizes from the DOF maps. Note that in
158 // mixed-topology meshes, despite there being multiple DOF maps, the
159 // index maps and block sizes are the same.
160 std::array<std::reference_wrapper<const DofMap>, 2> dofmaps{
161 *a.function_spaces().at(0)->dofmaps(0),
162 *a.function_spaces().at(1)->dofmaps(0)};
163
164 const std::array index_maps{dofmaps[0].get().index_map,
165 dofmaps[1].get().index_map};
166 const std::array bs
167 = {dofmaps[0].get().index_map_bs(), dofmaps[1].get().index_map_bs()};
168
169 la::SparsityPattern pattern(mesh->comm(), index_maps, bs);
170 build_sparsity_pattern(pattern, a);
171 return pattern;
172}

179template <dolfinx::scalar T, std::floating_point U>
180void build_sparsity_pattern(la::SparsityPattern& pattern, const Form<T, U>& a)
181{
182 if (a.rank() != 2)
183 {
184 throw std::runtime_error(
185 "Cannot create sparsity pattern. Form is not a bilinear.");
186 }
187
188 std::shared_ptr mesh = a.mesh();
189 assert(mesh);
190 std::shared_ptr mesh0 = a.function_spaces().at(0)->mesh();
191 assert(mesh0);
192 std::shared_ptr mesh1 = a.function_spaces().at(1)->mesh();
193 assert(mesh1);
194
195 const std::set<IntegralType> types = a.integral_types();
196 if (types.find(IntegralType::interior_facet) != types.end()
197 or types.find(IntegralType::exterior_facet) != types.end())
198 {
199 // FIXME: cleanup these calls? Some of the happen internally again.
200 int tdim = mesh->topology()->dim();
201 mesh->topology_mutable()->create_entities(tdim - 1);
202 mesh->topology_mutable()->create_connectivity(tdim - 1, tdim);
203 }
204
205 common::Timer t0("Build sparsity");
206
207 auto extract_cells = [](std::span<const std::int32_t> facets)
208 {
209 assert(facets.size() % 2 == 0);
210 std::vector<std::int32_t> cells;
211 cells.reserve(facets.size() / 2);
212 for (std::size_t i = 0; i < facets.size(); i += 2)
213 cells.push_back(facets[i]);
214 return cells;
215 };
216
217 const int num_cell_types = mesh->topology()->cell_types().size();
218 for (int cell_type_idx = 0; cell_type_idx < num_cell_types; ++cell_type_idx)
219 {
220 std::array<std::reference_wrapper<const DofMap>, 2> dofmaps{
221 *a.function_spaces().at(0)->dofmaps(cell_type_idx),
222 *a.function_spaces().at(1)->dofmaps(cell_type_idx)};
223
224 // Create and build sparsity pattern
225 for (auto type : types)
226 {
227 std::vector<int> ids = a.integral_ids(type);
228 switch (type)
229 {
230 case IntegralType::cell:
231 for (int id : ids)
232 {
233 sparsitybuild::cells(pattern,
234 {a.domain(type, id, cell_type_idx, *mesh0),
235 a.domain(type, id, cell_type_idx, *mesh1)},
236 {{dofmaps[0], dofmaps[1]}});
237 }
238 break;
239 case IntegralType::interior_facet:
240 for (int id : ids)
241 {
242 sparsitybuild::interior_facets(
243 pattern,
244 {extract_cells(a.domain(type, id, *mesh0)),
245 extract_cells(a.domain(type, id, *mesh1))},
246 {{dofmaps[0], dofmaps[1]}});
247 }
248 break;
249 case IntegralType::exterior_facet:
250 for (int id : ids)
251 {
252 sparsitybuild::cells(pattern,
253 {extract_cells(a.domain(type, id, *mesh0)),
254 extract_cells(a.domain(type, id, *mesh1))},
255 {{dofmaps[0], dofmaps[1]}});
256 }
257 break;
258 default:
259 throw std::runtime_error("Unsupported integral type");
260 }
261 }
262 }
263
264 t0.stop();
265}

268template <std::floating_point T>
269ElementDofLayout create_element_dof_layout(const fem::FiniteElement<T>& element,
270 const std::vector<int>& parent_map
271 = {})
272{
273 // Create subdofmaps and compute offset
274 std::vector<int> offsets(1, 0);
275 std::vector<dolfinx::fem::ElementDofLayout> sub_doflayout;
276 int bs = element.block_size();
277 for (int i = 0; i < element.num_sub_elements(); ++i)
278 {
279 // The ith sub-element. For mixed elements this is subelements()[i]. For
280 // blocked elements, the sub-element will always be the same, so we'll use
281 // sub_elements()[0]
282 std::shared_ptr<const fem::FiniteElement<T>> sub_e
283 = element.sub_elements()[bs > 1 ? 0 : i];
284
285 // In a mixed element DOFs are ordered element by element, so the offset to
286 // the next sub-element is sub_e->space_dimension(). Blocked elements use
287 // xxyyzz ordering, so the offset to the next sub-element is 1
288
289 std::vector<int> parent_map_sub(sub_e->space_dimension(), offsets.back());
290 for (std::size_t j = 0; j < parent_map_sub.size(); ++j)
291 parent_map_sub[j] += bs * j;
292 offsets.push_back(offsets.back() + (bs > 1 ? 1 : sub_e->space_dimension()));
293 sub_doflayout.push_back(
294 dolfinx::fem::create_element_dof_layout(*sub_e, parent_map_sub));
295 }
296
297 return ElementDofLayout(bs, element.entity_dofs(),
298 element.entity_closure_dofs(), parent_map,
299 sub_doflayout);
300}

361template <dolfinx::scalar T, std::floating_point U = scalar_value_type_t<T>>
362Form<T, U> create_form_factory(
363 const std::vector<std::reference_wrapper<const ufcx_form>>& ufcx_forms,
364 const std::vector<std::shared_ptr<const FunctionSpace<U>>>& spaces,
365 const std::vector<std::shared_ptr<const Function<T, U>>>& coefficients,
366 const std::vector<std::shared_ptr<const Constant<T>>>& constants,
367 const std::map<
368 IntegralType,
369 std::vector<std::pair<std::int32_t, std::span<const std::int32_t>>>>&
370 subdomains,
371 const std::map<std::shared_ptr<const mesh::Mesh<U>>,
372 std::span<const std::int32_t>>& entity_maps,
373 std::shared_ptr<const mesh::Mesh<U>> mesh = nullptr)
374{
375 for (const ufcx_form& ufcx_form : ufcx_forms)
376 {
377 if (ufcx_form.rank != (int)spaces.size())
378 throw std::runtime_error("Wrong number of argument spaces for Form.");
379 if (ufcx_form.num_coefficients != (int)coefficients.size())
380 {
381 throw std::runtime_error("Mismatch between number of expected and "
382 "provided Form coefficients.");
383 }
384 if (ufcx_form.num_constants != (int)constants.size())
385 {
386 throw std::runtime_error(
387 "Mismatch between number of expected and provided Form constants.");
388 }
389 }
390
391 // Check argument function spaces
392 for (std::size_t form_idx = 0; form_idx < ufcx_forms.size(); ++form_idx)
393 {
394 for (std::size_t i = 0; i < spaces.size(); ++i)
395 {
396 assert(spaces[i]->elements(form_idx));
397 if (auto element_hash
398 = ufcx_forms[form_idx].get().finite_element_hashes[i];
399 element_hash != 0
400 and element_hash
401 != spaces[i]->elements(form_idx)->basix_element().hash())
402 {
403 throw std::runtime_error(
404 "Cannot create form. Elements are different to "
405 "those used to compile the form.");
406 }
407 }
408 }
409
410 // Extract mesh from FunctionSpace, and check they are the same
411 if (!mesh and !spaces.empty())
412 mesh = spaces[0]->mesh();
413 for (auto& V : spaces)
414 {
415 if (mesh != V->mesh() and entity_maps.find(V->mesh()) == entity_maps.end())
416 throw std::runtime_error(
417 "Incompatible mesh. entity_maps must be provided.");
418 }
419 if (!mesh)
420 throw std::runtime_error("No mesh could be associated with the Form.");
421
422 auto topology = mesh->topology();
423 assert(topology);
424 const int tdim = topology->dim();
425
426 // NOTE: This assumes all forms in mixed-topology meshes have the same
427 // integral offsets. Since the UFL forms for each type of cell should be
428 // the same, I think this assumption is OK.
429 const int* integral_offsets = ufcx_forms[0].get().form_integral_offsets;
430 std::vector<int> num_integrals_type(3);
431 for (int i = 0; i < 3; ++i)
432 num_integrals_type[i] = integral_offsets[i + 1] - integral_offsets[i];
433
434 // Create facets, if required
435 if (num_integrals_type[exterior_facet] > 0
436 or num_integrals_type[interior_facet] > 0)
437 {
438 mesh->topology_mutable()->create_entities(tdim - 1);
439 mesh->topology_mutable()->create_connectivity(tdim - 1, tdim);
440 mesh->topology_mutable()->create_connectivity(tdim, tdim - 1);
441 }
442
443 // Get list of integral IDs, and load tabulate tensor into memory for
444 // each
445 using kern_t = std::function<void(T*, const T*, const T*, const U*,
446 const int*, const std::uint8_t*)>;
447 std::map<IntegralType, std::vector<integral_data<T, U>>> integrals;
448
449 // Attach cell kernels
450 bool needs_facet_permutations = false;
451 {
452 std::vector<std::int32_t> default_cells;
453 std::span<const int> ids(ufcx_forms[0].get().form_integral_ids
454 + integral_offsets[cell],
455 num_integrals_type[cell]);
456 auto itg = integrals.insert({IntegralType::cell, {}});
457 auto sd = subdomains.find(IntegralType::cell);
458 for (std::size_t form_idx = 0; form_idx < ufcx_forms.size(); ++form_idx)
459 {
460 const ufcx_form& ufcx_form = ufcx_forms[form_idx];
461 for (int i = 0; i < num_integrals_type[cell]; ++i)
462 {
463 const int id = ids[i];
464 ufcx_integral* integral
465 = ufcx_form.form_integrals[integral_offsets[cell] + i];
466 assert(integral);
467
468 // Build list of active coefficients
469 std::vector<int> active_coeffs;
470 for (int j = 0; j < ufcx_form.num_coefficients; ++j)
471 {
472 if (integral->enabled_coefficients[j])
473 active_coeffs.push_back(j);
474 }
475
476 kern_t k = nullptr;
477 if constexpr (std::is_same_v<T, float>)
478 k = integral->tabulate_tensor_float32;
479#ifndef DOLFINX_NO_STDC_COMPLEX_KERNELS
480 else if constexpr (std::is_same_v<T, std::complex<float>>)
481 {
482 k = reinterpret_cast<void (*)(
483 T*, const T*, const T*,
484 const typename scalar_value_type<T>::value_type*, const int*,
485 const unsigned char*)>(integral->tabulate_tensor_complex64);
486 }
487#endif // DOLFINX_NO_STDC_COMPLEX_KERNELS
488 else if constexpr (std::is_same_v<T, double>)
489 k = integral->tabulate_tensor_float64;
490#ifndef DOLFINX_NO_STDC_COMPLEX_KERNELS
491 else if constexpr (std::is_same_v<T, std::complex<double>>)
492 {
493 k = reinterpret_cast<void (*)(
494 T*, const T*, const T*,
495 const typename scalar_value_type<T>::value_type*, const int*,
496 const unsigned char*)>(integral->tabulate_tensor_complex128);
497 }
498#endif // DOLFINX_NO_STDC_COMPLEX_KERNELS
499
500 if (!k)
501 {
502 throw std::runtime_error(
503 "UFCx kernel function is NULL. Check requested types.");
504 }
505
506 // Build list of entities to assemble over
507 if (id == -1)
508 {
509 // Default kernel, operates on all (owned) cells
510 assert(topology->index_maps(tdim).at(form_idx));
511 default_cells.resize(
512 topology->index_maps(tdim).at(form_idx)->size_local(), 0);
513 std::iota(default_cells.begin(), default_cells.end(), 0);
514 itg.first->second.emplace_back(id, k, default_cells, active_coeffs);
515 }
516 else if (sd != subdomains.end())
517 {
518 // NOTE: This requires that pairs are sorted
519 auto it = std::ranges::lower_bound(sd->second, id, std::less<>{},
520 [](auto& a) { return a.first; });
521 if (it != sd->second.end() and it->first == id)
522 itg.first->second.emplace_back(id, k, it->second, active_coeffs);
523 }
524
525 if (integral->needs_facet_permutations)
526 needs_facet_permutations = true;
527 }
528 }
529 }
530
531 // Attach exterior facet kernels
532 std::vector<std::int32_t> default_facets_ext;
533 {
534 std::span<const int> ids(ufcx_forms[0].get().form_integral_ids
535 + integral_offsets[exterior_facet],
536 num_integrals_type[exterior_facet]);
537 auto itg = integrals.insert({IntegralType::exterior_facet, {}});
538 auto sd = subdomains.find(IntegralType::exterior_facet);
539 for (std::size_t form_idx = 0; form_idx < ufcx_forms.size(); ++form_idx)
540 {
541 const ufcx_form& ufcx_form = ufcx_forms[form_idx];
542 for (int i = 0; i < num_integrals_type[exterior_facet]; ++i)
543 {
544 const int id = ids[i];
545 ufcx_integral* integral
546 = ufcx_form.form_integrals[integral_offsets[exterior_facet] + i];
547 assert(integral);
548 std::vector<int> active_coeffs;
549 for (int j = 0; j < ufcx_form.num_coefficients; ++j)
550 {
551 if (integral->enabled_coefficients[j])
552 active_coeffs.push_back(j);
553 }
554
555 kern_t k = nullptr;
556 if constexpr (std::is_same_v<T, float>)
557 k = integral->tabulate_tensor_float32;
558#ifndef DOLFINX_NO_STDC_COMPLEX_KERNELS
559 else if constexpr (std::is_same_v<T, std::complex<float>>)
560 {
561 k = reinterpret_cast<void (*)(
562 T*, const T*, const T*,
563 const typename scalar_value_type<T>::value_type*, const int*,
564 const unsigned char*)>(integral->tabulate_tensor_complex64);
565 }
566#endif // DOLFINX_NO_STDC_COMPLEX_KERNELS
567 else if constexpr (std::is_same_v<T, double>)
568 k = integral->tabulate_tensor_float64;
569#ifndef DOLFINX_NO_STDC_COMPLEX_KERNELS
570 else if constexpr (std::is_same_v<T, std::complex<double>>)
571 {
572 k = reinterpret_cast<void (*)(
573 T*, const T*, const T*,
574 const typename scalar_value_type<T>::value_type*, const int*,
575 const unsigned char*)>(integral->tabulate_tensor_complex128);
576 }
577#endif // DOLFINX_NO_STDC_COMPLEX_KERNELS
578 assert(k);
579
580 // Build list of entities to assembler over
581 const std::vector bfacets = mesh::exterior_facet_indices(*topology);
582 auto f_to_c = topology->connectivity(tdim - 1, tdim);
583 assert(f_to_c);
584 auto c_to_f = topology->connectivity(tdim, tdim - 1);
585 assert(c_to_f);
586 if (id == -1)
587 {
588 // Default kernel, operates on all (owned) exterior facets
589 default_facets_ext.reserve(2 * bfacets.size());
590 for (std::int32_t f : bfacets)
591 {
592 // There will only be one pair for an exterior facet integral
593 auto pair
594 = impl::get_cell_facet_pairs<1>(f, f_to_c->links(f), *c_to_f);
595 default_facets_ext.insert(default_facets_ext.end(), pair.begin(),
596 pair.end());
597 }
598 itg.first->second.emplace_back(id, k, default_facets_ext,
599 active_coeffs);
600 }
601 else if (sd != subdomains.end())
602 {
603 // NOTE: This requires that pairs are sorted
604 auto it = std::ranges::lower_bound(sd->second, id, std::less<>{},
605 [](auto& a) { return a.first; });
606 if (it != sd->second.end() and it->first == id)
607 itg.first->second.emplace_back(id, k, it->second, active_coeffs);
608 }
609
610 if (integral->needs_facet_permutations)
611 needs_facet_permutations = true;
612 }
613 }
614 }
615
616 // Attach interior facet kernels
617 std::vector<std::int32_t> default_facets_int;
618 {
619 std::span<const int> ids(ufcx_forms[0].get().form_integral_ids
620 + integral_offsets[interior_facet],
621 num_integrals_type[interior_facet]);
622 auto itg = integrals.insert({IntegralType::interior_facet, {}});
623 auto sd = subdomains.find(IntegralType::interior_facet);
624 for (std::size_t form_idx = 0; form_idx < ufcx_forms.size(); ++form_idx)
625 {
626 const ufcx_form& ufcx_form = ufcx_forms[form_idx];
627 // Create indicator for interprocess facets
628 std::vector<std::int8_t> interprocess_marker;
629 if (num_integrals_type[interior_facet] > 0)
630 {
631 assert(topology->index_map(tdim - 1));
632 const std::vector<std::int32_t>& interprocess_facets
633 = topology->interprocess_facets();
634 std::int32_t num_facets = topology->index_map(tdim - 1)->size_local()
635 + topology->index_map(tdim - 1)->num_ghosts();
636 interprocess_marker.resize(num_facets, 0);
637 std::ranges::for_each(interprocess_facets,
638 [&interprocess_marker](auto f)
639 { interprocess_marker[f] = 1; });
640 }
641
642 for (int i = 0; i < num_integrals_type[interior_facet]; ++i)
643 {
644 const int id = ids[i];
645 ufcx_integral* integral
646 = ufcx_form.form_integrals[integral_offsets[interior_facet] + i];
647 assert(integral);
648 std::vector<int> active_coeffs;
649 for (int j = 0; j < ufcx_form.num_coefficients; ++j)
650 {
651 if (integral->enabled_coefficients[j])
652 active_coeffs.push_back(j);
653 }
654
655 kern_t k = nullptr;
656 if constexpr (std::is_same_v<T, float>)
657 k = integral->tabulate_tensor_float32;
658#ifndef DOLFINX_NO_STDC_COMPLEX_KERNELS
659 else if constexpr (std::is_same_v<T, std::complex<float>>)
660 {
661 k = reinterpret_cast<void (*)(
662 T*, const T*, const T*,
663 const typename scalar_value_type<T>::value_type*, const int*,
664 const unsigned char*)>(integral->tabulate_tensor_complex64);
665 }
666#endif // DOLFINX_NO_STDC_COMPLEX_KERNELS
667 else if constexpr (std::is_same_v<T, double>)
668 k = integral->tabulate_tensor_float64;
669#ifndef DOLFINX_NO_STDC_COMPLEX_KERNELS
670 else if constexpr (std::is_same_v<T, std::complex<double>>)
671 {
672 k = reinterpret_cast<void (*)(
673 T*, const T*, const T*,
674 const typename scalar_value_type<T>::value_type*, const int*,
675 const unsigned char*)>(integral->tabulate_tensor_complex128);
676 }
677#endif // DOLFINX_NO_STDC_COMPLEX_KERNELS
678 assert(k);
679
680 // Build list of entities to assembler over
681 auto f_to_c = topology->connectivity(tdim - 1, tdim);
682 assert(f_to_c);
683 auto c_to_f = topology->connectivity(tdim, tdim - 1);
684 assert(c_to_f);
685 if (id == -1)
686 {
687 // Default kernel, operates on all (owned) interior facets
688 assert(topology->index_map(tdim - 1));
689 std::int32_t num_facets = topology->index_map(tdim - 1)->size_local();
690 default_facets_int.reserve(4 * num_facets);
691 for (std::int32_t f = 0; f < num_facets; ++f)
692 {
693 if (f_to_c->num_links(f) == 2)
694 {
695 auto pairs
696 = impl::get_cell_facet_pairs<2>(f, f_to_c->links(f), *c_to_f);
697 default_facets_int.insert(default_facets_int.end(), pairs.begin(),
698 pairs.end());
699 }
700 else if (interprocess_marker[f])
701 {
702 throw std::runtime_error(
703 "Cannot compute interior facet integral over interprocess "
704 "facet. Please use ghost mode shared facet when creating the "
705 "mesh");
706 }
707 }
708 itg.first->second.emplace_back(id, k, default_facets_int,
709 active_coeffs);
710 }
711 else if (sd != subdomains.end())
712 {
713 auto it = std::ranges::lower_bound(sd->second, id, std::less{},
714 [](auto& a) { return a.first; });
715 if (it != sd->second.end() and it->first == id)
716 itg.first->second.emplace_back(id, k, it->second, active_coeffs);
717 }
718
719 if (integral->needs_facet_permutations)
720 needs_facet_permutations = true;
721 }
722 }
723 }
724
725 std::map<IntegralType,
726 std::vector<std::pair<std::int32_t, std::vector<std::int32_t>>>>
727 sd;
728 for (auto& [itg, data] : subdomains)
729 {
730 std::vector<std::pair<std::int32_t, std::vector<std::int32_t>>> x;
731 for (auto& [id, idx] : data)
732 x.emplace_back(id, std::vector(idx.data(), idx.data() + idx.size()));
733 sd.insert({itg, std::move(x)});
734 }
735
736 return Form<T, U>(spaces, integrals, coefficients, constants,
737 needs_facet_permutations, entity_maps, mesh);
738}

887template <int _bs, dolfinx::scalar T>
888void pack(std::span<T> coeffs, std::int32_t cell, int bs, std::span<const T> v,
889 std::span<const std::uint32_t> cell_info, const DofMap& dofmap,
890 auto transform)
891{
892 auto dofs = dofmap.cell_dofs(cell);
893 for (std::size_t i = 0; i < dofs.size(); ++i)
894 {
895 if constexpr (_bs < 0)
896 {
897 const int pos_c = bs * i;
898 const int pos_v = bs * dofs[i];
899 for (int k = 0; k < bs; ++k)
900 coeffs[pos_c + k] = v[pos_v + k];
901 }
902 else
903 {
904 assert(_bs == bs);
905 const int pos_c = _bs * i;
906 const int pos_v = _bs * dofs[i];
907 for (int k = 0; k < _bs; ++k)
908 coeffs[pos_c + k] = v[pos_v + k];
909 }
910 }
911
912 transform(coeffs, cell_info, cell, 1);
913}

936template <dolfinx::scalar T, std::floating_point U>
937void pack_coefficient_entity(std::span<T> c, int cstride,
938 const Function<T, U>& u,
939 std::span<const std::uint32_t> cell_info,
940 std::span<const std::int32_t> entities,
941 std::size_t estride, FetchCells auto&& fetch_cells,
942 std::int32_t offset)
943{
944 // Read data from coefficient Function u
945 std::span<const T> v = u.x()->array();
946 const DofMap& dofmap = *u.function_space()->dofmap();
947 auto element = u.function_space()->element();
948 assert(element);
949 int space_dim = element->space_dimension();
950
951 // Transformation from conforming degrees-of-freedom to reference
952 // degrees-of-freedom
953 auto transformation
954 = element->template dof_transformation_fn<T>(doftransform::transpose);
955 const int bs = dofmap.bs();
956 switch (bs)
957 {
958 case 1:
959 for (std::size_t e = 0; e < entities.size(); e += estride)
960 {
961 auto entity = entities.subspan(e, estride);
962 if (std::int32_t cell = fetch_cells(entity); cell >= 0)
963 {
964 auto cell_coeff
965 = c.subspan((e / estride) * cstride + offset, space_dim);
966 pack<1>(cell_coeff, cell, bs, v, cell_info, dofmap, transformation);
967 }
968 }
969 break;
970 case 2:
971 for (std::size_t e = 0; e < entities.size(); e += estride)
972 {
973 auto entity = entities.subspan(e, estride);
974 if (std::int32_t cell = fetch_cells(entity); cell >= 0)
975 {
976 auto cell_coeff
977 = c.subspan((e / estride) * cstride + offset, space_dim);
978 pack<2>(cell_coeff, cell, bs, v, cell_info, dofmap, transformation);
979 }
980 }
981 break;
982 case 3:
983 for (std::size_t e = 0; e < entities.size(); e += estride)
984 {
985 auto entity = entities.subspan(e, estride);
986 if (std::int32_t cell = fetch_cells(entity); cell >= 0)
987 {
988 auto cell_coeff = c.subspan(e / estride * cstride + offset, space_dim);
989 pack<3>(cell_coeff, cell, bs, v, cell_info, dofmap, transformation);
990 }
991 }
992 break;
993 default:
994 for (std::size_t e = 0; e < entities.size(); e += estride)
995 {
996 auto entity = entities.subspan(e, estride);
997 if (std::int32_t cell = fetch_cells(entity); cell >= 0)
998 {
999 auto cell_coeff
1000 = c.subspan((e / estride) * cstride + offset, space_dim);
1001 pack<-1>(cell_coeff, cell, bs, v, cell_info, dofmap, transformation);
1002 }
1003 }
1004 break;
1005 }
1006}

1071template <dolfinx::scalar T, std::floating_point U>
1072void pack_coefficients(const Form<T, U>& form, IntegralType integral_type,
1073 int id, std::span<T> c, int cstride)
1074{
1075 // Get form coefficient offsets and dofmaps
1076 const std::vector<std::shared_ptr<const Function<T, U>>>& coefficients
1077 = form.coefficients();
1078 const std::vector<int> offsets = form.coefficient_offsets();
1079
1080 // Indicator for packing coefficients
1081 std::vector<int> active_coefficient(coefficients.size(), 0);
1082 if (!coefficients.empty())
1083 {
1084 switch (integral_type)
1085 {
1086 case IntegralType::cell:
1087 {
1088 // Get indicator for all coefficients that are active in cell
1089 // integrals
1090 for (std::size_t i = 0; i < form.num_integrals(IntegralType::cell); ++i)
1091 {
1092 for (auto idx : form.active_coeffs(IntegralType::cell, i))
1093 active_coefficient[idx] = 1;
1094 }
1095
1096 // Iterate over coefficients
1097 for (std::size_t coeff = 0; coeff < coefficients.size(); ++coeff)
1098 {
1099 if (!active_coefficient[coeff])
1100 continue;
1101
1102 // Get coefficient mesh
1103 auto mesh = coefficients[coeff]->function_space()->mesh();
1104 assert(mesh);
1105
1106 // Other integrals in the form might have coefficients defined over
1107 // entities of codim > 0, which don't make sense for cell integrals, so
1108 // don't pack them.
1109 if (const int codim
1110 = form.mesh()->topology()->dim() - mesh->topology()->dim();
1111 codim > 0)
1112 {
1113 throw std::runtime_error("Should not be packing coefficients with "
1114 "codim>0 in a cell integral");
1115 }
1116
1117 std::vector<std::int32_t> cells
1118 = form.domain(IntegralType::cell, id, *mesh);
1119 std::span<const std::uint32_t> cell_info
1120 = impl::get_cell_orientation_info(*coefficients[coeff]);
1121 impl::pack_coefficient_entity(
1122 c, cstride, *coefficients[coeff], cell_info, cells, 1,
1123 [](auto entity) { return entity.front(); }, offsets[coeff]);
1124 }
1125 break;
1126 }
1127 case IntegralType::exterior_facet:
1128 {
1129 // Get indicator for all coefficients that are active in exterior
1130 // facet integrals
1131 for (std::size_t i = 0;
1132 i < form.num_integrals(IntegralType::exterior_facet); ++i)
1133 {
1134 for (auto idx : form.active_coeffs(IntegralType::exterior_facet, i))
1135 active_coefficient[idx] = 1;
1136 }
1137
1138 // Iterate over coefficients
1139 for (std::size_t coeff = 0; coeff < coefficients.size(); ++coeff)
1140 {
1141 if (!active_coefficient[coeff])
1142 continue;
1143
1144 auto mesh = coefficients[coeff]->function_space()->mesh();
1145 std::vector<std::int32_t> facets
1146 = form.domain(IntegralType::exterior_facet, id, *mesh);
1147 std::span<const std::uint32_t> cell_info
1148 = impl::get_cell_orientation_info(*coefficients[coeff]);
1149 impl::pack_coefficient_entity(
1150 c, cstride, *coefficients[coeff], cell_info, facets, 2,
1151 [](auto entity) { return entity.front(); }, offsets[coeff]);
1152 }
1153 break;
1154 }
1155 case IntegralType::interior_facet:
1156 {
1157 // Get indicator for all coefficients that are active in interior
1158 // facet integrals
1159 for (std::size_t i = 0;
1160 i < form.num_integrals(IntegralType::interior_facet); ++i)
1161 {
1162 for (auto idx : form.active_coeffs(IntegralType::interior_facet, i))
1163 active_coefficient[idx] = 1;
1164 }
1165
1166 // Iterate over coefficients
1167 for (std::size_t coeff = 0; coeff < coefficients.size(); ++coeff)
1168 {
1169 if (!active_coefficient[coeff])
1170 continue;
1171
1172 auto mesh = coefficients[coeff]->function_space()->mesh();
1173 std::vector<std::int32_t> facets
1174 = form.domain(IntegralType::interior_facet, id, *mesh);
1175
1176 std::span<const std::uint32_t> cell_info
1177 = impl::get_cell_orientation_info(*coefficients[coeff]);
1178
1179 // Pack coefficient ['+']
1180 impl::pack_coefficient_entity(
1181 c, 2 * cstride, *coefficients[coeff], cell_info, facets, 4,
1182 [](auto entity) { return entity[0]; }, 2 * offsets[coeff]);
1183
1184 // Pack coefficient ['-']
1185 impl::pack_coefficient_entity(
1186 c, 2 * cstride, *coefficients[coeff], cell_info, facets, 4,
1187 [](auto entity) { return entity[2]; },
1188 offsets[coeff] + offsets[coeff + 1]);
1189 }
1190 break;
1191 }
1192 default:
1193 throw std::runtime_error(
1194 "Could not pack coefficient. Integral type not supported.");
1195 }
1196 }
