38template <typename T, int _bs0 = -1, int _bs1 = -1>
39void _lift_bc_cells(
40 std::span<T> b, const mesh::Geometry<scalar_value_type_t<T>>& geometry,
41 FEkernel<T> auto kernel, std::span<const std::int32_t> cells,
42 const std::function<void(const std::span<T>&,
43 const std::span<const std::uint32_t>&,
44 std::int32_t, int)>& dof_transform,
45 const graph::AdjacencyList<std::int32_t>& dofmap0, int bs0,
46 const std::function<void(const std::span<T>&,
47 const std::span<const std::uint32_t>&,
48 std::int32_t, int)>& dof_transform_to_transpose,
49 const graph::AdjacencyList<std::int32_t>& dofmap1, int bs1,
50 std::span<const T> constants, std::span<const T> coeffs, int cstride,
51 std::span<const std::uint32_t> cell_info, std::span<const T> bc_values1,
52 std::span<const std::int8_t> bc_markers1, std::span<const T> x0, T scale)
53{
54 assert(_bs0 < 0 or _bs0 == bs0);
55 assert(_bs1 < 0 or _bs1 == bs1);
56
57 if (cells.empty())
58 return;
59
60 // Prepare cell geometry
61 const graph::AdjacencyList<std::int32_t>& x_dofmap = geometry.dofmap();
62 const std::size_t num_dofs_g = geometry.cmap().dim();
63 auto x = geometry.x();
64
65 // Data structures used in bc application
66 std::vector<scalar_value_type_t<T>> coordinate_dofs(3 * num_dofs_g);
67 std::vector<T> Ae, be;
68 for (std::size_t index = 0; index < cells.size(); ++index)
69 {
70 std::int32_t c = cells[index];
71
72 // Get dof maps for cell
73 auto dmap1 = dofmap1.links(c);
74
75 // Check if bc is applied to cell
76 bool has_bc = false;
77 for (std::size_t j = 0; j < dmap1.size(); ++j)
78 {
79 if constexpr (_bs1 > 0)
80 {
81 for (int k = 0; k < _bs1; ++k)
82 {
83 assert(_bs1 * dmap1[j] + k < (int)bc_markers1.size());
84 if (bc_markers1[_bs1 * dmap1[j] + k])
85 {
86 has_bc = true;
87 break;
88 }
89 }
90 }
91 else
92 {
93 for (int k = 0; k < bs1; ++k)
94 {
95 assert(bs1 * dmap1[j] + k < (int)bc_markers1.size());
96 if (bc_markers1[bs1 * dmap1[j] + k])
97 {
98 has_bc = true;
99 break;
100 }
101 }
102 }
103 }
104
105 if (!has_bc)
106 continue;
107
108 // Get cell coordinates/geometry
109 auto x_dofs = x_dofmap.links(c);
110 for (std::size_t i = 0; i < x_dofs.size(); ++i)
111 {
112 std::copy_n(std::next(x.begin(), 3 * x_dofs[i]), 3,
113 std::next(coordinate_dofs.begin(), 3 * i));
114 }
115
116 // Size data structure for assembly
117 auto dmap0 = dofmap0.links(c);
118
119 const int num_rows = bs0 * dmap0.size();
120 const int num_cols = bs1 * dmap1.size();
121
122 const T* coeff_array = coeffs.data() + index * cstride;
123 Ae.resize(num_rows * num_cols);
124 std::fill(Ae.begin(), Ae.end(), 0);
125 kernel(Ae.data(), coeff_array, constants.data(), coordinate_dofs.data(),
126 nullptr, nullptr);
127 dof_transform(Ae, cell_info, c, num_cols);
128 dof_transform_to_transpose(Ae, cell_info, c, num_rows);
129
130 // Size data structure for assembly
131 be.resize(num_rows);
132 std::fill(be.begin(), be.end(), 0);
133 for (std::size_t j = 0; j < dmap1.size(); ++j)
134 {
135 if constexpr (_bs1 > 0)
136 {
137 for (int k = 0; k < _bs1; ++k)
138 {
139 const std::int32_t jj = _bs1 * dmap1[j] + k;
140 assert(jj < (int)bc_markers1.size());
141 if (bc_markers1[jj])
142 {
143 const T bc = bc_values1[jj];
144 const T _x0 = x0.empty() ? 0.0 : x0[jj];
145 // const T _x0 = 0.0;
146 // be -= Ae.col(bs1 * j + k) * scale * (bc - _x0);
147 for (int m = 0; m < num_rows; ++m)
148 be[m] -= Ae[m * num_cols + _bs1 * j + k] * scale * (bc - _x0);
149 }
150 }
151 }
152 else
153 {
154 for (int k = 0; k < bs1; ++k)
155 {
156 const std::int32_t jj = bs1 * dmap1[j] + k;
157 assert(jj < (int)bc_markers1.size());
158 if (bc_markers1[jj])
159 {
160 const T bc = bc_values1[jj];
161 const T _x0 = x0.empty() ? 0.0 : x0[jj];
162 // be -= Ae.col(bs1 * j + k) * scale * (bc - _x0);
163 for (int m = 0; m < num_rows; ++m)
164 be[m] -= Ae[m * num_cols + bs1 * j + k] * scale * (bc - _x0);
165 }
166 }
167 }
168 }
169
170 for (std::size_t i = 0; i < dmap0.size(); ++i)
171 {
172 if constexpr (_bs0 > 0)
173 {
174 for (int k = 0; k < _bs0; ++k)
175 b[_bs0 * dmap0[i] + k] += be[_bs0 * i + k];
176 }
177 else
178 {
179 for (int k = 0; k < bs0; ++k)
180 b[bs0 * dmap0[i] + k] += be[bs0 * i + k];
181 }
182 }
183 }
184}

279 for (int m = 0; m < num_rows; ++m)

282 }

284
285 for (std::size_t i = 0; i < dmap0.size(); ++i)
286 for (int k = 0; k < bs0; ++k)
287 b[bs0 * dmap0[i] + k] += be[bs0 * i + k];

297template <typename T, int _bs = -1>
298void _lift_bc_interior_facets(
299 std::span<T> b, const mesh::Geometry<scalar_value_type_t<T>>& geometry,
300 int num_cell_facets, FEkernel<T> auto kernel,
301 std::span<const std::int32_t> facets,
302 const std::function<void(const std::span<T>&,
303 const std::span<const std::uint32_t>&,
304 std::int32_t, int)>& dof_transform,
305 const graph::AdjacencyList<std::int32_t>& dofmap0, int bs0,
306 const std::function<void(const std::span<T>&,
307 const std::span<const std::uint32_t>&,
308 std::int32_t, int)>& dof_transform_to_transpose,
309 const graph::AdjacencyList<std::int32_t>& dofmap1, int bs1,
310 std::span<const T> constants, std::span<const T> coeffs, int cstride,
311 std::span<const std::uint32_t> cell_info,
312 const std::function<std::uint8_t(std::size_t)>& get_perm,
313 std::span<const T> bc_values1, std::span<const std::int8_t> bc_markers1,
314 std::span<const T> x0, T scale)
315{
316 if (facets.empty())
317 return;
318
319 // Prepare cell geometry
320 const graph::AdjacencyList<std::int32_t>& x_dofmap = geometry.dofmap();
321 const std::size_t num_dofs_g = geometry.cmap().dim();
322 auto x = geometry.x();
323
324 // Data structures used in assembly
325 using X = scalar_value_type_t<T>;
326 std::vector<X> coordinate_dofs(2 * num_dofs_g * 3);
327 std::span<X> cdofs0(coordinate_dofs.data(), num_dofs_g * 3);
328 std::span<X> cdofs1(coordinate_dofs.data() + num_dofs_g * 3, num_dofs_g * 3);
329 std::vector<T> Ae, be;
330
331 // Temporaries for joint dofmaps
332 std::vector<std::int32_t> dmapjoint0, dmapjoint1;
333 assert(facets.size() % 4 == 0);
334
335 for (std::size_t index = 0; index < facets.size(); index += 4)
336 {
337 std::array<std::int32_t, 2> cells = {facets[index], facets[index + 2]};
338 std::array<std::int32_t, 2> local_facet
339 = {facets[index + 1], facets[index + 3]};
340
341 // Get cell geometry
342 auto x_dofs0 = x_dofmap.links(cells[0]);
343 for (std::size_t i = 0; i < x_dofs0.size(); ++i)
344 {
345 std::copy_n(std::next(x.begin(), 3 * x_dofs0[i]), 3,
346 std::next(cdofs0.begin(), 3 * i));
347 }
348 auto x_dofs1 = x_dofmap.links(cells[1]);
349 for (std::size_t i = 0; i < x_dofs1.size(); ++i)
350 {
351 std::copy_n(std::next(x.begin(), 3 * x_dofs1[i]), 3,
352 std::next(cdofs1.begin(), 3 * i));
353 }
354
355 // Get dof maps for cells and pack
356 std::span<const std::int32_t> dmap0_cell0 = dofmap0.links(cells[0]);
357 std::span<const std::int32_t> dmap0_cell1 = dofmap0.links(cells[1]);
358 dmapjoint0.resize(dmap0_cell0.size() + dmap0_cell1.size());
359 std::copy(dmap0_cell0.begin(), dmap0_cell0.end(), dmapjoint0.begin());
360 std::copy(dmap0_cell1.begin(), dmap0_cell1.end(),
361 std::next(dmapjoint0.begin(), dmap0_cell0.size()));
362
363 std::span<const std::int32_t> dmap1_cell0 = dofmap1.links(cells[0]);
364 std::span<const std::int32_t> dmap1_cell1 = dofmap1.links(cells[1]);
365 dmapjoint1.resize(dmap1_cell0.size() + dmap1_cell1.size());
366 std::copy(dmap1_cell0.begin(), dmap1_cell0.end(), dmapjoint1.begin());
367 std::copy(dmap1_cell1.begin(), dmap1_cell1.end(),
368 std::next(dmapjoint1.begin(), dmap1_cell0.size()));
369
370 // Check if bc is applied to cell0
371 bool has_bc = false;
372 for (std::size_t j = 0; j < dmap1_cell0.size(); ++j)
373 {
374 for (int k = 0; k < bs1; ++k)
375 {
376 if (bc_markers1[bs1 * dmap1_cell0[j] + k])
377 {
378 has_bc = true;
379 break;
380 }
381 }
382 }
383
384 // Check if bc is applied to cell1
385 for (std::size_t j = 0; j < dmap1_cell1.size(); ++j)
386 {
387 for (int k = 0; k < bs1; ++k)
388 {
389 if (bc_markers1[bs1 * dmap1_cell1[j] + k])
390 {
391 has_bc = true;
392 break;
393 }
394 }
395 }
396
397 if (!has_bc)
398 continue;
399
400 const int num_rows = bs0 * dmapjoint0.size();
401 const int num_cols = bs1 * dmapjoint1.size();
402
403 // Tabulate tensor
404 Ae.resize(num_rows * num_cols);
405 std::fill(Ae.begin(), Ae.end(), 0);
406 const std::array perm{
407 get_perm(cells[0] * num_cell_facets + local_facet[0]),
408 get_perm(cells[1] * num_cell_facets + local_facet[1])};
409 kernel(Ae.data(), coeffs.data() + index / 2 * cstride, constants.data(),
410 coordinate_dofs.data(), local_facet.data(), perm.data());
411
412 std::span<T> _Ae(Ae);
413 std::span<T> sub_Ae0 = _Ae.subspan(bs0 * dmap0_cell0.size() * num_cols,
414 bs0 * dmap0_cell1.size() * num_cols);
415 std::span<T> sub_Ae1
416 = _Ae.subspan(bs1 * dmap1_cell0.size(),
417 num_rows * num_cols - bs1 * dmap1_cell0.size());
418
419 dof_transform(_Ae, cell_info, cells[0], num_cols);
420 dof_transform(sub_Ae0, cell_info, cells[1], num_cols);
421 dof_transform_to_transpose(_Ae, cell_info, cells[0], num_rows);
422 dof_transform_to_transpose(sub_Ae1, cell_info, cells[1], num_rows);
423
424 be.resize(num_rows);
425 std::fill(be.begin(), be.end(), 0);
426
427 // Compute b = b - A*b for cell0
428 for (std::size_t j = 0; j < dmap1_cell0.size(); ++j)
429 {
430 for (int k = 0; k < bs1; ++k)
431 {
432 const std::int32_t jj = bs1 * dmap1_cell0[j] + k;
433 if (bc_markers1[jj])
434 {
435 const T bc = bc_values1[jj];
436 const T _x0 = x0.empty() ? 0.0 : x0[jj];
437 // be -= Ae.col(bs1 * j + k) * scale * (bc - _x0);
438 for (int m = 0; m < num_rows; ++m)
439 be[m] -= Ae[m * num_cols + bs1 * j + k] * scale * (bc - _x0);
440 }
441 }
442 }
443
444 // Compute b = b - A*b for cell1
445 const int offset = bs1 * dmap1_cell0.size();
446 for (std::size_t j = 0; j < dmap1_cell1.size(); ++j)
447 {
448 for (int k = 0; k < bs1; ++k)
449 {
450 const std::int32_t jj = bs1 * dmap1_cell1[j] + k;
451 if (bc_markers1[jj])
452 {
453 const T bc = bc_values1[jj];
454 const T _x0 = x0.empty() ? 0.0 : x0[jj];
455 // be -= Ae.col(offset + bs1 * j + k) * scale * (bc - x0[jj]);
456 for (int m = 0; m < num_rows; ++m)
457 {
458 be[m]
459 -= Ae[m * num_cols + offset + bs1 * j + k] * scale * (bc - _x0);
460 }
461 }
462 }
463 }
464
465 for (std::size_t i = 0; i < dmap0_cell0.size(); ++i)
466 for (int k = 0; k < bs0; ++k)
467 b[bs0 * dmap0_cell0[i] + k] += be[bs0 * i + k];
468
469 const int offset_be = bs0 * dmap0_cell0.size();
470 for (std::size_t i = 0; i < dmap0_cell1.size(); ++i)
471 for (int k = 0; k < bs0; ++k)
472 b[bs0 * dmap0_cell1[i] + k] += be[offset_be + bs0 * i + k];
473 }
474}
481template <typename T, int _bs = -1>
482void assemble_cells(
483 const std::function<void(const std::span<T>&,
484 const std::span<const std::uint32_t>&,
485 std::int32_t, int)>& dof_transform,
486 std::span<T> b, const mesh::Geometry<scalar_value_type_t<T>>& geometry,
487 std::span<const std::int32_t> cells,
488 const graph::AdjacencyList<std::int32_t>& dofmap, int bs,
489 FEkernel<T> auto kernel, std::span<const T> constants,
490 std::span<const T> coeffs, int cstride,
491 std::span<const std::uint32_t> cell_info)
492{
493 assert(_bs < 0 or _bs == bs);
494
495 if (cells.empty())
496 return;
497
498 // Prepare cell geometry
499 const graph::AdjacencyList<std::int32_t>& x_dofmap = geometry.dofmap();
500 const std::size_t num_dofs_g = geometry.cmap().dim();
501 auto x = geometry.x();
502
503 // FIXME: Add proper interface for num_dofs
504 // Create data structures used in assembly
505 const int num_dofs = dofmap.links(0).size();
506 std::vector<scalar_value_type_t<T>> coordinate_dofs(3 * num_dofs_g);
507 std::vector<T> be(bs * num_dofs);
508 std::span<T> _be(be);
509
510 // Iterate over active cells
511 for (std::size_t index = 0; index < cells.size(); ++index)
512 {
513 std::int32_t c = cells[index];
514
515 // Get cell coordinates/geometry
516 auto x_dofs = x_dofmap.links(c);
517 for (std::size_t i = 0; i < x_dofs.size(); ++i)
518 {
519 std::copy_n(std::next(x.begin(), 3 * x_dofs[i]), 3,
520 std::next(coordinate_dofs.begin(), 3 * i));
521 }
522
523 // Tabulate vector for cell
524 std::fill(be.begin(), be.end(), 0);
525 kernel(be.data(), coeffs.data() + index * cstride, constants.data(),
526 coordinate_dofs.data(), nullptr, nullptr);
527 dof_transform(_be, cell_info, c, 1);
528
529 // Scatter cell vector to 'global' vector array
530 auto dofs = dofmap.links(c);
531 if constexpr (_bs > 0)
532 {
533 for (int i = 0; i < num_dofs; ++i)
534 for (int k = 0; k < _bs; ++k)
535 b[_bs * dofs[i] + k] += be[_bs * i + k];
536 }
537 else
538 {
539 for (int i = 0; i < num_dofs; ++i)
540 for (int k = 0; k < bs; ++k)
541 b[bs * dofs[i] + k] += be[bs * i + k];
542 }
543 }
544}

552template <typename T, int _bs = -1>
553void assemble_exterior_facets(
554 const std::function<void(const std::span<T>&,
555 const std::span<const std::uint32_t>&,
556 std::int32_t, int)>& dof_transform,
557 std::span<T> b, const mesh::Geometry<scalar_value_type_t<T>>& geometry,
558 std::span<const std::int32_t> facets,
559 const graph::AdjacencyList<std::int32_t>& dofmap, int bs,
560 FEkernel<T> auto fn, std::span<const T> constants,
561 std::span<const T> coeffs, int cstride,
562 std::span<const std::uint32_t> cell_info)
563{
564 assert(_bs < 0 or _bs == bs);
565
566 if (facets.empty())
567 return;
568
569 // Prepare cell geometry
570 const graph::AdjacencyList<std::int32_t>& x_dofmap = geometry.dofmap();
571 const std::size_t num_dofs_g = geometry.cmap().dim();
572 auto x = geometry.x();
573
574 // FIXME: Add proper interface for num_dofs
575 // Create data structures used in assembly
576 const int num_dofs = dofmap.links(0).size();
577 std::vector<scalar_value_type_t<T>> coordinate_dofs(3 * num_dofs_g);
578 std::vector<T> be(bs * num_dofs);
579 std::span<T> _be(be);
580 assert(facets.size() % 2 == 0);
581 for (std::size_t index = 0; index < facets.size(); index += 2)
582 {
583 std::int32_t cell = facets[index];
584 std::int32_t local_facet = facets[index + 1];
585
586 // Get cell coordinates/geometry
587 auto x_dofs = x_dofmap.links(cell);
588 for (std::size_t i = 0; i < x_dofs.size(); ++i)
589 {
590 std::copy_n(std::next(x.begin(), 3 * x_dofs[i]), 3,
591 std::next(coordinate_dofs.begin(), 3 * i));
592 }
593
594 // Tabulate element vector
595 std::fill(be.begin(), be.end(), 0);
596 fn(be.data(), coeffs.data() + index / 2 * cstride, constants.data(),
597 coordinate_dofs.data(), &local_facet, nullptr);
598
599 dof_transform(_be, cell_info, cell, 1);
600
601 // Add element vector to global vector
602 auto dofs = dofmap.links(cell);
603 if constexpr (_bs > 0)
604 {
605 for (int i = 0; i < num_dofs; ++i)
606 for (int k = 0; k < _bs; ++k)
607 b[_bs * dofs[i] + k] += be[_bs * i + k];
608 }
609 else
610 {
611 for (int i = 0; i < num_dofs; ++i)
612 for (int k = 0; k < bs; ++k)
613 b[bs * dofs[i] + k] += be[bs * i + k];
614 }

624template <typename T, int _bs = -1>
625void assemble_interior_facets(
626 const std::function<void(const std::span<T>&,
627 const std::span<const std::uint32_t>&,
628 std::int32_t, int)>& dof_transform,
629 std::span<T> b, const mesh::Geometry<scalar_value_type_t<T>>& geometry,
630 int num_cell_facets, std::span<const std::int32_t> facets,
631 const fem::DofMap& dofmap, FEkernel<T> auto fn,
632 std::span<const T> constants, std::span<const T> coeffs, int cstride,
633 std::span<const std::uint32_t> cell_info,
634 const std::function<std::uint8_t(std::size_t)>& get_perm)
635{
636 // Prepare cell geometry
637 const graph::AdjacencyList<std::int32_t>& x_dofmap = geometry.dofmap();
638 const std::size_t num_dofs_g = geometry.cmap().dim();
639 auto x = geometry.x();
640
641 // Create data structures used in assembly
642 using X = scalar_value_type_t<T>;
643 std::vector<X> coordinate_dofs(2 * num_dofs_g * 3);
644 std::span<X> cdofs0(coordinate_dofs.data(), num_dofs_g * 3);
645 std::span<X> cdofs1(coordinate_dofs.data() + num_dofs_g * 3, num_dofs_g * 3);
646 std::vector<T> be;
647
648 const int bs = dofmap.bs();
649 assert(_bs < 0 or _bs == bs);
650 assert(facets.size() % 4 == 0);
651 for (std::size_t index = 0; index < facets.size(); index += 4)
652 {
653 std::array<std::int32_t, 2> cells = {facets[index], facets[index + 2]};
654 std::array<std::int32_t, 2> local_facet
655 = {facets[index + 1], facets[index + 3]};
656
657 // Get cell geometry
658 auto x_dofs0 = x_dofmap.links(cells[0]);
659 for (std::size_t i = 0; i < x_dofs0.size(); ++i)
660 {
661 std::copy_n(std::next(x.begin(), 3 * x_dofs0[i]), 3,
662 std::next(cdofs0.begin(), 3 * i));
663 }
664 auto x_dofs1 = x_dofmap.links(cells[1]);
665 for (std::size_t i = 0; i < x_dofs1.size(); ++i)
666 {
667 std::copy_n(std::next(x.begin(), 3 * x_dofs1[i]), 3,
668 std::next(cdofs1.begin(), 3 * i));
669 }
670
671 // Get dofmaps for cells
672 std::span<const std::int32_t> dmap0 = dofmap.cell_dofs(cells[0]);
673 std::span<const std::int32_t> dmap1 = dofmap.cell_dofs(cells[1]);
674
675 // Tabulate element vector
676 be.resize(bs * (dmap0.size() + dmap1.size()));
677 std::fill(be.begin(), be.end(), 0);
678 const std::array perm{
679 get_perm(cells[0] * num_cell_facets + local_facet[0]),
680 get_perm(cells[1] * num_cell_facets + local_facet[1])};
681 fn(be.data(), coeffs.data() + index / 2 * cstride, constants.data(),
682 coordinate_dofs.data(), local_facet.data(), perm.data());
683
684 std::span<T> _be(be);
685 std::span<T> sub_be = _be.subspan(bs * dmap0.size(), bs * dmap1.size());
686
687 dof_transform(be, cell_info, cells[0], 1);
688 dof_transform(sub_be, cell_info, cells[1], 1);
689
690 // Add element vector to global vector
691 if constexpr (_bs > 0)
692 {
693 for (std::size_t i = 0; i < dmap0.size(); ++i)
694 for (int k = 0; k < _bs; ++k)
695 b[_bs * dmap0[i] + k] += be[_bs * i + k];
696 for (std::size_t i = 0; i < dmap1.size(); ++i)
697 for (int k = 0; k < _bs; ++k)
698 b[_bs * dmap1[i] + k] += be[_bs * (i + dmap0.size()) + k];
699 }
700 else
701 {
702 for (std::size_t i = 0; i < dmap0.size(); ++i)
703 for (int k = 0; k < bs; ++k)
704 b[bs * dmap0[i] + k] += be[bs * i + k];
705 for (std::size_t i = 0; i < dmap1.size(); ++i)
706 for (int k = 0; k < bs; ++k)
707 b[bs * dmap1[i] + k] += be[bs * (i + dmap0.size()) + k];
708 }
709 }
710}

727template <typename T>

796 {
797 _lift_bc_cells(b, geometry, kernel, cells, dof_transform, dofmap0, bs0,

799 coeffs, cstride, cell_info, bc_values1, bc_markers1, x0,

801 }

866template <typename T>
867void apply_lifting(
868 std::span<T> b, const std::vector<std::shared_ptr<const Form<T>>> a,
869 const mesh::Geometry<scalar_value_type_t<T>>& geometry,
870 const std::vector<std::span<const T>>& constants,
871 const std::vector<std::map<std::pair<IntegralType, int>,
872 std::pair<std::span<const T>, int>>>& coeffs,
873 const std::vector<std::vector<std::shared_ptr<const DirichletBC<T>>>>& bcs1,
874 const std::vector<std::span<const T>>& x0, T scale)
875{
876 // FIXME: make changes to reactivate this check
877 if (!x0.empty() and x0.size() != a.size())
878 {
879 throw std::runtime_error(
880 "Mismatch in size between x0 and bilinear form in assembler.");
881 }
882
883 if (a.size() != bcs1.size())
884 {
885 throw std::runtime_error(
886 "Mismatch in size between a and bcs in assembler.");
887 }
888
889 for (std::size_t j = 0; j < a.size(); ++j)
890 {
891 std::vector<std::int8_t> bc_markers1;
892 std::vector<T> bc_values1;
893 if (a[j] and !bcs1[j].empty())
894 {
895 assert(a[j]->function_spaces().at(0));
896
897 auto V1 = a[j]->function_spaces()[1];
898 assert(V1);
899 auto map1 = V1->dofmap()->index_map;
900 const int bs1 = V1->dofmap()->index_map_bs();
901 assert(map1);
902 const int crange = bs1 * (map1->size_local() + map1->num_ghosts());
903 bc_markers1.assign(crange, false);
904 bc_values1.assign(crange, 0.0);
905 for (const std::shared_ptr<const DirichletBC<T>>& bc : bcs1[j])
906 {
907 bc->mark_dofs(bc_markers1);
908 bc->dof_values(bc_values1);
909 }
910
911 if (!x0.empty())
912 {
913 lift_bc<T>(b, *a[j], geometry, constants[j], coeffs[j], bc_values1,
914 bc_markers1, x0[j], scale);
915 }
916 else
917 {
918 lift_bc<T>(b, *a[j], geometry, constants[j], coeffs[j], bc_values1,
919 bc_markers1, std::span<const T>(), scale);
920 }
921 }
922 }
