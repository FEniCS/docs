25 template <typename T>
26 T assemble_cells(const mesh::Geometry& geometry,
27  std::span<const std::int32_t> cells,
28  const std::function<void(T*, const T*, const T*,
29  const scalar_value_type_t<T>*,
30  const int*, const std::uint8_t*)>& fn,
31  std::span<const T> constants, std::span<const T> coeffs,
32  int cstride)
33 {
34  T value(0);
35  if (cells.empty())
36  return value;
37 
38  // Prepare cell geometry
39  const graph::AdjacencyList<std::int32_t>& x_dofmap = geometry.dofmap();
40  const std::size_t num_dofs_g = geometry.cmap().dim();
41  std::span<const double> x_g = geometry.x();
42 
43  // Create data structures used in assembly
44  std::vector<scalar_value_type_t<T>> coordinate_dofs(3 * num_dofs_g);
45 
46  // Iterate over all cells
47  for (std::size_t index = 0; index < cells.size(); ++index)
48  {
49  std::int32_t c = cells[index];
50 
51  // Get cell coordinates/geometry
52  auto x_dofs = x_dofmap.links(c);
53  for (std::size_t i = 0; i < x_dofs.size(); ++i)
54  {
55  std::copy_n(std::next(x_g.begin(), 3 * x_dofs[i]), 3,
56  std::next(coordinate_dofs.begin(), 3 * i));
57  }
58 
59  const T* coeff_cell = coeffs.data() + index * cstride;
60  fn(&value, coeff_cell, constants.data(), coordinate_dofs.data(), nullptr,
61  nullptr);
62  }
63 
64  return value;
65 }

68 template <typename T>
69 T assemble_exterior_facets(
70  const mesh::Mesh& mesh, std::span<const std::int32_t> facets,
71  const std::function<void(T*, const T*, const T*,
72  const scalar_value_type_t<T>*, const int*,
73  const std::uint8_t*)>& fn,
74  std::span<const T> constants, std::span<const T> coeffs, int cstride)
75 {
76  T value(0);
77  if (facets.empty())
78  return value;
79 
80  // Prepare cell geometry
81  const graph::AdjacencyList<std::int32_t>& x_dofmap = mesh.geometry().dofmap();
82  const std::size_t num_dofs_g = mesh.geometry().cmap().dim();
83  std::span<const double> x_g = mesh.geometry().x();
84 
85  // Create data structures used in assembly
86  std::vector<scalar_value_type_t<T>> coordinate_dofs(3 * num_dofs_g);
87 
88  // Iterate over all facets
89  assert(facets.size() % 2 == 0);
90  for (std::size_t index = 0; index < facets.size(); index += 2)
91  {
92  std::int32_t cell = facets[index];
93  std::int32_t local_facet = facets[index + 1];
94 
95  // Get cell coordinates/geometry
96  auto x_dofs = x_dofmap.links(cell);
97  for (std::size_t i = 0; i < x_dofs.size(); ++i)
98  {
99  std::copy_n(std::next(x_g.begin(), 3 * x_dofs[i]), 3,
100  std::next(coordinate_dofs.begin(), 3 * i));
101  }
102 
103  const T* coeff_cell = coeffs.data() + index / 2 * cstride;
104  fn(&value, coeff_cell, constants.data(), coordinate_dofs.data(),
105  &local_facet, nullptr);
106  }
107 
108  return value;
109 }

112 template <typename T>
113 T assemble_interior_facets(
114  const mesh::Mesh& mesh, std::span<const std::int32_t> facets,
115  const std::function<void(T*, const T*, const T*,
116  const scalar_value_type_t<T>*, const int*,
117  const std::uint8_t*)>& fn,
118  std::span<const T> constants, std::span<const T> coeffs, int cstride,
119  std::span<const int> offsets, std::span<const std::uint8_t> perms)
120 {
121  T value(0);
122  if (facets.empty())
123  return value;
124 
125  const int tdim = mesh.topology().dim();
126 
127  // Prepare cell geometry
128  const graph::AdjacencyList<std::int32_t>& x_dofmap = mesh.geometry().dofmap();
129  const std::size_t num_dofs_g = mesh.geometry().cmap().dim();
130  std::span<const double> x_g = mesh.geometry().x();
131 
132  // Create data structures used in assembly
133  using X = scalar_value_type_t<T>;
134  std::vector<X> coordinate_dofs(2 * num_dofs_g * 3);
135  std::span<X> cdofs0(coordinate_dofs.data(), num_dofs_g * 3);
136  std::span<X> cdofs1(coordinate_dofs.data() + num_dofs_g * 3, num_dofs_g * 3);
137 
138  std::vector<T> coeff_array(2 * offsets.back());
139  assert(offsets.back() == cstride);
140 
141  const int num_cell_facets
142  = mesh::cell_num_entities(mesh.topology().cell_type(), tdim - 1);
143 
144  // Iterate over all facets
145  assert(facets.size() % 4 == 0);
146  for (std::size_t index = 0; index < facets.size(); index += 4)
147  {
148  std::array<std::int32_t, 2> cells = {facets[index], facets[index + 2]};
149  std::array<std::int32_t, 2> local_facet
150  = {facets[index + 1], facets[index + 3]};
151 
152  // Get cell geometry
153  auto x_dofs0 = x_dofmap.links(cells[0]);
154  for (std::size_t i = 0; i < x_dofs0.size(); ++i)
155  {
156  std::copy_n(std::next(x_g.begin(), 3 * x_dofs0[i]), 3,
157  std::next(cdofs0.begin(), 3 * i));
158  }
159  auto x_dofs1 = x_dofmap.links(cells[1]);
160  for (std::size_t i = 0; i < x_dofs1.size(); ++i)
161  {
162  std::copy_n(std::next(x_g.begin(), 3 * x_dofs1[i]), 3,
163  std::next(cdofs1.begin(), 3 * i));
164  }
165 
166  const std::array perm{perms[cells[0] * num_cell_facets + local_facet[0]],
167  perms[cells[1] * num_cell_facets + local_facet[1]]};
168  fn(&value, coeffs.data() + index / 2 * cstride, constants.data(),
169  coordinate_dofs.data(), local_facet.data(), perm.data());
170  }
171 
172  return value;
173 }

176 template <typename T>
177 T assemble_scalar(
178  const fem::Form<T>& M, std::span<const T> constants,
179  const std::map<std::pair<IntegralType, int>,
180  std::pair<std::span<const T>, int>>& coefficients)
181 {
182  std::shared_ptr<const mesh::Mesh> mesh = M.mesh();
183  assert(mesh);
184 
185  T value = 0;
186  for (int i : M.integral_ids(IntegralType::cell))
187  {
188  const auto& fn = M.kernel(IntegralType::cell, i);
189  const auto& [coeffs, cstride] = coefficients.at({IntegralType::cell, i});
190  const std::vector<std::int32_t>& cells = M.cell_domains(i);
191  value += impl::assemble_cells(mesh->geometry(), cells, fn, constants,
192  coeffs, cstride);
193  }
194 
195  for (int i : M.integral_ids(IntegralType::exterior_facet))
196  {
197  const auto& fn = M.kernel(IntegralType::exterior_facet, i);
198  const auto& [coeffs, cstride]
199  = coefficients.at({IntegralType::exterior_facet, i});
200  const std::vector<std::int32_t>& facets = M.exterior_facet_domains(i);
201  value += impl::assemble_exterior_facets(*mesh, facets, fn, constants,
202  coeffs, cstride);
203  }
204 
205  if (M.num_integrals(IntegralType::interior_facet) > 0)
206  {
207  mesh->topology_mutable().create_entity_permutations();
208 
209  const std::vector<std::uint8_t>& perms
210  = mesh->topology().get_facet_permutations();
211 
212  const std::vector<int> c_offsets = M.coefficient_offsets();
213  for (int i : M.integral_ids(IntegralType::interior_facet))
214  {
215  const auto& fn = M.kernel(IntegralType::interior_facet, i);
216  const auto& [coeffs, cstride]
217  = coefficients.at({IntegralType::interior_facet, i});
218  const std::vector<std::int32_t>& facets = M.interior_facet_domains(i);
219  value += impl::assemble_interior_facets(
220  *mesh, facets, fn, constants, coeffs, cstride, c_offsets, perms);
221  }
222  }
223 
224  return value;
225 }
