213 // Cell types for entites in Topology, as follows:
214 // [CellType::point, edge_types..., facet_types..., cell_types...]
215 // Only one type is expected for vertices, (and usually edges), but facets
216 // and cells can be a list of multiple types, e.g. [quadrilateral, triangle]
217 // for facets.
218 // Offsets are position in the list for each entity dimension, in
219 // AdjacencyList style.
220 std::vector<CellType> _entity_types;
221 std::vector<std::int8_t> _entity_type_offsets;
222
223 // Parallel layout of entities for each dimension and cell type
224 // flattened in the same layout as _entity_types above.
225 std::vector<std::shared_ptr<const common::IndexMap>> _index_map;
226
227 // Connectivity between entity dimensions and cell types, arranged as
228 // a 2D array. The indexing follows the order in _entity_types, i.e.
229 // increasing in topological dimension. There may be multiple types in each
230 // dimension, e.g. triangle and quadrilateral facets.
231 // Connectivity between different entity types of same dimension will always
232 // be nullptr.
233 std::vector<std::vector<std::shared_ptr<graph::AdjacencyList<std::int32_t>>>>
234 _connectivity;
235
236 // The facet permutations (local facet, cell))
237 // [cell0_0, cell0_1, ,cell0_2, cell1_0, cell1_1, ,cell1_2, ...,
238 // celln_0, celln_1, ,celln_2,]
239 std::vector<std::uint8_t> _facet_permutations;
240
241 // Cell permutation info. See the documentation for
242 // get_cell_permutation_info for documentation of how this is encoded.
243 std::vector<std::uint32_t> _cell_permutations;
244
245 // List of facets that are on the inter-process boundary for each facet type
246 std::vector<std::vector<std::int32_t>> _interprocess_facets;
