268 // Cell types for entities in Topology, as follows:
269 //
270 // [CellType::point, edge_types..., facet_types..., cell_types...]
271 //
272 // Only one type is expected for vertices, (and usually edges), but
273 // facets and cells can be a list of multiple types, e.g.
274 // [quadrilateral, triangle] for facets. Offsets are position in the
275 // list for each entity dimension, in AdjacencyList style.
276 std::vector<CellType> _entity_types;
277 std::vector<std::int8_t> _entity_type_offsets;
278
279 // Parallel layout of entities for each dimension and cell type
280 // flattened in the same layout as _entity_types above.
281 std::vector<std::shared_ptr<const common::IndexMap>> _index_map;
282
283 // Connectivity between entity dimensions and cell types, arranged as
284 // a 2D array. The indexing follows the order in _entity_types, i.e.
285 // increasing in topological dimension. There may be multiple types in
286 // each dimension, e.g. triangle and quadrilateral facets.
287 // Connectivity between different entity types of same dimension will
288 // always be nullptr.
289 std::vector<std::vector<std::shared_ptr<graph::AdjacencyList<std::int32_t>>>>
290 _connectivity;
291
292 // The facet permutations (local facet, cell))
293 // [cell0_0, cell0_1, ,cell0_2, cell1_0, cell1_1, ,cell1_2, ...,
294 // celln_0, celln_1, ,celln_2,]
295 std::vector<std::uint8_t> _facet_permutations;
296
297 // Cell permutation info. See the documentation for
298 // get_cell_permutation_info for documentation of how this is encoded.
299 std::vector<std::uint32_t> _cell_permutations;
300
301 // List of facets that are on the inter-process boundary for each
302 // facet type
303 std::vector<std::vector<std::int32_t>> _interprocess_facets;
