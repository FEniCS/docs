140  void eval(const std::span<const std::int32_t>& cells, std::span<T> values,
141  std::array<std::size_t, 2> vshape) const
142  {
143  // Extract data from Expression
144  assert(_mesh);
145 
146  // Prepare coefficients and constants
147  const auto [coeffs, cstride] = pack_coefficients(*this, cells);
148  const std::vector<T> constant_data = pack_constants(*this);
149  const auto& fn = this->get_tabulate_expression();
150 
151  // Prepare cell geometry
152  const graph::AdjacencyList<std::int32_t>& x_dofmap
153  = _mesh->geometry().dofmap();
154  const std::size_t num_dofs_g = _mesh->geometry().cmap().dim();
155  std::span<const double> x_g = _mesh->geometry().x();
156 
157  // Create data structures used in evaluation
158  std::vector<scalar_value_type_t> coordinate_dofs(3 * num_dofs_g);
159 
160  int num_argument_dofs = 1;
161  std::span<const std::uint32_t> cell_info;
162  std::function<void(const std::span<T>&,
163  const std::span<const std::uint32_t>&, std::int32_t,
164  int)>
165  dof_transform_to_transpose
166  = [](const std::span<T>&, const std::span<const std::uint32_t>&,
167  std::int32_t, int)
168  {
169  // Do nothing
170  };
171 
172  if (_argument_function_space)
173  {
174  num_argument_dofs
175  = _argument_function_space->dofmap()->element_dof_layout().num_dofs();
176  auto element = _argument_function_space->element();
177 
178  assert(element);
179  if (element->needs_dof_transformations())
180  {
181  _mesh->topology_mutable().create_entity_permutations();
182  cell_info = std::span(_mesh->topology().get_cell_permutation_info());
183  dof_transform_to_transpose
184  = element
185  ->template get_dof_transformation_to_transpose_function<T>();
186  }
187  }
188 
189  const int size0 = _x_ref.second[0] * value_size();
190  std::vector<T> values_local(size0 * num_argument_dofs, 0);
191  const std::span<T> _values_local(values_local);
192 
193  // Iterate over cells and 'assemble' into values
194  for (std::size_t c = 0; c < cells.size(); ++c)
195  {
196  const std::int32_t cell = cells[c];
197 
198  auto x_dofs = x_dofmap.links(cell);
199  for (std::size_t i = 0; i < x_dofs.size(); ++i)
200  {
201  std::copy_n(std::next(x_g.begin(), 3 * x_dofs[i]), 3,
202  std::next(coordinate_dofs.begin(), 3 * i));
203  }
204 
205  const T* coeff_cell = coeffs.data() + c * cstride;
206  std::fill(values_local.begin(), values_local.end(), 0);
207  _fn(values_local.data(), coeff_cell, constant_data.data(),
208  coordinate_dofs.data(), nullptr, nullptr);
209 
210  dof_transform_to_transpose(_values_local, cell_info, c, size0);
211  for (std::size_t j = 0; j < values_local.size(); ++j)
212  values[c * vshape[1] + j] = values_local[j];
213  }
214  }
