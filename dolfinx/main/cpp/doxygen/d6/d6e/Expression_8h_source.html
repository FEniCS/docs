134 void eval(std::span<const std::int32_t> cells, std::span<T> values,
135 std::array<std::size_t, 2> vshape) const
136 {
137 // Prepare coefficients and constants
138 const auto [coeffs, cstride] = pack_coefficients(*this, cells);
139 const std::vector<T> constant_data = pack_constants(*this);
140 const auto& fn = this->get_tabulate_expression();
141
142 // Prepare cell geometry
143 assert(_mesh);
144 const graph::AdjacencyList<std::int32_t>& x_dofmap
145 = _mesh->geometry().dofmap();
146
147 // Get geometry data
148 auto cmaps = _mesh->geometry().cmaps();
149 assert(cmaps.size() == 1);
150
151 const std::size_t num_dofs_g = cmaps.back().dim();
152 std::span<const U> x_g = _mesh->geometry().x();
153
154 // Create data structures used in evaluation
155 std::vector<dolfinx::scalar_value_type_t<T>> coord_dofs(3 * num_dofs_g);
156
157 int num_argument_dofs = 1;
158 std::span<const std::uint32_t> cell_info;
159 std::function<void(const std::span<T>&,
160 const std::span<const std::uint32_t>&, std::int32_t,
161 int)>
162 dof_transform_to_transpose
163 = [](const std::span<T>&, const std::span<const std::uint32_t>&,
164 std::int32_t, int)
165 {
166 // Do nothing
167 };
168
169 if (_argument_function_space)
170 {
171 num_argument_dofs
172 = _argument_function_space->dofmap()->element_dof_layout().num_dofs();
173 auto element = _argument_function_space->element();
174
175 assert(element);
176 if (element->needs_dof_transformations())
177 {
178 _mesh->topology_mutable()->create_entity_permutations();
179 cell_info = std::span(_mesh->topology()->get_cell_permutation_info());
180 dof_transform_to_transpose
181 = element
182 ->template get_dof_transformation_to_transpose_function<T>();
183 }
184 }
185
186 // Iterate over cells and 'assemble' into values
187 const int size0 = _x_ref.second[0] * value_size();
188 std::vector<T> values_local(size0 * num_argument_dofs, 0);
189 for (std::size_t c = 0; c < cells.size(); ++c)
190 {
191 const std::int32_t cell = cells[c];
192 auto x_dofs = x_dofmap.links(cell);
193 for (std::size_t i = 0; i < x_dofs.size(); ++i)
194 {
195 std::copy_n(std::next(x_g.begin(), 3 * x_dofs[i]), 3,
196 std::next(coord_dofs.begin(), 3 * i));
197 }
198
199 const T* coeff_cell = coeffs.data() + c * cstride;
200 std::fill(values_local.begin(), values_local.end(), 0);
201 _fn(values_local.data(), coeff_cell, constant_data.data(),
202 coord_dofs.data(), nullptr, nullptr);
203
204 dof_transform_to_transpose(values_local, cell_info, c, size0);
205 for (std::size_t j = 0; j < values_local.size(); ++j)
206 values[c * vshape[1] + j] = values_local[j];
207 }
208 }
