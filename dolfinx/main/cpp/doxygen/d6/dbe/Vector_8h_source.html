222template <class V>
223auto inner_product(const V& a, const V& b)
224{
225 using T = typename V::value_type;
226 const std::int32_t local_size = a.bs() * a.index_map()->size_local();
227 if (local_size != b.bs() * b.index_map()->size_local())
228 throw std::runtime_error("Incompatible vector sizes");
229 std::span<const T> x_a = a.array().subspan(0, local_size);
230 std::span<const T> x_b = b.array().subspan(0, local_size);
231
232 const T local = std::transform_reduce(
233 x_a.begin(), x_a.end(), x_b.begin(), static_cast<T>(0), std::plus{},
234 [](T a, T b) -> T
235 {
236 if constexpr (std::is_same<T, std::complex<double>>::value
237 or std::is_same<T, std::complex<float>>::value)
238 {
239 return std::conj(a) * b;
240 }
241 else
242 return a * b;
243 });
244
245 T result;
246 MPI_Allreduce(&local, &result, 1, dolfinx::MPI::mpi_type<T>(), MPI_SUM,
247 a.index_map()->comm());
248 return result;
249}

254auto squared_norm(const V& a)
255{
256 using T = typename V::value_type;
257 T result = inner_product(a, a);

265template <class V>
266auto norm(const V& x, Norm type = Norm::l2)
267{
268 using T = typename V::value_type;
269 switch (type)
270 {
271 case Norm::l1:
272 {
273 std::int32_t size_local = x.bs() * x.index_map()->size_local();
274 std::span<const T> data = x.array().subspan(0, size_local);
275 using U = typename dolfinx::scalar_value_type_t<T>;
276 U local_l1
277 = std::accumulate(data.begin(), data.end(), U(0),
278 [](auto norm, auto x) { return norm + std::abs(x); });
279 U l1(0);
280 MPI_Allreduce(&local_l1, &l1, 1, MPI::mpi_type<U>(), MPI_SUM,
281 x.index_map()->comm());
282 return l1;
283 }
284 case Norm::l2:
285 return std::sqrt(squared_norm(x));
286 case Norm::linf:
287 {
288 std::int32_t size_local = x.bs() * x.index_map()->size_local();
289 std::span<const T> data = x.array().subspan(0, size_local);
290 auto max_pos = std::max_element(data.begin(), data.end(),
291 [](T a, T b)
292 { return std::norm(a) < std::norm(b); });
293 auto local_linf = std::abs(*max_pos);
294 decltype(local_linf) linf = 0;
295 MPI_Allreduce(&local_linf, &linf, 1, MPI::mpi_type<decltype(linf)>(),
296 MPI_MAX, x.index_map()->comm());
297 return linf;
298 }
299 default:
300 throw std::runtime_error("Norm type not supported");
301 }

309template <class V>
310void orthonormalize(std::span<V> basis, double tol = 1.0e-10)
311{
312 using T = typename V::value_type;
313 // Loop over each vector in basis
314 for (std::size_t i = 0; i < basis.size(); ++i)
315 {
316 // Orthogonalize vector i with respect to previously orthonormalized
317 // vectors
318 for (std::size_t j = 0; j < i; ++j)
319 {
320 // basis_i <- basis_i - dot_ij basis_j
321 T dot_ij = inner_product(basis[i], basis[j]);
322 std::transform(basis[j].array().begin(), basis[j].array().end(),
323 basis[i].array().begin(), basis[i].mutable_array().begin(),
324 [dot_ij](auto xj, auto xi) { return xi - dot_ij * xj; });
325 }
326
327 // Normalise basis function
328 double norm = la::norm(basis[i], la::Norm::l2);
329 if (norm < tol)
330 {
331 throw std::runtime_error(
332 "Linear dependency detected. Cannot orthogonalize.");
333 }
334 std::transform(basis[i].array().begin(), basis[i].array().end(),
335 basis[i].mutable_array().begin(),
336 [norm](auto x) { return x / norm; });
337 }
338}

344template <class V>
345bool is_orthonormal(std::span<const V> basis, double tol = 1.0e-10)
346{
347 using T = typename V::value_type;
348 for (std::size_t i = 0; i < basis.size(); i++)
349 {
350 for (std::size_t j = i; j < basis.size(); j++)
351 {
352 const double delta_ij = (i == j) ? 1.0 : 0.0;
353 T dot_ij = inner_product(basis[i], basis[j]);
354 if (std::abs(delta_ij - dot_ij) > tol)
355 return false;
356 }
357 }
358
359 return true;
360}
