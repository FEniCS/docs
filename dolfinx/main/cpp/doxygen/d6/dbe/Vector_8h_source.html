221template <class V>
222auto inner_product(const V& a, const V& b)
223{
224 using T = typename V::value_type;
225 const std::int32_t local_size = a.bs() * a.index_map()->size_local();
226 if (local_size != b.bs() * b.index_map()->size_local())
227 throw std::runtime_error("Incompatible vector sizes");
228 std::span<const T> x_a = a.array().subspan(0, local_size);
229 std::span<const T> x_b = b.array().subspan(0, local_size);
230
231 const T local = std::transform_reduce(
232 x_a.begin(), x_a.end(), x_b.begin(), static_cast<T>(0), std::plus{},
233 [](T a, T b) -> T
234 {
235 if constexpr (std::is_same<T, std::complex<double>>::value
236 or std::is_same<T, std::complex<float>>::value)
237 {
238 return std::conj(a) * b;
239 }
240 else
241 return a * b;
242 });
243
244 T result;
245 MPI_Allreduce(&local, &result, 1, dolfinx::MPI::mpi_type<T>(), MPI_SUM,
246 a.index_map()->comm());
247 return result;
248}

252template <class V>
253auto squared_norm(const V& a)
254{
255 using T = typename V::value_type;
256 T result = inner_product(a, a);

264template <class V>
265auto norm(const V& a, Norm type = Norm::l2)
266{
267 using T = typename V::value_type;
268 switch (type)
269 {
270 case Norm::l2:
271 return std::sqrt(squared_norm(a));
272 case Norm::linf:
273 {
274 const std::int32_t size_local = a.bs() * a.index_map()->size_local();
275 std::span<const T> x_a = a.array().subspan(0, size_local);
276 auto max_pos = std::max_element(x_a.begin(), x_a.end(),
277 [](T a, T b)
278 { return std::norm(a) < std::norm(b); });
279 auto local_linf = std::abs(*max_pos);
280 decltype(local_linf) linf = 0;
281 MPI_Allreduce(&local_linf, &linf, 1, MPI::mpi_type<decltype(linf)>(),
282 MPI_MAX, a.index_map()->comm());
283 return linf;
284 }
285 default:
286 throw std::runtime_error("Norm type not supported");
287 }

295template <class V>
296void orthonormalize(std::span<V> basis, double tol = 1.0e-10)
297{
298 using T = typename V::value_type;
299 // Loop over each vector in basis
300 for (std::size_t i = 0; i < basis.size(); ++i)
301 {
302 // Orthogonalize vector i with respect to previously orthonormalized
303 // vectors
304 for (std::size_t j = 0; j < i; ++j)
305 {
306 // basis_i <- basis_i - dot_ij basis_j
307 T dot_ij = inner_product(basis[i], basis[j]);
308 std::transform(basis[j].array().begin(), basis[j].array().end(),
309 basis[i].array().begin(), basis[i].mutable_array().begin(),
310 [dot_ij](auto xj, auto xi) { return xi - dot_ij * xj; });
311 }
312
313 // Normalise basis function
314 double norm = la::norm(basis[i], la::Norm::l2);
315 if (norm < tol)
316 {
317 throw std::runtime_error(
318 "Linear dependency detected. Cannot orthogonalize.");
319 }
320 std::transform(basis[i].array().begin(), basis[i].array().end(),
321 basis[i].mutable_array().begin(),
322 [norm](auto x) { return x / norm; });
323 }
324}

330template <class V>
331bool is_orthonormal(std::span<const V> basis, double tol = 1.0e-10)
332{
333 using T = typename V::value_type;
334 for (std::size_t i = 0; i < basis.size(); i++)
335 {
336 for (std::size_t j = i; j < basis.size(); j++)
337 {
338 const double delta_ij = (i == j) ? 1.0 : 0.0;
339 T dot_ij = inner_product(basis[i], basis[j]);
340 if (std::abs(delta_ij - dot_ij) > tol)
341 return false;
342 }
343 }
344
345 return true;
346}
