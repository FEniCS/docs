39 template <typename U, typename V, typename T>
40 void interpolation_apply(const U& Pi, const V& data, std::vector<T>& coeffs,
41  int bs)
42 {
43  // Compute coefficients = Pi * x (matrix-vector multiply)
44  if (bs == 1)
45  {
46  assert(data.shape(0) * data.shape(1) == Pi.shape(1));
47  for (std::size_t i = 0; i < Pi.shape(0); ++i)
48  {
49  coeffs[i] = 0.0;
50  for (std::size_t k = 0; k < data.shape(1); ++k)
51  for (std::size_t j = 0; j < data.shape(0); ++j)
52  coeffs[i] += Pi(i, k * data.shape(0) + j) * data(j, k);
53  }
54  }
55  else
56  {
57  const std::size_t cols = Pi.shape(1);
58  assert(data.shape(0) == Pi.shape(1));
59  assert(data.shape(1) == bs);
60  for (int k = 0; k < bs; ++k)
61  {
62  for (std::size_t i = 0; i < Pi.shape(0); ++i)
63  {
64  T acc = 0;
65  for (std::size_t j = 0; j < cols; ++j)
66  acc += Pi(i, j) * data(j, k);
67  coeffs[bs * i + k] = acc;
68  }
69  }
70  }
71 }

83 template <typename T>
84 void interpolate_same_map(Function<T>& u1, const Function<T>& u0,
85  const xtl::span<const std::int32_t>& cells)
86 {
87  auto V0 = u0.function_space();
88  assert(V0);
89  auto V1 = u1.function_space();
90  assert(V1);
91  auto mesh = V0->mesh();
92  assert(mesh);
93 
94  std::shared_ptr<const FiniteElement> element0 = V0->element();
95  assert(element0);
96  std::shared_ptr<const FiniteElement> element1 = V1->element();
97  assert(element1);
98 
99  const int tdim = mesh->topology().dim();
100  auto map = mesh->topology().index_map(tdim);
101  assert(map);
102  xtl::span<T> u1_array = u1.x()->mutable_array();
103  xtl::span<const T> u0_array = u0.x()->array();
104 
105  xtl::span<const std::uint32_t> cell_info;
106  if (element1->needs_dof_transformations()
107  or element0->needs_dof_transformations())
108  {
109  mesh->topology_mutable().create_entity_permutations();
110  cell_info = xtl::span(mesh->topology().get_cell_permutation_info());
111  }
112 
113  // Get dofmaps
114  auto dofmap1 = V1->dofmap();
115  auto dofmap0 = V0->dofmap();
116 
117  // Create interpolation operator
118  const xt::xtensor<double, 2> i_m
119  = element1->create_interpolation_operator(*element0);
120 
121  // Get block sizes and dof transformation operators
122  const int bs1 = dofmap1->bs();
123  const int bs0 = dofmap0->bs();
124  auto apply_dof_transformation
125  = element0->get_dof_transformation_function<T>(false, true, false);
126  auto apply_inverse_dof_transform
127  = element1->get_dof_transformation_function<T>(true, true, false);
128 
129  // Creat working array
130  std::vector<T> local0(element0->space_dimension());
131  std::vector<T> local1(element1->space_dimension());
132 
133  // Iterate over mesh and interpolate on each cell
134  for (auto c : cells)
135  {
136  xtl::span<const std::int32_t> dofs0 = dofmap0->cell_dofs(c);
137  for (std::size_t i = 0; i < dofs0.size(); ++i)
138  for (int k = 0; k < bs0; ++k)
139  local0[bs0 * i + k] = u0_array[bs0 * dofs0[i] + k];
140 
141  apply_dof_transformation(local0, cell_info, c, 1);
142 
143  // FIXME: Get compile-time ranges from Basix
144  // Apply interpolation operator
145  std::fill(local1.begin(), local1.end(), 0);
146  for (std::size_t i = 0; i < i_m.shape(0); ++i)
147  for (std::size_t j = 0; j < i_m.shape(1); ++j)
148  local1[i] += i_m(i, j) * local0[j];
149 
150  apply_inverse_dof_transform(local1, cell_info, c, 1);
151 
152  xtl::span<const std::int32_t> dofs1 = dofmap1->cell_dofs(c);
153  for (std::size_t i = 0; i < dofs1.size(); ++i)
154  for (int k = 0; k < bs1; ++k)
155  u1_array[bs1 * dofs1[i] + k] = local1[bs1 * i + k];
156  }
157 }

168 template <typename T>
169 void interpolate_nonmatching_maps(Function<T>& u1, const Function<T>& u0,
170  const xtl::span<const std::int32_t>& cells)
171 {
172  // Get mesh
173  auto V0 = u0.function_space();
174  assert(V0);
175  auto mesh = V0->mesh();
176  assert(mesh);
177 
178  // Mesh dims
179  const int tdim = mesh->topology().dim();
180  const int gdim = mesh->geometry().dim();
181 
182  // Get elements
183  auto V1 = u1.function_space();
184  assert(V1);
185  std::shared_ptr<const FiniteElement> element0 = V0->element();
186  assert(element0);
187  std::shared_ptr<const FiniteElement> element1 = V1->element();
188  assert(element1);
189 
190  xtl::span<const std::uint32_t> cell_info;
191  if (element1->needs_dof_transformations()
192  or element0->needs_dof_transformations())
193  {
194  mesh->topology_mutable().create_entity_permutations();
195  cell_info = xtl::span(mesh->topology().get_cell_permutation_info());
196  }
197 
198  // Get dofmaps
199  auto dofmap0 = V0->dofmap();
200  auto dofmap1 = V1->dofmap();
201 
202  const xt::xtensor<double, 2> X = element1->interpolation_points();
203 
204  // Get block sizes and dof transformation operators
205  const int bs0 = element0->block_size();
206  const int bs1 = element1->block_size();
207  const auto apply_dof_transformation0
208  = element0->get_dof_transformation_function<double>(false, false, false);
209  const auto apply_inverse_dof_transform1
210  = element1->get_dof_transformation_function<T>(true, true, false);
211 
212  // Get sizes of elements
213  const std::size_t dim0 = element0->space_dimension() / bs0;
214  const std::size_t value_size_ref0 = element0->reference_value_size() / bs0;
215  const std::size_t value_size0 = element0->value_size() / bs0;
216 
217  // Get geometry data
218  const fem::CoordinateElement& cmap = mesh->geometry().cmap();
219  const graph::AdjacencyList<std::int32_t>& x_dofmap
220  = mesh->geometry().dofmap();
221  // FIXME: Add proper interface for num coordinate dofs
222  const std::size_t num_dofs_g = x_dofmap.num_links(0);
223  xtl::span<const double> x_g = mesh->geometry().x();
224 
225  // Evaluate coordinate map basis at reference interpolation points
226  xt::xtensor<double, 4> phi(cmap.tabulate_shape(1, X.shape(0)));
227  xt::xtensor<double, 2> dphi;
228  cmap.tabulate(1, X, phi);
229  dphi = xt::view(phi, xt::range(1, tdim + 1), 0, xt::all(), 0);
230 
231  // Evaluate v basis functions at reference interpolation points
232  xt::xtensor<double, 4> basis_derivatives_reference0(
233  {1, X.shape(0), dim0, value_size_ref0});
234  element0->tabulate(basis_derivatives_reference0, X, 0);
235 
236  // Create working arrays
237  std::vector<T> local1(element1->space_dimension());
238  std::vector<T> coeffs0(element0->space_dimension());
239  xt::xtensor<double, 3> basis0({X.shape(0), dim0, value_size0});
240  xt::xtensor<double, 3> basis_reference0({X.shape(0), dim0, value_size_ref0});
241  xt::xtensor<T, 3> values0({X.shape(0), 1, element1->value_size()});
242  xt::xtensor<T, 3> mapped_values0({X.shape(0), 1, element1->value_size()});
243  xt::xtensor<double, 2> coordinate_dofs({num_dofs_g, gdim});
244  xt::xtensor<double, 3> J({X.shape(0), gdim, tdim});
245  xt::xtensor<double, 3> K({X.shape(0), tdim, gdim});
246  std::vector<double> detJ(X.shape(0));
247 
248  // Get interpolation operator
249  const xt::xtensor<double, 2>& Pi_1 = element1->interpolation_operator();
250 
251  using u_t = xt::xview<decltype(basis_reference0)&, std::size_t,
252  xt::xall<std::size_t>, xt::xall<std::size_t>>;
253  using U_t = xt::xview<decltype(basis_reference0)&, std::size_t,
254  xt::xall<std::size_t>, xt::xall<std::size_t>>;
255  using J_t = xt::xview<decltype(J)&, std::size_t, xt::xall<std::size_t>,
256  xt::xall<std::size_t>>;
257  using K_t = xt::xview<decltype(K)&, std::size_t, xt::xall<std::size_t>,
258  xt::xall<std::size_t>>;
259  auto push_forward_fn0 = element0->map_fn<u_t, U_t, J_t, K_t>();
260 
261  using u1_t = xt::xview<decltype(values0)&, std::size_t, xt::xall<std::size_t>,
262  xt::xall<std::size_t>>;
263  using U1_t = xt::xview<decltype(mapped_values0)&, std::size_t,
264  xt::xall<std::size_t>, xt::xall<std::size_t>>;
265  auto pull_back_fn1 = element1->map_fn<U1_t, u1_t, K_t, J_t>();
266 
267  // Iterate over mesh and interpolate on each cell
268  xtl::span<const T> array0 = u0.x()->array();
269  xtl::span<T> array1 = u1.x()->mutable_array();
270  for (auto c : cells)
271  {
272  // Get cell geometry (coordinate dofs)
273  auto x_dofs = x_dofmap.links(c);
274  for (std::size_t i = 0; i < num_dofs_g; ++i)
275  {
276  const int pos = 3 * x_dofs[i];
277  for (std::size_t j = 0; j < gdim; ++j)
278  coordinate_dofs(i, j) = x_g[pos + j];
279  }
280 
281  // Compute Jacobians and reference points for current cell
282  J.fill(0);
283  for (std::size_t p = 0; p < X.shape(0); ++p)
284  {
285  auto _J = xt::view(J, p, xt::all(), xt::all());
286  cmap.compute_jacobian(dphi, coordinate_dofs, _J);
287  cmap.compute_jacobian_inverse(_J, xt::view(K, p, xt::all(), xt::all()));
288  detJ[p] = cmap.compute_jacobian_determinant(_J);
289  }
290 
291  // Get evaluated basis on reference, apply DOF transformations, and
292  // push forward to physical element
293  basis_reference0 = xt::view(basis_derivatives_reference0, 0, xt::all(),
294  xt::all(), xt::all());
295  for (std::size_t p = 0; p < X.shape(0); ++p)
296  {
297  apply_dof_transformation0(
298  xtl::span(basis_reference0.data() + p * dim0 * value_size_ref0,
299  dim0 * value_size_ref0),
300  cell_info, c, value_size_ref0);
301  }
302 
303  for (std::size_t i = 0; i < basis0.shape(0); ++i)
304  {
305  auto _K = xt::view(K, i, xt::all(), xt::all());
306  auto _J = xt::view(J, i, xt::all(), xt::all());
307  auto _u = xt::view(basis0, i, xt::all(), xt::all());
308  auto _U = xt::view(basis_reference0, i, xt::all(), xt::all());
309  push_forward_fn0(_u, _U, _J, detJ[i], _K);
310  }
311 
312  // Copy expansion coefficients for v into local array
313  const int dof_bs0 = dofmap0->bs();
314  xtl::span<const std::int32_t> dofs0 = dofmap0->cell_dofs(c);
315  for (std::size_t i = 0; i < dofs0.size(); ++i)
316  for (int k = 0; k < dof_bs0; ++k)
317  coeffs0[dof_bs0 * i + k] = array0[dof_bs0 * dofs0[i] + k];
318 
319  // Evaluate v at the interpolation points (physical space values)
320  for (std::size_t p = 0; p < X.shape(0); ++p)
321  {
322  for (int k = 0; k < bs0; ++k)
323  {
324  for (std::size_t j = 0; j < value_size0; ++j)
325  {
326  T acc = 0;
327  for (std::size_t i = 0; i < dim0; ++i)
328  acc += coeffs0[bs0 * i + k] * basis0(p, i, j);
329  values0(p, 0, j * bs0 + k) = acc;
330  }
331  }
332  }
333 
334  // Pull back the physical values to the u reference
335  for (std::size_t i = 0; i < values0.shape(0); ++i)
336  {
337  auto _K = xt::view(K, i, xt::all(), xt::all());
338  auto _J = xt::view(J, i, xt::all(), xt::all());
339  auto _u = xt::view(values0, i, xt::all(), xt::all());
340  auto _U = xt::view(mapped_values0, i, xt::all(), xt::all());
341  pull_back_fn1(_U, _u, _K, 1.0 / detJ[i], _J);
342  }
343 
344  auto _mapped_values0 = xt::view(mapped_values0, xt::all(), 0, xt::all());
345  interpolation_apply(Pi_1, _mapped_values0, local1, bs1);
346  apply_inverse_dof_transform1(local1, cell_info, c, 1);
347 
348  // Copy local coefficients to the correct position in u dof array
349  const int dof_bs1 = dofmap1->bs();
350  xtl::span<const std::int32_t> dofs1 = dofmap1->cell_dofs(c);
351  for (std::size_t i = 0; i < dofs1.size(); ++i)
352  for (int k = 0; k < dof_bs1; ++k)
353  array1[dof_bs1 * dofs1[i] + k] = local1[dof_bs1 * i + k];
354  }
355 }

383 template <typename T>
384 void interpolate(Function<T>& u, xt::xarray<T>& f,
385  const xtl::span<const std::int32_t>& cells)
386 {
387  const std::shared_ptr<const FiniteElement> element
388  = u.function_space()->element();
389  assert(element);
390  const int element_bs = element->block_size();
391  if (int num_sub = element->num_sub_elements();
392  num_sub > 0 and num_sub != element_bs)
393  {
394  throw std::runtime_error("Cannot directly interpolate a mixed space. "
395  "Interpolate into subspaces.");
396  }
397 
398  // Get mesh
399  assert(u.function_space());
400  auto mesh = u.function_space()->mesh();
401  assert(mesh);
402 
403  const int gdim = mesh->geometry().dim();
404  const int tdim = mesh->topology().dim();
405 
406  xtl::span<const std::uint32_t> cell_info;
407  if (element->needs_dof_transformations())
408  {
409  mesh->topology_mutable().create_entity_permutations();
410  cell_info = xtl::span(mesh->topology().get_cell_permutation_info());
411  }
412 
413  if (f.dimension() == 1)
414  {
415  if (element->value_size() != 1)
416  throw std::runtime_error("Interpolation data has the wrong shape.");
417  f.reshape({std::size_t(element->value_size()),
418  std::size_t(f.shape(0) / element->value_size())});
419  }
420 
421  if (f.shape(0) != element->value_size())
422  throw std::runtime_error("Interpolation data has the wrong shape.");
423 
424  // Get dofmap
425  const auto dofmap = u.function_space()->dofmap();
426  assert(dofmap);
427  const int dofmap_bs = dofmap->bs();
428 
429  // Loop over cells and compute interpolation dofs
430  const int num_scalar_dofs = element->space_dimension() / element_bs;
431  const int value_size = element->value_size() / element_bs;
432 
433  xtl::span<T> coeffs = u.x()->mutable_array();
434  std::vector<T> _coeffs(num_scalar_dofs);
435 
436  // This assumes that any element with an identity interpolation matrix
437  // is a point evaluation
438  if (element->interpolation_ident())
439  {
440  const std::function<void(const xtl::span<T>&,
441  const xtl::span<const std::uint32_t>&,
442  std::int32_t, int)>
443  apply_inv_transpose_dof_transformation
444  = element->get_dof_transformation_function<T>(true, true, true);
445  for (std::size_t c = 0; c < cells.size(); ++c)
446  {
447  const std::int32_t cell = cells[c];
448  xtl::span<const std::int32_t> dofs = dofmap->cell_dofs(cell);
449  for (int k = 0; k < element_bs; ++k)
450  {
451  for (int i = 0; i < num_scalar_dofs; ++i)
452  _coeffs[i] = f(k, c * num_scalar_dofs + i);
453  apply_inv_transpose_dof_transformation(_coeffs, cell_info, cell, 1);
454  for (int i = 0; i < num_scalar_dofs; ++i)
455  {
456  const int dof = i * element_bs + k;
457  std::div_t pos = std::div(dof, dofmap_bs);
458  coeffs[dofmap_bs * dofs[pos.quot] + pos.rem] = _coeffs[i];
459  }
460  }
461  }
462  }
463  else
464  {
465  // Get the interpolation points on the reference cells
466  const xt::xtensor<double, 2>& X = element->interpolation_points();
467  if (X.shape(0) == 0)
468  {
469  throw std::runtime_error(
470  "Interpolation into this space is not yet supported.");
471  }
472 
473  if (f.shape(1) != cells.size() * X.shape(0))
474  throw std::runtime_error("Interpolation data has the wrong shape.");
475 
476  // Get coordinate map
477  const fem::CoordinateElement& cmap = mesh->geometry().cmap();
478 
479  // Get geometry data
480  const graph::AdjacencyList<std::int32_t>& x_dofmap
481  = mesh->geometry().dofmap();
482  // FIXME: Add proper interface for num coordinate dofs
483  const int num_dofs_g = x_dofmap.num_links(0);
484  xtl::span<const double> x_g = mesh->geometry().x();
485 
486  // Create data structures for Jacobian info
487  xt::xtensor<double, 3> J = xt::empty<double>({int(X.shape(0)), gdim, tdim});
488  xt::xtensor<double, 3> K = xt::empty<double>({int(X.shape(0)), tdim, gdim});
489  xt::xtensor<double, 1> detJ = xt::empty<double>({X.shape(0)});
490 
491  xt::xtensor<double, 2> coordinate_dofs
492  = xt::empty<double>({num_dofs_g, gdim});
493 
494  xt::xtensor<T, 3> reference_data({X.shape(0), 1, value_size});
495  xt::xtensor<T, 3> _vals({X.shape(0), 1, value_size});
496 
497  // Tabulate 1st order derivatives of shape functions at interpolation coords
498  xt::xtensor<double, 3> dphi = xt::view(
499  cmap.tabulate(1, X), xt::range(1, tdim + 1), xt::all(), xt::all(), 0);
500 
501  const std::function<void(const xtl::span<T>&,
502  const xtl::span<const std::uint32_t>&,
503  std::int32_t, int)>
504  apply_inverse_transpose_dof_transformation
505  = element->get_dof_transformation_function<T>(true, true);
506 
507  // Get interpolation operator
508  const xt::xtensor<double, 2>& Pi = element->interpolation_operator();
509 
510  using U_t = xt::xview<decltype(reference_data)&, std::size_t,
511  xt::xall<std::size_t>, xt::xall<std::size_t>>;
512  using J_t = xt::xview<decltype(J)&, std::size_t, xt::xall<std::size_t>,
513  xt::xall<std::size_t>>;
514  auto pull_back_fn = element->map_fn<U_t, U_t, J_t, J_t>();
515  for (std::size_t c = 0; c < cells.size(); ++c)
516  {
517  const std::int32_t cell = cells[c];
518  auto x_dofs = x_dofmap.links(cell);
519  for (int i = 0; i < num_dofs_g; ++i)
520  {
521  const int pos = 3 * x_dofs[i];
522  for (int j = 0; j < gdim; ++j)
523  coordinate_dofs(i, j) = x_g[pos + j];
524  }
525 
526  // Compute J, detJ and K
527  J.fill(0);
528  for (std::size_t p = 0; p < X.shape(0); ++p)
529  {
530  cmap.compute_jacobian(xt::view(dphi, xt::all(), p, xt::all()),
531  coordinate_dofs,
532  xt::view(J, p, xt::all(), xt::all()));
533  cmap.compute_jacobian_inverse(xt::view(J, p, xt::all(), xt::all()),
534  xt::view(K, p, xt::all(), xt::all()));
535  detJ[p] = cmap.compute_jacobian_determinant(
536  xt::view(J, p, xt::all(), xt::all()));
537  }
538 
539  xtl::span<const std::int32_t> dofs = dofmap->cell_dofs(cell);
540  for (int k = 0; k < element_bs; ++k)
541  {
542  // Extract computed expression values for element block k
543  for (int m = 0; m < value_size; ++m)
544  {
545  std::copy_n(&f(k * value_size + m, c * X.shape(0)), X.shape(0),
546  xt::view(_vals, xt::all(), 0, m).begin());
547  }
548 
549  // Get element degrees of freedom for block
550  for (std::size_t i = 0; i < X.shape(0); ++i)
551  {
552  auto _K = xt::view(K, i, xt::all(), xt::all());
553  auto _J = xt::view(J, i, xt::all(), xt::all());
554  auto _u = xt::view(_vals, i, xt::all(), xt::all());
555  auto _U = xt::view(reference_data, i, xt::all(), xt::all());
556  pull_back_fn(_U, _u, _K, 1.0 / detJ[i], _J);
557  }
558 
559  auto ref_data = xt::view(reference_data, xt::all(), 0, xt::all());
560  impl::interpolation_apply(Pi, ref_data, _coeffs, element_bs);
561  apply_inverse_transpose_dof_transformation(_coeffs, cell_info, cell, 1);
562 
563  // Copy interpolation dofs into coefficient vector
564  assert(_coeffs.size() == num_scalar_dofs);
565  for (int i = 0; i < num_scalar_dofs; ++i)
566  {
567  const int dof = i * element_bs + k;
568  std::div_t pos = std::div(dof, dofmap_bs);
569  coeffs[dofmap_bs * dofs[pos.quot] + pos.rem] = _coeffs[i];
570  }
571  }
572  }
573  }
574 }

581 template <typename T>
582 void interpolate(Function<T>& u, const Function<T>& v,
583  const xtl::span<const std::int32_t>& cells)
584 {
585  assert(u.function_space());
586  assert(v.function_space());
587  std::shared_ptr<const mesh::Mesh> mesh = u.function_space()->mesh();
588  assert(mesh);
589 
590  auto cell_map0 = mesh->topology().index_map(mesh->topology().dim());
591  assert(cell_map0);
592  std::size_t num_cells0 = cell_map0->size_local() + cell_map0->num_ghosts();
593  if (u.function_space() == v.function_space() and cells.size() == num_cells0)
594  {
595  // Same function spaces and on whole mesh
596  xtl::span<T> u1_array = u.x()->mutable_array();
597  xtl::span<const T> u0_array = v.x()->array();
598  std::copy(u0_array.begin(), u0_array.end(), u1_array.begin());
599  }
600  else
601  {
602  // Get mesh and check that functions share the same mesh
603  if (mesh != v.function_space()->mesh())
604  {
605  throw std::runtime_error(
606  "Interpolation on different meshes not supported (yet).");
607  }
608 
609  // Get elements and check value shape
610  auto element0 = v.function_space()->element();
611  assert(element0);
612  auto element1 = u.function_space()->element();
613  assert(element1);
614  if (element0->value_shape() != element1->value_shape())
615  {
616  throw std::runtime_error(
617  "Interpolation: elements have different value dimensions");
618  }
619 
620  if (*element1 == *element0)
621  {
622  // Same element, different dofmaps (or just a subset of cells)
623 
624  const int tdim = mesh->topology().dim();
625  auto cell_map = mesh->topology().index_map(tdim);
626  assert(cell_map);
627 
628  assert(element1->block_size() == element0->block_size());
629 
630  // Get dofmaps
631  std::shared_ptr<const fem::DofMap> dofmap0 = v.function_space()->dofmap();
632  assert(dofmap0);
633  std::shared_ptr<const fem::DofMap> dofmap1 = u.function_space()->dofmap();
634  assert(dofmap1);
635 
636  xtl::span<T> u1_array = u.x()->mutable_array();
637  xtl::span<const T> u0_array = v.x()->array();
638 
639  // Iterate over mesh and interpolate on each cell
640  const int bs0 = dofmap0->bs();
641  const int bs1 = dofmap1->bs();
642  for (auto c : cells)
643  {
644  xtl::span<const std::int32_t> dofs0 = dofmap0->cell_dofs(c);
645  xtl::span<const std::int32_t> dofs1 = dofmap1->cell_dofs(c);
646  assert(bs0 * dofs0.size() == bs1 * dofs1.size());
647  for (std::size_t i = 0; i < dofs0.size(); ++i)
648  {
649  for (int k = 0; k < bs0; ++k)
650  {
651  int index = bs0 * i + k;
652  std::div_t dv1 = std::div(index, bs1);
653  u1_array[bs1 * dofs1[dv1.quot] + dv1.rem]
654  = u0_array[bs0 * dofs0[i] + k];
655  }
656  }
657  }
658  }
659  else if (element1->map_type() == element0->map_type())
660  {
661  // Different elements, same basis function map type
662  impl::interpolate_same_map(u, v, cells);
663  }
664  else
665  {
666  // Different elements with different maps for basis functions
667  impl::interpolate_nonmatching_maps(u, v, cells);
668  }
669  }
670 }
