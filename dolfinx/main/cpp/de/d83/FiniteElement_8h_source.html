257  template <typename T>
258  std::function<void(const xtl::span<T>&, const xtl::span<const std::uint32_t>&,
259  std::int32_t, int)>
260  get_dof_transformation_function(bool inverse = false, bool transpose = false,
261  bool scalar_element = false) const
262  {
263  if (!needs_dof_transformations())
264  {
265  // If no permutation needed, return function that does nothing
266  return [](const xtl::span<T>&, const xtl::span<const std::uint32_t>&,
267  std::int32_t, int)
268  {
269  // Do nothing
270  };
271  }
272 
273  if (_sub_elements.size() != 0)
274  {
275  if (_bs == 1)
276  {
277  // Mixed element
278  std::vector<std::function<void(const xtl::span<T>&,
279  const xtl::span<const std::uint32_t>&,
280  std::int32_t, int)>>
281  sub_element_functions;
282  std::vector<int> dims;
283  for (std::size_t i = 0; i < _sub_elements.size(); ++i)
284  {
285  sub_element_functions.push_back(
286  _sub_elements[i]->get_dof_transformation_function<T>(inverse,
287  transpose));
288  dims.push_back(_sub_elements[i]->space_dimension());
289  }
290 
291  return [dims, sub_element_functions](
292  const xtl::span<T>& data,
293  const xtl::span<const std::uint32_t>& cell_info,
294  std::int32_t cell, int block_size)
295  {
296  std::size_t offset = 0;
297  for (std::size_t e = 0; e < sub_element_functions.size(); ++e)
298  {
299  const std::size_t width = dims[e] * block_size;
300  sub_element_functions[e](data.subspan(offset, width), cell_info,
301  cell, block_size);
302  offset += width;
303  }
304  };
305  }
306  else if (!scalar_element)
307  {
308  // Vector element
309  const std::function<void(const xtl::span<T>&,
310  const xtl::span<const std::uint32_t>&,
311  std::int32_t, int)>
312  sub_function = _sub_elements[0]->get_dof_transformation_function<T>(
313  inverse, transpose);
314  const int ebs = _bs;
315  return
316  [ebs, sub_function](const xtl::span<T>& data,
317  const xtl::span<const std::uint32_t>& cell_info,
318  std::int32_t cell, int data_block_size)
319  { sub_function(data, cell_info, cell, ebs * data_block_size); };
320  }
321  }
322  if (transpose)
323  {
324  if (inverse)
325  {
326  return [this](const xtl::span<T>& data,
327  const xtl::span<const std::uint32_t>& cell_info,
328  std::int32_t cell, int block_size)
329  {
330  apply_inverse_transpose_dof_transformation(data, cell_info[cell],
331  block_size);
332  };
333  }
334  else
335  {
336  return [this](const xtl::span<T>& data,
337  const xtl::span<const std::uint32_t>& cell_info,
338  std::int32_t cell, int block_size) {
339  apply_transpose_dof_transformation(data, cell_info[cell], block_size);
340  };
341  }
342  }
343  else
344  {
345  if (inverse)
346  {
347  return [this](const xtl::span<T>& data,
348  const xtl::span<const std::uint32_t>& cell_info,
349  std::int32_t cell, int block_size) {
350  apply_inverse_dof_transformation(data, cell_info[cell], block_size);
351  };
352  }
353  else
354  {
355  return [this](const xtl::span<T>& data,
356  const xtl::span<const std::uint32_t>& cell_info,
357  std::int32_t cell, int block_size)
358  { apply_dof_transformation(data, cell_info[cell], block_size); };
359  }
360  }
361  }

381  template <typename T>
382  std::function<void(const xtl::span<T>&, const xtl::span<const std::uint32_t>&,
383  std::int32_t, int)>
384  get_dof_transformation_to_transpose_function(bool inverse = false,
385  bool transpose = false,
386  bool scalar_element
387  = false) const
388  {
389  if (!needs_dof_transformations())
390  {
391  // If no permutation needed, return function that does nothing
392  return [](const xtl::span<T>&, const xtl::span<const std::uint32_t>&,
393  std::int32_t, int)
394  {
395  // Do nothing
396  };
397  }
398  else if (_sub_elements.size() != 0)
399  {
400  if (_bs == 1)
401  {
402  // Mixed element
403  std::vector<std::function<void(const xtl::span<T>&,
404  const xtl::span<const std::uint32_t>&,
405  std::int32_t, int)>>
406  sub_element_functions;
407  for (std::size_t i = 0; i < _sub_elements.size(); ++i)
408  {
409  sub_element_functions.push_back(
410  _sub_elements[i]->get_dof_transformation_to_transpose_function<T>(
411  inverse, transpose));
412  }
413 
414  return [this, sub_element_functions](
415  const xtl::span<T>& data,
416  const xtl::span<const std::uint32_t>& cell_info,
417  std::int32_t cell, int block_size)
418  {
419  std::size_t offset = 0;
420  for (std::size_t e = 0; e < sub_element_functions.size(); ++e)
421  {
422  sub_element_functions[e](data.subspan(offset, data.size() - offset),
423  cell_info, cell, block_size);
424  offset += _sub_elements[e]->space_dimension();
425  }
426  };
427  }
428  else if (!scalar_element)
429  {
430  // Vector element
431  const std::function<void(const xtl::span<T>&,
432  const xtl::span<const std::uint32_t>&,
433  std::int32_t, int)>
434  sub_function = _sub_elements[0]->get_dof_transformation_function<T>(
435  inverse, transpose);
436  return [this,
437  sub_function](const xtl::span<T>& data,
438  const xtl::span<const std::uint32_t>& cell_info,
439  std::int32_t cell, int data_block_size)
440  {
441  const int ebs = block_size();
442  const std::size_t dof_count = data.size() / data_block_size;
443  for (int block = 0; block < data_block_size; ++block)
444  {
445  sub_function(data.subspan(block * dof_count, dof_count), cell_info,
446  cell, ebs);
447  }
448  };
449  }
450  }
451 
452  if (transpose)
453  {
454  if (inverse)
455  {
456  return [this](const xtl::span<T>& data,
457  const xtl::span<const std::uint32_t>& cell_info,
458  std::int32_t cell, int block_size)
459  {
460  apply_inverse_transpose_dof_transformation_to_transpose(
461  data, cell_info[cell], block_size);
462  };
463  }
464  else
465  {
466  return [this](const xtl::span<T>& data,
467  const xtl::span<const std::uint32_t>& cell_info,
468  std::int32_t cell, int block_size)
469  {
470  apply_transpose_dof_transformation_to_transpose(data, cell_info[cell],
471  block_size);
472  };
473  }
474  }
475  else
476  {
477  if (inverse)
478  {
479  return [this](const xtl::span<T>& data,
480  const xtl::span<const std::uint32_t>& cell_info,
481  std::int32_t cell, int block_size)
482  {
483  apply_inverse_dof_transformation_to_transpose(data, cell_info[cell],
484  block_size);
485  };
486  }
487  else
488  {
489  return [this](const xtl::span<T>& data,
490  const xtl::span<const std::uint32_t>& cell_info,
491  std::int32_t cell, int block_size) {
492  apply_dof_transformation_to_transpose(data, cell_info[cell],
493  block_size);
494  };
495  }
496  }
