210  template <typename T>
211  constexpr void interpolate(const xt::xtensor<T, 2>& values,
212  const xtl::span<T>& dofs) const
213  {
214  if (!_element)
215  {
216  throw std::runtime_error("No underlying element for interpolation. "
217  "Cannot interpolate mixed elements directly.");
218  }
219 
220  const std::size_t rows = space_dimension();
221 
222  const xt::xtensor<double, 2>& Pi = interpolation_operator();
223  assert(Pi.size() % rows == 0);
224  const std::size_t cols = Pi.size() / rows;
225 
226  // Compute dofs = Pi * x (matrix-vector multiply)
227  if (_bs == 1)
228  {
229  for (std::size_t i = 0; i < rows; ++i)
230  {
231  // Can be replaced with std::transform_reduce once GCC 8 series dies.
232  // Dot product between row i of the matrix and 'values'
233  dofs[i] = std::inner_product(std::next(Pi.data(), i * cols),
234  std::next(Pi.data(), i * cols + cols),
235  values.data(), T(0.0));
236  }
237  }
238  else
239  {
240  for (int b = 0; b < _bs; ++b)
241  {
242  for (std::size_t i = 0; i < rows; ++i)
243  {
244  dofs[_bs * i + b] = 0;
245  for (std::size_t j = 0; j < cols; ++j)
246  dofs[_bs * i + b] += Pi(i, j) * values(b * cols + j);
247  }
248  }
249  }
250  }

305  template <typename T>
306  std::function<void(const xtl::span<T>&, const xtl::span<const std::uint32_t>&,
307  std::int32_t, int)>
308  get_dof_transformation_function(bool inverse = false, bool transpose = false,
309  bool scalar_element = false) const
310  {
311  if (!needs_dof_transformations())
312  {
313  // If no permutation needed, return function that does nothing
314  return [](const xtl::span<T>&, const xtl::span<const std::uint32_t>&,
315  std::int32_t, int)
316  {
317  // Do nothing
318  };
319  }
320 
321  if (_sub_elements.size() != 0)
322  {
323  if (_bs == 1)
324  {
325  // Mixed element
326  std::vector<std::function<void(const xtl::span<T>&,
327  const xtl::span<const std::uint32_t>&,
328  std::int32_t, int)>>
329  sub_element_functions;
330  std::vector<int> dims;
331  for (std::size_t i = 0; i < _sub_elements.size(); ++i)
332  {
333  sub_element_functions.push_back(
334  _sub_elements[i]->get_dof_transformation_function<T>(inverse,
335  transpose));
336  dims.push_back(_sub_elements[i]->space_dimension());
337  }
338 
339  return [dims, sub_element_functions](
340  const xtl::span<T>& data,
341  const xtl::span<const std::uint32_t>& cell_info,
342  std::int32_t cell, int block_size)
343  {
344  std::size_t offset = 0;
345  for (std::size_t e = 0; e < sub_element_functions.size(); ++e)
346  {
347  const std::size_t width = dims[e] * block_size;
348  sub_element_functions[e](data.subspan(offset, width), cell_info,
349  cell, block_size);
350  offset += width;
351  }
352  };
353  }
354  else if (!scalar_element)
355  {
356  // Vector element
357  const std::function<void(const xtl::span<T>&,
358  const xtl::span<const std::uint32_t>&,
359  std::int32_t, int)>
360  sub_function = _sub_elements[0]->get_dof_transformation_function<T>(
361  inverse, transpose);
362  const int ebs = _bs;
363  return
364  [ebs, sub_function](const xtl::span<T>& data,
365  const xtl::span<const std::uint32_t>& cell_info,
366  std::int32_t cell, int data_block_size)
367  { sub_function(data, cell_info, cell, ebs * data_block_size); };
368  }
369  }
370  if (transpose)
371  {
372  if (inverse)
373  {
374  return [this](const xtl::span<T>& data,
375  const xtl::span<const std::uint32_t>& cell_info,
376  std::int32_t cell, int block_size)
377  {
378  apply_inverse_transpose_dof_transformation(data, cell_info[cell],
379  block_size);
380  };
381  }
382  else
383  {
384  return [this](const xtl::span<T>& data,
385  const xtl::span<const std::uint32_t>& cell_info,
386  std::int32_t cell, int block_size) {
387  apply_transpose_dof_transformation(data, cell_info[cell], block_size);
388  };
389  }
390  }
391  else
392  {
393  if (inverse)
394  {
395  return [this](const xtl::span<T>& data,
396  const xtl::span<const std::uint32_t>& cell_info,
397  std::int32_t cell, int block_size) {
398  apply_inverse_dof_transformation(data, cell_info[cell], block_size);
399  };
400  }
401  else
402  {
403  return [this](const xtl::span<T>& data,
404  const xtl::span<const std::uint32_t>& cell_info,
405  std::int32_t cell, int block_size)
406  { apply_dof_transformation(data, cell_info[cell], block_size); };
407  }
408  }

427  std::function<void(const xtl::span<T>&, const xtl::span<const std::uint32_t>&,
428  std::int32_t, int)>
429  get_dof_transformation_to_transpose_function(bool inverse = false,
430  bool transpose = false,
431  bool scalar_element
432  = false) const
433  {
434  if (!needs_dof_transformations())
435  {
436  // If no permutation needed, return function that does nothing
437  return [](const xtl::span<T>&, const xtl::span<const std::uint32_t>&,
438  std::int32_t, int)
439  {
440  // Do nothing
441  };
442  }
443  else if (_sub_elements.size() != 0)
444  {
445  if (_bs == 1)
446  {
447  // Mixed element
448  std::vector<std::function<void(const xtl::span<T>&,
449  const xtl::span<const std::uint32_t>&,
450  std::int32_t, int)>>
451  sub_element_functions;
452  for (std::size_t i = 0; i < _sub_elements.size(); ++i)
453  {
454  sub_element_functions.push_back(
455  _sub_elements[i]->get_dof_transformation_to_transpose_function<T>(
456  inverse, transpose));
457  }
458 
459  return [this, sub_element_functions](
460  const xtl::span<T>& data,
461  const xtl::span<const std::uint32_t>& cell_info,
462  std::int32_t cell, int block_size)
463  {
464  std::size_t offset = 0;
465  for (std::size_t e = 0; e < sub_element_functions.size(); ++e)
466  {
467  sub_element_functions[e](data.subspan(offset, data.size() - offset),
468  cell_info, cell, block_size);
469  offset += _sub_elements[e]->space_dimension();
470  }
471  };
472  }
473  else if (!scalar_element)
474  {
475  // Vector element
476  const std::function<void(const xtl::span<T>&,
477  const xtl::span<const std::uint32_t>&,
478  std::int32_t, int)>
479  sub_function = _sub_elements[0]->get_dof_transformation_function<T>(
480  inverse, transpose);
481  return [this,
482  sub_function](const xtl::span<T>& data,
483  const xtl::span<const std::uint32_t>& cell_info,
484  std::int32_t cell, int data_block_size)
485  {
486  const int ebs = block_size();
487  const std::size_t dof_count = data.size() / data_block_size;
488  for (int block = 0; block < data_block_size; ++block)
489  {
490  sub_function(data.subspan(block * dof_count, dof_count), cell_info,
491  cell, ebs);
492  }
493  };
494  }
495  }
496 
497  if (transpose)
498  {
499  if (inverse)
500  {
501  return [this](const xtl::span<T>& data,
502  const xtl::span<const std::uint32_t>& cell_info,
503  std::int32_t cell, int block_size)
504  {
505  apply_inverse_transpose_dof_transformation_to_transpose(
506  data, cell_info[cell], block_size);
507  };
508  }
509  else
510  {
511  return [this](const xtl::span<T>& data,
512  const xtl::span<const std::uint32_t>& cell_info,
513  std::int32_t cell, int block_size)
514  {
515  apply_transpose_dof_transformation_to_transpose(data, cell_info[cell],
516  block_size);
517  };
518  }
519  }
520  else
521  {
522  if (inverse)
523  {
524  return [this](const xtl::span<T>& data,
525  const xtl::span<const std::uint32_t>& cell_info,
526  std::int32_t cell, int block_size)
527  {
528  apply_inverse_dof_transformation_to_transpose(data, cell_info[cell],
529  block_size);
530  };
531  }
532  else
533  {
534  return [this](const xtl::span<T>& data,
535  const xtl::span<const std::uint32_t>& cell_info,
536  std::int32_t cell, int block_size) {
537  apply_dof_transformation_to_transpose(data, cell_info[cell],
538  block_size);
539  };
540  }
