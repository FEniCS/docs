151  template <typename T>
152  constexpr void interpolate(const xt::xtensor<T, 2>& values,
153  xtl::span<T> dofs) const
154  {
155  if (!_element)
156  {
157  throw std::runtime_error("No underlying element for interpolation. "
158  "Cannot interpolate mixed elements directly.");
159  }
160 
161  const std::size_t rows = _space_dim / _bs;
162  assert(_space_dim % _bs == 0);
163  assert(dofs.size() == rows);
164 
165  // Compute dofs = Pi * x (matrix-vector multiply)
166  const xt::xtensor<double, 2>& Pi = _element->interpolation_matrix();
167  assert(Pi.size() % rows == 0);
168  const std::size_t cols = Pi.size() / rows;
169  for (std::size_t i = 0; i < rows; ++i)
170  {
171  // Can be replaced with std::transform_reduce once GCC 8 series dies.
172  // Dot product between row i of the matrix and 'values'
173  dofs[i] = std::inner_product(std::next(Pi.data(), i * cols),
174  std::next(Pi.data(), i * cols + cols),
175  values.data(), T(0.0));
176  }
177  }

223  template <typename T>
224  std::function<void(xtl::span<T>, const xtl::span<const std::uint32_t>&,
225  const std::int32_t, const int)>
226  get_dof_transformation_function(bool inverse = false, bool transpose = false,
227  bool scalar_element = false) const
228  {
229  if (!needs_dof_transformations())
230  {
231  // If no permutation needed, return function that does nothing
232  return [](xtl::span<T>, const xtl::span<const std::uint32_t>&,
233  const std::int32_t, const int) {};
234  }
235 
236  if (_sub_elements.size() != 0)
237  {
238  if (_bs == 1)
239  {
240  // Mixed element
241  std::vector<std::function<void(xtl::span<T>,
242  const xtl::span<const std::uint32_t>&,
243  const std::int32_t, const int)>>
244  sub_element_functions;
245  std::vector<int> dims;
246  for (std::size_t i = 0; i < _sub_elements.size(); ++i)
247  {
248  sub_element_functions.push_back(
249  _sub_elements[i]->get_dof_transformation_function<T>(inverse,
250  transpose));
251  dims.push_back(_sub_elements[i]->space_dimension());
252  }
253 
254  return [dims, sub_element_functions](
255  xtl::span<T> data,
256  const xtl::span<const std::uint32_t> cell_info,
257  const std::int32_t cell, const int block_size)
258  {
259  std::size_t start = 0;
260  for (std::size_t e = 0; e < sub_element_functions.size(); ++e)
261  {
262  const std::size_t width = dims[e] * block_size;
263  sub_element_functions[e](data.subspan(start, width), cell_info,
264  cell, block_size);
265  start += width;
266  }
267  };
268  }
269  else if (!scalar_element)
270  {
271  // Vector element
272  std::function<void(xtl::span<T>, const xtl::span<const std::uint32_t>&,
273  const std::int32_t, const int)>
274  sub_function = _sub_elements[0]->get_dof_transformation_function<T>(
275  inverse, transpose);
276  const int ebs = _bs;
277  return
278  [ebs, sub_function](xtl::span<T> data,
279  const xtl::span<const std::uint32_t> cell_info,
280  const std::int32_t cell,
281  const int data_block_size)
282  { sub_function(data, cell_info, cell, ebs * data_block_size); };
283  }
284  }
285  if (transpose)
286  {
287  if (inverse)
288  {
289  return [this](xtl::span<T> data,
290  const xtl::span<const std::uint32_t>& cell_info,
291  const std::int32_t cell, const int block_size)
292  {
293  apply_inverse_transpose_dof_transformation(data, cell_info[cell],
294  block_size);
295  };
296  }
297  else
298  {
299  return [this](xtl::span<T> data,
300  const xtl::span<const std::uint32_t>& cell_info,
301  const std::int32_t cell, const int block_size) {
302  apply_transpose_dof_transformation(data, cell_info[cell], block_size);
303  };
304  }
305  }
306  else
307  {
308  if (inverse)
309  {
310  return [this](xtl::span<T> data,
311  const xtl::span<const std::uint32_t>& cell_info,
312  const std::int32_t cell, const int block_size) {
313  apply_inverse_dof_transformation(data, cell_info[cell], block_size);
314  };
315  }
316  else
317  {
318  return [this](xtl::span<T> data,
319  const xtl::span<const std::uint32_t>& cell_info,
320  const std::int32_t cell, const int block_size)
321  { apply_dof_transformation(data, cell_info[cell], block_size); };
322  }
323  }

341  template <typename T>
342  std::function<void(xtl::span<T>, const xtl::span<const std::uint32_t>&,
343  const std::int32_t, const int)>
344  get_dof_transformation_to_transpose_function(bool inverse = false,
345  bool transpose = false,
346  bool scalar_element
347  = false) const
348  {
349  if (!needs_dof_transformations())
350  {
351  // If no permutation needed, return function that does nothing
352  return [](xtl::span<T>, const xtl::span<const std::uint32_t>&,
353  const std::int32_t, const int) {};
354  }
355  else if (_sub_elements.size() != 0)
356  {
357  if (_bs == 1)
358  {
359  // Mixed element
360  std::vector<std::function<void(xtl::span<T>,
361  const xtl::span<const std::uint32_t>&,
362  const std::int32_t, const int)>>
363  sub_element_functions;
364  for (std::size_t i = 0; i < _sub_elements.size(); ++i)
365  {
366  sub_element_functions.push_back(
367  _sub_elements[i]->get_dof_transformation_to_transpose_function<T>(
368  inverse, transpose));
369  }
370 
371  return [this, sub_element_functions](
372  xtl::span<T> data,
373  const xtl::span<const std::uint32_t>& cell_info,
374  const std::int32_t cell, const int block_size)
375  {
376  std::size_t start = 0;
377  for (std::size_t e = 0; e < sub_element_functions.size(); ++e)
378  {
379  const std::size_t width
380  = _sub_elements[e]->space_dimension() * block_size;
381  sub_element_functions[e](data.subspan(start, width), cell_info,
382  cell, block_size);
383  start += width;
384  }
385  };
386  }
387  else if (!scalar_element)
388  {
389  // Vector element
390  std::function<void(xtl::span<T>, const xtl::span<const std::uint32_t>&,
391  const std::int32_t, const int)>
392  sub_function = _sub_elements[0]->get_dof_transformation_function<T>(
393  inverse, transpose);
394  return [this, sub_function](
395  xtl::span<T> data,
396  const xtl::span<const std::uint32_t>& cell_info,
397  const std::int32_t cell, const int data_block_size)
398  {
399  const int ebs = block_size();
400  const std::size_t dof_count = data.size() / data_block_size;
401  for (int block = 0; block < data_block_size; ++block)
402  {
403  sub_function(data.subspan(block * dof_count, dof_count), cell_info,
404  cell, ebs);
405  }
406  };
407  }
408  }
409 
410  if (transpose)
411  {
412  if (inverse)
413  {
414  return [this](xtl::span<T> data,
415  const xtl::span<const std::uint32_t>& cell_info,
416  const std::int32_t cell, const int block_size)
417  {
418  apply_inverse_transpose_dof_transformation_to_transpose(
419  data, cell_info[cell], block_size);
420  };
421  }
422  else
423  {
424  return [this](xtl::span<T> data,
425  const xtl::span<const std::uint32_t>& cell_info,
426  const std::int32_t cell, const int block_size)
427  {
428  apply_transpose_dof_transformation_to_transpose(data, cell_info[cell],
429  block_size);
430  };
431  }
432  }
433  else
434  {
435  if (inverse)
436  {
437  return [this](xtl::span<T> data,
438  const xtl::span<const std::uint32_t>& cell_info,
439  const std::int32_t cell, const int block_size)
440  {
441  apply_inverse_dof_transformation_to_transpose(data, cell_info[cell],
442  block_size);
443  };
444  }
445  else
446  {
447  return [this](xtl::span<T> data,
448  const xtl::span<const std::uint32_t>& cell_info,
449  const std::int32_t cell, const int block_size) {
450  apply_dof_transformation_to_transpose(data, cell_info[cell],
451  block_size);
452  };
453  }
454  }
