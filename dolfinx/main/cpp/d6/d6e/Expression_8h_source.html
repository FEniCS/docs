91  void eval(const xtl::span<const std::int32_t>& cells, U& values) const
92  {
93  static_assert(std::is_same<T, typename U::value_type>::value,
94  "Expression and array types must be the same");
95 
96  // Extract data from Expression
97  assert(_mesh);
98 
99  // Prepare coefficients and constants
100  const auto [coeffs, cstride] = pack_coefficients(*this, cells);
101  const std::vector<T> constant_data = pack_constants(*this);
102  const auto& fn = this->get_tabulate_expression();
103 
104  // Prepare cell geometry
105  const graph::AdjacencyList<std::int32_t>& x_dofmap
106  = _mesh->geometry().dofmap();
107 
108  // FIXME: Add proper interface for num coordinate dofs
109  const std::size_t num_dofs_g = x_dofmap.num_links(0);
110  xtl::span<const double> x_g = _mesh->geometry().x();
111 
112  // Create data structures used in evaluation
113  std::vector<double> coordinate_dofs(3 * num_dofs_g);
114 
115  // Iterate over cells and 'assemble' into values
116  std::vector<T> values_e(_x.shape(0) * this->value_size(), 0);
117  for (std::size_t c = 0; c < cells.size(); ++c)
118  {
119  const std::int32_t cell = cells[c];
120 
121  auto x_dofs = x_dofmap.links(cell);
122  for (std::size_t i = 0; i < x_dofs.size(); ++i)
123  {
124  common::impl::copy_N<3>(std::next(x_g.begin(), 3 * x_dofs[i]),
125  std::next(coordinate_dofs.begin(), 3 * i));
126  }
127 
128  const T* coeff_cell = coeffs.data() + c * cstride;
129  std::fill(values_e.begin(), values_e.end(), 0.0);
130  fn(values_e.data(), coeff_cell, constant_data.data(),
131  coordinate_dofs.data());
132 
133  for (std::size_t j = 0; j < values_e.size(); ++j)
134  values(c, j) = values_e[j];
135  }
136  }
