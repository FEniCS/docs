91  template <typename U>
92  void eval(const xtl::span<const std::int32_t>& active_cells, U& values) const
93  {
94  static_assert(std::is_same<T, typename U::value_type>::value,
95  "Expression and array types must be the same");
96 
97  // Extract data from Expression
98  assert(_mesh);
99 
100  // Prepare coefficients and constants
101  const auto [coeffs, cstride] = pack_coefficients(*this, active_cells);
102  const std::vector<T> constant_data = pack_constants(*this);
103 
104  const auto& fn = this->get_tabulate_expression();
105 
106  // Prepare cell geometry
107  const graph::AdjacencyList<std::int32_t>& x_dofmap
108  = _mesh->geometry().dofmap();
109 
110  // FIXME: Add proper interface for num coordinate dofs
111  const std::size_t num_dofs_g = x_dofmap.num_links(0);
112  const xt::xtensor<double, 2>& x_g = _mesh->geometry().x();
113 
114  // Create data structures used in evaluation
115  std::vector<double> coordinate_dofs(3 * num_dofs_g);
116 
117  // Iterate over cells and 'assemble' into values
118  std::vector<T> values_e(this->num_points() * this->value_size(), 0);
119  for (std::size_t c = 0; c < active_cells.size(); ++c)
120  {
121  const std::int32_t cell = active_cells[c];
122 
123  auto x_dofs = x_dofmap.links(cell);
124  for (std::size_t i = 0; i < x_dofs.size(); ++i)
125  {
126  std::copy_n(xt::row(x_g, x_dofs[i]).cbegin(), 3,
127  std::next(coordinate_dofs.begin(), 3 * i));
128  }
129 
130  const T* coeff_cell = coeffs.data() + c * cstride;
131  std::fill(values_e.begin(), values_e.end(), 0.0);
132  fn(values_e.data(), coeff_cell, constant_data.data(),
133  coordinate_dofs.data());
134 
135  for (std::size_t j = 0; j < values_e.size(); ++j)
136  values(c, j) = values_e[j];
137  }
138  }
