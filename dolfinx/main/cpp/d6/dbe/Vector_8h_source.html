79  T norm(la::Norm type = la::Norm::l2) const
80  {
81  switch (type)
82  {
83  case la::Norm::l2:
84  return std::sqrt(this->squared_norm());
85  case la::Norm::linf:
86  {
87  const std::int32_t size_local = _map->size_local();
88  double local_linf = 0.0;
89  if (size_local > 0)
90  {
91  auto local_max_entry = std::max_element(
92  _x.begin(), std::next(_x.begin(), size_local),
93  [](T a, T b) { return std::norm(a) < std::norm(b); });
94  local_linf = std::abs(*local_max_entry);
95  }
96 
97  double linf = 0.0;
98  MPI_Allreduce(&local_linf, &linf, 1, MPI_DOUBLE, MPI_MAX, _map->comm());
99  return linf;
100  }
101  default:
102  throw std::runtime_error("Norm type not supported");
103  }
104  }

108  double squared_norm() const
109  {
110  const std::int32_t size_local = _map->size_local();
111  double result = std::transform_reduce(
112  _x.begin(), std::next(_x.begin(), size_local), 0.0, std::plus<double>(),
113  [](T val) { return std::norm(val); });
114  double norm2;
115  MPI_Allreduce(&result, &norm2, 1, MPI_DOUBLE, MPI_SUM, _map->comm());

148 template <typename T>
149 T inner_product(const Vector<T>& a, const Vector<T>& b)
150 {
151  const std::int32_t local_size = a.bs() * a.map()->size_local();
152  if (local_size != b.bs() * b.map()->size_local())
153  throw std::runtime_error("Incompatible vector sizes");
154  const std::vector<T>& x_a = a.array();
155  const std::vector<T>& x_b = b.array();
156 
157  const T local = std::transform_reduce(
158  x_a.begin(), x_a.begin() + local_size, x_b.begin(), static_cast<T>(0),
159  std::plus<T>(),
160  [](T a, T b) -> T
161  {
162  if constexpr (std::is_same<T, std::complex<double>>::value
163  or std::is_same<T, std::complex<float>>::value)
164  return std::conj(a) * b;
165  else
166  return a * b;
167  });
168 
169  T result;
170  MPI_Allreduce(&local, &result, 1, dolfinx::MPI::mpi_type<T>(), MPI_SUM,
171  a.map()->comm());
172  return result;
173 }
