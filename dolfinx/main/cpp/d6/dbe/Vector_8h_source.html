84  {
85  assert(_map);

87  // Pack send buffer
88  const std::vector<std::int32_t>& indices
89  = _map->scatter_fwd_indices().array();

91  for (std::size_t i = 0; i < indices.size(); ++i)
92  {
93  std::copy_n(std::next(_x.cbegin(), _bs * indices[i]), _bs,
94  std::next(_buffer_send_fwd.begin(), _bs * i));
95  }

98  _map->scatter_fwd_begin(xtl::span<const T>(_buffer_send_fwd), _datatype,

160  void scatter_rev_end(common::IndexMap::Mode op)
161  {
162  // Complete scatter
163  _map->scatter_rev_end(_request);
164 
165  // Copy/accumulate into owned part of the vector
166  const std::vector<std::int32_t>& shared_indices
167  = _map->scatter_fwd_indices().array();
168  switch (op)
169  {
170  case common::IndexMap::Mode::insert:
171  for (std::size_t i = 0; i < shared_indices.size(); ++i)
172  {
173  std::copy_n(std::next(_buffer_send_fwd.cbegin(), _bs * i), _bs,
174  std::next(_x.begin(), _bs * shared_indices[i]));
175  }
176  break;
177  case common::IndexMap::Mode::add:
178  for (std::size_t i = 0; i < shared_indices.size(); ++i)
179  for (int j = 0; j < _bs; ++j)
180  _x[shared_indices[i] * _bs + j] += _buffer_send_fwd[i * _bs + j];
181  break;
182  }
183  }

199  double norm(Norm type = Norm::l2) const
200  {
201  switch (type)
202  {
203  case Norm::l2:
204  return std::sqrt(this->squared_norm());
205  case Norm::linf:
206  {
207  const std::int32_t size_local = _bs * _map->size_local();
208  double local_linf = 0.0;
209  if (size_local > 0)
210  {
211  auto local_max_entry = std::max_element(
212  _x.begin(), std::next(_x.begin(), size_local),
213  [](T a, T b) { return std::norm(a) < std::norm(b); });
214  local_linf = std::abs(*local_max_entry);
215  }
216 
217  double linf = 0.0;
218  MPI_Allreduce(&local_linf, &linf, 1, MPI_DOUBLE, MPI_MAX, _map->comm());
219  return linf;
220  }
221  default:
222  throw std::runtime_error("Norm type not supported");
223  }
224  }
225 
228  double squared_norm() const
229  {
230  const std::int32_t size_local = _bs * _map->size_local();
231  double result = std::transform_reduce(
232  _x.begin(), std::next(_x.begin(), size_local), double(0),
233  std::plus<double>(), [](T val) { return std::norm(val); });
234  double norm2;
235  MPI_Allreduce(&result, &norm2, 1, MPI_DOUBLE, MPI_SUM, _map->comm());
236  return norm2;
237  }

273 template <typename T, class Allocator = std::allocator<T>>
274 T inner_product(const Vector<T, Allocator>& a, const Vector<T, Allocator>& b)
275 {
276  const std::int32_t local_size = a.bs() * a.map()->size_local();
277  if (local_size != b.bs() * b.map()->size_local())
278  throw std::runtime_error("Incompatible vector sizes");
279  xtl::span<const T> x_a = a.array();
280  xtl::span<const T> x_b = b.array();
281 
282  const T local = std::transform_reduce(
283  x_a.begin(), std::next(x_a.begin(), local_size), x_b.begin(),
284  static_cast<T>(0), std::plus<T>(),
285  [](T a, T b) -> T
286  {
287  if constexpr (std::is_same<T, std::complex<double>>::value
288  or std::is_same<T, std::complex<float>>::value)
289  {
290  return std::conj(a) * b;
291  }
292  else
293  return a * b;
294  });
295 
296  T result;
297  MPI_Allreduce(&local, &result, 1, dolfinx::MPI::mpi_type<T>(), MPI_SUM,
298  a.map()->comm());
299  return result;
300 }

307 template <typename T, typename U>
308 void orthonormalize(const xtl::span<Vector<T, U>>& basis, double tol = 1.0e-10)
309 {
310  // Loop over each vector in basis
311  for (std::size_t i = 0; i < basis.size(); ++i)
312  {
313  // Orthogonalize vector i with respect to previously orthonormalized
314  // vectors
315  for (std::size_t j = 0; j < i; ++j)
316  {
317  // basis_i <- basis_i - dot_ij basis_j
318  T dot_ij = inner_product(basis[i], basis[j]);
319  std::transform(basis[j].array().begin(), basis[j].array().end(),
320  basis[i].array().begin(), basis[i].mutable_array().begin(),
321  [dot_ij](auto xj, auto xi) { return xi - dot_ij * xj; });
322  }
323 
324  // Normalise basis function
325  double norm = basis[i].norm(Norm::l2);
326  if (norm < tol)
327  {
328  throw std::runtime_error(
329  "Linear dependency detected. Cannot orthogonalize.");
330  }
331  std::transform(basis[i].array().begin(), basis[i].array().end(),
332  basis[i].mutable_array().begin(),
333  [norm](auto x) { return x / norm; });
334  }
335 }

341 template <typename T, typename U>
342 bool is_orthonormal(const xtl::span<const Vector<T, U>>& basis,
343  double tol = 1.0e-10)
344 {
345  for (std::size_t i = 0; i < basis.size(); i++)
346  {
347  for (std::size_t j = i; j < basis.size(); j++)
348  {
349  const double delta_ij = (i == j) ? 1.0 : 0.0;
350  T dot_ij = inner_product(basis[i], basis[j]);
351  if (std::abs(delta_ij - dot_ij) > tol)
352  return false;
353  }
354  }
355 
356  return true;
357 }
