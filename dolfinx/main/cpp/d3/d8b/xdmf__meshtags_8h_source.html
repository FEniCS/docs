27 template <typename T>
28 void add_meshtags(MPI_Comm comm, const mesh::MeshTags<T>& meshtags,
29  pugi::xml_node& xml_node, const hid_t h5_id,
30  const std::string name)
31 {
32  // Get mesh
33  assert(meshtags.mesh());
34  std::shared_ptr<const mesh::Mesh> mesh = meshtags.mesh();
35  const int dim = meshtags.dim();
36 
37  const std::int32_t num_local_entities
38  = mesh->topology().index_map(dim)->size_local();
39 
40  // Find number of tagged entities in local range
41  const int num_active_entities
42  = std::lower_bound(meshtags.indices().begin(), meshtags.indices().end(),
43  num_local_entities)
44  - meshtags.indices().begin();
45 
46  const std::string path_prefix = "/MeshTags/" + name;
47  xdmf_mesh::add_topology_data(
48  comm, xml_node, h5_id, path_prefix, mesh->topology(), mesh->geometry(),
49  dim,
50  xtl::span<const std::int32_t>(meshtags.indices().data(),
51  num_active_entities));
52 
53  // Add attribute node with values
54  pugi::xml_node attribute_node = xml_node.append_child("Attribute");
55  assert(attribute_node);
56  attribute_node.append_attribute("Name") = name.c_str();
57  attribute_node.append_attribute("AttributeType") = "Scalar";
58  attribute_node.append_attribute("Center") = "Cell";
59 
60  std::int64_t global_num_values = 0;
61  const std::int64_t local_num_values = num_active_entities;
62  MPI_Allreduce(&local_num_values, &global_num_values, 1, MPI_INT64_T, MPI_SUM,
63  comm);
64  const std::int64_t offset
65  = dolfinx::MPI::global_offset(comm, num_active_entities, true);
66  const bool use_mpi_io = (dolfinx::MPI::size(comm) > 1);
67 
68  xdmf_utils::add_data_item(
69  attribute_node, h5_id, path_prefix + "/Values",
70  xtl::span<const T>(meshtags.values().data(), num_active_entities), offset,
71  {global_num_values, 1}, "", use_mpi_io);
72 }
