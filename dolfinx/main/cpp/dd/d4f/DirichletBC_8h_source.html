126  template <typename U>
127  DirichletBC(const std::variant<std::shared_ptr<const Function<T>>,
128  std::shared_ptr<const Constant<T>>>& g,
129  U&& dofs, const std::shared_ptr<const FunctionSpace>& V, void*)
130  : _function_space(V), _g(g), _dofs0(std::forward<U>(dofs))
131  {
132  // Compute number of owned dofs indices in the full space (will
133  // contain 'gaps' for sub-spaces)
134  assert(_function_space);
135  const int map0_bs = _function_space->dofmap()->index_map_bs();
136  const int map0_size = _function_space->dofmap()->index_map->size_local();
137  const int owned_size0 = map0_bs * map0_size;
138 
139  // Find number of owned indices in _dofs0
140  auto it0 = std::lower_bound(_dofs0.begin(), _dofs0.end(), owned_size0);
141  _owned_indices0 = std::distance(_dofs0.begin(), it0);
142 
143  // Unroll _dofs0 for dofmap block size > 1
144  if (const int bs = _function_space->dofmap()->bs(); bs > 1)
145  {
146  _owned_indices0 *= bs;
147  const std::vector<std::int32_t> dof_tmp = _dofs0;
148  _dofs0.resize(bs * dof_tmp.size());
149  for (std::size_t i = 0; i < dof_tmp.size(); ++i)
150  {
151  for (int k = 0; k < bs; ++k)
152  _dofs0[bs * i + k] = bs * dof_tmp[i] + k;
153  }
154  }
155  }

320  void set(xtl::span<T> x, double scale = 1.0) const
321  {
322  if (std::holds_alternative<std::shared_ptr<const Function<T>>>(_g))
323  {
324  auto g = std::get<std::shared_ptr<const Function<T>>>(_g);
325  assert(g);
326  xtl::span<const T> values = g->x()->array();
327  auto dofs1_g = _dofs1_g.empty() ? xtl::span(_dofs0) : xtl::span(_dofs1_g);
328  std::int32_t x_size = x.size();
329  for (std::size_t i = 0; i < _dofs0.size(); ++i)
330  {
331  if (_dofs0[i] < x_size)
332  {
333  assert(dofs1_g[i] < (std::int32_t)values.size());
334  x[_dofs0[i]] = scale * values[dofs1_g[i]];
335  }
336  }
337  }
338  else if (std::holds_alternative<std::shared_ptr<const Constant<T>>>(_g))
339  {
340  auto g = std::get<std::shared_ptr<const Constant<T>>>(_g);
341  std::vector<T> value = g->value;
342  int bs = _function_space->dofmap()->bs();
343  std::int32_t x_size = x.size();
344  std::for_each(_dofs0.cbegin(), _dofs0.cend(),
345  [x_size, bs, scale, &value, &x](auto dof)
346  {
347  if (dof < x_size)
348  x[dof] = scale * value[dof % bs];
349  });
350  }
351  }

357  void set(xtl::span<T> x, const xtl::span<const T>& x0,
358  double scale = 1.0) const
359  {
360  if (std::holds_alternative<std::shared_ptr<const Function<T>>>(_g))
361  {
362  auto g = std::get<std::shared_ptr<const Function<T>>>(_g);
363  assert(g);
364  xtl::span<const T> values = g->x()->array();
365  assert(x.size() <= x0.size());
366  auto dofs1_g = _dofs1_g.empty() ? xtl::span(_dofs0) : xtl::span(_dofs1_g);
367  std::int32_t x_size = x.size();
368  for (std::size_t i = 0; i < _dofs0.size(); ++i)
369  {
370  if (_dofs0[i] < x_size)
371  {
372  assert(dofs1_g[i] < (std::int32_t)values.size());
373  x[_dofs0[i]] = scale * (values[dofs1_g[i]] - x0[_dofs0[i]]);
374  }
375  }
376  }
377  else if (std::holds_alternative<std::shared_ptr<const Constant<T>>>(_g))
378  {
379  auto g = std::get<std::shared_ptr<const Constant<T>>>(_g);
380  const std::vector<T>& value = g->value;
381  std::int32_t bs = _function_space->dofmap()->bs();
382  std::for_each(_dofs0.cbegin(), _dofs0.cend(),
383  [&x, &x0, &value, scale, bs](auto dof)
384  {
385  if (dof < (std::int32_t)x.size())
386  x[dof] = scale * (value[dof % bs] - x0[dof]);
387  });
388  }
389  }
