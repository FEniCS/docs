142  template <typename U, typename = std::enable_if_t<std::is_same_v<
143  std::decay_t<U>, std::vector<std::int32_t>>>>
144  DirichletBC(const std::variant<std::shared_ptr<const Function<T>>,
145  std::shared_ptr<const Constant<T>>>& g,
146  U&& dofs, const std::shared_ptr<const FunctionSpace>& V = nullptr)
147  : _function_space(V), _g(g), _dofs0(std::forward<U>(dofs))
148  {
149  if (auto _g = std::get_if<std::shared_ptr<const Function<T>>>(&g); _g)
150  {
151  if (V)
152  {
153  throw std::runtime_error("Function space argument not expected for "
154  "DirichletBC when using a Function.");
155  }
156  assert(*_g);
157  _function_space = (*_g)->function_space();
158  }
159 
160  if (auto _g = std::get_if<std::shared_ptr<const Constant<T>>>(&g); _g)
161  {
162  if (!V)
163  {
164  throw std::runtime_error("Creating a DirichletBC with a Constant "
165  "requires a FunctionSpace argument");
166  }
167 
168  if (!V->element()->interpolation_ident())
169  {
170  throw std::runtime_error(
171  "Constant can be used only with point-evaluation elements");
172  }
173  }
174 
175  // Compute number of owned dofs indices in the full space (will
176  // contain 'gaps' for sub-spaces)
177  assert(_function_space);
178  const int map0_bs = _function_space->dofmap()->index_map_bs();
179  const int map0_size = _function_space->dofmap()->index_map->size_local();
180  const int owned_size0 = map0_bs * map0_size;
181 
182  // Find number of owned indices in _dofs0
183  auto it0 = std::lower_bound(_dofs0.begin(), _dofs0.end(), owned_size0);
184  _owned_indices0 = std::distance(_dofs0.begin(), it0);
185 
186  // Unroll _dofs0 for dofmap block size > 1
187  if (const int bs = _function_space->dofmap()->bs(); bs > 1)
188  {
189  _owned_indices0 *= bs;
190  const std::vector<std::int32_t> dof_tmp = _dofs0;
191  _dofs0.resize(bs * dof_tmp.size());
192  for (std::size_t i = 0; i < dof_tmp.size(); ++i)
193  {
194  for (int k = 0; k < bs; ++k)
195  _dofs0[bs * i + k] = bs * dof_tmp[i] + k;
196  }
197  }
198  }

291  void set(xtl::span<T> x, double scale = 1.0) const
292  {
293  if (std::holds_alternative<std::shared_ptr<const Function<T>>>(_g))
294  {
295  auto g = std::get<std::shared_ptr<const Function<T>>>(_g);
296  assert(g);
297  xtl::span<const T> values = g->x()->array();
298  auto dofs1_g = _dofs1_g.empty() ? xtl::span(_dofs0) : xtl::span(_dofs1_g);
299  std::int32_t x_size = x.size();
300  for (std::size_t i = 0; i < _dofs0.size(); ++i)
301  {
302  if (_dofs0[i] < x_size)
303  {
304  assert(dofs1_g[i] < (std::int32_t)values.size());
305  x[_dofs0[i]] = scale * values[dofs1_g[i]];
306  }
307  }
308  }
309  else if (std::holds_alternative<std::shared_ptr<const Constant<T>>>(_g))
310  {
311  auto g = std::get<std::shared_ptr<const Constant<T>>>(_g);
312  std::vector<T> value = g->value;
313  int bs = _function_space->dofmap()->bs();
314  std::int32_t x_size = x.size();
315  std::for_each(_dofs0.cbegin(), _dofs0.cend(),
316  [x_size, bs, scale, &value, &x](auto dof)
317  {
318  if (dof < x_size)
319  x[dof] = scale * value[dof % bs];
320  });
321  }
322  }

328  void set(xtl::span<T> x, const xtl::span<const T>& x0,
329  double scale = 1.0) const
330  {
331  if (std::holds_alternative<std::shared_ptr<const Function<T>>>(_g))
332  {
333  auto g = std::get<std::shared_ptr<const Function<T>>>(_g);
334  assert(g);
335  xtl::span<const T> values = g->x()->array();
336  assert(x.size() <= x0.size());
337  auto dofs1_g = _dofs1_g.empty() ? xtl::span(_dofs0) : xtl::span(_dofs1_g);
338  std::int32_t x_size = x.size();
339  for (std::size_t i = 0; i < _dofs0.size(); ++i)
340  {
341  if (_dofs0[i] < x_size)
342  {
343  assert(dofs1_g[i] < (std::int32_t)values.size());
344  x[_dofs0[i]] = scale * (values[dofs1_g[i]] - x0[_dofs0[i]]);
345  }
346  }
347  }
348  else if (std::holds_alternative<std::shared_ptr<const Constant<T>>>(_g))
349  {
350  auto g = std::get<std::shared_ptr<const Constant<T>>>(_g);
351  std::vector<T> value = g->value;
352  std::int32_t bs = _function_space->dofmap()->bs();
353  std::for_each(_dofs0.cbegin(), _dofs0.cend(),
354  [&x, &x0, &value, scale, bs](auto dof)
355  {
356  if (dof < (std::int32_t)x.size())
357  x[dof] = scale * (value[dof % bs] - x0[dof]);
358  });
359  }
