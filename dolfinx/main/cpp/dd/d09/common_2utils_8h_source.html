27 template <typename U, typename V>
28 std::pair<U, V> sort_unique(const U& indices, const V& values)
29 {
30  if (indices.size() != values.size())
31  throw std::runtime_error("Cannot sort two arrays of different lengths");
32 
33  std::vector<std::pair<typename U::value_type, typename V::value_type>> data(
34  indices.size());
35  for (std::size_t i = 0; i < indices.size(); ++i)
36  data[i] = {indices[i], values[i]};
37 
38  // Sort make unique
39  std::sort(data.begin(), data.end());
40  auto it = std::unique(data.begin(), data.end(),
41  [](auto& a, auto& b) { return a.first == b.first; });
42 
43  U indices_new;
44  V values_new;
45  indices_new.reserve(data.size());
46  values_new.reserve(data.size());
47  for (auto d = data.begin(); d != it; ++d)
48  {
49  indices_new.push_back(d->first);
50  values_new.push_back(d->second);
51  }
52 
53  return {std::move(indices_new), std::move(values_new)};
54 }

88 template <class T>
89 std::int64_t hash_global(const MPI_Comm comm, const T& x)
90 {
91  // Compute local hash
92  int64_t local_hash = hash_local(x);
93 
94  // Gather hash keys on root process
95  std::vector<int64_t> all_hashes(dolfinx::MPI::size(comm));
96  MPI_Gather(&local_hash, 1, MPI_INT64_T, all_hashes.data(), 1, MPI_INT64_T, 0,
97  comm);
98 
99  // Hash the received hash keys
100  boost::hash<std::vector<int64_t>> hash;
101  int64_t global_hash = hash(all_hashes);
102 
103  // Broadcast hash key to all processes
104  MPI_Bcast(&global_hash, 1, MPI_INT64_T, 0, comm);
105 
106  return global_hash;
107 }
