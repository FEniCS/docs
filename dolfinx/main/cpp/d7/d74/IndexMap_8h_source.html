241  template <typename T>
242  void scatter_fwd_begin(const xtl::span<const T>& send_buffer,
243  MPI_Datatype& data_type, MPI_Request& request,
244  const xtl::span<T>& recv_buffer) const
245  {
246  // Send displacement
247  const std::vector<int32_t>& displs_send_fwd = _shared_indices->offsets();
248 
249  // Return early if there are no incoming or outgoing edges
250  if (_displs_recv_fwd.size() == 1 and displs_send_fwd.size() == 1)
251  return;
252 
253  // Get block size
254  int n;
255  MPI_Type_size(data_type, &n);
256  n /= sizeof(T);
257  if (static_cast<int>(send_buffer.size()) != n * displs_send_fwd.back())
258  throw std::runtime_error("Incompatible send buffer size.");
259  if (static_cast<int>(recv_buffer.size()) != n * _displs_recv_fwd.back())
260  throw std::runtime_error("Incompatible receive buffer size..");
261 
262  // Start send/receive
263  MPI_Ineighbor_alltoallv(send_buffer.data(), _sizes_send_fwd.data(),
264  displs_send_fwd.data(), data_type,
265  recv_buffer.data(), _sizes_recv_fwd.data(),
266  _displs_recv_fwd.data(), data_type,
267  _comm_owner_to_ghost.comm(), &request);
268  }

278  // Return early if there are no incoming or outgoing edges

280  if (_displs_recv_fwd.size() == 1 and displs_send_fwd.size() == 1)

283  // Wait for communication to complete

297  template <typename T>
298  void scatter_fwd(const xtl::span<const T>& local_data,
299  xtl::span<T> remote_data, int n) const
300  {
301  MPI_Datatype data_type;
302  if (n == 1)
303  data_type = dolfinx::MPI::mpi_type<T>();
304  else
305  {
306  MPI_Type_contiguous(n, dolfinx::MPI::mpi_type<T>(), &data_type);
307  MPI_Type_commit(&data_type);
308  }
309 
310  const std::vector<std::int32_t>& indices = _shared_indices->array();
311  std::vector<T> send_buffer(n * indices.size());
312  for (std::size_t i = 0; i < indices.size(); ++i)
313  {
314  std::copy_n(std::next(local_data.cbegin(), n * indices[i]), n,
315  std::next(send_buffer.begin(), n * i));
316  }
317 
318  MPI_Request request;
319  std::vector<T> buffer_recv(n * _displs_recv_fwd.back());
320  scatter_fwd_begin(xtl::span<const T>(send_buffer), data_type, request,
321  xtl::span<T>(buffer_recv));
322  scatter_fwd_end(request);
323 
324  // Copy into ghost area ("remote_data")
325  assert(remote_data.size() == n * _ghost_pos_recv_fwd.size());
326  for (std::size_t i = 0; i < _ghost_pos_recv_fwd.size(); ++i)
327  {
328  std::copy_n(std::next(buffer_recv.cbegin(), n * _ghost_pos_recv_fwd[i]),
329  n, std::next(remote_data.begin(), n * i));
330  }
331 
332  if (n != 1)
333  MPI_Type_free(&data_type);
334  }

411  template <typename T>
412  void scatter_rev(xtl::span<T> local_data,
413  const xtl::span<const T>& remote_data, int n,
414  IndexMap::Mode op) const
415  {
416  MPI_Datatype data_type;
417  if (n == 1)
418  data_type = MPI::mpi_type<T>();
419  else
420  {
421  MPI_Type_contiguous(n, dolfinx::MPI::mpi_type<T>(), &data_type);
422  MPI_Type_commit(&data_type);
423  }
424 
425  // Pack send buffer
426  std::vector<T> buffer_send;
427  buffer_send.resize(n * _displs_recv_fwd.back());
428  for (std::size_t i = 0; i < _ghost_pos_recv_fwd.size(); ++i)
429  {
430  std::copy_n(std::next(remote_data.cbegin(), n * i), n,
431  std::next(buffer_send.begin(), n * _ghost_pos_recv_fwd[i]));
432  }
433 
434  // Exchange data
435  MPI_Request request;
436  std::vector<T> buffer_recv(n * _shared_indices->array().size());
437  scatter_rev_begin(xtl::span<const T>(buffer_send), data_type, request,
438  xtl::span<T>(buffer_recv));
439  scatter_rev_end(request);
440 
441  // Copy or accumulate into "local_data"
442  assert(local_data.size() == n * this->size_local());
443  const std::vector<std::int32_t>& shared_indices = _shared_indices->array();
444  switch (op)
445  {
446  case Mode::insert:
447  for (std::size_t i = 0; i < shared_indices.size(); ++i)
448  {
449  std::copy_n(std::next(buffer_recv.cbegin(), n * i), n,
450  std::next(local_data.begin(), n * shared_indices[i]));
451  }
452  break;
453  case Mode::add:
454  for (std::size_t i = 0; i < shared_indices.size(); ++i)
455  {
456  for (int j = 0; j < n; ++j)
457  local_data[shared_indices[i] * n + j] += buffer_recv[i * n + j];
458  }
459  break;
460  }
461 
462  if (n != 1)
463  MPI_Type_free(&data_type);
464  }
