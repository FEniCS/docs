154 template <typename T>
155 Form<T> create_form(
156  const ufc_form& ufc_form,
157  const std::vector<std::shared_ptr<const fem::FunctionSpace>>& spaces,
158  const std::vector<std::shared_ptr<const fem::Function<T>>>& coefficients,
159  const std::vector<std::shared_ptr<const fem::Constant<T>>>& constants,
160  const std::map<IntegralType, const mesh::MeshTags<int>*>& subdomains,
161  const std::shared_ptr<const mesh::Mesh>& mesh = nullptr)
162 {
163  if (ufc_form.rank != (int)spaces.size())
164  throw std::runtime_error("Wrong number of argument spaces for Form.");
165  if (ufc_form.num_coefficients != (int)coefficients.size())
166  {
167  throw std::runtime_error(
168  "Mismatch between number of expected and provided Form coefficients.");
169  }
170  if (ufc_form.num_constants != (int)constants.size())
171  {
172  throw std::runtime_error(
173  "Mismatch between number of expected and provided Form constants.");
174  }
175 
176  // Check argument function spaces
177 #ifdef DEBUG
178  for (std::size_t i = 0; i < spaces.size(); ++i)
179  {
180  assert(spaces[i]->element());
181  ufc_finite_element* ufc_element = ufc_form.finite_elements[i];
182  assert(ufc_element);
183  if (std::string(ufc_element->signature)
184  != spaces[i]->element()->signature())
185  {
186  throw std::runtime_error(
187  "Cannot create form. Wrong type of function space for argument.");
188  }
189  }
190 #endif
191 
192  // Get list of integral IDs, and load tabulate tensor into memory for
193  // each
194  using kern = std::function<void(T*, const T*, const T*, const double*,
195  const int*, const std::uint8_t*)>;
196  std::map<IntegralType, std::pair<std::vector<std::pair<int, kern>>,
197  const mesh::MeshTags<int>*>>
198  integral_data;
199 
200  bool needs_facet_permutations = false;
201 
202  // Attach cell kernels
203  std::vector<int> cell_integral_ids(ufc_form.integral_ids(cell),
204  ufc_form.integral_ids(cell)
205  + ufc_form.num_integrals(cell));
206  for (int i = 0; i < ufc_form.num_integrals(cell); ++i)
207  {
208  ufc_integral* integral = ufc_form.integrals(cell)[i];
209  assert(integral);
210  integral_data[IntegralType::cell].first.emplace_back(
211  cell_integral_ids[i], integral->tabulate_tensor);
212  }
213 
214  // Attach cell subdomain data
215  if (auto it = subdomains.find(IntegralType::cell);
216  it != subdomains.end() and !cell_integral_ids.empty())
217  {
218  integral_data[IntegralType::cell].second = it->second;
219  }
220 
221  // FIXME: Can facets be handled better?
222 
223  // Create facets, if required
224  if (ufc_form.num_integrals(exterior_facet) > 0
225  or ufc_form.num_integrals(interior_facet) > 0)
226  {
227  needs_facet_permutations = true;
228  if (!spaces.empty())
229  {
230  auto mesh = spaces[0]->mesh();
231  const int tdim = mesh->topology().dim();
232  spaces[0]->mesh()->topology_mutable().create_entities(tdim - 1);
233  }
234  }
235 
236  // Attach exterior facet kernels
237  std::vector<int> exterior_facet_integral_ids(
238  ufc_form.integral_ids(exterior_facet),
239  ufc_form.integral_ids(exterior_facet)
240  + ufc_form.num_integrals(exterior_facet));
241  for (int i = 0; i < ufc_form.num_integrals(exterior_facet); ++i)
242  {
243  ufc_integral* integral = ufc_form.integrals(exterior_facet)[i];
244  assert(integral);
245  integral_data[IntegralType::exterior_facet].first.emplace_back(
246  exterior_facet_integral_ids[i], integral->tabulate_tensor);
247  }
248 
249  // Attach exterior facet subdomain data
250  if (auto it = subdomains.find(IntegralType::exterior_facet);
251  it != subdomains.end() and !exterior_facet_integral_ids.empty())
252  {
253  integral_data[IntegralType::exterior_facet].second = it->second;
254  }
255 
256  // Attach interior facet kernels
257  std::vector<int> interior_facet_integral_ids(
258  ufc_form.integral_ids(interior_facet),
259  ufc_form.integral_ids(interior_facet)
260  + ufc_form.num_integrals(interior_facet));
261  for (int i = 0; i < ufc_form.num_integrals(interior_facet); ++i)
262  {
263  ufc_integral* integral = ufc_form.integrals(interior_facet)[i];
264  assert(integral);
265  integral_data[IntegralType::interior_facet].first.emplace_back(
266  interior_facet_integral_ids[i], integral->tabulate_tensor);
267  }
268 
269  // Attach interior facet subdomain data
270  if (auto it = subdomains.find(IntegralType::interior_facet);
271  it != subdomains.end() and !interior_facet_integral_ids.empty())
272  {
273  integral_data[IntegralType::interior_facet].second = it->second;
274  }
275 
276  return fem::Form(spaces, integral_data, coefficients, constants,
277  needs_facet_permutations, mesh);
278 }

289 template <typename T>
290 Form<T> create_form(
291  const ufc_form& ufc_form,
292  const std::vector<std::shared_ptr<const fem::FunctionSpace>>& spaces,
293  const std::map<std::string, std::shared_ptr<const fem::Function<T>>>&
294  coefficients,
295  const std::map<std::string, std::shared_ptr<const fem::Constant<T>>>&
296  constants,
297  const std::map<IntegralType, const mesh::MeshTags<int>*>& subdomains,
298  const std::shared_ptr<const mesh::Mesh>& mesh = nullptr)
299 {
300  // Place coefficients in appropriate order
301  std::vector<std::shared_ptr<const fem::Function<T>>> coeff_map;
302  for (const std::string& name : get_coefficient_names(ufc_form))
303  {
304  if (auto it = coefficients.find(name); it != coefficients.end())
305  coeff_map.push_back(it->second);
306  else
307  {
308  throw std::runtime_error("Form coefficient \"" + name
309  + "\" not provided.");
310  }
311  }
312 
313  // Place constants in appropriate order
314  std::vector<std::shared_ptr<const fem::Constant<T>>> const_map;
315  for (const std::string& name : get_constant_names(ufc_form))
316  {
317  if (auto it = constants.find(name); it != constants.end())
318  const_map.push_back(it->second);
319  else
320  {
321  throw std::runtime_error("Form constant \"" + name + "\" not provided.");
322  }
323  }
324 
325  return create_form(ufc_form, spaces, coeff_map, const_map, subdomains, mesh);
326 }

377 // Pack a single coefficient
378 template <typename T, int _bs = -1>
379 void pack_coefficient(
380  array2d<T>& c, const std::vector<T>& v,
381  const xtl::span<const std::uint32_t>& cell_info, const fem::DofMap& dofmap,
382  std::int32_t num_cells, std::int32_t offset, int space_dim,
383  const std::function<void(const xtl::span<T>&,
384  const xtl::span<const std::uint32_t>&,
385  std::int32_t, int)>& transform)
386 {
387  const int bs = dofmap.bs();
388  assert(_bs < 0 or _bs == bs);
389  for (std::int32_t cell = 0; cell < num_cells; ++cell)
390  {
391  auto dofs = dofmap.cell_dofs(cell);
392  auto cell_coeff = c.row(cell).subspan(offset, space_dim);
393  for (std::size_t i = 0; i < dofs.size(); ++i)
394  {
395  if constexpr (_bs < 0)
396  {
397  const int pos_c = bs * i;
398  const int pos_v = bs * dofs[i];
399  for (int k = 0; k < bs; ++k)
400  cell_coeff[pos_c + k] = v[pos_v + k];
401  }
402  else
403  {
404  const int pos_c = _bs * i;
405  const int pos_v = _bs * dofs[i];
406  for (int k = 0; k < _bs; ++k)
407  cell_coeff[pos_c + k] = v[pos_v + k];
408  }
409  }
410  transform(cell_coeff, cell_info, cell, 1);
411  }
412 }

417 template <typename U>
418 array2d<typename U::scalar_type> pack_coefficients(const U& u)
419 {
420  using T = typename U::scalar_type;
421 
422  // Get form coefficient offsets and dofmaps
423  const std::vector<std::shared_ptr<const fem::Function<T>>> coefficients
424  = u.coefficients();
425  const std::vector<int> offsets = u.coefficient_offsets();
426  std::vector<const fem::DofMap*> dofmaps(coefficients.size());
427  std::vector<const fem::FiniteElement*> elements(coefficients.size());
428  std::vector<std::reference_wrapper<const std::vector<T>>> v;
429  v.reserve(coefficients.size());
430  for (std::size_t i = 0; i < coefficients.size(); ++i)
431  {
432  elements[i] = coefficients[i]->function_space()->element().get();
433  dofmaps[i] = coefficients[i]->function_space()->dofmap().get();
434  v.push_back(coefficients[i]->x()->array());
435  }
436 
437  // Get mesh
438  std::shared_ptr<const mesh::Mesh> mesh = u.mesh();
439  assert(mesh);
440  const int tdim = mesh->topology().dim();
441  const std::int32_t num_cells
442  = mesh->topology().index_map(tdim)->size_local()
443  + mesh->topology().index_map(tdim)->num_ghosts();
444 
445  // Copy data into coefficient array
446  array2d<T> c(num_cells, offsets.back());
447  if (!coefficients.empty())
448  {
449  bool needs_dof_transformations = false;
450  for (std::size_t coeff = 0; coeff < dofmaps.size(); ++coeff)
451  {
452  if (elements[coeff]->needs_dof_transformations())
453  {
454  needs_dof_transformations = true;
455  mesh->topology_mutable().create_entity_permutations();
456  }
457  }
458 
459  // Iterate over coefficients
460  xtl::span<const std::uint32_t> cell_info;
461  if (needs_dof_transformations)
462  cell_info = xtl::span(mesh->topology().get_cell_permutation_info());
463  for (std::size_t coeff = 0; coeff < dofmaps.size(); ++coeff)
464  {
465  const std::function<void(const xtl::span<T>&,
466  const xtl::span<const std::uint32_t>&,
467  std::int32_t, int)>
468  transformation
469  = elements[coeff]->get_dof_transformation_function<T>(false, true);
470  if (int bs = dofmaps[coeff]->bs(); bs == 1)
471  {
472  impl::pack_coefficient<T, 1>(
473  c, v[coeff], cell_info, *dofmaps[coeff], num_cells, offsets[coeff],
474  elements[coeff]->space_dimension(), transformation);
475  }
476  else if (bs == 2)
477  {
478  impl::pack_coefficient<T, 2>(
479  c, v[coeff], cell_info, *dofmaps[coeff], num_cells, offsets[coeff],
480  elements[coeff]->space_dimension(), transformation);
481  }
482  else if (bs == 3)
483  {
484  impl::pack_coefficient<T, 3>(
485  c, v[coeff], cell_info, *dofmaps[coeff], num_cells, offsets[coeff],
486  elements[coeff]->space_dimension(), transformation);
487  }
488  else
489  {
490  impl::pack_coefficient<T>(
491  c, v[coeff], cell_info, *dofmaps[coeff], num_cells, offsets[coeff],
492  elements[coeff]->space_dimension(), transformation);
493  }
494  }
495  }
496 
497  return c;
498 }

502 template <typename U>
503 std::vector<typename U::scalar_type> pack_constants(const U& u)
504 {
505  using T = typename U::scalar_type;
506  const std::vector<std::shared_ptr<const fem::Constant<T>>>& constants
507  = u.constants();
508 
509  // Calculate size of array needed to store packed constants
510  std::int32_t size = std::accumulate(constants.begin(), constants.end(), 0,
511  [](std::int32_t sum, const auto& constant)
512  { return sum + constant->value.size(); });
513 
514  // Pack constants
515  std::vector<T> constant_values(size);
516  std::int32_t offset = 0;
517  for (const auto& constant : constants)
518  {
519  const std::vector<T>& value = constant->value;
520  for (std::size_t i = 0; i < value.size(); ++i)
521  constant_values[offset + i] = value[i];
522  offset += value.size();
523  }
524 
525  return constant_values;
526 }
