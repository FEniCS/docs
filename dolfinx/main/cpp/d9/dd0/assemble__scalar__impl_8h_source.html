61 template <typename T>
62 T assemble_scalar(const fem::Form<T>& M, const xtl::span<const T>& constants,
63  const array2d<T>& coeffs)
64 {
65  std::shared_ptr<const mesh::Mesh> mesh = M.mesh();
66  assert(mesh);
67  const int tdim = mesh->topology().dim();
68  const std::int32_t num_cells
69  = mesh->topology().connectivity(tdim, 0)->num_nodes();
70 
71  const bool needs_permutation_data = M.needs_permutation_data();
72  if (needs_permutation_data)
73  mesh->topology_mutable().create_entity_permutations();
74  const std::vector<std::uint32_t>& cell_info
75  = needs_permutation_data ? mesh->topology().get_cell_permutation_info()
76  : std::vector<std::uint32_t>(num_cells);
77 
78  T value(0);
79  for (int i : M.integral_ids(IntegralType::cell))
80  {
81  const auto& fn = M.kernel(IntegralType::cell, i);
82  const std::vector<std::int32_t>& active_cells
83  = M.domains(IntegralType::cell, i);
84  value += impl::assemble_cells(mesh->geometry(), active_cells, fn, constants,
85  coeffs, cell_info);
86  }
87 
88  if (M.num_integrals(IntegralType::exterior_facet) > 0
89  or M.num_integrals(IntegralType::interior_facet) > 0)
90  {
91  // FIXME: cleanup these calls? Some of these happen internally again.
92  mesh->topology_mutable().create_entities(tdim - 1);
93  mesh->topology_mutable().create_connectivity(tdim - 1, tdim);
94  mesh->topology_mutable().create_entity_permutations();
95 
96  const std::vector<std::uint8_t>& perms
97  = mesh->topology().get_facet_permutations();
98 
99  for (int i : M.integral_ids(IntegralType::exterior_facet))
100  {
101  const auto& fn = M.kernel(IntegralType::exterior_facet, i);
102  const std::vector<std::int32_t>& active_facets
103  = M.domains(IntegralType::exterior_facet, i);
104  value += impl::assemble_exterior_facets(
105  *mesh, active_facets, fn, constants, coeffs, cell_info, perms);
106  }
107 
108  const std::vector<int> c_offsets = M.coefficient_offsets();
109  for (int i : M.integral_ids(IntegralType::interior_facet))
110  {
111  const auto& fn = M.kernel(IntegralType::interior_facet, i);
112  const std::vector<std::int32_t>& active_facets
113  = M.domains(IntegralType::interior_facet, i);
114  value += impl::assemble_interior_facets(*mesh, active_facets, fn,
115  constants, coeffs, c_offsets,
116  cell_info, perms);
117  }
118  }
119 
120  return value;
121 }

123 template <typename T>
124 T assemble_cells(
125  const mesh::Geometry& geometry,
126  const xtl::span<const std::int32_t>& active_cells,
127  const std::function<void(T*, const T*, const T*, const double*, const int*,
128  const std::uint8_t*, const std::uint32_t)>& fn,
129  const xtl::span<const T>& constants, const array2d<T>& coeffs,
130  const xtl::span<const std::uint32_t>& cell_info)
131 {
132  const std::size_t gdim = geometry.dim();
133 
134  // Prepare cell geometry
135  const graph::AdjacencyList<std::int32_t>& x_dofmap = geometry.dofmap();
136 
137  // FIXME: Add proper interface for num coordinate dofs
138  const std::size_t num_dofs_g = x_dofmap.num_links(0);
139  const xt::xtensor<double, 2>& x_g = geometry.x();
140 
141  // Create data structures used in assembly
142  std::vector<double> coordinate_dofs(num_dofs_g * gdim);
143 
144  // Iterate over all cells
145  T value(0);
146  for (std::int32_t c : active_cells)
147  {
148  // Get cell coordinates/geometry
149  auto x_dofs = x_dofmap.links(c);
150  for (std::size_t i = 0; i < x_dofs.size(); ++i)
151  {
152  std::copy_n(xt::row(x_g, x_dofs[i]).begin(), gdim,
153  std::next(coordinate_dofs.begin(), i * gdim));
154  }
155 
156  auto coeff_cell = coeffs.row(c);
157  fn(&value, coeff_cell.data(), constants.data(), coordinate_dofs.data(),
158  nullptr, nullptr, cell_info[c]);
159  }
160 
161  return value;
162 }

164 template <typename T>
165 T assemble_exterior_facets(
166  const mesh::Mesh& mesh, const xtl::span<const std::int32_t>& active_facets,
167  const std::function<void(T*, const T*, const T*, const double*, const int*,
168  const std::uint8_t*, const std::uint32_t)>& fn,
169  const xtl::span<const T>& constants, const array2d<T>& coeffs,
170  const xtl::span<const std::uint32_t>& cell_info,
171  const xtl::span<const std::uint8_t>& perms)
172 {
173  const std::size_t gdim = mesh.geometry().dim();
174  const int tdim = mesh.topology().dim();
175 
176  // Prepare cell geometry
177  const graph::AdjacencyList<std::int32_t>& x_dofmap = mesh.geometry().dofmap();
178 
179  // FIXME: Add proper interface for num coordinate dofs
180  const std::size_t num_dofs_g = x_dofmap.num_links(0);
181  const xt::xtensor<double, 2>& x_g = mesh.geometry().x();
182 
183  // Creat data structures used in assembly
184  std::vector<double> coordinate_dofs(num_dofs_g * gdim);
185 
186  auto f_to_c = mesh.topology().connectivity(tdim - 1, tdim);
187  assert(f_to_c);
188  auto c_to_f = mesh.topology().connectivity(tdim, tdim - 1);
189  assert(c_to_f);
190 
191  // Iterate over all facets
192  T value(0);
193  for (std::int32_t facet : active_facets)
194  {
195  // Create attached cell
196  assert(f_to_c->num_links(facet) == 1);
197  const int cell = f_to_c->links(facet)[0];
198 
199  // Get local index of facet with respect to the cell
200  auto facets = c_to_f->links(cell);
201  auto it = std::find(facets.begin(), facets.end(), facet);
202  assert(it != facets.end());
203  const int local_facet = std::distance(facets.data(), it);
204 
205  // Get cell coordinates/geometry
206  auto x_dofs = x_dofmap.links(cell);
207  for (std::size_t i = 0; i < x_dofs.size(); ++i)
208  {
209  std::copy_n(xt::row(x_g, x_dofs[i]).begin(), gdim,
210  std::next(coordinate_dofs.begin(), i * gdim));
211  }
212 
213  auto coeff_cell = coeffs.row(cell);
214  fn(&value, coeff_cell.data(), constants.data(), coordinate_dofs.data(),
215  &local_facet, &perms[cell * facets.size() + local_facet],
216  cell_info[cell]);
217  }
218 
219  return value;
220 }

222 template <typename T>
223 T assemble_interior_facets(
224  const mesh::Mesh& mesh, const xtl::span<const std::int32_t>& active_facets,
225  const std::function<void(T*, const T*, const T*, const double*, const int*,
226  const std::uint8_t*, const std::uint32_t)>& fn,
227  const xtl::span<const T>& constants, const array2d<T>& coeffs,
228  const xtl::span<const int>& offsets,
229  const xtl::span<const std::uint32_t>& cell_info,
230  const xtl::span<const std::uint8_t>& perms)
231 {
232  const std::size_t gdim = mesh.geometry().dim();
233  const int tdim = mesh.topology().dim();
234 
235  // Prepare cell geometry
236  const graph::AdjacencyList<std::int32_t>& x_dofmap = mesh.geometry().dofmap();
237 
238  // FIXME: Add proper interface for num coordinate dofs
239  const std::size_t num_dofs_g = x_dofmap.num_links(0);
240  const xt::xtensor<double, 2>& x_g = mesh.geometry().x();
241 
242  // Creat data structures used in assembly
243  xt::xtensor<double, 3> coordinate_dofs({2, num_dofs_g, gdim});
244  std::vector<T> coeff_array(2 * offsets.back());
245  assert(offsets.back() == coeffs.shape[1]);
246 
247  auto f_to_c = mesh.topology().connectivity(tdim - 1, tdim);
248  assert(f_to_c);
249  auto c_to_f = mesh.topology().connectivity(tdim, tdim - 1);
250  assert(c_to_f);
251 
252  // Iterate over all facets
253  T value = 0;
254  for (std::int32_t f : active_facets)
255  {
256  // Create attached cell
257  auto cells = f_to_c->links(f);
258  assert(cells.size() == 2);
259 
260  const int facets_per_cell = c_to_f->num_links(cells[0]);
261 
262  // Get local index of facet with respect to the cell
263  std::array<int, 2> local_facet;
264  for (int i = 0; i < 2; ++i)
265  {
266  auto facets = c_to_f->links(cells[i]);
267  auto it = std::find(facets.begin(), facets.end(), f);
268  assert(it != facets.end());
269  local_facet[i] = std::distance(facets.begin(), it);
270  }
271 
272  // Get cell geometry
273  auto x_dofs0 = x_dofmap.links(cells[0]);
274  for (std::size_t i = 0; i < x_dofs0.size(); ++i)
275  {
276  std::copy_n(xt::view(x_g, x_dofs0[i]).begin(), gdim,
277  xt::view(coordinate_dofs, 0, i, xt::all()).begin());
278  }
279  auto x_dofs1 = x_dofmap.links(cells[1]);
280  for (std::size_t i = 0; i < x_dofs1.size(); ++i)
281  {
282  std::copy_n(xt::view(x_g, x_dofs1[i]).begin(), gdim,
283  xt::view(coordinate_dofs, 1, i, xt::all()).begin());
284  }
285 
286  // Layout for the restricted coefficients is flattened
287  // w[coefficient][restriction][dof]
288  auto coeff_cell0 = coeffs.row(cells[0]);
289  auto coeff_cell1 = coeffs.row(cells[1]);
290 
291  // Loop over coefficients
292  for (std::size_t i = 0; i < offsets.size() - 1; ++i)
293  {
294  // Loop over entries for coefficient i
295  const int num_entries = offsets[i + 1] - offsets[i];
296  std::copy_n(coeff_cell0.data() + offsets[i], num_entries,
297  std::next(coeff_array.begin(), 2 * offsets[i]));
298  std::copy_n(coeff_cell1.data() + offsets[i], num_entries,
299  std::next(coeff_array.begin(), offsets[i + 1] + offsets[i]));
300  }
301 
302  const std::array perm{perms[cells[0] * facets_per_cell + local_facet[0]],
303  perms[cells[1] * facets_per_cell + local_facet[1]]};
304  fn(&value, coeff_array.data(), constants.data(), coordinate_dofs.data(),
305  local_facet.data(), perm.data(), cell_info[cells[0]]);
306  }
307 
308  return value;
309 }
