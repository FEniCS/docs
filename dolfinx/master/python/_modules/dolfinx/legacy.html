[docs]def FunctionSpace(mesh: _cpp.mesh.Mesh, element, degree=None):
    """Create a FunctionSpace from a mesh and an element"""
    if degree:
        return fem.FunctionSpace(mesh, (element, degree))
    else:
        return fem.FunctionSpace(mesh, element, degree)
