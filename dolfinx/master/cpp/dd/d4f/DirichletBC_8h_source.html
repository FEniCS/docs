136 public:
150  DirichletBC(
151  const std::shared_ptr<const fem::Function<T>>& g,
152  const Eigen::Ref<const Eigen::Array<std::int32_t, Eigen::Dynamic, 1>>&
153  dofs)
154  : _function_space(g->function_space()), _g(g), _dofs0(dofs),
155  _dofs1_g(dofs)
156  {
157  const int bs = _function_space->dofmap()->bs();
158  if (bs > 1)
159  {
160  // Unroll for the block size
161  _dofs0.resize(bs * dofs.rows());
162  for (Eigen::Index i = 0; i < dofs.rows(); ++i)
163  {
164  for (int k = 0; k < bs; ++k)
165  _dofs0[bs * i + k] = bs * dofs[i] + k;
166  }
167  _dofs1_g = _dofs0;
168  }
169 
170  // TODO: allows single dofs array (let one point to the other)
171  const int owned_size0 = _function_space->dofmap()->index_map->size_local();
172  const int map0_bs = _function_space->dofmap()->index_map_bs();
173  auto* it = std::lower_bound(_dofs0.data(), _dofs0.data() + _dofs0.rows(),
174  map0_bs * owned_size0);
175  _owned_indices0 = std::distance(_dofs0.data(), it);
176  }
177 
195  DirichletBC(const std::shared_ptr<const fem::Function<T>>& g,
196  const std::array<Eigen::Array<std::int32_t, Eigen::Dynamic, 1>,
197  2>& V_g_dofs,
198  std::shared_ptr<const fem::FunctionSpace> V)
199  : _function_space(V), _g(g), _dofs0(V_g_dofs[0]), _dofs1_g(V_g_dofs[1])
200  {
201  assert(_dofs0.rows() == _dofs1_g.rows());
202  assert(_function_space);
203  assert(_g);
204 
205  const int map0_bs = _function_space->dofmap()->index_map_bs();
206  const int map0_size = _function_space->dofmap()->index_map->size_local();
207  const int owned_size0 = (map0_bs * map0_size);
208  auto it0 = std::lower_bound(_dofs0.data(), _dofs0.data() + _dofs0.rows(),
209  owned_size0);
210  _owned_indices0 = std::distance(_dofs0.data(), it0);
211  }

267  void set(Eigen::Ref<Eigen::Matrix<T, Eigen::Dynamic, 1>> x,
268  double scale = 1.0) const

286  void set(Eigen::Ref<Eigen::Matrix<T, Eigen::Dynamic, 1>> x,
287  const Eigen::Ref<const Eigen::Matrix<T, Eigen::Dynamic, 1>>& x0,
288  double scale = 1.0) const
289  {
290  assert(_g);
291  auto& g = _g->x()->array();
292  assert(x.rows() <= x0.rows());
293  for (Eigen::Index i = 0; i < _dofs0.rows(); ++i)
294  {
295  if (_dofs0(i) < x.rows())
296  {
297  assert(_dofs1_g(i) < g.rows());
298  x[_dofs0(i)] = scale * (g[_dofs1_g(i)] - x0[_dofs0(i)]);
299  }
300  }
301  }
