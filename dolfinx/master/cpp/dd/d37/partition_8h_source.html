185 template <typename T>
186 Eigen::Array<T, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor>
187 build::distribute_data(
188  MPI_Comm comm, const std::vector<std::int64_t>& indices,
189  const Eigen::Ref<const Eigen::Array<T, Eigen::Dynamic, Eigen::Dynamic,
190  Eigen::RowMajor>>& x)
191 {
192  common::Timer timer("Fetch float data from remote processes");
193 
194  const std::int64_t num_points_local = x.rows();
195  const int size = dolfinx::MPI::size(comm);
196  const int rank = dolfinx::MPI::rank(comm);
197  std::vector<std::int64_t> global_sizes(size);
198  MPI_Allgather(&num_points_local, 1, MPI_INT64_T, global_sizes.data(), 1,
199  MPI_INT64_T, comm);
200  std::vector<std::int64_t> global_offsets(size + 1, 0);
201  std::partial_sum(global_sizes.begin(), global_sizes.end(),
202  global_offsets.begin() + 1);
203 
204  // Build index data requests
205  std::vector<int> number_index_send(size, 0);
206  std::vector<int> index_owner(indices.size());
207  std::vector<int> index_order(indices.size());
208  std::iota(index_order.begin(), index_order.end(), 0);
209  std::sort(index_order.begin(), index_order.end(),
210  [&indices](int a, int b) { return (indices[a] < indices[b]); });
211 
212  int p = 0;
213  for (std::size_t i = 0; i < index_order.size(); ++i)
214  {
215  int j = index_order[i];
216  while (indices[j] >= global_offsets[p + 1])
217  ++p;
218  index_owner[j] = p;
219  number_index_send[p]++;
220  }
221 
222  // Compute send displacements
223  std::vector<int> disp_index_send(size + 1, 0);
224  std::partial_sum(number_index_send.begin(), number_index_send.end(),
225  disp_index_send.begin() + 1);
226 
227  // Pack global index send data
228  std::vector<std::int64_t> indices_send(disp_index_send.back());
229  std::vector<int> disp_tmp = disp_index_send;
230  for (std::size_t i = 0; i < indices.size(); ++i)
231  {
232  const int owner = index_owner[i];
233  indices_send[disp_tmp[owner]++] = indices[i];
234  }
235 
236  // Send/receive number of indices to communicate to each process
237  std::vector<int> number_index_recv(size);
238  MPI_Alltoall(number_index_send.data(), 1, MPI_INT, number_index_recv.data(),
239  1, MPI_INT, comm);
240 
241  // Compute receive displacements
242  std::vector<int> disp_index_recv(size + 1, 0);
243  std::partial_sum(number_index_recv.begin(), number_index_recv.end(),
244  disp_index_recv.begin() + 1);
245 
246  // Send/receive global indices
247  std::vector<std::int64_t> indices_recv(disp_index_recv.back());
248  MPI_Alltoallv(indices_send.data(), number_index_send.data(),
249  disp_index_send.data(), MPI_INT64_T, indices_recv.data(),
250  number_index_recv.data(), disp_index_recv.data(), MPI_INT64_T,
251  comm);
252 
253  const int item_size = x.cols();
254  assert(item_size != 0);
255  // Pack point data to send back (transpose)
256  Eigen::Array<T, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor> x_return(
257  indices_recv.size(), item_size);
258  for (int p = 0; p < size; ++p)
259  {
260  for (int i = disp_index_recv[p]; i < disp_index_recv[p + 1]; ++i)
261  {
262  const std::int32_t index_local = indices_recv[i] - global_offsets[rank];
263  assert(index_local >= 0);
264  x_return.row(i) = x.row(index_local);
265  }
266  }
267 
268  MPI_Datatype compound_type;
269  MPI_Type_contiguous(item_size, dolfinx::MPI::mpi_type<T>(), &compound_type);
270  MPI_Type_commit(&compound_type);
271 
272  // Send back point data
273  Eigen::Array<T, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor> my_x(
274  disp_index_send.back(), item_size);
275  MPI_Alltoallv(x_return.data(), number_index_recv.data(),
276  disp_index_recv.data(), compound_type, my_x.data(),
277  number_index_send.data(), disp_index_send.data(), compound_type,
278  comm);
279 
280  return my_x;
281 }
