125  Function collapse() const
126  {
127  // Create new collapsed FunctionSpace
128  const auto [function_space_new, collapsed_map]
129  = _function_space->collapse();
130 
131  // Create new vector
132  assert(function_space_new);
133  auto vector_new = std::make_shared<la::Vector<T>>(
134  function_space_new->dofmap()->index_map,
135  function_space_new->dofmap()->index_map_bs());
136 
137  // Copy values into new vector
138  const Eigen::Matrix<T, Eigen::Dynamic, 1>& x_old = _x->array();
139  Eigen::Matrix<T, Eigen::Dynamic, 1>& x_new = vector_new->array();
140  for (std::size_t i = 0; i < collapsed_map.size(); ++i)
141  {
142  assert((int)i < x_new.size());
143  assert(collapsed_map[i] < x_old.size());
144  x_new[i] = x_old[collapsed_map[i]];
145  }
146 
147  return Function(function_space_new, vector_new);
148  }

222  void
223  eval(const Eigen::Ref<
224  const Eigen::Array<double, Eigen::Dynamic, 3, Eigen::RowMajor>>& x,
225  const Eigen::Ref<const Eigen::Array<int, Eigen::Dynamic, 1>>& cells,
226  Eigen::Ref<
227  Eigen::Array<T, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor>>
228  u) const
229  {
230  // TODO: This could be easily made more efficient by exploiting points
231  // being ordered by the cell to which they belong.
232 
233  if (x.rows() != cells.rows())
234  {
235  throw std::runtime_error(
236  "Number of points and number of cells must be equal.");
237  }
238  if (x.rows() != u.rows())
239  {
240  throw std::runtime_error(
241  "Length of array for Function values must be the "
242  "same as the number of points.");
243  }
244 
245  // Get mesh
246  assert(_function_space);
247  std::shared_ptr<const mesh::Mesh> mesh = _function_space->mesh();
248  assert(mesh);
249  const int gdim = mesh->geometry().dim();
250  const int tdim = mesh->topology().dim();
251 
252  // Get geometry data
253  const graph::AdjacencyList<std::int32_t>& x_dofmap
254  = mesh->geometry().dofmap();
255 
256  // FIXME: Add proper interface for num coordinate dofs
257  const int num_dofs_g = x_dofmap.num_links(0);
258  const Eigen::Array<double, Eigen::Dynamic, 3, Eigen::RowMajor>& x_g
259  = mesh->geometry().x();
260  Eigen::Array<double, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor>
261  coordinate_dofs(num_dofs_g, gdim);
262 
263  // Get coordinate map
264  const fem::CoordinateElement& cmap = mesh->geometry().cmap();
265 
266  // Get element
267  assert(_function_space->element());
268  std::shared_ptr<const fem::FiniteElement> element
269  = _function_space->element();
270  assert(element);
271  const int bs_element = element->block_size();
272  const int reference_value_size
273  = element->reference_value_size() / bs_element;
274  const int value_size = element->value_size() / bs_element;
275  const int space_dimension = element->space_dimension() / bs_element;
276 
277  // If the space has sub elements, concatenate the evaluations on the sub
278  // elements
279  const int num_sub_elements = element->num_sub_elements();
280  if (num_sub_elements > 1 and num_sub_elements != bs_element)
281  {
282  if (bs_element != 1)
283  {
284  throw std::runtime_error(
285  "Blocked elements of mixed spaces are not yet supported.");
286  }
287  int offset = 0;
288  for (int sub_e = 0; sub_e < num_sub_elements; ++sub_e)
289  {
290  std::shared_ptr<const fem::FiniteElement> sub_element
291  = element->extract_sub_element({sub_e});
292 
293  const int sub_value_size = sub_element->value_size();
294  const Function sub_f = this->sub(sub_e);
295  Eigen::Array<T, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor> sub_u(
296  u.rows(), sub_value_size);
297  sub_f.eval(x, cells, sub_u);
298 
299  for (int i = 0; i < sub_value_size; ++i)
300  u.col(offset + i) = sub_u.col(i);
301  offset += sub_value_size;
302  }
303 
304  return;
305  }
306 
307  // Prepare geometry data structures
308  Eigen::Tensor<double, 3, Eigen::RowMajor> J(1, gdim, tdim);
309  Eigen::Array<double, Eigen::Dynamic, 1> detJ(1);
310  Eigen::Tensor<double, 3, Eigen::RowMajor> K(1, tdim, gdim);
311  Eigen::Array<double, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor> X(
312  1, tdim);
313 
314  // Prepare basis function data structures
315  Eigen::Tensor<double, 3, Eigen::RowMajor> basis_reference_values(
316  1, space_dimension, reference_value_size);
317  Eigen::Tensor<double, 3, Eigen::RowMajor> basis_values(1, space_dimension,
318  value_size);
319 
320  // Create work vector for expansion coefficients
321  Eigen::Matrix<T, 1, Eigen::Dynamic> coefficients(space_dimension
322  * bs_element);
323 
324  // Get dofmap
325  std::shared_ptr<const fem::DofMap> dofmap = _function_space->dofmap();
326  assert(dofmap);
327  const int bs_dof = dofmap->bs();
328 
329  mesh->topology_mutable().create_entity_permutations();
330  const Eigen::Array<std::uint32_t, Eigen::Dynamic, 1>& cell_info
331  = mesh->topology().get_cell_permutation_info();
332 
333  // Loop over points
334  u.setZero();
335  const Eigen::Matrix<T, Eigen::Dynamic, 1>& _v = _x->array();
336  for (Eigen::Index p = 0; p < cells.rows(); ++p)
337  {
338  const int cell_index = cells(p);
339 
340  // Skip negative cell indices
341  if (cell_index < 0)
342  continue;
343 
344  // Get cell geometry (coordinate dofs)
345  auto x_dofs = x_dofmap.links(cell_index);
346  for (int i = 0; i < num_dofs_g; ++i)
347  coordinate_dofs.row(i) = x_g.row(x_dofs[i]).head(gdim);
348 
349  // Compute reference coordinates X, and J, detJ and K
350  cmap.compute_reference_geometry(X, J, detJ, K, x.row(p).head(gdim),
351  coordinate_dofs);
352 
353  // Compute basis on reference element
354  element->evaluate_reference_basis(basis_reference_values, X);
355 
356  // Push basis forward to physical element
357  element->transform_reference_basis(basis_values, basis_reference_values,
358  X, J, detJ, K, cell_info[cell_index]);
359 
360  // Get degrees of freedom for current cell
361  auto dofs = dofmap->cell_dofs(cell_index);
362  for (Eigen::Index i = 0; i < dofs.size(); ++i)
363  for (int k = 0; k < bs_dof; ++k)
364  coefficients[bs_dof * i + k] = _v[bs_dof * dofs[i] + k];
365 
366  // Compute expansion
367  auto u_row = u.row(p);
368  for (int k = 0; k < bs_element; ++k)
369  {
370  for (int i = 0; i < space_dimension; ++i)
371  {
372  for (int j = 0; j < value_size; ++j)
373  {
374  // TODO: Find an Eigen shortcut for this operation?
375  u_row[j * bs_element + k]
376  += coefficients[bs_element * i + k] * basis_values(0, i, j);
377  }
378  }
379  }
380  }
381  }

385  Eigen::Array<T, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor>
386  compute_point_values() const
387  {
388  assert(_function_space);
389  std::shared_ptr<const mesh::Mesh> mesh = _function_space->mesh();
390  assert(mesh);
391  const int tdim = mesh->topology().dim();
392 
393  // Compute in tensor (one for scalar function, . . .)
394  const int value_size_loc = _function_space->element()->value_size();
395 
396  // Resize Array for holding point values
397  Eigen::Array<T, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor>
398  point_values(mesh->geometry().x().rows(), value_size_loc);
399 
400  // Prepare cell geometry
401  const graph::AdjacencyList<std::int32_t>& x_dofmap
402  = mesh->geometry().dofmap();
403 
404  // FIXME: Add proper interface for num coordinate dofs
405  const int num_dofs_g = x_dofmap.num_links(0);
406  const Eigen::Array<double, Eigen::Dynamic, 3, Eigen::RowMajor>& x_g
407  = mesh->geometry().x();
408 
409  // Interpolate point values on each cell (using last computed value if
410  // not continuous, e.g. discontinuous Galerkin methods)
411  Eigen::Array<double, Eigen::Dynamic, 3, Eigen::RowMajor> x(num_dofs_g, 3);
412  Eigen::Array<T, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor> values(
413  num_dofs_g, value_size_loc);
414  auto map = mesh->topology().index_map(tdim);
415  assert(map);
416  const std::int32_t num_cells = map->size_local() + map->num_ghosts();
417  for (std::int32_t c = 0; c < num_cells; ++c)
418  {
419  // Get coordinates for all points in cell
420  auto dofs = x_dofmap.links(c);
421  for (int i = 0; i < num_dofs_g; ++i)
422  x.row(i) = x_g.row(dofs[i]);
423 
424  values.resize(x.rows(), value_size_loc);
425 
426  // Call evaluate function
427  Eigen::Array<int, Eigen::Dynamic, 1> cells(x.rows());
428  cells = c;
429  eval(x, cells, values);
430 
431  // Copy values to array of point values
432  for (int i = 0; i < x.rows(); ++i)
433  point_values.row(dofs[i]) = values.row(i);
434  }
435 
436  return point_values;
437  }
