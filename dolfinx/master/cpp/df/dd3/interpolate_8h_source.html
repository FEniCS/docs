95 template <typename T>
96 void interpolate_from_any(Function<T>& u, const Function<T>& v)
97 {
98  assert(v.function_space());
99  const auto element = u.function_space()->element();
100  assert(element);
101  if (v.function_space()->element()->hash() != element->hash())
102  {
103  throw std::runtime_error("Restricting finite elements function in "
104  "different elements not supported.");
105  }
106 
107  const auto mesh = u.function_space()->mesh();
108  assert(mesh);
109  assert(v.function_space()->mesh());
110  if (mesh->id() != v.function_space()->mesh()->id())
111  {
112  throw std::runtime_error(
113  "Interpolation on different meshes not supported (yet).");
114  }
115  const int tdim = mesh->topology().dim();
116 
117  // Get dofmaps
118  assert(v.function_space());
119  std::shared_ptr<const fem::DofMap> dofmap_v = v.function_space()->dofmap();
120  assert(dofmap_v);
121  auto map = mesh->topology().index_map(tdim);
122  assert(map);
123 
124  std::vector<T>& coeffs = u.x()->mutable_array();
125 
126  // Iterate over mesh and interpolate on each cell
127  const auto dofmap_u = u.function_space()->dofmap();
128  const std::vector<T>& v_array = v.x()->array();
129  const int num_cells = map->size_local() + map->num_ghosts();
130  const int bs = dofmap_v->bs();
131  assert(bs == dofmap_u->bs());
132  for (int c = 0; c < num_cells; ++c)
133  {
134  tcb::span<const std::int32_t> dofs_v = dofmap_v->cell_dofs(c);
135  tcb::span<const std::int32_t> cell_dofs = dofmap_u->cell_dofs(c);
136  assert(dofs_v.size() == cell_dofs.size());
137  for (std::size_t i = 0; i < dofs_v.size(); ++i)
138  {
139  for (int k = 0; k < bs; ++k)
140  coeffs[bs * cell_dofs[i] + k] = v_array[bs * dofs_v[i] + k];
141  }
142  }
143 }

147 //----------------------------------------------------------------------------
148 template <typename T>
149 void interpolate(Function<T>& u, const Function<T>& v)
150 {
151  assert(u.function_space());
152  const auto element = u.function_space()->element();
153  assert(element);
154 
155  // Check that function ranks match
156  if (int rank_v = v.function_space()->element()->value_rank();
157  element->value_rank() != rank_v)
158  {
159  throw std::runtime_error("Cannot interpolate function into function space. "
160  "Rank of function ("
161  + std::to_string(rank_v)
162  + ") does not match rank of function space ("
163  + std::to_string(element->value_rank()) + ")");
164  }
165 
166  // Check that function dimension match
167  for (int i = 0; i < element->value_rank(); ++i)
168  {
169  if (int v_dim = v.function_space()->element()->value_dimension(i);
170  element->value_dimension(i) != v_dim)
171  {
172  throw std::runtime_error(
173  "Cannot interpolate function into function space. "
174  "Dimension "
175  + std::to_string(i) + " of function (" + std::to_string(v_dim)
176  + ") does not match dimension " + std::to_string(i)
177  + " of function space(" + std::to_string(element->value_dimension(i))
178  + ")");
179  }
180  }
181 
182  detail::interpolate_from_any(u, v);
183 }
184 //----------------------------------------------------------------------------
185 template <typename T>
186 void interpolate(
187  Function<T>& u,
188  const std::function<
189  Eigen::Array<T, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor>(
190  const Eigen::Ref<const Eigen::Array<double, 3, Eigen::Dynamic,
191  Eigen::RowMajor>>&)>& f,
192  const Eigen::Array<double, 3, Eigen::Dynamic, Eigen::RowMajor>& x,
193  const tcb::span<std::int32_t>& cells)
194 {
195  const auto element = u.function_space()->element();
196  assert(element);
197  const int element_bs = element->block_size();
198  if (int num_sub = element->num_sub_elements();
199  num_sub > 0 and num_sub != element_bs)
200  {
201  throw std::runtime_error("Cannot directly interpolate a mixed space. "
202  "Interpolate into subspaces.");
203  }
204 
205  // Get mesh
206  assert(u.function_space());
207  auto mesh = u.function_space()->mesh();
208  assert(mesh);
209 
210  // Get the interpolation points on the reference cells
211  const common::array2d<double> X = element->interpolation_points();
212 
213  mesh->topology_mutable().create_entity_permutations();
214  const std::vector<std::uint32_t>& cell_info
215  = mesh->topology().get_cell_permutation_info();
216 
217  // Evaluate function at physical points. The returned array has a
218  // number of rows equal to the number of components of the function,
219  // and the number of columns is equal to the number of evaluation
220  // points. Scalar case needs special handling as pybind11 will return
221  // a column array when we need a row array.
222  Eigen::Array<T, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor> values;
223  values = f(x);
224 
225  // FIXME: This is hack for NumPy/pybind11/Eigen that returns 1D arrays a
226  // column vectors. Fix in the pybind11 layer?
227  if (element->value_size() == 1 and values.rows() > 1)
228  values = values.transpose().eval();
229 
230  // Get dofmap
231  const auto dofmap = u.function_space()->dofmap();
232  assert(dofmap);
233  const int dofmap_bs = dofmap->bs();
234 
235  // NOTE: The below loop over cells could be skipped for some elements,
236  // e.g. Lagrange, where the interpolation is just the identity.
237 
238  // Loop over cells and compute interpolation dofs
239  const int num_scalar_dofs = element->space_dimension() / element_bs;
240  const int value_size = element->value_size() / element_bs;
241 
242  // Check that return type from f is the correct shape
243  if ((values.rows() != value_size * element_bs)
244  or (values.cols() != cells.size() * X.shape[0]))
245  {
246  throw std::runtime_error("Interpolation data has the wrong shape.");
247  }
248 
249  std::vector<T>& coeffs = u.x()->mutable_array();
250  std::vector<T> _coeffs(num_scalar_dofs);
251  common::array2d<T> _vals(value_size, X.shape[0]);
252  for (std::int32_t c : cells)
253  {
254  auto dofs = dofmap->cell_dofs(c);
255  for (int k = 0; k < element_bs; ++k)
256  {
257  // Extract computed expression values for element block k
258  for (int m = 0; m < value_size; ++m)
259  {
260  std::copy_n(&values(k * value_size + m, c * X.shape[0]), X.shape[0],
261  _vals.row(m).begin());
262  }
263 
264  // Get element degrees of freedom for block
265  element->interpolate(_vals, cell_info[c], tcb::make_span(_coeffs));
266  assert(_coeffs.size() == num_scalar_dofs);
267 
268  // Copy interpolation dofs into coefficient vector
269  for (int i = 0; i < num_scalar_dofs; ++i)
270  {
271  const int dof = i * element_bs + k;
272  std::div_t pos = std::div(dof, dofmap_bs);
273  coeffs[dofmap_bs * dofs[pos.quot] + pos.rem] = _coeffs[i];
274  }
275  }
276  }
277 }
278 //----------------------------------------------------------------------------
279 template <typename T>
280 void interpolate_c(
281  Function<T>& u,
282  const std::function<void(
283  Eigen::Ref<
284  Eigen::Array<T, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor>>,
285  const Eigen::Ref<const Eigen::Array<double, Eigen::Dynamic, 3,
286  Eigen::RowMajor>>&)>& f,
287  const Eigen::Array<double, 3, Eigen::Dynamic, Eigen::RowMajor>& x,
288  const tcb::span<std::int32_t>& cells)
289 {
290  const auto element = u.function_space()->element();
291  assert(element);
292  std::vector<int> vshape(element->value_rank(), 1);
293  for (std::size_t i = 0; i < vshape.size(); ++i)
294  vshape[i] = element->value_dimension(i);
295  const int value_size = std::accumulate(std::begin(vshape), std::end(vshape),
296  1, std::multiplies<>());
297 
298  auto fn =
299  [value_size,
300  &f](const Eigen::Ref<
301  const Eigen::Array<double, 3, Eigen::Dynamic, Eigen::RowMajor>>& x) {
302  Eigen::Array<T, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor> values(
303  x.cols(), value_size);
304  f(values, x.transpose());
305  return values;
306  };
307 
308  interpolate<T>(u, fn, x, cells);
309 }
