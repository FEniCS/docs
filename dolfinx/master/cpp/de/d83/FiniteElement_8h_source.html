157  template <typename T>
158  constexpr void interpolate(const common::array2d<T>& values,
159  std::uint32_t cell_permutation,
160  tcb::span<T> dofs) const
161  {
162  const std::size_t rows = _space_dim / _bs;
163  assert(_space_dim % _bs == 0);
164  assert(dofs.size() == rows);
165 
166  // Compute dofs = Pi * x (matrix-vector multiply)
167  assert(_interpolation_matrix.size() % rows == 0);
168  const std::size_t cols = _interpolation_matrix.size() / rows;
169  for (std::size_t i = 0; i < rows; ++i)
170  {
171  // Dot product between row i of the matrix and 'values'
172  dofs[i] = std::transform_reduce(
173  std::next(_interpolation_matrix.begin(), i * cols),
174  std::next(_interpolation_matrix.begin(), i * cols + cols),
175  values.data(), T(0.0));
176  }
177 
178  _apply_dof_transformation_to_scalar(dofs.data(), cell_permutation, 1);
179  }
