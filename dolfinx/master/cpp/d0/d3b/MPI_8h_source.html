201 template <typename T>
202 graph::AdjacencyList<T>
203 dolfinx::MPI::all_to_all(MPI_Comm comm,
204  const graph::AdjacencyList<T>& send_data)
205 {
206  const std::vector<std::int32_t>& send_offsets = send_data.offsets();
207  const std::vector<T>& values_in = send_data.array();
208 
209  const int comm_size = MPI::size(comm);
210  assert(send_data.num_nodes() == comm_size);
211 
212  // Data size per destination rank
213  std::vector<int> send_size(comm_size);
214  std::adjacent_difference(std::next(send_offsets.begin(), +1),
215  send_offsets.end(), send_size.begin());
216 
217  // Get received data sizes from each rank
218  std::vector<int> recv_size(comm_size);
219  MPI_Alltoall(send_size.data(), 1, mpi_type<int>(), recv_size.data(), 1,
220  mpi_type<int>(), comm);
221 
222  // Compute receive offset
223  std::vector<std::int32_t> recv_offset(comm_size + 1);
224  recv_offset[0] = 0;
225  std::partial_sum(recv_size.begin(), recv_size.end(), recv_offset.data() + 1);
226 
227  // Send/receive data
228  std::vector<T> recv_values(recv_offset[comm_size]);
229  MPI_Alltoallv(values_in.data(), send_size.data(), send_offsets.data(),
230  mpi_type<T>(), recv_values.data(), recv_size.data(),
231  recv_offset.data(), mpi_type<T>(), comm);
232 
233  return graph::AdjacencyList<T>(std::move(recv_values),
234  std::move(recv_offset));
235 }

237 template <typename T>
238 graph::AdjacencyList<T>
239 dolfinx::MPI::neighbor_all_to_all(MPI_Comm neighbor_comm,
240  const std::vector<int>& send_offsets,
241  const std::vector<T>& send_data)
242 {
243  // Get neighbor processes
244  int indegree(-1), outdegree(-2), weighted(-1);
245  MPI_Dist_graph_neighbors_count(neighbor_comm, &indegree, &outdegree,
246  &weighted);
247 
248  assert((int)send_data.size() == send_offsets.back());
249  assert(send_offsets[0] == 0);
250 
251  // Get receive sizes
252  std::vector<int> send_sizes(outdegree, 0);
253  std::vector<int> recv_sizes(indegree);
254  std::adjacent_difference(send_offsets.begin() + 1, send_offsets.end(),
255  send_sizes.begin());
256  MPI_Neighbor_alltoall(send_sizes.data(), 1, MPI::mpi_type<int>(),
257  recv_sizes.data(), 1, MPI::mpi_type<int>(),
258  neighbor_comm);
259 
260  // Work out recv offsets
261  std::vector<int> recv_offsets(recv_sizes.size() + 1);
262  recv_offsets[0] = 0;
263  std::partial_sum(recv_sizes.begin(), recv_sizes.end(),
264  recv_offsets.data() + 1);
265 
266  std::vector<T> recv_data(recv_offsets[recv_offsets.size() - 1]);
267  MPI_Neighbor_alltoallv(
268  send_data.data(), send_sizes.data(), send_offsets.data(),
269  MPI::mpi_type<T>(), recv_data.data(), recv_sizes.data(),
270  recv_offsets.data(), MPI::mpi_type<T>(), neighbor_comm);
271 
272  return graph::AdjacencyList<T>(std::move(recv_data), std::move(recv_offsets));
273 }
