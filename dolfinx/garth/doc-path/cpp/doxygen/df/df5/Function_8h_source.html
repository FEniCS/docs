176  void interpolate(
177  const std::function<xt::xarray<T>(const xt::xtensor<double, 2>&)>& f,
178  const xtl::span<const std::int32_t>& cells)
179  {
180  assert(_function_space);
181  assert(_function_space->element());
182  assert(_function_space->mesh());
183  const xt::xtensor<double, 2> x = fem::interpolation_coords(
184  *_function_space->element(), *_function_space->mesh(), cells);
185  auto fx = f(x);
186  if (int vs = _function_space->element()->value_size();
187  vs == 1 and fx.dimension() == 1)
188  {
189  // Check for scalar-valued functions
190  if (fx.shape(0) != x.shape(1))
191  throw std::runtime_error("Data returned by callable has wrong length");

193  else
194  {
195  // Check for vector/tensor value
196  if (fx.dimension() != 2)
197  throw std::runtime_error("Expected 2D array of data");
198  if (fx.shape(0) != vs)
199  {
200  throw std::runtime_error(
201  "Data returned by callable has wrong shape(0) size");
202  }
203  if (fx.shape(1) != x.shape(1))
204  {
205  throw std::runtime_error(
206  "Data returned by callable has wrong shape(1) size");
207  }
208  }

235  void interpolate(const Expression<T>& e,
236  const xtl::span<const std::int32_t>& cells)
237  {
238  // Check that spaces are compatible
239  assert(_function_space);
240  assert(_function_space->element());
241  std::size_t value_size = e.value_size();
242  if (e.argument_function_space())
243  throw std::runtime_error("Cannot interpolate Expression with Argument");
244 
245  if (value_size != _function_space->element()->value_size())
246  {
247  throw std::runtime_error(
248  "Function value size not equal to Expression value size");
249  }
250 
251  if (!xt::allclose(e.X(),
252  _function_space->element()->interpolation_points()))
253  {
254  throw std::runtime_error("Function element interpolation points not "
255  "equal to Expression interpolation points");
256  }
257 
258  // Array to hold evaluted Expression
259  std::size_t num_cells = cells.size();
260  std::size_t num_points = e.X().shape(0);
261  xt::xtensor<T, 3> f({num_cells, num_points, value_size});
262 
263  // Evaluate Expression at points
264  auto f_view = xt::reshape_view(f, {num_cells, num_points * value_size});
265  e.eval(cells, f_view);
266 
267  // Reshape evaluated data to fit interpolate
268  // Expression returns matrix of shape (num_cells, num_points *
269  // value_size), i.e. xyzxyz ordering of dof values per cell per point.
270  // The interpolation uses xxyyzz input, ordered for all points of each
271  // cell, i.e. (value_size, num_cells*num_points)
272  xt::xarray<T> _f = xt::reshape_view(xt::transpose(f, {2, 0, 1}),
273  {value_size, num_cells * num_points});
274 
275  // Interpolate values into appropriate space
276  fem::interpolate(*this, _f, cells);
277  }

304  void eval(const xt::xtensor<double, 2>& x,
305  const xtl::span<const std::int32_t>& cells,
306  xt::xtensor<T, 2>& u) const
307  {
308  // TODO: This could be easily made more efficient by exploiting points
309  // being ordered by the cell to which they belong.
310 
311  if (x.shape(0) != cells.size())
312  {
313  throw std::runtime_error(
314  "Number of points and number of cells must be equal.");
315  }
316  if (x.shape(0) != u.shape(0))
317  {
318  throw std::runtime_error(
319  "Length of array for Function values must be the "
320  "same as the number of points.");
321  }
322 
323  // Get mesh
324  assert(_function_space);
325  std::shared_ptr<const mesh::Mesh> mesh = _function_space->mesh();
326  assert(mesh);
327  const std::size_t gdim = mesh->geometry().dim();
328  const std::size_t tdim = mesh->topology().dim();
329  auto map = mesh->topology().index_map(tdim);
330 
331  // Get geometry data
332  const graph::AdjacencyList<std::int32_t>& x_dofmap
333  = mesh->geometry().dofmap();
334  // FIXME: Add proper interface for num coordinate dofs
335  const std::size_t num_dofs_g = x_dofmap.num_links(0);
336  xtl::span<const double> x_g = mesh->geometry().x();
337 
338  // Get coordinate map
339  const fem::CoordinateElement& cmap = mesh->geometry().cmap();
340 
341  // Get element
342  assert(_function_space->element());
343  std::shared_ptr<const fem::FiniteElement> element
344  = _function_space->element();
345  assert(element);
346  const int bs_element = element->block_size();
347  const std::size_t reference_value_size
348  = element->reference_value_size() / bs_element;
349  const std::size_t value_size = element->value_size() / bs_element;
350  const std::size_t space_dimension = element->space_dimension() / bs_element;
351 
352  // If the space has sub elements, concatenate the evaluations on the sub
353  // elements
354  const int num_sub_elements = element->num_sub_elements();
355  if (num_sub_elements > 1 and num_sub_elements != bs_element)
356  {
357  throw std::runtime_error("Function::eval is not supported for mixed "
358  "elements. Extract subspaces.");
359  }
360 
361  // Create work vector for expansion coefficients
362  std::vector<T> coefficients(space_dimension * bs_element);
363 
364  // Get dofmap
365  std::shared_ptr<const fem::DofMap> dofmap = _function_space->dofmap();
366  assert(dofmap);
367  const int bs_dof = dofmap->bs();
368 
369  xtl::span<const std::uint32_t> cell_info;
370  if (element->needs_dof_transformations())
371  {
372  mesh->topology_mutable().create_entity_permutations();
373  cell_info = xtl::span(mesh->topology().get_cell_permutation_info());
374  }
375 
376  xt::xtensor<double, 2> coordinate_dofs
377  = xt::zeros<double>({num_dofs_g, gdim});
378  xt::xtensor<double, 2> xp = xt::zeros<double>({std::size_t(1), gdim});
379 
380  // Loop over points
381  std::fill(u.data(), u.data() + u.size(), 0.0);
382  const xtl::span<const T>& _v = _x->array();
383 
384  // -- Lambda function for affine pull-backs
385  xt::xtensor<double, 4> data(cmap.tabulate_shape(1, 1));
386  const xt::xtensor<double, 2> X0(xt::zeros<double>({std::size_t(1), tdim}));
387  cmap.tabulate(1, X0, data);
388  const xt::xtensor<double, 2> dphi_i
389  = xt::view(data, xt::range(1, tdim + 1), 0, xt::all(), 0);
390  auto pull_back_affine = [dphi_i](auto&& X, const auto& cell_geometry,
391  auto&& J, auto&& K, const auto& x) mutable
392  {
393  fem::CoordinateElement::compute_jacobian(dphi_i, cell_geometry, J);
394  fem::CoordinateElement::compute_jacobian_inverse(J, K);
395  fem::CoordinateElement::pull_back_affine(
396  X, K, fem::CoordinateElement::x0(cell_geometry), x);
397  };
398 
399  xt::xtensor<double, 2> dphi;
400  xt::xtensor<double, 2> X({x.shape(0), tdim});
401  xt::xtensor<double, 3> J = xt::zeros<double>({x.shape(0), gdim, tdim});
402  xt::xtensor<double, 3> K = xt::zeros<double>({x.shape(0), tdim, gdim});
403  xt::xtensor<double, 1> detJ = xt::zeros<double>({x.shape(0)});
404  xt::xtensor<double, 4> phi(cmap.tabulate_shape(1, 1));
405 
406  xt::xtensor<double, 2> _Xp({1, tdim});
407  for (std::size_t p = 0; p < cells.size(); ++p)
408  {
409  const int cell_index = cells[p];
410  // Skip negative cell indices
411  if (cell_index < 0)
412  continue;
413 
414  // Get cell geometry (coordinate dofs)
415  auto x_dofs = x_dofmap.links(cell_index);
416  for (std::size_t i = 0; i < num_dofs_g; ++i)
417  {
418  const int pos = 3 * x_dofs[i];
419  for (std::size_t j = 0; j < gdim; ++j)
420  coordinate_dofs(i, j) = x_g[pos + j];
421  }
422 
423  for (std::size_t j = 0; j < gdim; ++j)
424  xp(0, j) = x(p, j);
425 
426  auto _J = xt::view(J, p, xt::all(), xt::all());
427  auto _K = xt::view(K, p, xt::all(), xt::all());
428  // Compute reference coordinates X, and J, detJ and K
429  if (cmap.is_affine())
430  {
431  pull_back_affine(_Xp, coordinate_dofs, _J, _K, xp);
432  detJ[p]
433  = dolfinx::fem::CoordinateElement::compute_jacobian_determinant(_J);
434  }
435  else
436  {
437  cmap.pull_back_nonaffine(_Xp, xp, coordinate_dofs);
438  cmap.tabulate(1, _Xp, phi);
439  dphi = xt::view(phi, xt::range(1, tdim + 1), 0, xt::all(), 0);
440  dolfinx::fem::CoordinateElement::compute_jacobian(dphi, coordinate_dofs,
441  _J);
442  dolfinx::fem::CoordinateElement::compute_jacobian_inverse(_J, _K);
443  detJ[p]
444  = dolfinx::fem::CoordinateElement::compute_jacobian_determinant(_J);
445  }
446  xt::row(X, p) = xt::row(_Xp, 0);
447  }
448 
449  // Prepare basis function data structures
450  xt::xtensor<double, 4> basis_derivatives_reference_values(
451  {1, x.shape(0), space_dimension, reference_value_size});
452  xt::xtensor<double, 3> basis_values(
453  {static_cast<std::size_t>(1), space_dimension, value_size});
454 
455  // Compute basis on reference element
456  element->tabulate(basis_derivatives_reference_values, X, 0);
457 
458  using u_t = xt::xview<decltype(basis_values)&, std::size_t,
459  xt::xall<std::size_t>, xt::xall<std::size_t>>;
460  using U_t
461  = xt::xview<decltype(basis_derivatives_reference_values)&, std::size_t,
462  std::size_t, xt::xall<std::size_t>, xt::xall<std::size_t>>;
463  using J_t = xt::xview<decltype(J)&, std::size_t, xt::xall<std::size_t>,
464  xt::xall<std::size_t>>;
465  using K_t = xt::xview<decltype(K)&, std::size_t, xt::xall<std::size_t>,
466  xt::xall<std::size_t>>;
467  auto push_forward_fn = element->map_fn<u_t, U_t, J_t, K_t>();
468  const std::function<void(const xtl::span<double>&,
469  const xtl::span<const std::uint32_t>&,
470  std::int32_t, int)>
471  apply_dof_transformation
472  = element->get_dof_transformation_function<double>();
473  const std::size_t num_basis_values = space_dimension * reference_value_size;
474 
475  for (std::size_t p = 0; p < cells.size(); ++p)
476  {
477  const int cell_index = cells[p];
478  // Skip negative cell indices
479  if (cell_index < 0)
480  continue;
481 
482  // Permute the reference values to account for the cell's orientation
483  apply_dof_transformation(
484  xtl::span(basis_derivatives_reference_values.data()
485  + p * num_basis_values,
486  num_basis_values),
487  cell_info, cell_index, reference_value_size);
488 
489  auto _K = xt::view(K, p, xt::all(), xt::all());
490  auto _J = xt::view(J, p, xt::all(), xt::all());
491  auto _U = xt::view(basis_derivatives_reference_values, (std::size_t)0, p,
492  xt::all(), xt::all());
493  auto _u = xt::view(basis_values, (std::size_t)0, xt::all(), xt::all());
494  push_forward_fn(_u, _U, _J, detJ[p], _K);
495 
496  // Get degrees of freedom for current cell
497  xtl::span<const std::int32_t> dofs = dofmap->cell_dofs(cell_index);
498  for (std::size_t i = 0; i < dofs.size(); ++i)
499  for (int k = 0; k < bs_dof; ++k)
500  coefficients[bs_dof * i + k] = _v[bs_dof * dofs[i] + k];
501 
502  // Compute expansion
503  auto u_row = xt::row(u, p);
504  for (int k = 0; k < bs_element; ++k)
505  {
506  for (std::size_t i = 0; i < space_dimension; ++i)
507  {
508  for (std::size_t j = 0; j < value_size; ++j)
509  {
510  u_row[j * bs_element + k]
511  += coefficients[bs_element * i + k] * basis_values(0, i, j);
512  }
513  }
514  }
515  }
516  }
517 
519  std::string name = "u";
520 
522  std::size_t id() const { return _id; }
523 
524 private:
525  // ID
526  std::size_t _id;
527 
528  // The function space
529  std::shared_ptr<const FunctionSpace> _function_space;
530 
531  // The vector of expansion coefficients (local)
532  std::shared_ptr<la::Vector<T>> _x;
533 };
