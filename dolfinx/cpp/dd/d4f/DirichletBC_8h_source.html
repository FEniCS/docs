144 public:
152  DirichletBC(
153  const std::shared_ptr<const function::Function<T>>& g,
154  const Eigen::Ref<const Eigen::Array<std::int32_t, Eigen::Dynamic, 1>>&
155  dofs)
156  : _function_space(g->function_space()), _g(g), _dofs0(dofs),
157  _dofs1_g(dofs)
158  {
159  const int bs = _function_space->dofmap()->bs();
160  if (bs > 1)
161  {
162  // Unroll for the block size
163  _dofs0.resize(bs * dofs.rows());
164  for (Eigen::Index i = 0; i < dofs.rows(); ++i)
165  {
166  for (int k = 0; k < bs; ++k)
167  _dofs0[bs * i + k] = bs * dofs[i] + k;
168  }
169  _dofs1_g = _dofs0;
170  }
171 
172  // TODO: allows single dofs array (let one point to the other)
173  const int owned_size0 = _function_space->dofmap()->index_map->size_local();
174  const int map0_bs = _function_space->dofmap()->index_map_bs();
175  auto* it = std::lower_bound(_dofs0.data(), _dofs0.data() + _dofs0.rows(),
176  map0_bs * owned_size0);
177  _owned_indices0 = std::distance(_dofs0.data(), it);
178  }
179 
190  DirichletBC(const std::shared_ptr<const function::Function<T>>& g,
191  const std::array<Eigen::Array<std::int32_t, Eigen::Dynamic, 1>,
192  2>& V_g_dofs,
193  std::shared_ptr<const function::FunctionSpace> V)
194  : _function_space(V), _g(g), _dofs0(V_g_dofs[0]), _dofs1_g(V_g_dofs[1])
195  {
196  assert(_dofs0.rows() == _dofs1_g.rows());
197  assert(_function_space);
198  assert(_g);
199 
200  const int map0_bs = _function_space->dofmap()->index_map_bs();
201  const int map0_size = _function_space->dofmap()->index_map->size_local();
202  const int owned_size0 = (map0_bs * map0_size);
203  auto it0 = std::lower_bound(_dofs0.data(), _dofs0.data() + _dofs0.rows(),
204  owned_size0);
205  _owned_indices0 = std::distance(_dofs0.data(), it0);
206  }

258  void set(Eigen::Ref<Eigen::Matrix<T, Eigen::Dynamic, 1>> x,
259  double scale = 1.0) const
260  {
261  assert(_g);
262  auto& g = _g->x()->array();

264  {

277  void set(Eigen::Ref<Eigen::Matrix<T, Eigen::Dynamic, 1>> x,
278  const Eigen::Ref<const Eigen::Matrix<T, Eigen::Dynamic, 1>>& x0,
279  double scale = 1.0) const
280  {
281  assert(_g);
282  auto& g = _g->x()->array();
283  assert(x.rows() <= x0.rows());
284  for (Eigen::Index i = 0; i < _dofs0.rows(); ++i)

290  }
