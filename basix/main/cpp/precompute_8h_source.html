117 void apply_permutation(const std::span<const std::size_t>& perm,
118  const std::span<E>& data, std::size_t offset = 0,
119  std::size_t block_size = 1)
120 {
121  for (std::size_t i = 0; i < perm.size(); ++i)
122  {
123  for (std::size_t b = 0; b < block_size; ++b)
124  {
125  std::swap(data[block_size * (offset + i) + b],
126  data[block_size * (offset + perm[i]) + b]);
127  }
128  }

225 template <typename T, typename E>
226 void apply_matrix(const std::span<const std::size_t>& v_size_t,
227  const std::experimental::mdspan<
228  const T, std::experimental::dextents<std::size_t, 2>>& M,
229  const std::span<E>& data, std::size_t offset = 0,
230  std::size_t block_size = 1)
231 {
232  const std::size_t dim = v_size_t.size();
233  apply_permutation(v_size_t, data, offset, block_size);
234  for (std::size_t b = 0; b < block_size; ++b)
235  {
236  for (std::size_t i = 0; i < dim; ++i)
237  {
238  for (std::size_t j = i + 1; j < dim; ++j)
239  {
240  data[block_size * (offset + i) + b]
241  += M(i, j) * data[block_size * (offset + j) + b];
242  }
243  }
244  for (std::size_t i = 1; i <= dim; ++i)
245  {
246  data[block_size * (offset + dim - i) + b] *= M(dim - i, dim - i);
247  for (std::size_t j = 0; j < dim - i; ++j)
248  {
249  data[block_size * (offset + dim - i) + b]
250  += M(dim - i, j) * data[block_size * (offset + j) + b];
251  }
252  }
253  }
254 }
