291 void precompute::apply_permutation(const std::vector<std::size_t>& perm,
292  xtl::span<E>& data, std::size_t offset,
293  std::size_t block_size)
294 {
295  for (std::size_t b = 0; b < block_size; ++b)
296  {
297  for (std::size_t i = 0; i < perm.size(); ++i)
298  {
299  std::swap(data[block_size * (offset + i) + b],
300  data[block_size * (offset + perm[i]) + b]);
301  }
302  }
303 }

305 template <typename U>
306 std::tuple<std::vector<std::size_t>, std::vector<typename U::value_type>,
307  xt::xtensor<typename U::value_type, 2>>
308 precompute::prepare_matrix(const U& matrix)
309 {
310  using T = typename U::value_type;
311  const std::size_t dim = matrix.shape(0);
312  std::vector<std::size_t> perm(dim);
313  xt::xtensor<T, 2> permuted_matrix({dim, dim});
314  std::vector<T> diag(dim);
315 
316  // Permute the matrix so that all the top left blocks are invertible
317  for (std::size_t i = 0; i < dim; ++i)
318  {
319  double max_det = 0;
320  std::size_t col = -1;
321  for (std::size_t j = 0; j < dim; ++j)
322  {
323  bool used = false;
324  for (std::size_t k = 0; k < i; ++k)
325  if (perm[k] == j)
326  used = true;
327 
328  if (!used)
329  {
330  xt::view(permuted_matrix, xt::all(), i)
331  = xt::view(matrix, xt::all(), j);
332  double det = std::abs(xt::linalg::det(xt::view(
333  permuted_matrix, xt::range(0, i + 1), xt::range(0, i + 1))));
334  if (det > max_det)
335  {
336  max_det = det;
337  col = j;
338  }
339  }
340  }
341  xt::view(permuted_matrix, xt::all(), i) = xt::view(matrix, xt::all(), col);
342  perm[i] = col;
343  }
344 
345  // Create the precomputed representation of the matrix
346  xt::xtensor<T, 2> prepared_matrix({dim, dim});
347 
348  for (std::size_t i = 0; i < dim; ++i)
349  {
350  diag[i] = permuted_matrix(i, i);
351  prepared_matrix(i, i) = 0;
352  if (i < dim - 1)
353  {
354  xt::view(prepared_matrix, i, xt::range(i + 1, dim))
355  = xt::view(permuted_matrix, i, xt::range(i + 1, dim));
356  }
357  if (i > 0)
358  {
359  xt::xtensor<T, 1> v = xt::linalg::solve(
360  xt::transpose(
361  xt::view(permuted_matrix, xt::range(0, i), xt::range(0, i))),
362  xt::view(permuted_matrix, i, xt::range(0, i)));
363 
364  xt::view(prepared_matrix, i, xt::range(0, i)) = v;
365 
366  diag[i] -= xt::linalg::dot(
367  v, xt::view(permuted_matrix, xt::range(0, i), i))(0);
368  for (std::size_t j = i + 1; j < dim; ++j)
369  {
370  prepared_matrix(i, j) -= xt::linalg::dot(
371  v, xt::view(permuted_matrix, xt::range(0, i), j))(0);
372  }
373  }
374  }
375 
376  return {prepare_permutation(perm), std::move(diag),
377  std::move(prepared_matrix)};
378 }

380 template <typename T, typename E>
381 void precompute::apply_matrix(
382  const std::tuple<std::vector<std::size_t>, std::vector<T>,
383  xt::xtensor<T, 2>>& matrix,
384  xtl::span<E>& data, std::size_t offset, std::size_t block_size)
385 {
386  const std::vector<std::size_t>& v_size_t = std::get<0>(matrix);
387  const std::vector<T>& v_t = std::get<1>(matrix);
388  const xt::xtensor<T, 2>& M = std::get<2>(matrix);
389 
390  const std::size_t dim = v_size_t.size();
391  apply_permutation(v_size_t, data, offset, block_size);
392  for (std::size_t b = 0; b < block_size; ++b)
393  {
394  for (std::size_t i = 0; i < dim; ++i)
395  {
396  data[block_size * (offset + i) + b] *= v_t[i];
397  for (std::size_t j = 0; j < dim; ++j)
398  {
399  data[block_size * (offset + i) + b]
400  += M(i, j) * data[block_size * (offset + j) + b];
401  }
402  }
403  }
404 }
