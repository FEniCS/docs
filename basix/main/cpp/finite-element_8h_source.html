648  template <typename O, typename P, typename Q, typename R>
649  std::function<void(O&, const P&, const Q&, double, const R&)> map_fn() const
650  {
651  switch (_map_type)
652  {
653  case maps::type::identity:
654  return [](O& u, const P& U, const Q&, double, const R&)
655  {
656  // std::cout << "ident" << std::endl;
657  assert(U.extent(0) == u.extent(0));
658  assert(U.extent(1) == u.extent(1));
659  for (std::size_t i = 0; i < U.extent(0); ++i)
660  for (std::size_t j = 0; j < U.extent(1); ++j)
661  u(i, j) = U(i, j);
662  };
663  case maps::type::covariantPiola:
664  return [](O& u, const P& U, const Q& J, double detJ, const R& K)
665  {
666  // std::cout << "map 1" << std::endl;
667  maps::covariant_piola(u, U, J, detJ, K);
668  };
669  case maps::type::contravariantPiola:
670  return [](O& u, const P& U, const Q& J, double detJ, const R& K)
671  {
672  // std::cout << "map 2" << std::endl;
673  maps::contravariant_piola(u, U, J, detJ, K);
674  };
675  case maps::type::doubleCovariantPiola:
676  return [](O& u, const P& U, const Q& J, double detJ, const R& K)
677  {
678  // std::cout << "map 3" << std::endl;
679  maps::double_covariant_piola(u, U, J, detJ, K);
680  };
681  case maps::type::doubleContravariantPiola:
682  return [](O& u, const P& U, const Q& J, double detJ, const R& K)
683  {
684  // std::cout << "map 4" << std::endl;
685  maps::double_contravariant_piola(u, U, J, detJ, K);
686  };
687  default:
688  throw std::runtime_error("Map not implemented");
689  }
690  }

1942 template <typename T>
1943 void FiniteElement::apply_inverse_dof_transformation_to_transpose(
1944  const xtl::span<T>& data, int block_size, std::uint32_t cell_info) const
1945 {
1946  if (_dof_transformations_are_identity)
1947  return;
1948 
1949  if (_cell_tdim >= 2)
1950  {
1951  // This assumes 3 bits are used per face. This will need updating if
1952  // 3D cells with faces with more than 4 sides are implemented
1953  int face_start = _cell_tdim == 3 ? 3 * _num_edofs[2].size() : 0;
1954  int dofstart
1955  = std::accumulate(_num_edofs[0].cbegin(), _num_edofs[0].cend(), 0);
1956 
1957  // Transform DOFs on edges
1958  for (std::size_t e = 0; e < _num_edofs[1].size(); ++e)
1959  {
1960  // Reverse an edge
1961  if (cell_info >> (face_start + e) & 1)
1962  {
1963  const auto& m = _etrans_inv.at(cell::type::interval)[0];
1964  const auto& v_size_t = std::get<0>(m);
1965  const auto& v_t = std::get<1>(m);
1966  const auto& matrix = std::get<2>(m);
1967  precompute::apply_matrix_to_transpose(
1968  xtl::span(v_size_t), xtl::span(v_t),
1969  impl::cmdspan2_t(matrix.data(), matrix.shape(0), matrix.shape(1)),
1970  data, dofstart, block_size);
1971  }
1972  dofstart += _num_edofs[1][e];
1973  }
1974 
1975  if (_cell_tdim == 3)
1976  {
1977  // Permute DOFs on faces
1978  for (std::size_t f = 0; f < _num_edofs[2].size(); ++f)
1979  {
1980  // Rotate a face
1981  for (std::uint32_t r = 0; r < (cell_info >> (3 * f + 1) & 3); ++r)
1982  {
1983  const auto& m = _etrans_inv.at(_cell_subentity_types[2][f])[0];
1984  const auto& v_size_t = std::get<0>(m);
1985  const auto& v_t = std::get<1>(m);
1986  const auto& matrix = std::get<2>(m);
1987  precompute::apply_matrix_to_transpose(
1988  xtl::span(v_size_t), xtl::span(v_t),
1989  impl::cmdspan2_t(matrix.data(), matrix.shape(0), matrix.shape(1)),
1990  data, dofstart, block_size);
1991  }
1992 
1993  // Reflect a face
1994  if (cell_info >> (3 * f) & 1)
1995  {
1996  const auto& m = _etrans_inv.at(_cell_subentity_types[2][f])[1];
1997  const auto& v_size_t = std::get<0>(m);
1998  const auto& v_t = std::get<1>(m);
1999  const auto& matrix = std::get<2>(m);
2000  precompute::apply_matrix_to_transpose(
2001  xtl::span(v_size_t), xtl::span(v_t),
2002  impl::cmdspan2_t(matrix.data(), matrix.shape(0), matrix.shape(1)),
2003  data, dofstart, block_size);
2004  }
2005  dofstart += _num_edofs[2][f];
2006  }
2007  }
2008  }
2009 }
