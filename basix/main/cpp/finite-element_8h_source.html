551 template <typename T, typename E>
552 void FiniteElement::map_push_forward_m(const xt::xtensor<T, 3>& U,
553  const xt::xtensor<double, 3>& J,
554  const tcb::span<const double>& detJ,
555  const xt::xtensor<double, 3>& K,
556  E&& u) const
557 {
558  // FIXME: Should U.shape(2) be replaced by the physical value size?
559  // Can it differ?
560  // std::array<std::size_t, 3> s = {U.shape(0), U.shape(1), U.shape(2)};
561  // auto _u = xt::adapt(u, s[0] * s[1] * s[2], xt::no_ownership(), s);
562 
563  // Loop over each point
564  for (std::size_t p = 0; p < U.shape(0); ++p)
565  {
566  auto J_p = xt::view(J, p, xt::all(), xt::all());
567  auto K_p = xt::view(K, p, xt::all(), xt::all());
568 
569  // Loop over values at each point
570  for (std::size_t i = 0; i < U.shape(1); ++i)
571  {
572  auto U_data = xt::view(U, p, i, xt::all());
573  auto u_data = xt::view(u, p, i, xt::all());
574  maps::apply_map(u_data, U_data, J_p, detJ[p], K_p, map_type);
575  }
576  }
577 }

579 template <typename T, typename E>
580 void FiniteElement::map_pull_back_m(const xt::xtensor<T, 3>& u,
581  const xt::xtensor<double, 3>& J,
582  const tcb::span<const double>& detJ,
583  const xt::xtensor<double, 3>& K,
584  E&& U) const
585 {
586  // Loop over each point
587  for (std::size_t p = 0; p < u.shape(0); ++p)
588  {
589  auto J_p = xt::view(J, p, xt::all(), xt::all());
590  auto K_p = xt::view(K, p, xt::all(), xt::all());
591 
592  // Loop over each item at point to be transformed
593  for (std::size_t i = 0; i < u.shape(1); ++i)
594  {
595  auto u_data = xt::view(u, p, i, xt::all());
596  auto U_data = xt::view(U, p, i, xt::all());
597  maps::apply_map(U_data, u_data, K_p, 1.0 / detJ[p], J_p, map_type);
598  }
599  }
