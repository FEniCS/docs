527 template <typename T, typename E>
528 void FiniteElement::map_push_forward_m(const xt::xtensor<T, 3>& U,
529  const xt::xtensor<double, 3>& J,
530  const tcb::span<const double>& detJ,
531  const xt::xtensor<double, 3>& K,
532  E&& u) const
533 {
534  // FIXME: Should U.shape(2) be replaced by the physical value size?
535  // Can it differ?
536  // std::array<std::size_t, 3> s = {U.shape(0), U.shape(1), U.shape(2)};
537  // auto _u = xt::adapt(u, s[0] * s[1] * s[2], xt::no_ownership(), s);
538 
539  // Loop over each point
540  for (std::size_t p = 0; p < U.shape(0); ++p)
541  {
542  auto J_p = xt::view(J, p, xt::all(), xt::all());
543  auto K_p = xt::view(K, p, xt::all(), xt::all());
544 
545  // Loop over values at each point
546  for (std::size_t i = 0; i < U.shape(1); ++i)
547  {
548  auto U_data = xt::view(U, p, i, xt::all());
549  auto u_data = xt::view(u, p, i, xt::all());
550  maps::apply_map(u_data, U_data, J_p, detJ[p], K_p, map_type);
551  }
552  }
553 }

555 template <typename T, typename E>
556 void FiniteElement::map_pull_back_m(const xt::xtensor<T, 3>& u,
557  const xt::xtensor<double, 3>& J,
558  const tcb::span<const double>& detJ,
559  const xt::xtensor<double, 3>& K,
560  E&& U) const
561 {
562  // Loop over each point
563  for (std::size_t p = 0; p < u.shape(0); ++p)
564  {
565  auto J_p = xt::view(J, p, xt::all(), xt::all());
566  auto K_p = xt::view(K, p, xt::all(), xt::all());
567 
568  // Loop over each item at point to be transformed
569  for (std::size_t i = 0; i < u.shape(1); ++i)
570  {
571  auto u_data = xt::view(u, p, i, xt::all());
572  auto U_data = xt::view(U, p, i, xt::all());
573  maps::apply_map(U_data, u_data, K_p, 1.0 / detJ[p], J_p, map_type);
574  }
575  }
576 }
