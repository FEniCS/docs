478 template <typename T>
479 void FiniteElement::map_push_forward_m(
480  const Eigen::Array<T, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor>&
481  reference_data,
482  const Eigen::Array<double, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor>&
483  J,
484  const Eigen::ArrayXd& detJ,
485  const Eigen::Array<double, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor>&
486  K,
487  T* physical_data) const
488 {
489  const int reference_dim = cell::topological_dimension(_cell_type);
490  const int physical_dim = J.cols() / reference_dim;
491  const int physical_value_size
492  = compute_value_size(_mapping_type, physical_dim);
493  const int reference_value_size = value_size();
494  const int nresults = reference_data.cols() / reference_value_size;
495  const int npoints = reference_data.rows();
496 
497  for (int pt = 0; pt < npoints; ++pt)
498  {
499  Eigen::Map<
500  const Eigen::Array<T, Eigen::Dynamic, Eigen::Dynamic, Eigen::ColMajor>>
501  reference_block(reference_data.row(pt).data(), reference_value_size,
502  nresults);
503  Eigen::Map<Eigen::Array<T, Eigen::Dynamic, Eigen::Dynamic, Eigen::ColMajor>>
504  physical_block(physical_data + pt * physical_value_size * nresults,
505  physical_value_size, nresults);
506  Eigen::Map<const Eigen::Matrix<double, Eigen::Dynamic, Eigen::Dynamic,
507  Eigen::RowMajor>>
508  current_J(J.row(pt).data(), physical_dim, reference_dim);
509  Eigen::Map<const Eigen::Matrix<double, Eigen::Dynamic, Eigen::Dynamic,
510  Eigen::RowMajor>>
511  current_K(K.row(pt).data(), reference_dim, physical_dim);
512  for (int i = 0; i < reference_block.cols(); ++i)
513  {
514  if constexpr (std::is_same<T, double>::value)
515  {
516  for (int i = 0; i < reference_block.cols(); ++i)
517  physical_block.col(i) = _map_push_forward(
518  reference_block.col(i), current_J, detJ[pt], current_K);
519  }
520  else
521  {
522  physical_block.col(i).real() = _map_push_forward(
523  reference_block.col(i).real(), current_J, detJ[pt], current_K);
524  physical_block.col(i).imag() = _map_push_forward(
525  reference_block.col(i).imag(), current_J, detJ[pt], current_K);
526  }
527  }
528  }
529 }

531 template <typename T>
532 void FiniteElement::map_pull_back_m(
533  const Eigen::Array<T, Eigen::Dynamic, Eigen::Dynamic>& physical_data,
534  const Eigen::Array<double, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor>&
535  J,
536  const Eigen::ArrayXd& detJ,
537  const Eigen::Array<double, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor>&
538  K,
539  T* reference_data) const
540 {
541  const int reference_dim = cell::topological_dimension(_cell_type);
542  const int physical_dim = J.cols() / reference_dim;
543  const int physical_value_size
544  = compute_value_size(_mapping_type, physical_dim);
545  const int reference_value_size = value_size();
546  const int nresults = physical_data.cols() / physical_value_size;
547  const int npoints = physical_data.rows();
548 
549  Eigen::Map<Eigen::Array<T, Eigen::Dynamic, Eigen::Dynamic, Eigen::ColMajor>>
550  reference_array(reference_data, nresults * npoints, reference_value_size);
551 
552  for (int pt = 0; pt < npoints; ++pt)
553  {
554  Eigen::Map<const Eigen::Matrix<double, Eigen::Dynamic, Eigen::Dynamic,
555  Eigen::RowMajor>>
556  current_J(J.row(pt).data(), physical_dim, reference_dim);
557  Eigen::Map<const Eigen::Matrix<double, Eigen::Dynamic, Eigen::Dynamic,
558  Eigen::RowMajor>>
559  current_K(K.row(pt).data(), reference_dim, physical_dim);
560  for (int i = 0; i < nresults; ++i)
561  {
562  if constexpr (std::is_same<T, double>::value)
563  {
564  reference_array.row(pt * nresults + i)
565  = _map_push_forward(physical_data.row(pt * nresults + i), current_K,
566  1 / detJ[pt], current_J);
567  }
568  else
569  {
570  reference_array.row(pt * nresults + i).real()
571  = _map_push_forward(physical_data.row(pt * nresults + i).real(),
572  current_K, 1 / detJ[pt], current_J);
573  reference_array.row(pt * nresults + i).imag()
574  = _map_push_forward(physical_data.row(pt * nresults + i).imag(),
575  current_K, 1 / detJ[pt], current_J);
576  }
577  }
578  }
579 }
