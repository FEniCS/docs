685 template <typename T, typename E>
686 void FiniteElement::map_push_forward_m(const xt::xtensor<T, 3>& U,
687  const xt::xtensor<double, 3>& J,
688  const xtl::span<const double>& detJ,
689  const xt::xtensor<double, 3>& K,
690  E&& u) const
691 {
692  // FIXME: Should U.shape(2) be replaced by the physical value size?
693  // Can it differ?
694  // std::array<std::size_t, 3> s = {U.shape(0), U.shape(1), U.shape(2)};
695  // auto _u = xt::adapt(u, s[0] * s[1] * s[2], xt::no_ownership(), s);
696 
697  // Loop over each point
698  for (std::size_t p = 0; p < U.shape(0); ++p)
699  {
700  auto J_p = xt::view(J, p, xt::all(), xt::all());
701  auto K_p = xt::view(K, p, xt::all(), xt::all());
702 
703  // Loop over values at each point
704  for (std::size_t i = 0; i < U.shape(1); ++i)
705  {
706  auto U_data = xt::view(U, p, i, xt::all());
707  auto u_data = xt::view(u, p, i, xt::all());
708  maps::apply_map(u_data, U_data, J_p, detJ[p], K_p, map_type);
709  }
710  }
711 }

713 template <typename T, typename E>
714 void FiniteElement::map_pull_back_m(const xt::xtensor<T, 3>& u,
715  const xt::xtensor<double, 3>& J,
716  const xtl::span<const double>& detJ,
717  const xt::xtensor<double, 3>& K,
718  E&& U) const
719 {
720  // Loop over each point
721  for (std::size_t p = 0; p < u.shape(0); ++p)
722  {
723  auto J_p = xt::view(J, p, xt::all(), xt::all());
724  auto K_p = xt::view(K, p, xt::all(), xt::all());
725 
726  // Loop over each item at point to be transformed
727  for (std::size_t i = 0; i < u.shape(1); ++i)
728  {
729  auto u_data = xt::view(u, p, i, xt::all());
730  auto U_data = xt::view(U, p, i, xt::all());
731  maps::apply_map(U_data, u_data, K_p, 1.0 / detJ[p], J_p, map_type);
732  }
733  }

736 template <typename T>
737 void FiniteElement::apply_dof_transformation(const xtl::span<T>& data,
738  int block_size,
739  std::uint32_t cell_info) const
740 {
741  if (_dof_transformations_are_identity)
742  return;
743 
744  if (_cell_tdim >= 2)
745  {
746  // This assumes 3 bits are used per face. This will need updating if
747  // 3D cells with faces with more than 4 sides are implemented
748  int face_start = _cell_tdim == 3 ? 3 * _num_edofs[2].size() : 0;
749  int dofstart
750  = std::accumulate(_num_edofs[0].cbegin(), _num_edofs[0].cend(), 0);
751 
752  // Transform DOFs on edges
753  for (std::size_t e = 0; e < _num_edofs[1].size(); ++e)
754  {
755  // Reverse an edge
756  if (cell_info >> (face_start + e) & 1)
757  precompute::apply_matrix(_etrans.at(cell::type::interval)[0], data,
758  dofstart, block_size);
759  dofstart += _num_edofs[1][e];
760  }
761 
762  if (_cell_tdim == 3)
763  {
764  // Permute DOFs on faces
765  for (std::size_t f = 0; f < _num_edofs[2].size(); ++f)
766  {
767  // Reflect a face
768  if (cell_info >> (3 * f) & 1)
769  precompute::apply_matrix(_etrans.at(_cell_subentity_types[2][f])[1],
770  data, dofstart, block_size);
771 
772  // Rotate a face
773  for (std::uint32_t r = 0; r < (cell_info >> (3 * f + 1) & 3); ++r)
774  precompute::apply_matrix(_etrans.at(_cell_subentity_types[2][f])[0],
775  data, dofstart, block_size);
776  dofstart += _num_edofs[2][f];
777  }
778  }
779  }
780 }
