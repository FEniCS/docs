436  template <typename O, typename P, typename Q, typename R>
437  std::function<void(O&, const P&, const Q&, double, const R&)> map_fn() const
438  {
439  switch (_map_type)
440  {
441  case maps::type::identity:
442  return [](O& u, const P& U, const Q&, double, const R&) { u.assign(U); };
443  case maps::type::covariantPiola:
444  return [](O& u, const P& U, const Q& J, double detJ, const R& K)
445  { maps::covariant_piola(u, U, J, detJ, K); };
446  case maps::type::contravariantPiola:
447  return [](O& u, const P& U, const Q& J, double detJ, const R& K)
448  { maps::contravariant_piola(u, U, J, detJ, K); };
449  case maps::type::doubleCovariantPiola:
450  return [](O& u, const P& U, const Q& J, double detJ, const R& K)
451  { maps::double_covariant_piola(u, U, J, detJ, K); };
452  case maps::type::doubleContravariantPiola:
453  return [](O& u, const P& U, const Q& J, double detJ, const R& K)
454  { maps::double_contravariant_piola(u, U, J, detJ, K); };
455  default:
456  throw std::runtime_error("Map not implemented");
457  }

1037 template <typename T>
1038 void FiniteElement::apply_dof_transformation(const xtl::span<T>& data,
1039  int block_size,
1040  std::uint32_t cell_info) const
1041 {
1042  if (_dof_transformations_are_identity)
1043  return;
1044 
1045  if (_cell_tdim >= 2)
1046  {
1047  // This assumes 3 bits are used per face. This will need updating if
1048  // 3D cells with faces with more than 4 sides are implemented
1049  int face_start = _cell_tdim == 3 ? 3 * _num_edofs[2].size() : 0;
1050  int dofstart
1051  = std::accumulate(_num_edofs[0].cbegin(), _num_edofs[0].cend(), 0);
1052 
1053  // Transform DOFs on edges
1054  for (std::size_t e = 0; e < _num_edofs[1].size(); ++e)
1055  {
1056  // Reverse an edge
1057  if (cell_info >> (face_start + e) & 1)
1058  precompute::apply_matrix(_etrans.at(cell::type::interval)[0], data,
1059  dofstart, block_size);
1060  dofstart += _num_edofs[1][e];
1061  }
1062 
1063  if (_cell_tdim == 3)
1064  {
1065  // Permute DOFs on faces
1066  for (std::size_t f = 0; f < _num_edofs[2].size(); ++f)
1067  {
1068  // Reflect a face
1069  if (cell_info >> (3 * f) & 1)
1070  precompute::apply_matrix(_etrans.at(_cell_subentity_types[2][f])[1],
1071  data, dofstart, block_size);
1072 
1073  // Rotate a face
1074  for (std::uint32_t r = 0; r < (cell_info >> (3 * f + 1) & 3); ++r)
1075  precompute::apply_matrix(_etrans.at(_cell_subentity_types[2][f])[0],
1076  data, dofstart, block_size);
1077  dofstart += _num_edofs[2][f];
1078  }
1079  }
1080  }
1081 }
