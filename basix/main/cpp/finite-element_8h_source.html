601  template <typename O, typename P, typename Q, typename R>
602  std::function<void(O&, const P&, const Q&, double, const R&)> map_fn() const
603  {
604  switch (_map_type)
605  {
606  case maps::type::identity:
607  return [](O& u, const P& U, const Q&, double, const R&)
608  {
609  assert(U.extent(0) == u.extent(0));
610  assert(U.extent(1) == u.extent(1));
611  for (std::size_t i = 0; i < U.extent(0); ++i)
612  for (std::size_t j = 0; j < U.extent(1); ++j)
613  u(i, j) = U(i, j);
614  };
615  case maps::type::covariantPiola:
616  return [](O& u, const P& U, const Q& J, double detJ, const R& K)
617  { maps::covariant_piola(u, U, J, detJ, K); };
618  case maps::type::contravariantPiola:
619  return [](O& u, const P& U, const Q& J, double detJ, const R& K)
620  { maps::contravariant_piola(u, U, J, detJ, K); };
621  case maps::type::doubleCovariantPiola:
622  return [](O& u, const P& U, const Q& J, double detJ, const R& K)
623  { maps::double_covariant_piola(u, U, J, detJ, K); };
624  case maps::type::doubleContravariantPiola:
625  return [](O& u, const P& U, const Q& J, double detJ, const R& K)
626  { maps::double_contravariant_piola(u, U, J, detJ, K); };
627  default:
628  throw std::runtime_error("Map not implemented");
629  }
630  }

1281 template <typename T, bool post>
1282 void FiniteElement::permute_data(
1283  std::span<T> data, int block_size, std::uint32_t cell_info,
1284  const std::map<cell::type, std::vector<std::vector<std::size_t>>>& eperm)
1285  const
1286 {
1287  if (_cell_tdim >= 2)
1288  {
1289  // This assumes 3 bits are used per face. This will need updating if 3D
1290  // cells with faces with more than 4 sides are implemented
1291  int face_start = _cell_tdim == 3 ? 3 * _edofs[2].size() : 0;
1292 
1293  // Permute DOFs on edges
1294  {
1295  auto& trans = eperm.at(cell::type::interval)[0];
1296  for (std::size_t e = 0; e < _edofs[1].size(); ++e)
1297  {
1298  // Reverse an edge
1299  if (cell_info >> (face_start + e) & 1)
1300  precompute::apply_permutation_mapped(trans, data, _edofs[1][e],
1301  block_size);
1302  }
1303  }
1304 
1305  if (_cell_tdim == 3)
1306  {
1307  // Permute DOFs on faces
1308  for (std::size_t f = 0; f < _edofs[2].size(); ++f)
1309  {
1310  auto& trans = eperm.at(_cell_subentity_types[2][f]);
1311 
1312  // Reflect a face (pre rotate)
1313  if (!post and cell_info >> (3 * f) & 1)
1314  precompute::apply_permutation_mapped(trans[1], data, _edofs[2][f],
1315  block_size);
1316 
1317  // Rotate a face
1318  for (std::uint32_t r = 0; r < (cell_info >> (3 * f + 1) & 3); ++r)
1319  precompute::apply_permutation_mapped(trans[0], data, _edofs[2][f],
1320  block_size);
1321 
1322  // Reflect a face (post rotate)
1323  if (post and cell_info >> (3 * f) & 1)
1324  precompute::apply_permutation_mapped(trans[1], data, _edofs[2][f],
1325  block_size);
1326  }
1327  }
1328  }
1329 }

1331 template <typename T, bool post, typename OP>
1332 void FiniteElement::transform_data(
1333  std::span<T> data, int block_size, std::uint32_t cell_info,
1334  const std::map<cell::type, trans_data_t>& etrans, OP op) const
1335 {
1336  if (_cell_tdim >= 2)
1337  {
1338  // This assumes 3 bits are used per face. This will need updating if
1339  // 3D cells with faces with more than 4 sides are implemented
1340  int face_start = _cell_tdim == 3 ? 3 * _edofs[2].size() : 0;
1341  int dofstart = 0;
1342  for (auto& edofs0 : _edofs[0])
1343  dofstart += edofs0.size();
