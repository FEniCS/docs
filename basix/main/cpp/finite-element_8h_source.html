408  template <typename O, typename P, typename Q, typename R>
409  std::function<void(O&, const P&, const Q&, double, const R&)> map_fn() const
410  {
411  switch (_map_type)
412  {
413  case maps::type::identity:
414  return [](O& u, const P& U, const Q&, double, const R&) { u.assign(U); };
415  case maps::type::covariantPiola:
416  return [](O& u, const P& U, const Q& J, double detJ, const R& K)
417  { maps::covariant_piola(u, U, J, detJ, K); };
418  case maps::type::contravariantPiola:
419  return [](O& u, const P& U, const Q& J, double detJ, const R& K)
420  { maps::contravariant_piola(u, U, J, detJ, K); };
421  case maps::type::doubleCovariantPiola:
422  return [](O& u, const P& U, const Q& J, double detJ, const R& K)
423  { maps::double_covariant_piola(u, U, J, detJ, K); };
424  case maps::type::doubleContravariantPiola:
425  return [](O& u, const P& U, const Q& J, double detJ, const R& K)
426  { maps::double_contravariant_piola(u, U, J, detJ, K); };
427  default:
428  throw std::runtime_error("Map not implemented");
429  }

899 template <typename T>
900 void FiniteElement::apply_dof_transformation(const xtl::span<T>& data,
901  int block_size,
902  std::uint32_t cell_info) const
903 {
904  if (_dof_transformations_are_identity)
905  return;
906 
907  if (_cell_tdim >= 2)
908  {
909  // This assumes 3 bits are used per face. This will need updating if
910  // 3D cells with faces with more than 4 sides are implemented
911  int face_start = _cell_tdim == 3 ? 3 * _num_edofs[2].size() : 0;
912  int dofstart
913  = std::accumulate(_num_edofs[0].cbegin(), _num_edofs[0].cend(), 0);
914 
915  // Transform DOFs on edges
916  for (std::size_t e = 0; e < _num_edofs[1].size(); ++e)
917  {
918  // Reverse an edge
919  if (cell_info >> (face_start + e) & 1)
920  precompute::apply_matrix(_etrans.at(cell::type::interval)[0], data,
921  dofstart, block_size);
922  dofstart += _num_edofs[1][e];
923  }
924 
925  if (_cell_tdim == 3)
926  {
927  // Permute DOFs on faces
928  for (std::size_t f = 0; f < _num_edofs[2].size(); ++f)
929  {
930  // Reflect a face
931  if (cell_info >> (3 * f) & 1)
932  precompute::apply_matrix(_etrans.at(_cell_subentity_types[2][f])[1],
933  data, dofstart, block_size);
934 
935  // Rotate a face
936  for (std::uint32_t r = 0; r < (cell_info >> (3 * f + 1) & 3); ++r)
937  precompute::apply_matrix(_etrans.at(_cell_subentity_types[2][f])[0],
938  data, dofstart, block_size);
939  dofstart += _num_edofs[2][f];
940  }
941  }
942  }
943 }
