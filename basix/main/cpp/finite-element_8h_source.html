510 template <typename T, typename E>
511 void FiniteElement::map_push_forward_m(const xt::xtensor<T, 3>& U,
512  const xt::xtensor<double, 3>& J,
513  const tcb::span<const double>& detJ,
514  const xt::xtensor<double, 3>& K,
515  E&& u) const
516 {
517  // FIXME: Should U.shape(2) be replaced by the physical value size?
518  // Can it differ?
519  // std::array<std::size_t, 3> s = {U.shape(0), U.shape(1), U.shape(2)};
520  // auto _u = xt::adapt(u, s[0] * s[1] * s[2], xt::no_ownership(), s);
521 
522  // Loop over each point
523  for (std::size_t p = 0; p < U.shape(0); ++p)
524  {
525  auto J_p = xt::view(J, p, xt::all(), xt::all());
526  auto K_p = xt::view(K, p, xt::all(), xt::all());
527 
528  // Loop over values at each point
529  for (std::size_t i = 0; i < U.shape(1); ++i)
530  {
531  // Note: we assign here to a xt::xtensor<double, 1> until the
532  // maps are updated to accept xtensor objects rather than spans
533  // auto U_data = xt::row(U_b, i);
534  xt::xtensor<T, 1> U_data = xt::view(U, p, i, xt::all());
535  std::vector<T> f
536  = mapping::apply_map<T>(U_data, J_p, detJ[p], K_p, map_type);
537  for (std::size_t j = 0; j < f.size(); ++j)
538  u(p, i, j) = f[j];
539  }
540  }

543 template <typename T, typename E>
544 void FiniteElement::map_pull_back_m(const xt::xtensor<T, 3>& u,
545  const xt::xtensor<double, 3>& J,
546  const tcb::span<const double>& detJ,
547  const xt::xtensor<double, 3>& K,
548  E&& U) const
549 {
550  // Loop over each point
551  for (std::size_t p = 0; p < u.shape(0); ++p)
552  {
553  auto J_p = xt::view(J, p, xt::all(), xt::all());
554  auto K_p = xt::view(K, p, xt::all(), xt::all());
555 
556  // Loop over each item at point to be transformed
557  for (std::size_t i = 0; i < u.shape(1); ++i)
558  {
559  // Note: we assign here to a xt::xtensor<double, 1> until the
560  // maps are updated to accept xtensor objects rather than spans
561  // auto U_data = xt::row(U_b, i);
562 
563  // Map data
564  xt::xtensor<T, 1> u_data = xt::view(u, p, i, xt::all());
565  std::vector<T> f
566  = mapping::apply_map<T>(u_data, K_p, 1.0 / detJ[p], J_p, map_type);
567  for (std::size_t j = 0; j < f.size(); ++j)
568  U(p, i, j) = f[j];
569  }
570  }
571 }
