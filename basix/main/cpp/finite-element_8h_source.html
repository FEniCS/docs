557  template <typename O, typename P, typename Q, typename R>
558  std::function<void(O&, const P&, const Q&, double, const R&)> map_fn() const
559  {
560  switch (_map_type)
561  {
562  case maps::type::identity:
563  return [](O& u, const P& U, const Q&, double, const R&) { u.assign(U); };
564  case maps::type::L2Piola:
565  return [](O& u, const P& U, const Q& J, double detJ, const R& K)
566  { maps::l2_piola(u, U, J, detJ, K); };
567  case maps::type::covariantPiola:
568  return [](O& u, const P& U, const Q& J, double detJ, const R& K)
569  { maps::covariant_piola(u, U, J, detJ, K); };
570  case maps::type::contravariantPiola:
571  return [](O& u, const P& U, const Q& J, double detJ, const R& K)
572  { maps::contravariant_piola(u, U, J, detJ, K); };
573  case maps::type::doubleCovariantPiola:
574  return [](O& u, const P& U, const Q& J, double detJ, const R& K)
575  { maps::double_covariant_piola(u, U, J, detJ, K); };
576  case maps::type::doubleContravariantPiola:
577  return [](O& u, const P& U, const Q& J, double detJ, const R& K)
578  { maps::double_contravariant_piola(u, U, J, detJ, K); };
579  default:
580  throw std::runtime_error("Map not implemented");
581  }

1279 template <typename T>
1280 void FiniteElement::apply_dof_transformation(const xtl::span<T>& data,
1281  int block_size,
1282  std::uint32_t cell_info) const
1283 {
1284  if (_dof_transformations_are_identity)
1285  return;
1286 
1287  if (_cell_tdim >= 2)
1288  {
1289  // This assumes 3 bits are used per face. This will need updating if
1290  // 3D cells with faces with more than 4 sides are implemented
1291  int face_start = _cell_tdim == 3 ? 3 * _num_edofs[2].size() : 0;
1292  int dofstart
1293  = std::accumulate(_num_edofs[0].cbegin(), _num_edofs[0].cend(), 0);
1294 
1295  // Transform DOFs on edges
1296  for (std::size_t e = 0; e < _num_edofs[1].size(); ++e)
1297  {
1298  // Reverse an edge
1299  if (cell_info >> (face_start + e) & 1)
1300  precompute::apply_matrix(_etrans.at(cell::type::interval)[0], data,
1301  dofstart, block_size);
1302  dofstart += _num_edofs[1][e];
1303  }
1304 
1305  if (_cell_tdim == 3)
1306  {
1307  // Permute DOFs on faces
1308  for (std::size_t f = 0; f < _num_edofs[2].size(); ++f)
1309  {
1310  // Reflect a face
1311  if (cell_info >> (3 * f) & 1)
1312  precompute::apply_matrix(_etrans.at(_cell_subentity_types[2][f])[1],
1313  data, dofstart, block_size);
1314 
1315  // Rotate a face
1316  for (std::uint32_t r = 0; r < (cell_info >> (3 * f + 1) & 3); ++r)
1317  precompute::apply_matrix(_etrans.at(_cell_subentity_types[2][f])[0],
1318  data, dofstart, block_size);
1319  dofstart += _num_edofs[2][f];
1320  }
1321  }
1322  }
1323 }

1331 
1332  if (_cell_tdim >= 2)

1334  // This assumes 3 bits are used per face. This will need updating if

1336  int face_start = _cell_tdim == 3 ? 3 * _num_edofs[2].size() : 0;
1337  int dofstart

1339 
1340  // Transform DOFs on edges
