398  template <typename O, typename P, typename Q, typename S, typename T>
399  void map_push_forward_m(const O& U, const P& J, const Q& detJ, const S& K,
400  T&& u) const
401  {
402  // FIXME: Should U.shape(2) be replaced by the physical value size?
403  // Can it differ?
404 
405  // Loop over points that share J
406  for (std::size_t i = 0; i < U.shape(0); ++i)
407  {
408  auto _J = xt::view(J, i, xt::all(), xt::all());
409  auto _K = xt::view(K, i, xt::all(), xt::all());
410  auto _U = xt::view(U, i, xt::all(), xt::all());
411  auto _u = xt::view(u, i, xt::all(), xt::all());
412  maps::apply_map(_u, _U, _J, detJ[i], _K, map_type);
413  }

447  void map_pull_back_m(const O& u, const P& J, const Q& detJ, const S& K,
448  T&& U) const
449  {
450  // Loop over points that share K and K
451  for (std::size_t i = 0; i < u.shape(0); ++i)
452  {
453  auto _J = xt::view(J, i, xt::all(), xt::all());
454  auto _K = xt::view(K, i, xt::all(), xt::all());
455  auto _u = xt::view(u, i, xt::all(), xt::all());
456  auto _U = xt::view(U, i, xt::all(), xt::all());
457  maps::apply_map(_U, _u, _K, 1.0 / detJ[i], _J, map_type);
458  }

892 template <typename T>
893 void FiniteElement::apply_dof_transformation(const xtl::span<T>& data,
894  int block_size,
895  std::uint32_t cell_info) const
896 {
897  if (_dof_transformations_are_identity)
898  return;
899 
900  if (_cell_tdim >= 2)
901  {
902  // This assumes 3 bits are used per face. This will need updating if
903  // 3D cells with faces with more than 4 sides are implemented
904  int face_start = _cell_tdim == 3 ? 3 * _num_edofs[2].size() : 0;
905  int dofstart
906  = std::accumulate(_num_edofs[0].cbegin(), _num_edofs[0].cend(), 0);
907 
908  // Transform DOFs on edges
909  for (std::size_t e = 0; e < _num_edofs[1].size(); ++e)
910  {
911  // Reverse an edge
912  if (cell_info >> (face_start + e) & 1)
913  precompute::apply_matrix(_etrans.at(cell::type::interval)[0], data,
914  dofstart, block_size);
915  dofstart += _num_edofs[1][e];
916  }
917 
918  if (_cell_tdim == 3)
919  {
920  // Permute DOFs on faces
921  for (std::size_t f = 0; f < _num_edofs[2].size(); ++f)
922  {
923  // Reflect a face
924  if (cell_info >> (3 * f) & 1)
925  precompute::apply_matrix(_etrans.at(_cell_subentity_types[2][f])[1],
926  data, dofstart, block_size);
927 
928  // Rotate a face
929  for (std::uint32_t r = 0; r < (cell_info >> (3 * f + 1) & 3); ++r)
930  precompute::apply_matrix(_etrans.at(_cell_subentity_types[2][f])[0],
931  data, dofstart, block_size);
932  dofstart += _num_edofs[2][f];
933  }
934  }
935  }
936 }

1279 
1280  // Compute coefficients = Pi * x (matrix-vector multiply)

1282  assert(Pi.size() % rows == 0);

1284  for (std::size_t i = 0; i < rows; ++i)
1285  {

1287  // Dot product between row i of the matrix and 'data'

1289  std::next(Pi.data(), i * cols + cols),
