51 template <typename U, typename V>
52 xt::xtensor<typename U::value_type, 2> dot(const U& A, const V& B)
53 {
54  xt::xtensor<typename U::value_type, 2> C
55  = xt::zeros<typename U::value_type>({A.shape(0), B.shape(1)});
56 
57  assert(A.shape(1) == B.shape(0));
58  for (std::size_t i = 0; i < A.shape(0); i++)
59  for (std::size_t j = 0; j < B.shape(1); j++)
60  for (std::size_t k = 0; k < A.shape(1); k++)
61  C(i, j) += A(i, k) * B(k, j);
62 
63  return C;
64 }
