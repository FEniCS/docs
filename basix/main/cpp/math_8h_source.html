134 template <typename U, typename V>
135 std::pair<std::vector<double>, std::array<std::size_t, 2>> dot_new(const U& A,
136  const V& B)
137 {
138  std::vector<double> C(A.extent(0) * B.extent(1));
139  std::array<std::size_t, 2> shape = {A.extent(0), B.extent(1)};
140  std::experimental::mdspan<double, std::experimental::dextents<std::size_t, 2>>
141  _C(C.data(), shape);
142  for (std::size_t i = 0; i < shape[0]; ++i)
143  for (std::size_t j = 0; j < shape[1]; ++j)
144  for (std::size_t k = 0; k < A.extent(1); ++k)
145  _C(i, j) += A(i, k) * B(k, j);
146  return {std::move(C), std::move(shape)};
147 }
