37 template <typename U, typename V>
38 std::pair<std::vector<typename U::value_type>, std::array<std::size_t, 2>>
39 outer(const U& u, const V& v)
40 {
41  std::vector<typename U::value_type> result(u.size() * v.size());
42  for (std::size_t i = 0; i < u.size(); ++i)
43  for (std::size_t j = 0; j < v.size(); ++j)
44  result[i * v.size() + j] = u[i] * v[j];
45 
46  return {std::move(result), {u.size(), v.size()}};
47 }

92 template <typename U, typename V, typename W>
93 void dot(const U& A, const V& B, W&& C)
94 {
95  assert(A.extent(1) == B.extent(0));
96  assert(C.extent(0) == C.extent(0));
97  assert(C.extent(1) == B.extent(1));
98  if (A.extent(0) * B.extent(1) * A.extent(1) < 4096)
99  {
100  for (std::size_t i = 0; i < A.extent(0); ++i)
101  for (std::size_t j = 0; j < B.extent(1); ++j)
102  for (std::size_t k = 0; k < A.extent(1); ++k)
103  C(i, j) += A(i, k) * B(k, j);
104  }
105  else
106  {
107  impl::dot_blas(A, {A.extent(0), A.extent(1)}, B, {B.extent(0), B.extent(1)},
108  C);
109  }
110 }
