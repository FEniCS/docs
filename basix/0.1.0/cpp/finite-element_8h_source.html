585 void FiniteElement::map_push_forward_m(const xt::xtensor<T, 3>& U,
586  const xt::xtensor<double, 3>& J,
587  const xtl::span<const double>& detJ,
588  const xt::xtensor<double, 3>& K,
589  E&& u) const
590 {
591  // FIXME: Should U.shape(2) be replaced by the physical value size?
592  // Can it differ?
593  // std::array<std::size_t, 3> s = {U.shape(0), U.shape(1), U.shape(2)};
594  // auto _u = xt::adapt(u, s[0] * s[1] * s[2], xt::no_ownership(), s);
595 
596  // Loop over each point
597  for (std::size_t p = 0; p < U.shape(0); ++p)
598  {
599  auto J_p = xt::view(J, p, xt::all(), xt::all());
600  auto K_p = xt::view(K, p, xt::all(), xt::all());
601 
602  // Loop over values at each point
603  for (std::size_t i = 0; i < U.shape(1); ++i)
604  {
605  auto U_data = xt::view(U, p, i, xt::all());
606  auto u_data = xt::view(u, p, i, xt::all());
607  maps::apply_map(u_data, U_data, J_p, detJ[p], K_p, map_type);
608  }
609  }

612 template <typename T, typename E>
613 void FiniteElement::map_pull_back_m(const xt::xtensor<T, 3>& u,
614  const xt::xtensor<double, 3>& J,
615  const xtl::span<const double>& detJ,
616  const xt::xtensor<double, 3>& K,
617  E&& U) const
618 {
619  // Loop over each point
620  for (std::size_t p = 0; p < u.shape(0); ++p)
621  {
622  auto J_p = xt::view(J, p, xt::all(), xt::all());
623  auto K_p = xt::view(K, p, xt::all(), xt::all());
624 
625  // Loop over each item at point to be transformed
626  for (std::size_t i = 0; i < u.shape(1); ++i)
627  {
628  auto u_data = xt::view(u, p, i, xt::all());
629  auto U_data = xt::view(U, p, i, xt::all());
630  maps::apply_map(U_data, u_data, K_p, 1.0 / detJ[p], J_p, map_type);
631  }
632  }

635 template <typename T>
636 void FiniteElement::apply_dof_transformation(const xtl::span<T>& data,
637  int block_size,
638  std::uint32_t cell_info) const
639 {
640  if (_dof_transformations_are_identity)
641  return;
642 
643  if (_cell_tdim >= 2)
644  {
645  // This assumes 3 bits are used per face. This will need updating if
646  // 3D cells with faces with more than 4 sides are implemented
647  int face_start = _cell_tdim == 3 ? 3 * _edofs[2].size() : 0;
648  int dofstart = std::accumulate(_edofs[0].cbegin(), _edofs[0].cend(), 0);
649 
650  // Transform DOFs on edges
651  for (std::size_t e = 0; e < _edofs[1].size(); ++e)
652  {
653  // Reverse an edge
654  if (cell_info >> (face_start + e) & 1)
655  precompute::apply_matrix(_etrans[0], data, dofstart, block_size);
656  dofstart += _edofs[1][e];
657  }
658 
659  if (_cell_tdim == 3)
660  {
661  // Permute DOFs on faces
662  for (std::size_t f = 0; f < _edofs[2].size(); ++f)
663  {
664  // Reflect a face
665  if (cell_info >> (3 * f) & 1)
666  precompute::apply_matrix(_etrans[2], data, dofstart, block_size);
667 
668  // Rotate a face
669  for (std::uint32_t r = 0; r < (cell_info >> (3 * f + 1) & 3); ++r)
670  precompute::apply_matrix(_etrans[1], data, dofstart, block_size);
671  dofstart += _edofs[2][f];
672  }
673  }
674  }
675 }
